74void Nyx::ext_src_force(
75 amrex::Box const& bx,
76 amrex::Array4<const amrex::Real> const& old_state,
77 amrex::Array4<const amrex::Real> const& new_state,
78 amrex::Array4<const amrex::Real> const& old_diag,
79 amrex::Array4<amrex::Real> const& new_diag,

81 const amrex::Real* problo,
82 const amrex::Real* dx,
83 const amrex::Real time,
84 const amrex::Real z,
85 const amrex::Real dt)
86{
87 // Make a copy of the state so we can evolve it then throw it away
88 amrex::FArrayBox tmp_state_fab(bx, QVAR);
89 amrex::Elixir tmp_state_eli = tmp_state_fab.elixir();
90 auto const& tmp_state = tmp_state_fab.array();
91
92 // tmp_state_fab.copy<RunOn::Device>(new_state);
93 amrex::ParallelFor(bx, QVAR, [=] AMREX_GPU_DEVICE(int i, int j, int k, int n) noexcept {
94 tmp_state(i,j,k,n) = new_state(i,j,k,n);
95 });
96 amrex::Real a = 1.0 / (1.0+z);
97 amrex::Gpu::streamSynchronize();
98 // Note that when we call this routine to compute the "old" source,
99 // both "old_state" and "new_state" are acutally the "old" state.
100 // When we call this routine to compute the "new" source,
101 // both "old_state" is in fact the "old" state and
102 // "new_state" is in fact the "new" state
103
104 amrex::Real half_dt = 0.5 * dt;
105 integrate_state_force(bx, tmp_state, new_diag,
106 dx, time, a, half_dt);
107
108 // Recall that this routine is called from a tiled MFIter
109 // ! For old source: lo(:), hi(:) are the bounds of the growntilebox(src.nGrow9))
110 // For new source: lo(:), hi(:) are the bounds of the tilebox, e.g. valid region only
111 amrex::ParallelFor(bx, [=] AMREX_GPU_DEVICE(int i, int j, int k) noexcept {
112 src(i,j,k,0) = 0.0;
113 src(i,j,k,UMX) = (tmp_state(i,j,k,UMX) - new_state(i,j,k,UMX)) * a / half_dt;
114 src(i,j,k,UMY) = (tmp_state(i,j,k,UMY) - new_state(i,j,k,UMY)) * a / half_dt;
115 src(i,j,k,UMZ) = (tmp_state(i,j,k,UMZ) - new_state(i,j,k,UMZ)) * a / half_dt;
116 src(i,j,k,UEINT) = (tmp_state(i,j,k,UEINT) - new_state(i,j,k,UEINT)) * a / half_dt;
117 src(i,j,k,UEDEN) = (tmp_state(i,j,k,UEDEN) - new_state(i,j,k,UEDEN)) * a / half_dt;
118 });
