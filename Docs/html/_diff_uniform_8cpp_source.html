206int
207finestLevelCoveringDomain(const AmrData& amr_data)
208{
209 // Find the finest level covering the entire domain. Return
210 // -1 if there isn't one suitable
211 int finest_level = amr_data.FinestLevel();
212 const Vector<Box>& domain_array = amr_data.ProbDomain();
213
214 for (int iLevel=finest_level; iLevel>=0; --iLevel)
215 {
216 const BoxArray& ba = amr_data.boxArray(iLevel);
217 BoxDomain bd;
218 bd.add(BoxList(ba));
219 BoxDomain complement = amrex::complementIn(domain_array[iLevel],bd);
220 if (complement.isEmpty())
221 return iLevel;
222 }
223 return -1;
224}

226IntVect
227getRefRatio(const Box& crse,
228 const Box& fine)
229{
230 // Compute refinement ratio between crse and fine boxes, return invalid
231 // IntVect if there is none suitable
232 ParmParse pp("");
233 Vector<int> rr_in(BL_SPACEDIM,-1);
234 int Nrr = 0;
235 Nrr = pp.countval("ref_ratio",Nrr);
236 BL_ASSERT(Nrr==0 || Nrr==BL_SPACEDIM || Nrr==1);
237 if (Nrr>0)
238 {
239 pp.queryarr("ref_ratio",rr_in,0,Nrr);
240 return IntVect(rr_in);
241 }
242
243 IntVect ref_ratio;
244 for (int i=0; i<BL_SPACEDIM; ++i)
245 ref_ratio[i] = fine.size()[i] / crse.size()[i];
246
247 // Check results
248 Box test1 = ::Box(fine).coarsen(ref_ratio);
249 Box test2 = ::Box(test1).refine(ref_ratio);
250 if (test1 != crse || test2 != fine)
251 ref_ratio = IntVect();
252 return ref_ratio;
253}

256amrDatasHaveSameDerives(const AmrData& amrd1,
257 const AmrData& amrd2)
258{

260 const Vector<std::string>& derives2 = amrd2.PlotVarNames();
261 int length = derives1.size();
262 if (length != derives2.size())
263 return false;
264 for (int i=0; i<length; ++i)
