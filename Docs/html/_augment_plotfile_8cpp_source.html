65template <class T>
66static Vector<T> concatVectors(const Vector<T>& a, const Vector<T>& b)
67{
68 Vector<T> ab(a.size() + b.size());
69 for (int i = 0; i < a.size(); i++)
70 ab[i] = a[i];
71 for (int i = 0; i < b.size(); i++)
72 ab[i + a.size()] = b[i];
73 return ab;
74}

76template <class T>

79 int oldSize = a.size();
80 a.resize(oldSize + b.size());
