15 void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
16 {
17 int nx = nppc[0];
18 int ny = nppc[1];
19 int nz = nppc[2];
20
21 int ix_part = i_part/(ny * nz);
22 int iy_part = (i_part % (ny * nz)) % ny;
23 int iz_part = (i_part % (ny * nz)) / ny;
24
25 r[0] = (0.5+ix_part)/nx;
26 r[1] = (0.5+iy_part)/ny;
27 r[2] = (0.5+iz_part)/nz;
28 }

72 for (IntVect iv = tile_box.smallEnd(); iv <= tile_box.bigEnd(); tile_box.next(iv))

95 p.rdata(RealData::vz) = u[2];

106CellSortedParticleContainer::UpdateCellVectors()
107{
108 BL_PROFILE("CellSortedParticleContainer::UpdateCellVectors");
109
110 const int lev = 0;
111
112 bool needs_update = false;
113 if (not m_vectors_initialized)
114 {
115 // this is the first call, so we must update
116 m_vectors_initialized = true;
117 needs_update = true;
118 }
119 else if ((m_BARef != this->ParticleBoxArray(lev).getRefID()) or
120 (m_DMRef != this->ParticleDistributionMap(lev).getRefID()))
121 {
122 // the grids have changed, so we must update
123 m_BARef = this->ParticleBoxArray(lev).getRefID();
124 m_DMRef = this->ParticleDistributionMap(lev).getRefID();
125 needs_update = true;
126 }
127
128 if (not needs_update) return;
129
130 // clear old data
131 m_cell_vectors.clear();
132 m_vector_size.clear();
133 m_vector_ptrs.clear();
134
135 // allocate storage for cell vectors. NOTE - do not tile this loop
136 for(MFIter mfi = MakeMFIter(lev, false); mfi.isValid(); ++mfi)
137 {
138 const Box& box = mfi.validbox();
139 const int grid_id = mfi.index();
140 m_cell_vectors[grid_id].resize(box);
141 m_vector_size[grid_id].resize(box);
142 m_vector_ptrs[grid_id].resize(box);
143 }
144
145 // insert particles into vectors - this can be tiled
146#ifdef _OPENMP
147#pragma omp parallel
148#endif
149 for (MyParIter pti(*this, lev); pti.isValid(); ++pti)
150 {
151 auto& particles = pti.GetArrayOfStructs();
152 const int np = pti.numParticles();
153 for(int pindex = 0; pindex < np; ++pindex) {
154 ParticleType& p = particles[pindex];
155 const IntVect& iv = this->Index(p, lev);
156 p.idata(IntData::sorted) = 1;
157 p.idata(IntData::i) = iv[0];
158 p.idata(IntData::j) = iv[1];
159 p.idata(IntData::k) = iv[2];
160 // note - use 1-based indexing for convenience with Fortran
161 m_cell_vectors[pti.index()](iv).push_back(pindex + 1);
162 }
163 }
164
165 UpdateFortranStructures();
