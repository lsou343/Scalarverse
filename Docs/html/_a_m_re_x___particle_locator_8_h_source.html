63 int iy_hi = amrex::min((lo.y + nGrow - m_lo.y) / m_bin_size.y, m_num_bins.y-1);
64 int iz_hi = amrex::min((lo.z + nGrow - m_lo.z) / m_bin_size.z, m_num_bins.z-1);
65
66 for (int ii = ix_lo; ii <= ix_hi; ++ii) {
67 for (int jj = iy_lo; jj <= iy_hi; ++jj) {
68 for (int kk = iz_lo; kk <= iz_hi; ++kk) {
69 int index = (ii * m_num_bins.y + jj) * m_num_bins.z + kk;
