38 const Geometry& geom_, MPI_Comm comm_)
39 : comm(comm_),
40 geom(geom_)
41{
42 static_assert(AMREX_SPACEDIM > 1, "PETScABecLap: 1D not supported");
43 static_assert(std::is_same<Real, PetscScalar>::value, "amrex::Real != PetscScalar");
44#ifdef AMREX_USE_HYPRE
45 static_assert(std::is_same<HYPRE_Int, PetscInt>::value, "HYPRE_Int != PetscInt");

48 const int ncomp = 1;
49 int ngrow = 0;

54 ngrow = 1;
55#endif
56

58 BoxArray edge_boxes(grids);
59 edge_boxes.surroundingNodes(i);
60 bcoefs[i].define(edge_boxes, dmap, ncomp, ngrow);
61 bcoefs[i].setVal(0.0);
62 }
63
64 diaginv.define(grids,dmap,ncomp,0);

66 PETSC_COMM_WORLD = comm;

119void
120PETScABecLap::solve (MultiFab& soln, const MultiFab& rhs, Real rel_tol, Real abs_tol,
121 int max_iter, const BndryData& bndry, int max_bndry_order)
122{
123 Gpu::LaunchSafeGuard lsg(false); // xxxxx TODO: gpu
124 BL_PROFILE("PETScABecLap::solve()");
125
126 if (solver == nullptr || m_bndry != &bndry || m_maxorder != max_bndry_order)
127 {
128 m_bndry = &bndry;
129 m_maxorder = max_bndry_order;
130 m_factory = &(rhs.Factory());
131 prepareSolver();
132 }
133 else

135 m_factory = &(rhs.Factory());
136 }
137
138 loadVectors(soln, rhs);
139 //
140 VecAssemblyBegin(x);
141 VecAssemblyEnd(x);

146 KSPSolve(solver, b, x);
147 if (verbose >= 2)

153 amrex::Print() <<"\n" << niters << " PETSc Iterations, Residual Norm " << res << std::endl;

162 int num_procs, myid;
163 MPI_Comm_size(PETSC_COMM_WORLD, &num_procs);
164 MPI_Comm_rank(PETSC_COMM_WORLD, &myid);
165
166 const BoxArray& ba = acoefs.boxArray();
167 const DistributionMapping& dm = acoefs.DistributionMap();
168
169#if defined(AMREX_DEBUG) || defined(AMREX_TESTING)
170 if (sizeof(PetscInt) < sizeof(Long)) {
171 Long ncells_grids = ba.numPts();
172 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(ncells_grids < static_cast<Long>(std::numeric_limits<PetscInt>::max()),
173 "PetscInt is too short");
174 }
175#endif
176
177 // how many non-covered cells do we have?
178 ncells_grid.define(ba,dm);
179 cell_id.define(ba,dm,1,1);
180 cell_id_vec.define(ba,dm);
181
182#ifdef AMREX_USE_EB
183 auto ebfactory = dynamic_cast<EBFArrayBoxFactory const*>(m_factory);
184 const FabArray<EBCellFlagFab>* flags = (ebfactory) ? &(ebfactory->getMultiEBCellFlagFab()) : nullptr;
185 const MultiFab* vfrac = (ebfactory) ? &(ebfactory->getVolFrac()) : nullptr;
186 auto area = (ebfactory) ? ebfactory->getAreaFrac()
187 : Array<const MultiCutFab*,AMREX_SPACEDIM>{AMREX_D_DECL(nullptr,nullptr,nullptr)};
188 auto fcent = (ebfactory) ? ebfactory->getFaceCent()
189 : Array<const MultiCutFab*,AMREX_SPACEDIM>{AMREX_D_DECL(nullptr,nullptr,nullptr)};
190 auto barea = (ebfactory) ? &(ebfactory->getBndryArea()) : nullptr;
191 auto bcent = (ebfactory) ? &(ebfactory->getBndryCent()) : nullptr;
192#endif
193
194 PetscInt ncells_proc = 0;
195#ifdef _OPENMP
196#pragma omp parallel reduction(+:ncells_proc)
197#endif
198 { BaseFab<PetscInt> ifab;
199 for (MFIter mfi(cell_id); mfi.isValid(); ++mfi)
200 {
201 const Box& bx = mfi.validbox();
202 BaseFab<PetscInt>& cid_fab = cell_id[mfi];
203 cid_fab.setVal<RunOn::Host>(std::numeric_limits<PetscInt>::lowest());
204#ifdef AMREX_USE_EB
205 auto fabtyp = (flags) ? (*flags)[mfi].getType(bx) : FabType::regular;
206 if (fabtyp == FabType::covered)
207 {
208 ncells_grid[mfi] = 0;
209 }
210 else if (fabtyp == FabType::singlevalued)
211 {
212 amrex_hpeb_fill_cellid(bx,
213 ncells_grid[mfi],
214 cid_fab,
215 (*flags)[mfi]);
216 ncells_proc += ncells_grid[mfi];
217 }
218 else
219#endif
220 {
221 Long npts = bx.numPts();
222 ncells_grid[mfi] = npts;
223 ncells_proc += npts;
224
225 ifab.resize(bx);
226 PetscInt* p = ifab.dataPtr();
227 for (Long i = 0; i < npts; ++i) {
228 *p++ = i;
229 }
230 cid_fab.copy<RunOn::Host>(ifab,bx);
231 }
232 }
233 }
234
235 Vector<PetscInt> ncells_allprocs(num_procs);
236 MPI_Allgather(&ncells_proc, sizeof(PetscInt), MPI_CHAR,
237 ncells_allprocs.data(), sizeof(PetscInt), MPI_CHAR,
238 PETSC_COMM_WORLD);
239 PetscInt proc_begin = 0;
240 for (int i = 0; i < myid; ++i) {
241 proc_begin += ncells_allprocs[i];
242 }
243 PetscInt ncells_world = 0;
244 for (auto i : ncells_allprocs) {
245 ncells_world += i;
246 }
247
248 LayoutData<PetscInt> offset(ba,dm);
249 PetscInt proc_end = proc_begin;
250 for (MFIter mfi(ncells_grid); mfi.isValid(); ++mfi)
251 {
252 offset[mfi] = proc_end;
253 proc_end += ncells_grid[mfi];
254 }
255 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(proc_end == proc_begin+ncells_proc,
256 "PETScABecLap::prepareSolver: how did this happen?");
257
258#ifdef _OPENMP
259#pragma omp parallel
260#endif
261 for (MFIter mfi(cell_id,true); mfi.isValid(); ++mfi)
262 {
263 cell_id[mfi].plus<RunOn::Host>(offset[mfi], mfi.tilebox());
264 }
265
266 cell_id.FillBoundary(geom.periodicity());
267
268 // estimated amount of block diag elements
269 PetscInt d_nz = (eb_stencil_size + regular_stencil_size) / 2;
270 // estimated amount of block off diag elements
271 PetscInt o_nz = d_nz / 2;
272 MatCreate(PETSC_COMM_WORLD, &A);
273 MatSetType(A, MATMPIAIJ);
274 MatSetSizes(A, ncells_proc, ncells_proc, ncells_world, ncells_world);
275 MatMPIAIJSetPreallocation(A, d_nz, NULL, o_nz, NULL );
276 //Maybe an over estimate of the diag/off diag #of non-zero entries, so we turn off malloc warnings
277 MatSetUp(A);
278 MatSetOption(A, MAT_NEW_NONZERO_LOCATION_ERR, PETSC_FALSE);
279
280 // A.SetValues
281 const Real* dx = geom.CellSize();
282 const int bho = (m_maxorder > 2) ? 1 : 0;
283 FArrayBox foo(Box::TheUnitBox());
284 const int is_eb_dirichlet = m_eb_b_coeffs != nullptr;
285 for (MFIter mfi(acoefs); mfi.isValid(); ++mfi)
286 {
287 const Box& bx = mfi.validbox();
288
289#ifdef AMREX_USE_EB
290 const auto fabtyp = (flags) ? (*flags)[mfi].getType(bx) : FabType::regular;
291#else
292 const auto fabtyp = FabType::regular;
293#endif
294 if (fabtyp != FabType::covered)
295 {
296 const PetscInt max_stencil_size = (fabtyp == FabType::regular) ?
297 regular_stencil_size : eb_stencil_size;
298
299 const PetscInt nrows = ncells_grid[mfi];
300 cell_id_vec[mfi].resize(nrows);
301
302 Gpu::ManagedDeviceVector<PetscInt> ncolsg(nrows,0);
303 Gpu::ManagedDeviceVector<PetscInt> colsg(nrows*(AMREX_SPACEDIM*2+1),0);
304 Gpu::ManagedDeviceVector<Real> matg(nrows*(AMREX_SPACEDIM*2+1),0.0);
305
306 GpuArray<int,AMREX_SPACEDIM*2> bctype;
307 GpuArray<Real,AMREX_SPACEDIM*2> bcl;
308 const Vector< Vector<BoundCond> > & bcs_i = m_bndry->bndryConds(mfi);
309 const BndryData::RealTuple & bcl_i = m_bndry->bndryLocs(mfi);
310 for (OrientationIter oit; oit; oit++) {
311 int cdir(oit());
312 bctype[cdir] = bcs_i[cdir][0];
313 bcl[cdir] = bcl_i[cdir];
314 }
315
316 if (fabtyp == FabType::regular)
317 {
318 amrex_hpijmatrix(bx,
319 nrows, ncolsg.dataPtr(),
320 cell_id_vec[mfi].dataPtr(),
321 colsg.dataPtr(), matg.dataPtr(),
322 cell_id[mfi],
323 offset[mfi],
324 diaginv[mfi],
325 acoefs[mfi],
326 bcoefs[0][mfi],
327 bcoefs[1][mfi],
328#if (AMREX_SPACEDIM == 3)
329 bcoefs[2][mfi],
330#endif
331 scalar_a, scalar_b, dx,
332 bctype, bcl, bho, nullptr);
333 }
334#ifdef AMREX_USE_EB
335 else
336 {
337 FArrayBox const& beb = (is_eb_dirichlet) ? (*m_eb_b_coeffs)[mfi] : foo;
338 int size_vec = std::pow(3,AMREX_SPACEDIM);
339 colsg.resize(nrows*size_vec);
340 matg.resize(nrows*size_vec);
341 amrex_hpeb_ijmatrix(bx,
342 nrows, ncolsg.dataPtr(),
343 cell_id_vec[mfi].dataPtr(),
344 colsg.dataPtr(), matg.dataPtr(),
345 cell_id[mfi],
346 offset[mfi], diaginv[mfi],
347 acoefs[mfi], bcoefs[0][mfi],
348 bcoefs[1][mfi],
349#if (AMREX_SPACEDIM == 3)
350 bcoefs[2][mfi],
351#endif
352 (*flags)[mfi],
353 (*vfrac)[mfi],
354 (*area[0])[mfi], (*area[1])[mfi],
355#if (AMREX_SPACEDIM == 3)
356 (*area[2])[mfi],
357#endif
358 (*fcent[0])[mfi], (*fcent[1])[mfi],
359#if (AMREX_SPACEDIM == 3)
360 (*fcent[2])[mfi],
361#endif
362 (*barea)[mfi],
363 (*bcent)[mfi],
364 beb, is_eb_dirichlet,
365 scalar_a, scalar_b, dx,
366 bctype, bcl, bho);
367 }
368#endif
369
370 PetscInt* rows = cell_id_vec[mfi].data();
371 PetscInt* ncols = (PetscInt*) ncolsg.dataPtr();
372 PetscInt* cols = (PetscInt*) colsg.dataPtr();
373 Real* mat = (Real*) matg.dataPtr();
374
375 //Load in by row!
376 int matid = 0;
377 for (int rit = 0; rit < nrows; ++rit)
378 {
379 MatSetValues(A, 1, &rows[rit], ncols[rit], &cols[matid], &mat[matid], INSERT_VALUES);
380 matid += ncols[rit];
381 }
382 }
383 }
384
385 MatAssemblyBegin(A, MAT_FINAL_ASSEMBLY);
386 MatAssemblyEnd(A, MAT_FINAL_ASSEMBLY);
387 // create solver
388 KSPCreate(PETSC_COMM_WORLD, &solver);
389 KSPSetOperators(solver, A, A);
390
391 // Set up preconditioner
392 PC pc;
393 KSPGetPC(solver, &pc);
394
395 // Classic AMG
396 PCSetType(pc, PCGAMG);
397 PCGAMGSetType(pc, PCGAMGAGG);
398 PCGAMGSetNSmooths(pc,0);
399// PCSetType(pc, PCJACOBI);
400
401
402// we are not using command line options KSPSetFromOptions(solver);
403 // create b & x
404 VecCreateMPI(PETSC_COMM_WORLD, ncells_proc, ncells_world, &x);
405 VecDuplicate(x, &b);
406}

452 bfab = &rhsfab;
