28void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
29{
30 int nx = nppc[0];
31#if AMREX_SPACEDIM > 1
32 int ny = nppc[1];
33#else
34 int ny = 1;
35#endif
36#if AMREX_SPACEDIM > 2
37 int nz = nppc[2];
38#else
39 int nz = 1;
40#endif
41
42 int ix_part = i_part/(ny * nz);
43 int iy_part = (i_part % (ny * nz)) % ny;
44 int iz_part = (i_part % (ny * nz)) / ny;
45
46 r[0] = (0.5+ix_part)/nx;
47 r[1] = (0.5+iy_part)/ny;
48 r[2] = (0.5+iz_part)/nz;
49}
