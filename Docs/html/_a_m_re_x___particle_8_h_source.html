35 ParticleIDWrapper (const ParticleIDWrapper& rhs) = default;

37 AMREX_GPU_HOST_DEVICE
38 ParticleIDWrapper& operator= (const ParticleIDWrapper& pidw) noexcept

40 return this->operator=(Long(pidw));

42
43 AMREX_GPU_HOST_DEVICE
44 ParticleIDWrapper& operator= (const Long id) noexcept
45 {
46 // zero out the 40 leftmost bits, which store the sign and the abs of the id;
47 m_idata &= 0x00FFFFFF;
48
49 uint64_t val;
50 uint64_t sign = id >= 0;
51 if (sign)
52 {
53 // 2**39-1, the max value representible in this fashion
54 AMREX_ASSERT(id <= 549755813887L);
55 val = id;
56 }
57 else
58 {
59 // -2**39-1, the min value representible in this fashion
60 AMREX_ASSERT(id >= -549755813887L);
61 val = -id;
62 }
63
64 m_idata |= (sign << 63); // put the sign in the leftmost bit
65 m_idata |= (val << 24); // put the val in the next 39
66 return *this;
67 }
68
69 AMREX_GPU_HOST_DEVICE
70 operator Long () const noexcept
71 {
72 Long r = 0;
73
74 uint64_t sign = m_idata >> 63; // extract leftmost sign bit
75 uint64_t val = ((m_idata >> 24) & 0x7FFFFFFFFF); // extract next 39 id bits
76
77 Long lval = static_cast<Long>(val); // bc we take -
78 r = (sign) ? lval : -lval;
79 return r;
80 }

409template<int NReal, int NInt>
410inline
411void
412Particle<NReal, NInt>::CIC_Fracs (const Real* frac, Real* fracs)
413{
414 //
415 // "frac" should be dimensioned: Real frac[AMREX_SPACEDIM]
416 //
417 // "fracs" should be dimensioned: Real fracs[AMREX_D_TERM(2,+2,+4)]
418 //
419#if (AMREX_SPACEDIM == 1)
420
422 fracs[0] = frac[0];
423
424 // Low
425 fracs[1] = (1-frac[0]);
426
427#elif (AMREX_SPACEDIM == 2)
428
430 fracs[0] = frac[0] * frac[1] ;
431
432 // LH
433 fracs[1] = (1-frac[0]) * frac[1];
434
435 // LL
436 fracs[2] = (1-frac[0]) * (1-frac[1]);
437
438 // HL
439 fracs[3] = frac[0] * (1-frac[1]);
440
441#elif (AMREX_SPACEDIM == 3)
442
444 fracs[0] = frac[0] * frac[1] * frac[2];
445
446 // LHH
447 fracs[1] = (1-frac[0]) * frac[1] * frac[2];
448
449 // LLH
450 fracs[2] = (1-frac[0]) * (1-frac[1]) * frac[2];
451
452 // HLH
453 fracs[3] = frac[0] * (1-frac[1]) * frac[2];
454
455 // HHL
456 fracs[4] = frac[0] * frac[1] * (1-frac[2]);
457
458 // LHL
459 fracs[5] = (1-frac[0]) * frac[1] * (1-frac[2]);
460
461 // LLL
462 fracs[6] = (1-frac[0]) * (1-frac[1]) * (1-frac[2]);
463
464 // HLL
465 fracs[7] = frac[0] * (1-frac[1]) * (1-frac[2]);
466#endif
467}

581template <int NReal, int NInt>
582int
583Particle<NReal, NInt>::CIC_Cells_Fracs (const Particle<NReal, NInt>& p,
584 const Real* plo,
585 const Real* dx_geom,
586 const Real* dx_part,
587 Vector<Real>& fracs,
588 Vector<IntVect>& cells)
589{
590 BL_PROFILE("Particle<NReal, NInt>::CIC_Cells_Fracs()");
591
592 if (dx_geom == dx_part)
593 {
594 const int M = AMREX_D_TERM(2,+2,+4);
595 fracs.resize(M);
596 cells.resize(M);
597 Particle<NReal, NInt>::CIC_Cells_Fracs_Basic(p,plo,dx_geom,fracs.dataPtr(),cells.dataPtr());
598 return M;
599 }
600 //
601 // The first element in fracs and cells is the lowest corner, the last is the highest.
602 //
603 const Real hilen[AMREX_SPACEDIM] = { AMREX_D_DECL((p.m_pos[0]-plo[0]+dx_part[0]/2)/dx_geom[0],
604 (p.m_pos[1]-plo[1]+dx_part[1]/2)/dx_geom[1],
605 (p.m_pos[2]-plo[2]+dx_part[2]/2)/dx_geom[2]) };
606
607 const Real lolen[AMREX_SPACEDIM] = { AMREX_D_DECL((p.m_pos[0]-plo[0]-dx_part[0]/2)/dx_geom[0],
608 (p.m_pos[1]-plo[1]-dx_part[1]/2)/dx_geom[1],
609 (p.m_pos[2]-plo[2]-dx_part[2]/2)/dx_geom[2]) };
610
611 const IntVect hicell(AMREX_D_DECL(static_cast<int>(amrex::Math::floor(hilen[0])),
612 static_cast<int>(amrex::Math::floor(hilen[1])),
613 static_cast<int>(amrex::Math::floor(hilen[2]))));
614 const IntVect locell(AMREX_D_DECL(static_cast<int>(amrex::Math::floor(lolen[0])),
615 static_cast<int>(amrex::Math::floor(lolen[1])),
616 static_cast<int>(amrex::Math::floor(lolen[2]))));
617
618 const Real cell_density = AMREX_D_TERM(dx_geom[0]/dx_part[0],*dx_geom[1]/dx_part[1],*dx_geom[2]/dx_part[2]);
619
620 const int M = AMREX_D_TERM((hicell[0]-locell[0]+1),*(hicell[1]-locell[1]+1),*(hicell[2]-locell[2]+1));
621
622 fracs.resize(M);
623 cells.resize(M);
624
626 int i = 0;
627#if (AMREX_SPACEDIM == 1)
628 for (int xi = locell[0]; xi <= hicell[0]; xi++)
629 {
630 cells[i][0] = xi;
631 fracs[i] = (std::min(hilen[0]-xi,Real(1.0))-std::max(lolen[0]-xi,Real(0.0)))*cell_density;
632 i++;
633 }
634#elif (AMREX_SPACEDIM == 2)
635 for (int yi = locell[1]; yi <= hicell[1]; yi++)
636 {
637 const Real yf = std::min(hilen[1]-yi,Real(1.0))-std::max(lolen[1]-yi,Real(0.0));
638 for (int xi = locell[0]; xi <= hicell[0]; xi ++)
639 {
640 cells[i][0] = xi;
641 cells[i][1] = yi;
642 fracs[i] = yf * (std::min(hilen[0]-xi,Real(1.0))-std::max(lolen[0]-xi,Real(0.0)))*cell_density;
643 i++;
644 }
645 }
646#elif (AMREX_SPACEDIM == 3)
647 for (int zi = locell[2]; zi <= hicell[2]; zi++)
648 {
649 const Real zf = std::min(hilen[2]-zi,Real(1.0))-std::max(lolen[2]-zi,Real(0.0));
650 for (int yi = locell[1]; yi <= hicell[1]; yi++)
651 {
652 const Real yf = std::min(hilen[1]-yi,Real(1.0))-std::max(lolen[1]-yi,Real(0.0));
653 for (int xi = locell[0]; xi <= hicell[0]; xi++)
654 {
655 cells[i][0] = xi;
656 cells[i][1] = yi;
657 cells[i][2] = zi;
658 fracs[i] = zf * yf * (std::min(hilen[0]-xi,Real(1.0))-std::max(lolen[0]-xi,Real(0.0))) * cell_density;
659 i++;
660 }
661 }
662 }
663#endif
664
665 return M;
666}
