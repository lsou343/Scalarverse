331void
332FabArrayBase::CPC::define (const BoxArray& ba_dst, const DistributionMapping& dm_dst,
333 const Vector<int>& imap_dst,
334 const BoxArray& ba_src, const DistributionMapping& dm_src,
335 const Vector<int>& imap_src,
336 int MyProc)
337{
338 BL_PROFILE("FabArrayBase::CPC::define()");
339
340 BL_ASSERT(ba_dst.size() > 0 && ba_src.size() > 0);
341 BL_ASSERT(ba_dst.ixType() == ba_src.ixType());
342
343 m_LocTags.reset(new CopyComTag::CopyComTagsContainer);
344 m_SndTags.reset(new CopyComTag::MapOfCopyComTagContainers);
345 m_RcvTags.reset(new CopyComTag::MapOfCopyComTagContainers);
346
347 if (!(imap_dst.empty() && imap_src.empty()))
348 {
349 const int nlocal_src = imap_src.size();
350 const IntVect& ng_src = m_srcng;
351 const int nlocal_dst = imap_dst.size();
352 const IntVect& ng_dst = m_dstng;
353
354 std::vector< std::pair<int,Box> > isects;
355
356 const std::vector<IntVect>& pshifts = m_period.shiftIntVect();
357
358 auto& send_tags = *m_SndTags;
359
360 for (int i = 0; i < nlocal_src; ++i)
361 {
362 const int k_src = imap_src[i];
363 const Box& bx_src = amrex::grow(ba_src[k_src], ng_src);
364
365 for (std::vector<IntVect>::const_iterator pit=pshifts.begin(); pit!=pshifts.end(); ++pit)
366 {
367 ba_dst.intersections(bx_src+(*pit), isects, false, ng_dst);
368
369 for (int j = 0, M = isects.size(); j < M; ++j)
370 {
371 const int k_dst = isects[j].first;
372 const Box& bx = isects[j].second;
373 const int dst_owner = dm_dst[k_dst];
374
375 if (ParallelDescriptor::sameTeam(dst_owner)) {
376 continue; // local copy will be dealt with later
377 } else if (MyProc == dm_src[k_src]) {
378 send_tags[dst_owner].push_back(CopyComTag(bx, bx-(*pit), k_dst, k_src));
379 }
380 }
381 }
382 }
383
384 auto& recv_tags = *m_RcvTags;
385
386 BaseFab<int> localtouch(The_Cpu_Arena()), remotetouch(The_Cpu_Arena());
387 bool check_local = false, check_remote = false;
388#if defined(_OPENMP)
389 if (omp_get_max_threads() > 1) {
390 check_local = true;
391 check_remote = true;
392 }
393#elif defined(AMREX_USE_GPU)
394 check_local = true;
395 check_remote = true;
396#endif
397
398 if (ParallelDescriptor::TeamSize() > 1) {
399 check_local = true;
400 }
401
402 m_threadsafe_loc = not check_local;
403 m_threadsafe_rcv = not check_remote;
404
405 for (int i = 0; i < nlocal_dst; ++i)
406 {
407 const int k_dst = imap_dst[i];
408 const Box& bx_dst = amrex::grow(ba_dst[k_dst], ng_dst);
409
410 if (check_local) {
411 localtouch.resize(bx_dst);
412 localtouch.setVal<RunOn::Host>(0);
413 }
414
415 if (check_remote) {
416 remotetouch.resize(bx_dst);
417 remotetouch.setVal<RunOn::Host>(0);
418 }
419
420 for (std::vector<IntVect>::const_iterator pit=pshifts.begin(); pit!=pshifts.end(); ++pit)
421 {
422 ba_src.intersections(bx_dst+(*pit), isects, false, ng_src);
423
424 for (int j = 0, M = isects.size(); j < M; ++j)
425 {
426 const int k_src = isects[j].first;
427 const Box& bx = isects[j].second - *pit;
428 const int src_owner = dm_src[k_src];
429
430 if (ParallelDescriptor::sameTeam(src_owner, MyProc)) { // local copy
431 const BoxList tilelist(bx, FabArrayBase::comm_tile_size);
432 for (BoxList::const_iterator
433 it_tile = tilelist.begin(),
434 End_tile = tilelist.end(); it_tile != End_tile; ++it_tile)
435 {
436 m_LocTags->push_back(CopyComTag(*it_tile, (*it_tile)+(*pit), k_dst, k_src));
437 }
438 if (check_local) {
439 localtouch.plus<RunOn::Host>(1, bx);
440 }
441 } else if (MyProc == dm_dst[k_dst]) {
442 recv_tags[src_owner].push_back(CopyComTag(bx, bx+(*pit), k_dst, k_src));
443 if (check_remote) {
444 remotetouch.plus<RunOn::Host>(1, bx);
445 }
446 }
447 }
448 }
449
450 if (check_local) {
451 // safe if a cell is touched no more than once
452 // keep checking thread safety if it is safe so far
453 check_local = m_threadsafe_loc = localtouch.max<RunOn::Host>() <= 1;
454 }
455
456 if (check_remote) {
457 check_remote = m_threadsafe_rcv = remotetouch.max<RunOn::Host>() <= 1;
458 }
459 }
460
461 for (int ipass = 0; ipass < 2; ++ipass) // pass 0: send; pass 1: recv
462 {
463 CopyComTag::MapOfCopyComTagContainers & Tags = (ipass == 0) ? *m_SndTags : *m_RcvTags;
464 for (auto& kv : Tags)
465 {
466 std::vector<CopyComTag>& cctv = kv.second;
467 // We need to fix the order so that the send and recv processes match.
468 std::sort(cctv.begin(), cctv.end());
469 }
470 }
471 }
472}

519{
520 amrex::ignore_unused(no_assertion);
521 BL_ASSERT(no_assertion || getBDKey() == m_bdkey);
522
523 std::vector<CPCacheIter> others;
524
525 std::pair<CPCacheIter,CPCacheIter> er_it = m_TheCPCache.equal_range(m_bdkey);
526
527 for (CPCacheIter it = er_it.first; it != er_it.second; ++it)
528 {
529 const BDKey& srckey = it->second->m_srcbdk;
530 const BDKey& dstkey = it->second->m_dstbdk;
531
532 BL_ASSERT((srckey==dstkey && srckey==m_bdkey) ||
533 (m_bdkey==srckey) || (m_bdkey==dstkey));
534
535 if (srckey != dstkey) {
536 const BDKey& otherkey = (m_bdkey == srckey) ? dstkey : srckey;
537 std::pair<CPCacheIter,CPCacheIter> o_er_it = m_TheCPCache.equal_range(otherkey);
538
539 for (CPCacheIter oit = o_er_it.first; oit != o_er_it.second; ++oit)
540 {
541 if (it->second == oit->second)
542 others.push_back(oit);
543 }
544 }
545
546#ifdef AMREX_MEM_PROFILING
547 m_CPC_stats.bytes -= it->second->bytes();
548#endif
549 m_CPC_stats.recordErase(it->second->m_nuse);
550 delete it->second;
551 }
552
553 m_TheCPCache.erase(er_it.first, er_it.second);
554
555 for (std::vector<CPCacheIter>::iterator it = others.begin(),
556 End = others.end(); it != End; ++it)
557 {
558 m_TheCPCache.erase(*it);
559 }

669 const int nlocal = imap.size();
670 const IntVect& ng = m_ngrow;
671 const IntVect ng_ng = m_ngrow - 1;
672 std::vector< std::pair<int,Box> > isects;
673
674 const std::vector<IntVect>& pshifts = m_period.shiftIntVect();
675
676 auto& send_tags = *m_SndTags;
677
678 for (int i = 0; i < nlocal; ++i)
679 {
680 const int ksnd = imap[i];
681 const Box& vbx = ba[ksnd];
682 const Box& vbx_ng = amrex::grow(vbx,1);
683
684 for (auto pit=pshifts.cbegin(); pit!=pshifts.cend(); ++pit)
685 {
686 ba.intersections(vbx+(*pit), isects, false, ng);
687
688 for (int j = 0, M = isects.size(); j < M; ++j)
689 {
690 const int krcv = isects[j].first;
691 const Box& bx = isects[j].second;
692 const int dst_owner = dm[krcv];
693
694 if (ParallelDescriptor::sameTeam(dst_owner)) {
695 continue; // local copy will be dealt with later
696 } else if (MyProc == dm[ksnd]) {
697 BoxList bl = amrex::boxDiff(bx, ba[krcv]);
698 if (m_multi_ghost)
699 {
700 // In the case where ngrow>1, augment the send/rcv box list
701 // with boxes for overlapping ghost nodes.
702 const Box& ba_krcv = amrex::grow(ba[krcv],1);
703 const Box& dst_bx_ng = (amrex::grow(ba_krcv,ng_ng) & (vbx_ng + (*pit)));
704 const BoxList &bltmp = ba.complementIn(dst_bx_ng);
705 for (auto const& btmp : bltmp)
706 {
707 bl.join(amrex::boxDiff(btmp,ba_krcv));
708 }
709 bl.simplify();
710 }
711 for (BoxList::const_iterator lit = bl.begin(); lit != bl.end(); ++lit)
712 send_tags[dst_owner].push_back(CopyComTag(*lit, (*lit)-(*pit), krcv, ksnd));
713 }
714 }
715 }
716 }
717
718 auto& recv_tags = *m_RcvTags;
719
720 BaseFab<int> localtouch(The_Cpu_Arena()), remotetouch(The_Cpu_Arena());
721 bool check_local = false, check_remote = false;
722#if defined(_OPENMP)
723 if (omp_get_max_threads() > 1) {
724 check_local = true;
725 check_remote = true;
726 }
727#elif defined(AMREX_USE_GPU)
728 check_local = true;
729 check_remote = true;
730#endif
731
732 if (ParallelDescriptor::TeamSize() > 1) {
733 check_local = true;
734 }
735
736 m_threadsafe_loc = not check_local;
737 m_threadsafe_rcv = not check_remote;
738
739 for (int i = 0; i < nlocal; ++i)
740 {
741 const int krcv = imap[i];
742 const Box& vbx = ba[krcv];
743 const Box& vbx_ng = amrex::grow(vbx,1);
744 const Box& bxrcv = amrex::grow(vbx, ng);
745
746 if (check_local) {
747 localtouch.resize(bxrcv);
748 localtouch.setVal<RunOn::Host>(0);
749 }
750
751 if (check_remote) {
752 remotetouch.resize(bxrcv);
753 remotetouch.setVal<RunOn::Host>(0);
754 }
755
756 for (auto pit=pshifts.cbegin(); pit!=pshifts.cend(); ++pit)
757 {
758 ba.intersections(bxrcv+(*pit), isects);
759
760 for (int j = 0, M = isects.size(); j < M; ++j)
761 {
762 const int ksnd = isects[j].first;
763 const Box& dst_bx = isects[j].second - *pit;
764 const int src_owner = dm[ksnd];
765
766 BoxList bl = amrex::boxDiff(dst_bx, vbx);
767
768 if (m_multi_ghost)
769 {
770 // In the case where ngrow>1, augment the send/rcv box list
771 // with boxes for overlapping ghost nodes.
772 Box ba_ksnd = ba[ksnd];
773 ba_ksnd.grow(1);
774 const Box dst_bx_ng = (ba_ksnd & (bxrcv + (*pit))) - (*pit);
775 const BoxList &bltmp = ba.complementIn(dst_bx_ng);
776 for (auto const& btmp : bltmp)
777 {
778 bl.join(amrex::boxDiff(btmp,vbx_ng));
779 }
780 bl.simplify();
781 }
782 for (BoxList::const_iterator lit = bl.begin(); lit != bl.end(); ++lit)
783 {
784 const Box& blbx = *lit;
785
786 if (ParallelDescriptor::sameTeam(src_owner)) { // local copy
787 const BoxList tilelist(blbx, FabArrayBase::comm_tile_size);
788 for (BoxList::const_iterator
789 it_tile = tilelist.begin(),
790 End_tile = tilelist.end(); it_tile != End_tile; ++it_tile)
791 {
792 m_LocTags->push_back(CopyComTag(*it_tile, (*it_tile)+(*pit), krcv, ksnd));
793 }
794 if (check_local) {
795 localtouch.plus<RunOn::Host>(1, blbx);
796 }
797 } else if (MyProc == dm[krcv]) {
798 recv_tags[src_owner].push_back(CopyComTag(blbx, blbx+(*pit), krcv, ksnd));
799 if (check_remote) {
800 remotetouch.plus<RunOn::Host>(1, blbx);
801 }
802 }
803 }
804 }
805 }
806
807 if (check_local) {
808 // safe if a cell is touched no more than once
809 // keep checking thread safety if it is safe so far
810 check_local = m_threadsafe_loc = localtouch.max<RunOn::Host>() <= 1;
811 }
812
813 if (check_remote) {
814 check_remote = m_threadsafe_rcv = remotetouch.max<RunOn::Host>() <= 1;
815 }
816 }
817
818 for (int ipass = 0; ipass < 2; ++ipass) // pass 0: send; pass 1: recv
819 {
820 CopyComTag::MapOfCopyComTagContainers & Tags = (ipass == 0) ? *m_SndTags : *m_RcvTags;
821
822 for (auto& kv : Tags)
823 {
824 std::vector<CopyComTag>& cctv = kv.second;
825
826 // We need to fix the order so that the send and recv processes match.
827 std::sort(cctv.begin(), cctv.end());
828
829 std::vector<CopyComTag> cctv_tags_cross;
830 cctv_tags_cross.reserve(cctv.size());
831
832 for (auto const& tag : cctv)
833 {
834 const Box& bx = tag.dbox;
835 const IntVect& d2s = tag.sbox.smallEnd() - tag.dbox.smallEnd();
836
837 std::vector<Box> boxes;
838 if (m_cross) {
839 const Box& dstvbx = ba[tag.dstIndex];
840 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
841 {
842 Box lo = dstvbx;
843 lo.setSmall(dir, dstvbx.smallEnd(dir) - ng[dir]);
844 lo.setBig (dir, dstvbx.smallEnd(dir) - 1);
845 lo &= bx;
846 if (lo.ok()) {
847 boxes.push_back(lo);
848 }
849
850 Box hi = dstvbx;
851 hi.setSmall(dir, dstvbx.bigEnd(dir) + 1);
852 hi.setBig (dir, dstvbx.bigEnd(dir) + ng[dir]);
853 hi &= bx;
854 if (hi.ok()) {
855 boxes.push_back(hi);
856 }
857 }
858 } else {
859 boxes.push_back(bx);
860 }
861
862 if (!boxes.empty())
863 {
864 for (auto const& cross_box : boxes)
865 {
866 if (m_cross)
867 {
868 cctv_tags_cross.push_back(CopyComTag(cross_box, cross_box+d2s,
869 tag.dstIndex, tag.srcIndex));
870 }
871 }
872 }
873 }
874
875 if (!cctv_tags_cross.empty()) {
876 cctv.swap(cctv_tags_cross);
877 }
878 }
879 }

1138 const DistributionMapping& dm = fa.DistributionMap();
1139 const Vector<int>& imap = fa.IndexArray();
1140
1141 const int nlocal = imap.size();
1142 std::vector<std::pair<int,Box> > isects;
1143
1144 Box loxbox = amrex::adjCellLo(m_domain, 0, m_ngrow[0]).growHi(1, m_ngrow[1]);
1145 Box loybox = amrex::adjCellLo(m_domain, 1, m_ngrow[1]).growHi(0, m_ngrow[0]);
1146#if (AMREX_SPACEDIM == 3)
1147 loxbox.grow(2, m_ngrow[2]);
1148 loybox.grow(2, m_ngrow[2]);
1149#endif
1150
1151 Array<Box,2> const domain_dst{loxbox, loybox};
1152 auto const src_to_dst = amrex::makeTuple(NonLocalBC::Rotate90CounterClockWise{},
1153 NonLocalBC::Rotate90ClockWise{});
1154 auto const dst_to_src = amrex::makeTuple(NonLocalBC::Rotate90ClockWise{},
1155 NonLocalBC::Rotate90CounterClockWise{});
1156 Array<Box,2> const domain_src{amrex::get<0>(dst_to_src)(domain_dst[0]),
1157 amrex::get<1>(dst_to_src)(domain_dst[1])};
1158
1159 auto& send_tags = *m_SndTags;
1160
1161 for (int i = 0; i < nlocal; ++i)
1162 {
1163 const int ksnd = imap[i];
1164 Box gbx = ba[ksnd];
1165 if (gbx.bigEnd(0) == m_domain.bigEnd(0)) {
1166 gbx.growHi(0, m_ngrow[0]);
1167 }
1168 if (gbx.bigEnd(1) == m_domain.bigEnd(1)) {
1169 gbx.growHi(1, m_ngrow[1]);
1170 }
1171#if (AMREX_SPACEDIM == 3)
1172 if (gbx.smallEnd(2) == m_domain.smallEnd(2)) {
1173 gbx.growLo(2, m_ngrow[2]);
1174 }
1175 if (gbx.bigEnd(2) == m_domain.bigEnd(2)) {
1176 gbx.growHi(2, m_ngrow[2]);
1177 }
1178#endif
1179
1180 Array<Box,2> const src_boxes{gbx & domain_src[0], gbx & domain_src[1]};
1181 Array<Box,2> const dst_boxes{amrex::get<0>(src_to_dst)(src_boxes[0]),
1182 amrex::get<1>(src_to_dst)(src_boxes[1])};
1183
1184 for (int n = 0; n < 2; ++n) {

1186 ba.intersections(dst_boxes[n], isects, false, m_ngrow);
1187 for (int j = 0, M = isects.size(); j < M; ++j) {
1188 const int krcv = isects[j].first;
1189 const Box& bxrcv = isects[j].second;
1190 const int dst_owner = dm[krcv];
1191 if (dst_owner != myproc) // local copy will be dealt with later
1192 {
1193 Box bxsnd = (n==0) ? amrex::get<0>(dst_to_src)(bxrcv)
1194 : amrex::get<1>(dst_to_src)(bxrcv);
1195 send_tags[dst_owner].push_back(FabArrayBase::CopyComTag(bxrcv, bxsnd,
1196 krcv, ksnd));
1197 }
1198 }
1199 }
1200 }
1201 }
1202
1203 auto& recv_tags = *m_RcvTags;
1204
1205 for (int i = 0; i < nlocal; ++i)
1206 {
1207 const int krcv = imap[i];
1208 Box const& gbx = amrex::grow(ba[krcv],m_ngrow);
1209
1210 Array<Box,2> const dst_boxes{gbx & domain_dst[0], gbx & domain_dst[1]};
1211 Array<Box,2> const src_boxes{amrex::get<0>(dst_to_src)(dst_boxes[0]),
1212 amrex::get<1>(dst_to_src)(dst_boxes[1])};
1213
1214 for (int n = 0; n < 2; ++n) {
1215 if (dst_boxes[n].ok()) {
1216 ba.intersections(src_boxes[n], isects);
1217 for (int j = 0, M = isects.size(); j < M; ++j) {
1218 const int ksnd = isects[j].first;
1219 Box bxsnd = isects[j].second;
1220 // the ghost cells at hi-x, hi-y, lo-z, and hi-z
1221 // boundares are also the source
1222 if (bxsnd.bigEnd(n) == m_domain.bigEnd(n)) {
1223 bxsnd.growHi(n,m_ngrow[n]);
1224 }
1225#if (AMREX_SPACEDIM == 3)
1226 if (bxsnd.smallEnd(2) == m_domain.smallEnd(2)) {
1227 bxsnd.growLo(2,m_ngrow[2]);
1228 }
1229 if (bxsnd.bigEnd(2) == m_domain.bigEnd(2)) {
1230 bxsnd.growHi(2,m_ngrow[2]);
1231 }
1232#endif
1233 Box bxrcv = (n==0) ? amrex::get<0>(src_to_dst)(bxsnd)
1234 : amrex::get<1>(src_to_dst)(bxsnd);
1235 const int src_owner = dm[ksnd];
1236 FabArrayBase::CopyComTag cct(bxrcv, bxsnd, krcv, ksnd);
1237 if (src_owner == myproc) {
1238 m_LocTags->push_back(cct);
1239 } else {
1240 recv_tags[src_owner].push_back(cct);
1241 }
1242 }
1243 }
1244 }
1245 }

1247 for (int ipass = 0; ipass < 2; ++ipass) // pass 0: send; pass 1: recv
1248 {
1249 auto& Tags = (ipass == 0) ? *m_SndTags : *m_RcvTags;
1250 for (auto& kv : Tags)
1251 {
1252 auto& cctv = kv.second;
1253 // We need to fix the order so that the send and recv processes match.
1254 std::sort(cctv.begin(), cctv.end());
1255 }

1328 std::vector<std::pair<int,Box> > isects;
1329
1330 Box domain_dst = amrex::adjCellLo(m_domain, 0, m_ngrow[0]).grow(1, m_ngrow[1]);
1331#if (AMREX_SPACEDIM == 3)
1332 domain_dst.grow(2, m_ngrow[2]);
1333#endif
1334
1335 auto const convert = NonLocalBC::Rotate180Fn{m_domain.length(1)};
1336

1338
1339 auto& send_tags = *m_SndTags;
1340
1341 for (int i = 0; i < nlocal; ++i)
1342 {
1343 const int ksnd = imap[i];
1344 Box gbx = ba[ksnd];
1345 for (int idim = 1; idim < AMREX_SPACEDIM; ++idim) {
1346 if (gbx.smallEnd(idim) == m_domain.smallEnd(idim)) {
1347 gbx.growLo(idim, m_ngrow[idim]);

1349 if (gbx.bigEnd(idim) == m_domain.bigEnd(idim)) {
1350 gbx.growHi(idim, m_ngrow[idim]);

1360 const int krcv = isects[j].first;
1361 const Box& bxrcv = isects[j].second;
1362 const int dst_owner = dm[krcv];
1363 if (dst_owner != myproc) // local copy will be dealt with later
1364 {
1365 Box const& bxsnd = convert(bxrcv);
1366 send_tags[dst_owner].push_back(FabArrayBase::CopyComTag(bxrcv, bxsnd,
1367 krcv, ksnd));
1368 }
1369 }
1370 }
1371 }
1372

1374
1375 for (int i = 0; i < nlocal; ++i)
1376 {
1377 const int krcv = imap[i];
1378 Box const& gbx = amrex::grow(ba[krcv],m_ngrow);
1379
1380 Box const& dst_box = gbx & domain_dst;
1381 Box const& src_box = convert(dst_box);
1382
1383 if (dst_box.ok()) {
1384 ba.intersections(src_box, isects);
1385 for (int j = 0, M = isects.size(); j < M; ++j) {
1386 const int ksnd = isects[j].first;
1387 Box bxsnd = isects[j].second;
1388 // the ghost cells at lo-y, hi-y, lo-z, and hi-z
1389 // boundares are also the source
1390 for (int idim = 1; idim < AMREX_SPACEDIM; ++idim) {
1391 if (bxsnd.smallEnd(idim) == m_domain.smallEnd(idim)) {
1392 bxsnd.growLo(idim, m_ngrow[idim]);
1393 }
1394 if (bxsnd.bigEnd(idim) == m_domain.bigEnd(idim)) {
1395 bxsnd.growHi(idim, m_ngrow[idim]);
1396 }
1397 }

1399 const int src_owner = dm[ksnd];
1400 FabArrayBase::CopyComTag cct(bxrcv, bxsnd, krcv, ksnd);
1401 if (src_owner == myproc) {
1402 m_LocTags->push_back(cct);
1403 } else {
1404 recv_tags[src_owner].push_back(cct);
1405 }
1406 }
1407 }
1408 }
1409
1410 for (int ipass = 0; ipass < 2; ++ipass) // pass 0: send; pass 1: recv
1411 {
1412 auto& Tags = (ipass == 0) ? *m_SndTags : *m_RcvTags;
1413 for (auto& kv : Tags)
1414 {
1415 auto& cctv = kv.second;
1416 // We need to fix the order so that the send and recv processes match.
1417 std::sort(cctv.begin(), cctv.end());
1418 }
1419 }

1652FabArrayBase::FPinfo::FPinfo (const FabArrayBase& srcfa,
1653 const FabArrayBase& dstfa,
1654 const Box& dstdomain,
1655 const IntVect& dstng,
1656 const BoxConverter& coarsener,
1657 const Box& fdomain,
1658 const Box& cdomain,
1659 const EB2::IndexSpace* index_space)
1660 : m_srcbdk (srcfa.getBDKey()),
1661 m_dstbdk (dstfa.getBDKey()),
1662 m_dstdomain(dstdomain),
1663 m_dstng (dstng),
1664 m_coarsener(coarsener.clone()),
1665 m_nuse (0)
1666{
1667 amrex::ignore_unused(fdomain,cdomain,index_space);
1668 BL_PROFILE("FPinfo::FPinfo()");
1669
1670 const BoxArray& srcba = srcfa.boxArray();
1671 const BoxArray& dstba = dstfa.boxArray();
1672 BL_ASSERT(srcba.ixType() == dstba.ixType());
1673
1674 BoxArray srcba_simplified = srcba.simplified();
1675 BoxArray dstba_simplified = dstba.simplified();
1676
1677 const IndexType& boxtype = dstba.ixType();
1678 BL_ASSERT(boxtype == dstdomain.ixType());
1679
1680 BL_ASSERT(dstng.allLE(dstfa.nGrowVect()));
1681
1682 BoxList bl(boxtype);
1683 const int Ndst = dstba_simplified.size();
1684 const int nprocs = ParallelContext::NProcsSub();
1685 int iboxlo, iboxhi;
1686 bool parallel_ci;
1687 if (Ndst > 8) {
1688 parallel_ci = true;
1689 const int navg = Ndst / nprocs;
1690 const int nextra = Ndst - navg*nprocs;
1691 const int myproc = ParallelContext::MyProcSub();
1692 iboxlo = (myproc < nextra) ? myproc*(navg+1) : myproc*navg+nextra;
1693 iboxhi = (myproc < nextra) ? iboxlo+navg+1-1 : iboxlo+navg-1;
1694 } else {
1695 parallel_ci = false;
1696 iboxlo = 0;
1697 iboxhi = Ndst-1;
1698 }
1699 for (int i = iboxlo; i <= iboxhi; ++i) {
1700 Box bx = dstba_simplified[i];
1701 bx.grow(m_dstng);
1702 bx &= m_dstdomain;
1703 BoxList const& leftover = srcba_simplified.complementIn(bx);
1704 if (leftover.isNotEmpty()) {
1705 bl.join(leftover);
1706 }
1707 }
1708
1709 if (parallel_ci) {
1710 amrex::AllGatherBoxes(bl.data());
1711 }
1712
1713 if (bl.isEmpty()) return;
1714
1715 Long ncells_total = 0L;
1716 Long ncells_max = 0L;
1717 for (auto const& b : bl) {
1718 auto n = b.numPts();
1719 ncells_total += n;
1720 ncells_max = std::max(ncells_max, n);
1721 }
1722
1723 Long ncells_avg = ncells_total / ParallelContext::NProcsSub();
1724 Long ncells_target = std::max(2*ncells_avg, Long(8*8*8));
1725 if (ncells_max > ncells_target) {
1726 BoxList bltmp(boxtype);
1727 Vector<Box>& bltmpvec = bltmp.data();
1728 for (Box const& b : bl) {
1729 Long const npts = b.numPts();
1730 if (npts <= ncells_target) {
1731 bltmp.push_back(b);
1732 } else {
1733 IntVect const len = b.length();
1734 IntVect numblk{1};
1735 while (npts > (AMREX_D_TERM(numblk[0],*numblk[1],*numblk[2])) * ncells_target) {
1736#if (AMREX_SPACEDIM == 3)
1737 int longdir = (len[2] >= len[0] && len[2] >= len[1]) ? 2 :
1738 (len[1] >= len[0]) ? 1 : 0;
1739#elif (AMREX_SPACEDIM == 2)
1740 int longdir = (len[1] >= len[0]) ? 1 : 0;
1741#elif (AMREX_SPACEDIM == 1)
1742 int longdir = 0;
1743#else
1744 static_assert(false, "FabArrayBase::FPinfo() unsupported AMREX_SPACEDIM");
1745#endif
1746 numblk[longdir] *= 2;
1747 }
1748 numblk.min(len);
1749 IntVect sz, extra;
1750 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
1751 sz[idim] = len[idim] / numblk[idim];
1752 extra[idim] = len[idim] - sz[idim] * numblk[idim];
1753 }
1754 if (numblk == 1) {
1755 bltmp.push_back(b);
1756 } else {
1757 IntVect const& boxlo = b.smallEnd();
1758#if (AMREX_SPACEDIM == 3)
1759 for (int k = 0; k < numblk[2]; ++k) {
1760 int klo = (k < extra[2]) ? k*(sz[2]+1) : (k*sz[2]+extra[2]);
1761 int khi = (k < extra[2]) ? klo+(sz[2]+1)-1 : klo+sz[2]-1;
1762 klo += boxlo[2];
1763 khi += boxlo[2];
1764#endif
1765#if (AMREX_SPACEDIM >= 2)
1766 for (int j = 0; j < numblk[1]; ++j) {
1767 int jlo = (j < extra[1]) ? j*(sz[1]+1) : (j*sz[1]+extra[1]);
1768 int jhi = (j < extra[1]) ? jlo+(sz[1]+1)-1 : jlo+sz[1]-1;
1769 jlo += boxlo[1];
1770 jhi += boxlo[1];
1771#endif
1772 for (int i = 0; i < numblk[0]; ++i) {
1773 int ilo = (i < extra[0]) ? i*(sz[0]+1) : (i*sz[0]+extra[0]);
1774 int ihi = (i < extra[0]) ? ilo+(sz[0]+1)-1 : ilo+sz[0]-1;
1775 ilo += boxlo[0];
1776 ihi += boxlo[0];
1777 bltmpvec.emplace_back(IntVect(AMREX_D_DECL(ilo,jlo,klo)),
1778 IntVect(AMREX_D_DECL(ihi,jhi,khi)),
1779 boxtype);
1780 AMREX_D_TERM(},},})
1781 }
1782 }
1783 }
1784 std::swap(bl,bltmp);
1785 }
1786
1787 BoxList blcrse(boxtype);
1788 blcrse.reserve(bl.size());
1789 for (auto const& b : bl) {
1790 blcrse.push_back(coarsener.doit(b));
1791 }
1792
1793 ba_crse_patch.define(std::move(blcrse));
1794 ba_fine_patch.define(std::move(bl));
1795 dm_patch.KnapSackProcessorMap(ba_fine_patch, ParallelContext::NProcsSub());
1796
1797#ifdef AMREX_USE_EB
1798 if (index_space)
1799 {
1800 fact_crse_patch = makeEBFabFactory(index_space,
1801 index_space->getGeometry(cdomain),
1802 ba_crse_patch,
1803 dm_patch,
1804 {0,0,0}, EBSupport::basic);
1805 int ng = boxtype.cellCentered() ? 0 : 1; // to avoid dengerate box
1806 fact_fine_patch = makeEBFabFactory(index_space,
1807 index_space->getGeometry(fdomain),
1808 ba_fine_patch,
1809 dm_patch,
1810 {ng,ng,ng}, EBSupport::basic);
1811 }
1812 else
1813#endif
1814 {
1815 fact_crse_patch.reset(new FArrayBoxFactory());
1816 fact_fine_patch.reset(new FArrayBoxFactory());
1817 }
1818}
