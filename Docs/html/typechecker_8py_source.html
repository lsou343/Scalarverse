129def c_to_f_type(ctyp):
130 # supported C types: char, short, int, long, float, double, void, _Bool
131 if ctyp == 'char':
132 return 'CHARACTER 1 1'
133 elif ctyp == 'int8_t':
134 return 'INTEGER 1'
135 elif ctyp == 'short':
136 return 'INTEGER 2'
137 elif ctyp == 'int':
138 return 'INTEGER 4'
139 elif ctyp == 'long':
140 return 'INTEGER 8' # yes, this is our assumption
141 elif ctyp == 'float':
142 return 'REAL 4'
143 elif ctyp == 'double':
144 return 'REAL 8'
145 elif ctyp == 'amrex_real':
146 return 'REAL 8'
147 elif ctyp == 'void':
148 return 'void'
149 elif ctyp == '_Bool':
150 return 'LOGICAL 1'
