76bool BLProfStats::AddPiece(std::list<TimeRange> &addToHere,
77 const TimeRange &pieceToAdd)
78{
79 // ---- add the new piece, sort, then remove overlap
80 addToHere.push_back(pieceToAdd);
81 addToHere.sort(BLProfStats::TimeRangeCompare());
82
83 std::list<TimeRange>::iterator it, itnext;
84 std::list<std::list<TimeRange>::iterator> overlaps;
85 for(it = addToHere.begin(); it != addToHere.end(); ++it) {
86 itnext = it;
87 ++itnext;
88 if(itnext != addToHere.end()) {
89 TimeRange &tRange = *it;
90 TimeRange &tRangeNext = *itnext;
91 if(tRange.stopTime >= tRangeNext.startTime) {
92 tRangeNext.startTime = tRange.startTime;
93 tRangeNext.stopTime = std::max(tRange.stopTime, tRangeNext.stopTime);
94 overlaps.push_back(it);
95 }
96 }
97 }
98 std::list<std::list<TimeRange>::iterator >::iterator oit;
99 for(oit = overlaps.begin(); oit != overlaps.end(); ++oit) {
100 addToHere.erase(*oit);
101 }
102 return true;
103}

107std::list<BLProfStats::TimeRange> BLProfStats::RangeIntersection(
108 std::list<TimeRange> &rangeList,
109 const TimeRange &pieceToIntersect)
110{
111 std::list<TimeRange> intersectList, intersectListComp;
112 intersectList.push_back(pieceToIntersect);
113 intersectListComp.push_back(pieceToIntersect);
114
115 std::list<TimeRange>::iterator it;
116 for(it = rangeList.begin(); it != rangeList.end(); ++it) {
117 RemovePiece(intersectListComp, *it);
118 }
119 for(it = intersectListComp.begin(); it != intersectListComp.end(); ++it) {
120 RemovePiece(intersectList, *it);
121 }
122 return intersectList;

126// ----------------------------------------------------------------------
127bool BLProfStats::RemovePiece(std::list<TimeRange> &removeFromHere,
128 const TimeRange &pieceToRemove)
129{
130
131
132 bool piecesRemoved(false);
133 std::list<TimeRange>::iterator it;
134 std::list<std::list<TimeRange>::iterator> eraseThese;
135
136 for(it = removeFromHere.begin(); it != removeFromHere.end(); ++it) {
137 TimeRange &tRangeFrom = *it;
138 bool bothLow(pieceToRemove.stopTime < tRangeFrom.startTime);
139 bool loInRange((pieceToRemove.startTime > tRangeFrom.startTime) &&
140 (pieceToRemove.startTime < tRangeFrom.stopTime));
141 bool hiInRange((pieceToRemove.stopTime > tRangeFrom.startTime) &&
142 (pieceToRemove.stopTime < tRangeFrom.stopTime));
143 bool covered((pieceToRemove.startTime <= tRangeFrom.startTime) &&
144 (pieceToRemove.stopTime >= tRangeFrom.stopTime));
145 bool bothHigh(pieceToRemove.startTime > tRangeFrom.stopTime);
146 // Warning! covered test may require an epsilon in future
147 // implementations. Works fine now given fixed nature of
148 // presently used data.
149
150 // ---- there are six cases
151 if(bothLow) { // ---- do nothing
152 } else if( ! loInRange && hiInRange) { // ---- remove low end piece
153 tRangeFrom.startTime = pieceToRemove.stopTime;
154 piecesRemoved = true;
155
156 } else if(loInRange && hiInRange) { // ---- remove middle piece
157 removeFromHere.insert(it,
158 TimeRange(tRangeFrom.startTime, pieceToRemove.startTime));
159 tRangeFrom.startTime = pieceToRemove.stopTime; // reuse this part
160 piecesRemoved = true;
161
162 } else if(loInRange && ! hiInRange) { // ---- remove high end piece
163 tRangeFrom.stopTime = pieceToRemove.startTime;
164 piecesRemoved = true;
165
166 } else if(covered) { // ---- remove the whole range
167 eraseThese.push_back(it);
168 piecesRemoved = true;
169
170 } else if(bothHigh) { // ---- do nothing
171 }
172
173 }
174 std::list<std::list<TimeRange>::iterator >::iterator eit;
175 for(eit = eraseThese.begin(); eit != eraseThese.end(); ++eit) {
176 removeFromHere.erase(*eit);
177 }
178
179 return piecesRemoved;
180}

445// ----------------------------------------------------------------------
446void BLProfStats::ReadBlock(BLPDataBlock &dBlock) {
447 int nFuncs(blpFNames.size());
448 if(dBlock.nCalls.size() != nFuncs) {
449 dBlock.nCalls.resize(nFuncs);
450 }
451 if(dBlock.totalTime.size() != nFuncs) {
452 dBlock.totalTime.resize(nFuncs);
453 }
454
455 std::string fullFileName(dirName + '/' + dBlock.fileName);
456 std::ifstream instr(fullFileName.c_str());
457
458 long dataSizeNC(nFuncs * sizeof(long));
459 long dataSizeTT(nFuncs * sizeof(Real));
460
461 instr.seekg(dBlock.seekpos);
462
463 instr.read(reinterpret_cast<char *>(dBlock.nCalls.dataPtr()), dataSizeNC);
464 instr.read(reinterpret_cast<char *>(dBlock.totalTime.dataPtr()), dataSizeTT);
465
466 if(instr.fail()) {
467 cout << "******** Error::ReadBlock: instr failed: fsize seekpos = "
468 << FileSize(fullFileName) << " " << instr.tellg() << endl;
469 }
470
471 instr.close();
472}
