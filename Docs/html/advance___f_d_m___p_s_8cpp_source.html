401inline void drift(hacc::Dfft &dfft, MultiFab &Ax_new,
402 int const gridsize, Real const dt, Real const h, Real const a_half,
403 Real const hbaroverm,std::vector<complex_t, hacc::AlignedAllocator<complex_t, ALIGN> >* a,std::vector<complex_t, hacc::AlignedAllocator<complex_t, ALIGN> >* b)
404{
405 if(!dt) return;
406
407 const Real pi = 4 * std::atan(1.0);
408 const Real tpi = 2 * pi;
409 const Real hsq = h*h;
410 const std::complex<double> imagi(0.0,1.0);
411 const int *self = dfft.self_kspace();
412 const int *local_ng = dfft.local_ng_kspace();
413 const int *global_ng = dfft.global_ng();
414 size_t local_indx = 0;
415
416 dfft.forward(&((*a)[0]));
417#ifdef _OPENMP
418#pragma omp parallel for
419#endif
420 for(size_t i=0; i<(size_t)local_ng[0]; i++) {
421 int global_i = local_ng[0]*self[0] + i;
422 if (global_i > global_ng[0]/2.){
423 global_i = global_i - global_ng[0];
424 }
425
426
427 for(size_t j=0; j<(size_t)local_ng[1]; j++) {
428 int global_j = local_ng[1]*self[1] + j;
429 if (global_j > global_ng[1]/2.){
430 global_j = global_j - global_ng[1];
431 }
432
433 AMREX_PRAGMA_SIMD
434 for(size_t k=0; k<(size_t)local_ng[2]; k++) {
435 int global_k = local_ng[2]*self[2] + k;
436 size_t local_indx_threaded = (size_t)local_ng[1]*(size_t)local_ng[2]*i+(size_t)local_ng[2]*j+k;
437 if (global_k > global_ng[2]/2.){
438 global_k = global_k - global_ng[2];
439 }
440
441 double kx = tpi * double(global_i)/double(global_ng[0]);
442 double ky = tpi * double(global_j)/double(global_ng[1]);
443 double kz = tpi * double(global_k)/double(global_ng[2]);
444 double k2 = (kx*kx + ky*ky + kz*kz)/h/h;
445
446 (*a)[local_indx_threaded] *= std::exp(- imagi * hbaroverm * k2 / a_half / a_half / 2.0 * dt );
447 (*a)[local_indx_threaded] /= dfft.global_size();
448 }
449 }
450 }
451 dfft.backward(&((*a)[0]));
