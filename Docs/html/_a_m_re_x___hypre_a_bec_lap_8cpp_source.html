32void
33HypreABecLap::solve(MultiFab& soln, const MultiFab& rhs, Real reltol, Real abstol,
34 int maxiter, const BndryData& bndry, int max_bndry_order)
35{
36 if (solver == NULL || m_bndry != &bndry || m_maxorder != max_bndry_order)
37 {
38 m_bndry = &bndry;
39 m_maxorder = max_bndry_order;
40 m_factory = &(rhs.Factory());
41 prepareSolver();
42 }
43 else
44 {
45 m_factory = &(rhs.Factory());
46 }
47
48 // do this repeatedly to avoid memory leak
49 HYPRE_StructVectorCreate(comm, grid, &b);
50 HYPRE_StructVectorCreate(comm, grid, &x);
51 //
52 HYPRE_StructVectorInitialize(b);
53 HYPRE_StructVectorInitialize(x);
54 //
55 loadVectors(soln, rhs);
56 //
57 HYPRE_StructVectorAssemble(x);
58 HYPRE_StructVectorAssemble(b);
59
60 HYPRE_StructPFMGSetMaxIter(solver, maxiter);
61 HYPRE_StructPFMGSetTol(solver, reltol);
62 if (abstol > 0.0)
63 {
64 Real bnorm;
65 bnorm = hypre_StructInnerProd((hypre_StructVector *) b,
66 (hypre_StructVector *) b);
67 bnorm = std::sqrt(bnorm);
68
69 const BoxArray& grids = rhs.boxArray();
70 Real volume = grids.d_numPts();
71 Real reltol_new = (bnorm > 0.0) ? (abstol / bnorm * std::sqrt(volume)) : reltol;
72
73 if (reltol_new > reltol) {
74 HYPRE_StructPFMGSetTol(solver, reltol_new);
75 }
76 }
77
78 HYPRE_StructPFMGSolve(solver, A, b, x);
79
80 if (verbose >= 2)
81 {
82 HYPRE_Int num_iterations;
83 Real res;
84 HYPRE_StructPFMGGetNumIterations(solver, &num_iterations);
85 HYPRE_StructPFMGGetFinalRelativeResidualNorm(solver, &res);
86
87 amrex::Print() << "\n" << num_iterations
88 << " Hypre PFMG Iterations, Relative Residual "
89 << res << std::endl;
90 }
91
92 getSolution(soln);
93
94 // do this repeatedly to avoid memory leak
95 HYPRE_StructVectorDestroy(b);
96 b = NULL;
97 HYPRE_StructVectorDestroy(x);
98 x = NULL;
99}

102HypreABecLap::getSolution (MultiFab& soln)
103{
104 FArrayBox rfab;
105 for (MFIter mfi(soln); mfi.isValid(); ++mfi)
106 {
107 const Box &reg = mfi.validbox();
108
109 FArrayBox *xfab;
110 if (soln.nGrow() == 0) { // need a temporary if soln is the wrong size
111 xfab = &soln[mfi];
112 }
113 else {
114 xfab = &rfab;
115 xfab->resize(reg);
116 }
117
118 auto reglo = Hypre::loV(reg);
119 auto reghi = Hypre::hiV(reg);
120 HYPRE_StructVectorGetBoxValues(x, reglo.data(), reghi.data(), xfab->dataPtr());
121
122 if (soln.nGrow() != 0) {
123 soln[mfi].copy<RunOn::Host>(*xfab, 0, 0, 1);
124 }
125 }

128void
129HypreABecLap::prepareSolver ()
130{
131 BL_PROFILE("HypreABecLap::prepareSolver()");
132
133 HYPRE_StructGridCreate(comm, AMREX_SPACEDIM, &grid);
134
135 Array<HYPRE_Int,AMREX_SPACEDIM> is_periodic {AMREX_D_DECL(0,0,0)};
136 for (int i = 0; i < AMREX_SPACEDIM; i++) {
137 if (geom.isPeriodic(i)) {
138 is_periodic[i] = geom.period(i);
139 AMREX_ASSERT(Hypre::ispow2(is_periodic[i]));
140 AMREX_ASSERT(geom.Domain().smallEnd(i) == 0);
141 }
142 }
143 if (geom.isAnyPeriodic()) {
144 HYPRE_StructGridSetPeriodic(grid, is_periodic.data());
145 }
146
147 for (MFIter mfi(acoefs); mfi.isValid(); ++mfi)
148 {
149 const Box& bx = mfi.validbox();
150 auto lo = Hypre::loV(bx);
151 auto hi = Hypre::hiV(bx);
152 HYPRE_StructGridSetExtents(grid, lo.data(), hi.data());
153 }
154
155 HYPRE_StructGridAssemble(grid);
156
157#if (AMREX_SPACEDIM == 2)
158 HYPRE_Int offsets[regular_stencil_size][2] = {{ 0, 0}, // 0
159 {-1, 0}, // 1
160 { 1, 0}, // 2
161 { 0, -1}, // 3
162 { 0, 1}}; // 4
163#elif (AMREX_SPACEDIM == 3)
164 HYPRE_Int offsets[regular_stencil_size][3] = {{ 0, 0, 0}, // 0
165 {-1, 0, 0}, // 1
166 { 1, 0, 0}, // 2
167 { 0, -1, 0}, // 3
168 { 0, 1, 0}, // 4
169 { 0, 0, -1}, // 5
170 { 0, 0, 1}}; // 6
171#endif
172
173 HYPRE_StructStencil stencil;
174 HYPRE_StructStencilCreate(AMREX_SPACEDIM, regular_stencil_size, &stencil);
175
176 for (int i = 0; i < regular_stencil_size; ++i) {
177 HYPRE_StructStencilSetElement(stencil, i, offsets[i]);
178 }
179
180 HYPRE_StructMatrixCreate(comm, grid, stencil, &A);
181 HYPRE_StructMatrixInitialize(A);
182
183 HYPRE_StructStencilDestroy(stencil);
184
185 HYPRE_StructVectorCreate(comm, grid, &b);
186 HYPRE_StructVectorCreate(comm, grid, &x);
187
188 HYPRE_StructVectorInitialize(b);
189 HYPRE_StructVectorInitialize(x);
190
191 // A.SetValues() & A.assemble()
192 Array<HYPRE_Int,regular_stencil_size> stencil_indices;
193 std::iota(stencil_indices.begin(), stencil_indices.end(), 0);
194 const Real* dx = geom.CellSize();
195 const int bho = (m_maxorder > 2) ? 1 : 0;
196 BaseFab<GpuArray<Real,regular_stencil_size>> rfab;
197 for (MFIter mfi(acoefs); mfi.isValid(); ++mfi)
198 {
199 const Box &reg = mfi.validbox();
200
201 rfab.resize(reg);
202 amrex_hpacoef(reg, rfab, acoefs[mfi], scalar_a);
203
204 for (int idim = 0; idim < AMREX_SPACEDIM; idim++) {
205 amrex_hpbcoef(reg, rfab, bcoefs[idim][mfi], scalar_b, dx, idim);
206 }
207
208 const Vector< Vector<BoundCond> > & bcs_i = m_bndry->bndryConds(mfi);
209 const BndryData::RealTuple & bcl_i = m_bndry->bndryLocs(mfi);
210
211 // add b.c.'s for A matrix
212 for (OrientationIter oit; oit; oit++)
213 {
214 Orientation ori = oit();
215 int cdir(ori);
216 int idim = ori.coordDir();
217 const int bctype = bcs_i[cdir][0];
218 const Real &bcl = bcl_i[cdir];
219 const Mask &msk = m_bndry->bndryMasks(ori)[mfi];
220
221 amrex_hpmat(reg, rfab, bcoefs[idim][mfi], msk, scalar_b, dx, cdir, bctype, bcl, bho);
222 }
223
224 amrex_hpdiag(reg, rfab, diaginv[mfi]);
225 Real* mat = (Real*) rfab.dataPtr();
226
227 auto reglo = Hypre::loV(reg);
228 auto reghi = Hypre::hiV(reg);
229 HYPRE_StructMatrixSetBoxValues(A, reglo.data(), reghi.data(),
230 regular_stencil_size, stencil_indices.data(),
231 mat);
232 }
233 HYPRE_StructMatrixAssemble(A);
234
235 // create solver
236 HYPRE_StructPFMGCreate(comm, &solver);
237 int logging = (verbose >= 2) ? 1 : 0;
238 HYPRE_StructPFMGSetLogging(solver, logging);
239 HYPRE_StructPFMGSetup(solver, A, b, x);
240
241 HYPRE_StructVectorDestroy(b);
242 b = NULL;
243 HYPRE_StructVectorDestroy(x);
244 x = NULL;
245}

250{
251 BL_PROFILE("HypreABecLap::loadVectors()");
252
253 soln.setVal(0.0);
254
255 FArrayBox rhsfab;

263

265 rhsfab.copy<RunOn::Host>(rhs[mfi],reg);
266 rhsfab.mult<RunOn::Host>(diaginv[mfi]);
267

269 }
270}
