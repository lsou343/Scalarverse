47void
48MLMGBndry::setBoxBC (RealTuple& bloc, BCTuple& bctag, const Box& bx, const Box& domain,
49 const Array<LinOpBCType,AMREX_SPACEDIM>& lo,
50 const Array<LinOpBCType,AMREX_SPACEDIM>& hi,
51 const Real* dx, int ratio,
52 const RealVect& interior_bloc,
53 const Array<Real,AMREX_SPACEDIM>& domain_bloc_lo,
54 const Array<Real,AMREX_SPACEDIM>& domain_bloc_hi,
55 const GpuArray<int,AMREX_SPACEDIM>& is_periodic)
56{
57 for (OrientationIter fi; fi; ++fi)
58 {
59 const Orientation face = fi();
60 const int dir = face.coordDir();
61
62 if (domain[face] == bx[face] && !is_periodic[dir])
63 {
64 // All physical bc values are located on face.
65 bloc[face] = face.isLow() ? domain_bloc_lo[dir] : domain_bloc_hi[dir];
66 const auto linop_bc = face.isLow() ? lo[dir] : hi[dir];
67 if (linop_bc == LinOpBCType::Dirichlet) {
68 bctag[face] = AMREX_LO_DIRICHLET;
69 } else if (linop_bc == LinOpBCType::Neumann) {
70 bctag[face] = AMREX_LO_NEUMANN;
71 } else if (linop_bc == LinOpBCType::reflect_odd) {
72 bctag[face] = AMREX_LO_REFLECT_ODD;
73 } else {
74 amrex::Abort("MLMGBndry::setBoxBC: Unknown LinOpBCType");
75 }
76 }
77 else
78 {
79 // Internal bndry.
80 bctag[face] = AMREX_LO_DIRICHLET;
81 bloc[face] = ratio > 0 ? 0.5*ratio*dx[dir] : interior_bloc[dir];
82 // If this is next to another same level box, bloc is
83 // wrong. But it doesn't matter, because we also have
84 // mask. It is used only if mask says it is next to
85 // coarse cells.
86 }
87 }
88}
