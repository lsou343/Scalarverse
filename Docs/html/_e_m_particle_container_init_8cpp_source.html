8 AMREX_GPU_HOST_DEVICE void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
9 {
10 int nx = nppc[0];
11 int ny = nppc[1];
12 int nz = nppc[2];
13
14 int ix_part = i_part/(ny * nz);
15 int iy_part = (i_part % (ny * nz)) % ny;
16 int iz_part = (i_part % (ny * nz)) / ny;
17
18 r[0] = (0.5+ix_part)/nx;
19 r[1] = (0.5+iy_part)/ny;
20 r[2] = (0.5+iz_part)/nz;
21 }
