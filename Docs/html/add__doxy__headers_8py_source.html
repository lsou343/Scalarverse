31def make_method_header(description="", parameters=[], space=""):
32
33 # if both are empty, do nothing
34 if description == "":
35 return ""
36
37 # remove // from description
38 description = re.sub(r"//", "", description).strip()
39 description = re.sub(r"\n[ ]*", f"\n{space}* ", description)
40 # we need to escape <> or else they get confused for html tags
41 description = re.sub(r"<", "<", description)
42 description = re.sub(r">", ">", description)
43
44 if description == "":
45 # description was just //, so just return a new line
46 return """
47"""
48
49 boilerplate = ""
50
51 if description != "":
52
53 boilerplate += f"""{space}
54{space}/**
55{space}* \\brief {description}
56"""
57 elif parameters != []:
58 boilerplate += f"""{space}
59{space}/**
60"""
61
62 if parameters != []:
63 boilerplate += f"""{space}*
64"""
65 for param in parameters:
66 param_name = (param.split('=')[0].strip()).split(' ')[-1]
67 # if param name is surrounded by /* */ ignore it
68 re_commented = re.compile(r"\/\*(\w*)\*\/")
69 # if the name ends in & or *, then the prototype only lists
70 # the function type, not the parameter name
71 if param_name[-1] == '&' or param_name[-1] == '*' or re_commented.search(param_name):
72 continue
73 boilerplate += f"""{space}* \param {param_name}
74"""
75
76 boilerplate += f"""{space}*/
77"""
78
79 return boilerplate

340 description = re.sub(r"!", "", description).strip()
341 description = re.sub(r"\n[ ]*", "\n!! ", description)

408 re_subroutine = re.compile(
409 r"^[ \t]*(?:pure )?subroutine \S+\‍([\w ,&\n]*\‍)(?:[ &\n]*bind\‍(C, *name *= ?\"(\w+)\")?", flags=re.MULTILINE)
410
411 re_end_subroutine = re.compile(
412 r"^[ \t]*end subroutine", flags=re.MULTILINE)
413
414 re_comments = re.compile(
415 r"^[ \t]*!([\S \n]*?)(?=^[ \t]*[^!]*$)", flags=re.MULTILINE)
416
417 re_parameters = re.compile(
418 r"^[ \t]*([\w() ]+) *, *intent *\‍( *(in|out|inout) *\‍) *:: *([\w, ]+)", flags=re.MULTILINE)
419
420 with open(filename) as input_file:
421 data = input_file.read()
422
423 last_index = 0
424
425 for m in re.finditer(re_subroutine, data):
426
427 subroutine_end = re.search(
428 re_end_subroutine, data[m.end():]).start() + m.end()
429
430 # assume comments are after the prototype
431 comments = re.search(re_comments, data[m.end():subroutine_end])
432
433 parameters = re.findall(
434 re_parameters, data[m.end():subroutine_end])

443 comments.group(1), parameters=parameters)
