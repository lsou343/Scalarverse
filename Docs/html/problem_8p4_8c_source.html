6void evaluateBeta(double x, double y, double z, double *B, double *Bx, double *By, double *Bz){
7 double Bmin = 1.0;
8 double Bmax = 10.0;
9 double c2 = (Bmax-Bmin)/2; // coefficients to affect this transition
10 double c1 = (Bmax+Bmin)/2;
11 double c3 = 10.0; // how sharply (B)eta transitions
12 double xcenter = 0.50;
13 double ycenter = 0.50;
14 double zcenter = 0.50;
15 // calculate distance from center of the domain (0.5,0.5,0.5)
16 double r2 = pow((x-xcenter),2) + pow((y-ycenter),2) + pow((z-zcenter),2);
17 double r2x = 2.0*(x-xcenter);
18 double r2y = 2.0*(y-ycenter);
19 double r2z = 2.0*(z-zcenter);
20//double r2xx = 2.0;
21//double r2yy = 2.0;
22//double r2zz = 2.0;
23 double r = pow(r2,0.5);
24 double rx = 0.5*r2x*pow(r2,-0.5);
25 double ry = 0.5*r2y*pow(r2,-0.5);
26 double rz = 0.5*r2z*pow(r2,-0.5);
27//double rxx = 0.5*r2xx*pow(r2,-0.5) - 0.25*r2x*r2x*pow(r2,-1.5);
28//double ryy = 0.5*r2yy*pow(r2,-0.5) - 0.25*r2y*r2y*pow(r2,-1.5);
29//double rzz = 0.5*r2zz*pow(r2,-0.5) - 0.25*r2z*r2z*pow(r2,-1.5);
30 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
31 *B = c1+c2*tanh( c3*(r-0.25) );
32 *Bx = c2*c3*rx*(1-pow(tanh( c3*(r-0.25) ),2));
33 *By = c2*c3*ry*(1-pow(tanh( c3*(r-0.25) ),2));
34 *Bz = c2*c3*rz*(1-pow(tanh( c3*(r-0.25) ),2));
35}

39void evaluateU(double x, double y, double z, double *U, double *Ux, double *Uy, double *Uz, double *Uxx, double *Uyy, double *Uzz, int isPeriodic){
40 // should be continuous in u, u', and u''
41 // v(w) = w^4 - 2w^3 + w^2 + c
42 // u(x,y,z) = v(x)v(y)v(z)
43 // If Periodic, then the integral of the RHS should sum to zero.
44 // Setting shift=1/30 should ensure that the integrals of X, Y, or Z should sum to zero...
45 // That should(?) make the integrals of u,ux,uy,uz,uxx,uyy,uzz sum to zero and thus make the integral of f sum to zero
46 // If dirichlet, then w(0)=w(1) = 0.0
47 // Setting shift to 0 should ensure that U(x,y,z) = 0 on boundary
48 double shift = 0.0;if(isPeriodic)shift= -1.0/30.0;
49 double X = 1.0*pow(x,4) - 2.0*pow(x,3) + 1.0*pow(x,2) + shift;
50 double Y = 1.0*pow(y,4) - 2.0*pow(y,3) + 1.0*pow(y,2) + shift;
51 double Z = 1.0*pow(z,4) - 2.0*pow(z,3) + 1.0*pow(z,2) + shift;
52 double Xx = 4.0*pow(x,3) - 6.0*pow(x,2) + 2.0*x;
53 double Yy = 4.0*pow(y,3) - 6.0*pow(y,2) + 2.0*y;
54 double Zz = 4.0*pow(z,3) - 6.0*pow(z,2) + 2.0*z;
55 double Xxx = 12.0*pow(x,2) - 12.0*x + 2.0;
56 double Yyy = 12.0*pow(y,2) - 12.0*y + 2.0;
57 double Zzz = 12.0*pow(z,2) - 12.0*z + 2.0;
58 *U = X*Y*Z;
59 *Ux = Xx*Y*Z;
60 *Uy = X*Yy*Z;
61 *Uz = X*Y*Zz;
62 *Uxx = Xxx*Y*Z;
63 *Uyy = X*Yyy*Z;
64 *Uzz = X*Y*Zzz;
65}
