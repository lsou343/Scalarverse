31void
32writePlotFile (const std::string& dir,
33 std::ostream& os,
34 int level,
35 const MultiFab& mf,
36 const Geometry& geom,
37 const IntVect& refRatio,
38 Real bgVal,
39 const Vector<std::string>& names)
40{
41 //
42 // Faked data
43 //
44 const int NUM_STATE = mf.nComp();
45 const Real cumTime = 0.0;
46 const Real curTime = cumTime;
47 const int finestLevel = 1;
48 Vector< Box > domain(finestLevel+1);
49 const IndexType& ixType = mf.boxArray().ixType();
50 if (ixType != IndexType::TheCellType())
51 amrex::Error("writePlotfile unable to handle non cell-centered data for now");
52 Box tmpb = Box(geom.Domain()).convert(ixType);
53 Vector<int> corr(BL_SPACEDIM);
54 for (int d = 0; d < BL_SPACEDIM; d++)
55 {
56 corr[d] = (ixType.ixType(d) == IndexType::CELL ? 1 : 0);
57 }
58 for (int M = 0; M < finestLevel; M++)
59 {
60 tmpb.coarsen(refRatio);
61 }
62 domain[0] = tmpb;
63 const int levelSteps = 0;
64 Vector< Vector< Real > > dx(finestLevel+1);
65 Vector< Vector<RealBox> > grid_loc(finestLevel+1);
66 const BoxArray& grids = mf.boxArray();
67 for (int j = 0; j<= finestLevel; j++)
68 {
69 if (j!=0)
70 domain[j] = Box( domain[j-1] ).refine(refRatio);
71 dx[j].resize(BL_SPACEDIM);
72 for (int k = 0; k < BL_SPACEDIM; k++)
73 {
74 dx[j][k] = (geom.ProbHi(k) - geom.ProbLo(k))/domain[j].length(k);
75 }
76 if (j==0)
77 {
78 grid_loc[j].resize(1);
79 grid_loc[j][0] = RealBox(geom.ProbLo(),geom.ProbHi());
80 } else {
81 grid_loc[j].resize(grids.size());
82 for (int L=0; L < grids.size(); L++)
83 {
84 const Box& bx = grids[L];
85 grid_loc[j][L] = RealBox(AMREX_D_DECL( dx[j][0]*bx.smallEnd(0),
86 dx[j][1]*bx.smallEnd(1),
87 dx[j][2]*bx.smallEnd(2) ),
88 AMREX_D_DECL( dx[j][0]*(bx.bigEnd(0)+corr[0]),
89 dx[j][1]*(bx.bigEnd(1)+corr[1]),
90 dx[j][2]*(bx.bigEnd(2)+corr[2]) ));
91 }
92 }
93 }
94 const int Coord = 0;
95 BoxArray tba = BoxArray(&tmpb,1);
96 DistributionMapping dm {tba};
97 MultiFab level0_dat(tba,dm,mf.nComp(),mf.nGrow());
98 for (int j=0; j<mf.nComp(); ++j)
99 level0_dat.setVal(0.5*(mf.min(j)+mf.max(j)),j,1);
100 //level0_dat.setVal(bgVal);
101
102 int i, n;
103 //
104 // There is only one MultiFab written out at each level in HyperCLaw.
105 //
106 static const std::string MultiFabBaseName("/MultiFab");
107 //
108 // Build the directory to hold the MultiFabs at this level.
109 // The name is relative to the directory containing the Header file.
110 //
111 std::string Level = amrex::Concatenate("Level_", level, 1);
112 //
113 // Now for the full pathname of that directory.
114 //
115 std::string FullPath = dir;
116 if (!FullPath.empty() && FullPath[FullPath.length()-1] != '/')
117 FullPath += '/';
118 FullPath += Level;
119 //
120 // Only the I/O processor makes the directory if it doesn't already exist.
121 //
122 if (ParallelDescriptor::IOProcessor())
123 if (!amrex::UtilCreateDirectory(FullPath, 0755))
124 amrex::CreateDirectoryFailed(FullPath);
125 //
126 // Force other processors to wait till directory is built.
127 //
128 ParallelDescriptor::Barrier();
129
130 if (ParallelDescriptor::IOProcessor())
131 {
132 if (level == 0)
133 {
134 //
135 // The first thing we write out is the plot file type.
136 //
137 os << thePlotFileType() << '\n';
138 // Number of components, and names
139 int n_var = NUM_STATE;
140 os << n_var << '\n';
141 for (n = 0; n < NUM_STATE; n++)
142 {
143 if (names.size()==0)
144 {
145 std::string name = amrex::Concatenate("state_", n, 1);
146
147 os << name << '\n';
148 }
149 else
150 {
151 os << names[n] << '\n';
152 }
153 }
154 // dimensionality
155 os << BL_SPACEDIM << '\n';
156 // time
157 os << cumTime << '\n';
158 // finest amr level
159 os << finestLevel << '\n';
160 // prob domain
161 for (i = 0; i < BL_SPACEDIM; i++) os << geom.ProbLo(i) << ' ';
162 os << '\n';
163 for (i = 0; i < BL_SPACEDIM; i++) os << geom.ProbHi(i) << ' ';
164 os << '\n';
165 // refinement ratio
166 for (i = 0; i < finestLevel; i++) os << refRatio << ' ';
167 os << '\n';
168 // int domain
169 for (i = 0; i <= finestLevel; i++) os << domain[i] << ' ';
170 os << '\n';
171 // level steps
172 for (i = 0; i <= finestLevel; i++) os << levelSteps << ' ';
173 os << '\n';
174 for (i = 0; i <= finestLevel; i++)
175 {
176 // cell size
177 const Real* dx_lev = dx[i].dataPtr();
178 for (int k = 0; k < BL_SPACEDIM; k++) os << dx_lev[k] << ' ';
179 os << '\n';
180 }
181 // coordinate system
182 os << Coord << '\n';
183 os << "0\n"; // The bndry data.
184 }
185 //
186 // Now write state data.
187 //
188 int ngrds = (level==0 ? 1 : grids.size());
189 Real cur_time = curTime;
190 const MultiFab& cell_dat = (level==0 ? level0_dat : mf);
191
192 os << level << ' ' << ngrds << ' ' << cur_time << '\n';
193 // level steps
194 os << levelSteps << '\n';
195
196 for (i = 0; i < cell_dat.boxArray().size(); ++i)
197 {
198 for (n = 0; n < BL_SPACEDIM; n++)
199 // lo/hi position of this grid
200 os <<grid_loc[level][i].lo(n) << ' ' << grid_loc[level][i].hi(n) << '\n';
201 }
202
203 //
204 // Finally, the full relative pathname of the MultiFab.
205 // The name is relative to the Header file containing this name.
206 // It's the name that gets written into the Header.
207 //
208 std::string PathNameInHeader = Level;
209 PathNameInHeader += MultiFabBaseName;
210
211 os << PathNameInHeader << '\n';
212 }
213 //
214 // Now amend FullPath to contain full pathname of MF.
215 //
216 FullPath += MultiFabBaseName;
217
218 //RunStats::addBytes(VisMF::Write(state[State_Type].newData(),FullPath,how));
219 const MultiFab& cell_dat = (level==0 ? level0_dat : mf);
220 VisMF::Write(cell_dat,FullPath,VisMF::OneFilePerCPU);
221}

223void
224writePlotFile (const char* name,
225 const MultiFab& mf,
226 const Geometry& geom,
227 const IntVect& refRatio,
228 Real bgVal,
229 const Vector<std::string>& names)
230{
231
232 Real dPlotFileTime0(ParallelDescriptor::second());
233 //std::string pltfile = Concatenate(root, num);
234 std::string pltfile(name);
235
236 //
237 // Only the I/O processor makes the directory if it doesn't already exist.
238 //
239 if (ParallelDescriptor::IOProcessor())
240 if (!amrex::UtilCreateDirectory(pltfile, 0755))
241 amrex::CreateDirectoryFailed(pltfile);
242 //
243 // Force other processors to wait till directory is built.
244 //
245 ParallelDescriptor::Barrier();
246
247 std::string HeaderFileName = pltfile + "/Header";
248
249 VisMF::IO_Buffer io_buffer(VisMF::IO_Buffer_Size);
250
251 std::ofstream HeaderFile;
252
253 HeaderFile.rdbuf()->pubsetbuf(io_buffer.dataPtr(), io_buffer.size());
254
255 int old_prec;
256
257 if (ParallelDescriptor::IOProcessor())
258 {
259 //
260 // Only the IOProcessor() writes to the header file.
261 //
262 HeaderFile.open(HeaderFileName.c_str(), std::ios::out|std::ios::trunc);
263
264 if (!HeaderFile.good())
265 amrex::FileOpenFailed(HeaderFileName);
266
267 old_prec = HeaderFile.precision(30);
268 }
269
270 int finest_level = 1;
271 for (int k = 0; k <= finest_level; k++)
272 {
273 writePlotFile(pltfile, HeaderFile, k, mf, geom, refRatio, bgVal, names);
274 }
275
276 if (ParallelDescriptor::IOProcessor())
277 {
278 HeaderFile.precision(old_prec);
279
280 if (!HeaderFile.good())
281 amrex::Error("Amr::writePlotFile() failed");
282 }
283
284 Real dPlotFileTime1(ParallelDescriptor::second());
285 Real dPlotFileTime(dPlotFileTime1 - dPlotFileTime0);
286 ParallelDescriptor::ReduceRealMax(dPlotFileTime);
287 if(ParallelDescriptor::IOProcessor()) {
288 std::cout << "Write plotfile time = " << dPlotFileTime << " seconds." << std::endl;
289 }
290
291}

293void WritePlotFile(const Vector<MultiFab*>& mfa,
294 const Vector<Box>& probDomain,
295 AmrData& amrdToMimic,
296 const std::string& oFile,
297 bool verbose,
298 const Vector<std::string>& varNames)
299{
300 // If varnames not provided, use names in original plotfile
301 const Vector<std::string>& derives = (varNames.size()==0 ? amrdToMimic.PlotVarNames() : varNames);
302 AMREX_ASSERT(derives.size()==(*mfa[0]).nComp());
303 int ntype = derives.size();
304 int finestLevel = mfa.size() - 1;
305 AMREX_ALWAYS_ASSERT(finestLevel >= 0);
306
307 if (ParallelDescriptor::IOProcessor())
308 if (!amrex::UtilCreateDirectory(oFile,0755))
309 amrex::CreateDirectoryFailed(oFile);
310 //
311 // Force other processors to wait till directory is built.
312 //
313 ParallelDescriptor::Barrier();
314
315 std::string oFileHeader(oFile);
316 oFileHeader += "/Header";
317
318 VisMF::IO_Buffer io_buffer(VisMF::IO_Buffer_Size);
319
320 std::ofstream os;
321
322 os.rdbuf()->pubsetbuf(io_buffer.dataPtr(), io_buffer.size());
323
324 if (verbose && ParallelDescriptor::IOProcessor())
325 std::cout << "Opening file = " << oFileHeader << '\n';
326
327 os.open(oFileHeader.c_str(), std::ios::out|std::ios::binary);
328
329 if (os.fail())
330 amrex::FileOpenFailed(oFileHeader);
331 //
332 // Start writing plotfile.
333 //
334 os << amrdToMimic.PlotFileVersion() << '\n';
335 int n_var = ntype;
336 os << n_var << '\n';
337 for (int n = 0; n < ntype; n++) os << derives[n] << '\n';
338 os << BL_SPACEDIM << '\n';
339 os << amrdToMimic.Time() << '\n';
340 os << finestLevel << '\n';
341 int i;
342 for (i = 0; i < BL_SPACEDIM; i++) os << amrdToMimic.ProbLo()[i] << ' ';
343 os << '\n';
344 for (i = 0; i < BL_SPACEDIM; i++) os << amrdToMimic.ProbHi()[i] << ' ';
345 os << '\n';
346 for (i = 0; i < finestLevel; i++) os << amrdToMimic.RefRatio()[i] << ' ';
347 os << '\n';
348 for (i = 0; i <= finestLevel; i++) os << probDomain[i] << ' ';
349 os << '\n';
350 for (i = 0; i <= finestLevel; i++) os << 0 << ' ';
351 os << '\n';
352 for (i = 0; i <= finestLevel; i++)
353 {
354 for (int k = 0; k < BL_SPACEDIM; k++)
355 os << amrdToMimic.DxLevel()[i][k] << ' ';
356 os << '\n';
357 }
358 os << amrdToMimic.CoordSys() << '\n';
359 os << "0\n"; // The bndry data width.
360 //
361 // Write out level by level.
362 //
363 for (int iLevel = 0; iLevel <= finestLevel; ++iLevel)
364 {
365 //
366 // Write state data.
367 //
368 int nGrids = amrdToMimic.boxArray(iLevel).size();
369
370 std::string LevelStr = amrex::Concatenate("Level_", iLevel, 1);
371
372 if (ParallelDescriptor::IOProcessor())
373 {
374 os << iLevel << ' ' << nGrids << ' ' << amrdToMimic.Time() << '\n';
375 os << 0 << '\n';
376
377 for (i = 0; i < nGrids; ++i)
378 {
379 for (int n = 0; n < BL_SPACEDIM; n++)
380 {
381 os << amrdToMimic.GridLocLo()[iLevel][i][n]
382 << ' '
383 << amrdToMimic.GridLocHi()[iLevel][i][n]
384 << '\n';
385 }
386 }
387 //
388 // Build the directory to hold the MultiFabs at this level.
389 //
390 std::string Level(oFile);
391 Level += '/';
392 Level += LevelStr;
393
394 if (!amrex::UtilCreateDirectory(Level, 0755))
395 amrex::CreateDirectoryFailed(Level);
396 }
397 //
398 // Force other processors to wait till directory is built.
399 //
400 ParallelDescriptor::Barrier();
401 //
402 // Now build the full relative pathname of the MultiFab.
403 //
404 static const std::string MultiFabBaseName("/MultiFab");
405
406 std::string PathName(oFile);
407 PathName += '/';
408 PathName += LevelStr;
409 PathName += MultiFabBaseName;
410
411 if (ParallelDescriptor::IOProcessor())
412 {
413 //
414 // The full name relative to the Header file.
415 //
416 std::string RelativePathName(LevelStr);
417 RelativePathName += '/';
418 RelativePathName += MultiFabBaseName;
419 os << RelativePathName << '\n';
420 }
421 VisMF::Write(*mfa[iLevel], PathName, VisMF::OneFilePerCPU);
422 }
423
424 os.close();
425}
