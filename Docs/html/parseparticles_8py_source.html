1"""
2 a module for parsing MAESTRO particle files
3
4 Particle histories in MAESTRO are stored in a set of files, e.g.
5 timestamp_#, where the # in the file name is the number of the CPU
6 (actually MPI process) with particles. As particles advect, they
7 may jump to different CPUs, so a given particle's history may
8 initially be written out in one file and then switch to a different
9 file. This module sorts the particle data to be in proper time
10 order.
11
12 This module should be imported in your own python script, for example:
13
14 >>> import parseparticles
15 >>> myparticles = parseparticles.parseParticleFile(myParticleFiles)
16
17 where myParticleFiles is a list of the timestamp files containing
18 the particle data.
19
20 This returns a dictionary whose keys are internal ids (comprised
21 of the particle's id and its origin CPU) and values are particle
22 objects, which contain all of the associated information about a
23 particle's history
24
25
26 Usually we don't care about the original particle ID information, so
27 we just strip out the particle objects into a list:
28
29 >>> particles = myparticles.values()
30
31 Here, each item in the list is a particle object.
32
33
34 Each particle object contains data describing the particle
35 identity, as well as a history. The history itself is a list of
36 particleInstance objects. Each particleInstance object stores
37 the state of the particle at an instance in time. For example:
38
39 >>> time = particles[n].history[i].t
40
41 is the time associated with particleInstance i of particle n.
42 There is a unique particleInstance for particle n for every
43 timestep of this particle's existence.
44
45 Coordinate information is accessed as:
46
47 >>> x = particles[n].history[i].xyz[0]
48
49 for the x-coordinate (xyz[1] would be y, etc.).
50
51 State data is stored in a numpy array. The state names
52 corresponding to each data element are stored in the main particle
53 object and can be looked-up via the getVarIndex() method. For
54 example, to store the density of particle n at timestep i into the
55 dens variable, we would do:
56
57 >>> idens = particles[n].getVarIndex("density")
58 >>> dens = particles[n].history[i].data[idens]
59
60 Any field that was stored in the original timestamp file would be
61 present in the data array (and the name should be available
62 via getVarIndex())
63
64
65 The total number of particles can be found simply as len(particles),
66 and the number of particleInstances for a given particle can be
67 found as len(particles[n].history)
68"""

295def parseParticleFile(maestroParticleFiles):
296 """
297 read in all the particle data from a file and add each particle
298 instance to the _particleDict dictionary, grouping all of the
299 history for a unique particle (identified by the pid and
300 originCPU) together in a particle object in the dictionary
301 """
302
303 haveHeader = 0
304
305 for maestroParticleFile in ensure_list(maestroParticleFiles):
306
307 # read the file line by line
308 mf = open(maestroParticleFile, "r")
309
310 for line in mf:
311
312 # skip blank lines
313 if line.isspace(): continue
314
315 # look for a header information
316 if (line.startswith("#")):
317
318 # store old header, if it exists
319 if (haveHeader == 1):
320 oldHeader = list(dataNames) # list() makes a copy
321
322
323 fields = string.split(line[1:])
324
325 # make sure we know what we are doing -- the first 2
326 # fields should be the particle ID and origin CPU
327 if (fields[0] == "part-ID" and fields[1] == "origin-CPU"):
328 ipid = 0
329 ioCPU = 1
330
331 else:
332 print "ERROR: particle file columns not in expected order"
333 sys.exit(2)
334
335
336 # the next fields should be x, y, and z, depending on the
337 # dimensionality
338 if (fields[2] == "x" and fields[3] == "y" and
339 fields[4] == "z"):
340 dim = 3
341 ix = 2; iy = 3; iz = 4
342
343 elif (fields[2] == "x" and fields[3] == "y"):
344 dim = 2
345 ix = 2; iy = 3
346
347 elif (fields[2] == "x"):
348 dim = 1
349 ix = 2
350
351 else:
352 print "ERROR: particle file columns not in expected order"
353 sys.exit(2)
354
355
356 # then comes time
357 if (fields[2 + dim] == "time"):
358 it = 2 + dim
359
360 else:
361 print "ERROR: particle file columns not in expected order"
362 sys.exit(2)
363
364
365 # everything else is associated data
366 if (len(fields) > 3 + dim):
367 idata = 3 + dim
368 ndata = len(fields) - idata
369 dataNames = fields[idata:]
370 else:
371 ndata = 0
372
373
374 # did the header change?
375 if (haveHeader == 1):
376 if (not oldHeader == dataNames):
377 print "ERROR: header changed while parsing files"
378 sys.exit(2)
379
380
381 # done with the header
382 haveHeader = 1
383 numData = len(dataNames)
384 continue
385
386
387 # normal particle data -- first find the required particle data
388 fields = map(float,string.split(line))
389
390 pid = int(fields[ipid])
391 originCPU = int(fields[ioCPU])
392
393 id = pid*_idFactor + originCPU
394
395 xyz = fields[ix:ix+dim]
396 time = fields[it]
397 dataValues = fields[idata:]
398
399 if (not len(dataValues) == numData):
400 print "ERROR: number of data values not equal to number of names"
401 sys.exit(2)
402
403
404 if not id in _particleDict:
405 particle(pid, originCPU, dim, dataNames)
406
407 _particleDict[id].addInstance(xyz=xyz,
408 t=time,
409 dataValues=dataValues)
410
411
412 # finish up by making sure that each particle's history is
413 # in proper time-order
414 for id in _particleDict:
415 _particleDict[id].finalize()
416
417
418 return _particleDict
