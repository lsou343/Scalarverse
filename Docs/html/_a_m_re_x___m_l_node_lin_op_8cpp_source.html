140Real
141MLNodeLinOp::xdoty (int amrlev, int mglev, const MultiFab& x, const MultiFab& y, bool local) const
142{
143 amrex::ignore_unused(amrlev);
144 AMREX_ASSERT(amrlev==0);
145 AMREX_ASSERT(mglev+1==m_num_mg_levels[0] || mglev==0);
146 const auto& mask = (mglev+1 == m_num_mg_levels[0]) ? m_bottom_dot_mask : m_coarse_dot_mask;
147 const int ncomp = y.nComp();
148 const int nghost = 0;
149 MultiFab tmp(x.boxArray(), x.DistributionMap(), ncomp, 0);
150 MultiFab::Copy(tmp, x, 0, 0, ncomp, nghost);
151 for (int i = 0; i < ncomp; i++) {
152 MultiFab::Multiply(tmp, mask, 0, i, 1, nghost);
153 }
154 Real result = MultiFab::Dot(tmp,0,y,0,ncomp,nghost,true);
155 if (!local) {
156 ParallelAllReduce::Sum(result, ParallelContext::CommunicatorSub());
157 }
158 return result;
159}

216
217 m_masks_built = true;
218
219 m_is_bottom_singular = false;
220 auto itlo = std::find(m_lobc[0].begin(), m_lobc[0].end(), BCType::Dirichlet);
221 auto ithi = std::find(m_hibc[0].begin(), m_hibc[0].end(), BCType::Dirichlet);
222 if (itlo == m_lobc[0].end() && ithi == m_hibc[0].end())
223 { // No Dirichlet

232 for (int mglev = 0; mglev < m_num_mg_levels[amrlev]; ++mglev)
233 {
234 const Geometry& geom = m_geom[amrlev][mglev];
235 const auto& period = geom.periodicity();
236 const Box& ccdomain = geom.Domain();
237 const Box& nddomain = amrex::surroundingNodes(ccdomain);
238
239 auto& dmask = *m_dirichlet_mask[amrlev][mglev];
240
241 iMultiFab ccm(m_grids[amrlev][mglev],m_dmap[amrlev][mglev],1,1);
242 ccm.BuildMask(ccdomain,period,0,1,2,0);
243
244 MFItInfo mfi_info;
245 if (Gpu::notInLaunchRegion()) mfi_info.SetDynamic(true);
246
247 if (m_overset_dirichlet_mask and mglev > 0) {
248 const auto& dmask_fine = *m_dirichlet_mask[amrlev][mglev-1];
249 amrex::average_down_nodal(dmask_fine, dmask, IntVect(2));
250 }
251#ifdef _OPENMP
252#pragma omp parallel if (Gpu::notInLaunchRegion())
253#endif
254 for (MFIter mfi(dmask, mfi_info); mfi.isValid(); ++mfi)

259 AMREX_LAUNCH_HOST_DEVICE_LAMBDA ( ndbx, tbx,

263 }
264 }
