321 Index src_start, Index dst_start, N n) noexcept
322{
323 auto np = src.numParticles();
324 Gpu::DeviceVector<Index> offsets(np);
325 Gpu::exclusive_scan(mask, mask+np, offsets.begin());
326
327 Index last_mask, last_offset;
328 Gpu::copyAsync(Gpu::deviceToHost, mask+np-1, mask + np, &last_mask);
329 Gpu::copyAsync(Gpu::deviceToHost, offsets.data()+np-1, offsets.data()+np, &last_offset);
330
331 auto p_offsets = offsets.dataPtr();
332
333 const auto src_data = src.getConstParticleTileData();
334 auto dst_data = dst.getParticleTileData();
335
336 AMREX_HOST_DEVICE_FOR_1D( n, i,
337 {
338 if (mask[src_start+i]) copyParticle(dst_data, src_data, src_start+i,
339 dst_start+p_offsets[src_start+i]);
340 });

342 Gpu::synchronize();
