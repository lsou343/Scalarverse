56/*
57 A Collection of Fortran Array-like Objects
58
59
60 The FabArray<FAB> class implements a collection (stored as an array) of
61 Fortran array-like objects. The parameterized type FAB is intended to be
62 any class derived from BaseFab<T>. For example, FAB may be a BaseFab of
63 integers, so we could write:
64
65 FabArray<BaseFab<int> > int_fabs;
66
67 Then int_fabs is a FabArray that can hold a collection of BaseFab<int>
68 objects.
69
70 FabArray is not just a general container class for Fortran arrays. It is
71 intended to hold "grid" data for use in finite difference calculations in
72 which the data is defined on a union of (usually disjoint) rectangular
73 regions embedded in a uniform index space. This region, called the valid
74 region, is represented by a BoxArray. For the purposes of this discussion,
75 the Kth Box in the BoxArray represents the interior region of the Kth grid.
76
77 Since the intent is to be used with finite difference calculations a
78 FabArray also includes the notion of a boundary region for each grid. The
79 boundary region is specified by the ngrow parameter which tells the FabArray
80 to allocate each FAB to be ngrow cells larger in all directions than the
81 underlying Box. The larger region covered by the union of all the FABs is
82 called the region of definition. The underlying notion is that the valid
83 region contains the grid interior data and the region of definition includes
84 the interior region plus the boundary areas.
85
86 Operations are available to copy data from the valid regions into these
87 boundary areas where the two overlap. The number of components, that is,
88 the number of values that can be stored in each cell of a FAB, is either
89 given as an argument to the constructor or is inherent in the definition of
90 the underlying FAB. Each FAB in the FabArray will have the same number of
91 components.
92
93 In summary, a FabArray is an array of FABs. The Kth element contains a FAB
94 that holds the data for the Kth grid, a Box that defines the valid region
95 of the Kth grid.
96
97 A typical use for a FabArray would be to hold the solution vector or
98 right-hand-side when solving a linear system of equations on a union of
99 rectangular grids. The copy operations would be used to copy data from the
100 valid regions of neighboring grids into the boundary regions after each
101 relaxation step of the iterative method. If a multigrid method is used, a
102 FabArray could be used to hold the data at each level in the multigrid
103 hierarchy.
104
105 This class is a concrete class not a polymorphic one.
106
107 This class does NOT provide a copy constructor or assignment operator.
108*/

1291template <class FAB>
1292void
1293FabArray<FAB>::AllocFabs (const FabFactory<FAB>& factory, Arena* ar,
1294 const Vector<std::string>& tags)
1295{
1296 const int n = indexArray.size();
1297 const int nworkers = ParallelDescriptor::TeamSize();
1298 shmem.alloc = (nworkers > 1);
1299
1300 bool alloc = !shmem.alloc;
1301
1302 FabInfo fab_info;
1303 fab_info.SetAlloc(alloc).SetShared(shmem.alloc).SetArena(ar);
1304
1305 m_fabs_v.reserve(n);
1306
1307 Long nbytes = 0L;
1308 for (int i = 0; i < n; ++i)
1309 {
1310 int K = indexArray[i];
1311 const Box& tmpbox = fabbox(K);
1312 m_fabs_v.push_back(factory.create(tmpbox, n_comp, fab_info, K));
1313 nbytes += amrex::nBytesOwned(*m_fabs_v.back());
1314 }
1315
1316 m_tags.clear();
1317 m_tags.emplace_back("All");
1318 for (auto const& t : m_region_tag) {
1319 m_tags.push_back(t);
1320 }
1321 for (auto const& t : tags) {
1322 m_tags.push_back(t);
1323 }
1324 for (auto const& t: m_tags) {
1325 updateMemUsage(t, nbytes, ar);
1326 }
1327
1328#ifdef BL_USE_TEAM
1329 if (shmem.alloc)
1330 {
1331 const int teamlead = ParallelDescriptor::MyTeamLead();
1332
1333 shmem.n_values = 0;
1334 shmem.n_points = 0;
1335 Vector<Long> offset(n,0);
1336 Vector<Long> nextoffset(nworkers,-1);
1337 for (int i = 0; i < n; ++i) {
1338 int K = indexArray[i];
1339 int owner = distributionMap[K] - teamlead;
1340 Long s = m_fabs_v[i]->size();
1341 if (ownership[i]) {
1342 shmem.n_values += s;
1343 shmem.n_points += m_fabs_v[i]->numPts();
1344 }
1345 if (nextoffset[owner] < 0) {
1346 offset[i] = 0;
1347 nextoffset[owner] = s;
1348 } else {
1349 offset[i] = nextoffset[owner];
1350 nextoffset[owner] += s;
1351 }
1352 }
1353
1354 size_t bytes = shmem.n_values*sizeof(value_type);
1355
1356 value_type *mfp;
1357 Vector<value_type*> dps;
1358
1359#if defined (BL_USE_MPI3)
1360
1361 static MPI_Info info = MPI_INFO_NULL;
1362 if (info == MPI_INFO_NULL) {
1363 MPI_Info_create(&info);
1364 MPI_Info_set(info, "alloc_shared_noncontig", "true");
1365 }
1366
1367 const MPI_Comm& team_comm = ParallelDescriptor::MyTeam().get();
1368
1369 BL_MPI_REQUIRE( MPI_Win_allocate_shared(bytes, sizeof(value_type),
1370 info, team_comm, &mfp, &shmem.win) );
1371
1372 for (int w = 0; w < nworkers; ++w) {
1373 MPI_Aint sz;
1374 int disp;
1375 value_type *dptr = 0;
1376 BL_MPI_REQUIRE( MPI_Win_shared_query(shmem.win, w, &sz, &disp, &dptr) );
1377 // BL_ASSERT(disp == sizeof(value_type));
1378 dps.push_back(dptr);
1379 }
1380
1381#else
1382
1383 amrex::Abort("BaseFab::define: to allocate shared memory, USE_MPI3 must be true");
1384
1385#endif
1386
1387 for (int i = 0; i < n; ++i) {
1388 int K = indexArray[i];
1389 int owner = distributionMap[K] - teamlead;
1390 value_type *p = dps[owner] + offset[i];
1391 m_fabs_v[i]->setPtr(p, m_fabs_v[i]->size());
1392 }
1393
1394 for (Long i = 0; i < shmem.n_values; i++, mfp++) {
1395 new (mfp) value_type;
1396 }
1397
1398 amrex::update_fab_stats(shmem.n_points, shmem.n_values, sizeof(value_type));
1399 }
1400#endif
1401}

1965FabArray<FAB>::BuildMask (const Box& phys_domain, const Periodicity& period,
1966 value_type covered, value_type notcovered,
1967 value_type physbnd, value_type interior)
1968{
1969 int ncomp = this->nComp();
1970 const IntVect& ngrow = this->nGrowVect();
1971
1972 Box domain = amrex::convert(phys_domain, boxArray().ixType());
1973 for (int i = 0; i < AMREX_SPACEDIM; ++i) {
1974 if (period.isPeriodic(i)) {
1975 domain.grow(i, ngrow[i]);
1976 }
1977 }
1978
1979#ifdef _OPENMP
1980#pragma omp parallel if (Gpu::notInLaunchRegion())
1981#endif
1982 for (MFIter mfi(*this,TilingIfNotGPU()); mfi.isValid(); ++mfi)
1983 {
1984 auto const& fab = this->array(mfi);
1985 Box const& fbx = mfi.growntilebox();
1986 Box const& gbx = fbx & domain;
1987 Box const& vbx = mfi.validbox();
1988 AMREX_HOST_DEVICE_FOR_4D_FUSIBLE(fbx, ncomp, i, j, k, n,
1989 {
1990 IntVect iv(AMREX_D_DECL(i,j,k));
1991 if (vbx.contains(iv)) {
1992 fab(i,j,k,n) = interior;
1993 } else if (gbx.contains(iv)) {
1994 fab(i,j,k,n) = notcovered;
1995 } else {
1996 fab(i,j,k,n) = physbnd;
1997 }
1998 });
1999 }
2000
2001 const FabArrayBase::FB& TheFB = this->getFB(ngrow,period);
2002 setVal(covered, TheFB, 0, ncomp);
2003}
