343CoordSys::GetEdgeVolCoord (Vector<Real>& vc,
344 const Box& region,
345 int dir) const
346{
347 //
348 // In cartesian and Z direction of RZ volume coordinates
349 // are idential to physical distance from axis.
350 //
351 GetEdgeLoc(vc,region,dir);
352 //
353 // In R direction of RZ, vol coord = (r^2)/2
354 //
355#if (AMREX_SPACEDIM == 2)
356 if (dir == 0 && c_sys == RZ)
357 {
358 int len = vc.size();
359 AMREX_PRAGMA_SIMD
360 for (int i = 0; i < len; i++)
361 {
362 Real r = vc[i];
363 vc[i] = 0.5*r*r;
364 }
365 }
366#elif (AMREX_SPACEDIM == 1)
367 if (c_sys == SPHERICAL)
368 {
369 int len = vc.size();
370 AMREX_PRAGMA_SIMD
371 for (int i = 0; i < len; i++) {
372 Real r = vc[i];
373 vc[i] = FOURPI/3.*r*r*r;
374 }
375 }
376#endif

474CoordSys::Volume (const Real xlo[AMREX_SPACEDIM],
475 const Real xhi[AMREX_SPACEDIM]) const
476{
477 switch (c_sys)
