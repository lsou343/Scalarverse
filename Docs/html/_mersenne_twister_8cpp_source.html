5 and a description of the Mersenne Twister random number generator can be found here:
6 http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html
7 It has a claimed periodicity of 2^{19937}-1, which is pretty incredible. The algorithm
8 as-is is mainly intended for Monte Carlo realizations, and is not intended to be used for

61unsigned long int mt_random() {
62 unsigned long int * b = mt_buffer;
63 int idx = mt_index;
64 unsigned long int s;
65 int i;
66
67 if (idx == MT_LEN*sizeof(unsigned long int))
68 {
69 idx = 0;
70 i = 0;
71 for (; i < MT_IB; i++) {
72 s = TWIST(b, i, i+1);
73 b[i] = b[i + MT_IA] ^ (s >> 1) ^ MAGIC(s);
74 }
75 for (; i < MT_LEN-1; i++) {
76 s = TWIST(b, i, i+1);
77 b[i] = b[i - MT_IB] ^ (s >> 1) ^ MAGIC(s);
78 }
79
80 s = TWIST(b, MT_LEN-1, 0);
81 b[MT_LEN-1] = b[MT_IA-1] ^ (s >> 1) ^ MAGIC(s);
82 }
83 mt_index = idx + sizeof(unsigned long int);
84 return *(unsigned long int *)((unsigned char *)b + idx);
