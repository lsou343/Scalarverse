71def read_amrex_binary_particle_file(fn, ptype="particle0"):
72 '''
73
74 This function returns the particle data stored in a particular
75 plot file and particle type. It returns two numpy arrays, the
76 first containing the particle integer data, and the second the
77 particle real data. For example, if a dataset has 3000 particles,
78 which have two integer and five real components, this function will
79 return two numpy arrays, one with the shape (3000, 2) and the other
80 with the shape (3000, 5).
81
82 Usage:
83
84 idata, rdata = read_particle_data("plt00000", "particle0")
85
86 '''
87 base_fn = fn + "/" + ptype
88 header = AMReXParticleHeader(base_fn + "/Header")
89
90 idtype = "(%d,)i4" % header.num_int
91 if header.real_type == np.float64:
92 fdtype = "(%d,)f8" % header.num_real
93 elif header.real_type == np.float32:
94 fdtype = "(%d,)f4" % header.num_real
95
96 idata = np.empty((header.num_particles, header.num_int ))
97 rdata = np.empty((header.num_particles, header.num_real))
98
99 ip = 0
100 for lvl, level_grids in enumerate(header.grids):
101 for (which, count, where) in level_grids:
102 if count == 0: continue
103 fn = base_fn + "/Level_%d/DATA_%05d" % (lvl, which)
104
105 with open(fn, 'rb') as f:
106 f.seek(where)
107 if header.is_checkpoint:
108 ints = np.fromfile(f, dtype = idtype, count=count)
109 idata[ip:ip+count] = ints
110
111 floats = np.fromfile(f, dtype = fdtype, count=count)
112 rdata[ip:ip+count] = floats
113 ip += count
114
115 return idata, rdata
