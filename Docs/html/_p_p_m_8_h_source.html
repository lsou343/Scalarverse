28 // Compute van Leer slopes
29
30 Real dsl = 2.0_rt * (s[im1] - s[im2]);
31 Real dsr = 2.0_rt * (s[i0] - s[im1]);
32
33 Real dsvl_l = 0.0_rt;
34 if (dsl*dsr > 0.0_rt) {
35 Real dsc = 0.5_rt * (s[i0] - s[im2]);
36 dsvl_l = amrex::Math::copysign(1.0_rt, dsc) * amrex::min(amrex::Math::abs(dsc),amrex::min(amrex::Math::abs(dsl),amrex::Math::abs(dsr)));
37 }
38
39 dsl = 2.0_rt * (s[i0] - s[im1]);
40 dsr = 2.0_rt * (s[ip1] - s[i0]);
41
42 Real dsvl_r = 0.0_rt;
43 if (dsl*dsr > 0.0_rt) {
44 Real dsc = 0.5_rt * (s[ip1] - s[im1]);
45 dsvl_r = amrex::Math::copysign(1.0_rt, dsc) * amrex::min(amrex::Math::abs(dsc),amrex::min(amrex::Math::abs(dsl),amrex::Math::abs(dsr)));
46 }
47
48 // Interpolate s to edges
49
50 sm = 0.5_rt * (s[i0] + s[im1]) - (1.0_rt/6.0_rt) * (dsvl_r - dsvl_l);
51
52 // Make sure sedge lies in between adjacent cell-centered values
53
54 sm = amrex::max(sm, amrex::min(s[i0], s[im1]));
55 sm = amrex::min(sm, amrex::max(s[i0], s[im1]));
56
57
58 // Compute van Leer slopes
59
60 dsl = 2.0_rt * (s[i0] - s[im1]);
61 dsr = 2.0_rt * (s[ip1] - s[i0]);
62
63 dsvl_l = 0.0_rt;
64 if (dsl*dsr > 0.0_rt) {
65 Real dsc = 0.5_rt * (s[ip1] - s[im1]);
66 dsvl_l = amrex::Math::copysign(1.0_rt, dsc) * amrex::min(amrex::Math::abs(dsc), amrex::min(amrex::Math::abs(dsl),amrex::Math::abs(dsr)));
67 }
68
69 dsl = 2.0_rt * (s[ip1] - s[i0]);
70 dsr = 2.0_rt * (s[ip2] - s[ip1]);
71
72 dsvl_r = 0.0_rt;
73 if (dsl*dsr > 0.0_rt) {
74 Real dsc = 0.5_rt * (s[ip2] - s[i0]);
75 dsvl_r = amrex::Math::copysign(1.0_rt, dsc) * amrex::min(amrex::Math::abs(dsc), amrex::min(amrex::Math::abs(dsl),amrex::Math::abs(dsr)));
76 }
77
78 // Interpolate s to edges
79
80 sp = 0.5_rt * (s[ip1] + s[i0]) - (1.0_rt/6.0_rt) * (dsvl_r - dsvl_l);
81
82 // Make sure sedge lies in between adjacent cell-centered values
83
84 sp = amrex::max(sp, amrex::min(s[ip1], s[i0]));
85 sp = amrex::min(sp, amrex::max(s[ip1], s[i0]));
86
87
88 // Flatten the parabola
89
90 sm = flatn * sm + (1.0_rt - flatn) * s[i0];
91 sp = flatn * sp + (1.0_rt - flatn) * s[i0];
92
93 // Modify using quadratic limiters -- note this version of the limiting comes
94 // from Colella and Sekora (2008), not the original PPM paper.
95
96 if ((sp - s[i0]) * (s[i0] - sm) <= 0.0_rt) {
97 sp = s[i0];
98 sm = s[i0];
99
100 } else if (amrex::Math::abs(sp - s[i0]) >= 2.0_rt * amrex::Math::abs(sm - s[i0])) {
101 sp = 3.0_rt * s[i0] - 2.0_rt * sm;
102
103 } else if (amrex::Math::abs(sm - s[i0]) >= 2.0_rt * amrex::Math::abs(sp - s[i0])) {
104 sm = 3.0_rt * s[i0] - 2.0_rt * sp;
105 }

107}

125AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
126void
127ppm_int_profile(const Real sm, const Real sp, const Real sc,
128 const Real u, const Real c, const Real dtdx,
129 Real* Ip, Real* Im) {
130
131 // Integrate the parabolic profile to the edge of the cell.
132
133 // compute x-component of Ip and Im
134 Real s6 = 6.0_rt * sc - 3.0_rt * (sm + sp);
135
136 // Ip/m is the integral under the parabola for the extent
137 // that a wave can travel over a timestep
138 //
139 // Ip integrates to the right edge of a cell
140 // Im integrates to the left edge of a cell
141
142 // u-c wave
143 Real speed = u - c;
144 Real sigma = amrex::Math::abs(speed) * dtdx;
145
146 // if speed == ZERO, then either branch is the same
147 if (speed <= 0.0_rt) {
148 Ip[0] = sp;
149 Im[0] = sm + 0.5_rt * sigma * (sp - sm + (1.0_rt - (2.0_rt/3.0_rt) * sigma) * s6);
150 } else {
151 Ip[0] = sp - 0.5_rt * sigma * (sp - sm - (1.0_rt - (2.0_rt/3.0_rt) * sigma) * s6);
152 Im[0] = sm;
153 }
154
155 // u wave
156 speed = u;
157 sigma = amrex::Math::abs(speed) * dtdx;
158
159 if (speed <= 0.0_rt) {
160 Ip[1] = sp;
161 Im[1] = sm + 0.5_rt * sigma * (sp - sm + (1.0_rt - (2.0_rt/3.0_rt) * sigma) * s6);
162 } else {
163 Ip[1] = sp - 0.5_rt * sigma * (sp - sm - (1.0_rt - (2.0_rt/3.0_rt) * sigma) * s6);
164 Im[1] = sm;
165 }
166
167 // u+c wave
168 speed = u + c;
169 sigma = amrex::Math::abs(speed) * dtdx;
170
171 if (speed <= 0.0_rt) {
172 Ip[2] = sp;
173 Im[2] = sm + 0.5_rt * sigma * (sp - sm + (1.0_rt - (2.0_rt/3.0_rt) * sigma) * s6);
174 } else {
175 Ip[2] = sp - 0.5_rt * sigma * (sp - sm - (1.0_rt - (2.0_rt/3.0_rt) * sigma) * s6);
176 Im[2] = sm;
177 }
178
179}
