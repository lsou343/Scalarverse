268template<class T> void amrex::BroadcastArray(Vector<T> &aT, int myLocalId, int rootId, const MPI_Comm &localComm)
269{
270 int aT_Size(-2);
271 if(myLocalId == rootId) {
272 aT_Size = aT.size();
273 }
274 ParallelDescriptor::Bcast(&aT_Size, 1, rootId, localComm);
275 BL_ASSERT(aT_Size >= 0);
276 if(myLocalId != rootId) {
277 aT.resize(aT_Size);
278 }
279 if(aT_Size > 0) {
280 ParallelDescriptor::Bcast(aT.dataPtr(), aT.size(), rootId, localComm);
281 }

317 routine. This is set automatically by the make subsystem.
318
319 For example, if the Fortran routine is named abcxyx, then it will
320 be called in C++ code as ABCXYZ.
321*/
322#define BL_FORT_USE_UPPERCASE 1
323#endif /*BL_FORT_USE_UPPERCASE*/

396amrex::hash_combine (uint64_t & seed, const T & val) noexcept
