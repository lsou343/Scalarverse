318void AGNParticleContainer::
319update_agn_particle_single (ParticleType& p,
320 amrex::Array4<amrex::Real const> const& acc,
321 amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& plo,
322 amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& dxi,
323 const amrex::Real& dt, const amrex::Real& a_prev,
324 const amrex::Real& a_cur, const int& do_move)
325{
326 if (p.id() <= 0) return;
327
328 amrex::Real half_dt = 0.5 * dt;
329 amrex::Real a_cur_inv = 1.0 / a_cur;
330 amrex::Real dt_a_cur_inv = dt * a_cur_inv;
331
332 amrex::Real lx = (p.pos(0) - plo[0]) * dxi[0] + 0.5;
333 amrex::Real ly = (p.pos(1) - plo[1]) * dxi[1] + 0.5;
334 amrex::Real lz = (p.pos(2) - plo[2]) * dxi[2] + 0.5;
335
336 int i = amrex::Math::floor(lx);
337 int j = amrex::Math::floor(ly);
338 int k = amrex::Math::floor(lz);
339
340 amrex::Real xint = lx - i;
341 amrex::Real yint = ly - j;
342 amrex::Real zint = lz - k;
343
344 amrex::Real sx[] = {1.-xint, xint};
345 amrex::Real sy[] = {1.-yint, yint};
346 amrex::Real sz[] = {1.-zint, zint};
347
348 for (int d=0; d < AMREX_SPACEDIM; ++d)
349 {
350 amrex::Real val = 0.0;
351 for (int kk = 0; kk<=1; ++kk)
352 {
353 for (int jj = 0; jj <= 1; ++jj)
354 {
355 for (int ii = 0; ii <= 1; ++ii)
356 {
357 val += sx[amrex::Math::abs(ii-1)]*
358 sy[amrex::Math::abs(jj-1)]*
359 sz[amrex::Math::abs(kk-1)]*acc(i-ii,j-jj,k-kk,d);
360 }
361 }
362 }
363
364 p.rdata(d+1)=a_prev*p.rdata(d+1)+half_dt * val;
365 p.rdata(d+1)*=a_cur_inv;
366 }
367
368 if (do_move == 1)
369 {
370 for (int comp=0; comp < AMREX_SPACEDIM; ++comp) {
371 p.pos(comp) = p.pos(comp) + dt_a_cur_inv * p.rdata(comp+1);
372 }
373 }
