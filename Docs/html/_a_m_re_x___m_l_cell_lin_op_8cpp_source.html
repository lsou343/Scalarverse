235MLCellLinOp::makeNGrids (int grid_size) const
236{
237 const Box& dombx = m_geom[0].back().Domain();
238
239 const BoxArray& old_ba = m_grids[0].back();
240 const int N = old_ba.size();
241 Vector<Box> bv;
242 bv.reserve(N);
243 for (int i = 0; i < N; ++i)
244 {
245 Box b = old_ba[i];
246 b.coarsen(grid_size);
247 b.refine(grid_size);
248 IntVect sz = b.size();
249 const IntVect nblks {AMREX_D_DECL(sz[0]/grid_size, sz[1]/grid_size, sz[2]/grid_size)};
250
251 IntVect big = b.smallEnd() + grid_size - 1;
252 b.setBig(big);
253
254#if (AMREX_SPACEDIM == 3)
255 for (int kk = 0; kk < nblks[2]; ++kk) {
256#endif
257#if (AMREX_SPACEDIM >= 2)
258 for (int jj = 0; jj < nblks[1]; ++jj) {
259#endif
260 for (int ii = 0; ii < nblks[0]; ++ii)
261 {
262 IntVect shft{AMREX_D_DECL(ii*grid_size,jj*grid_size,kk*grid_size)};
263 Box bb = amrex::shift(b,shft);
264 bb &= dombx;
265 bv.push_back(bb);
266 }
267#if (AMREX_SPACEDIM >= 2)
268 }
269#endif
270#if (AMREX_SPACEDIM == 3)
271 }
272#endif
273 }
274
275 std::sort(bv.begin(), bv.end());
276 bv.erase(std::unique(bv.begin(), bv.end()), bv.end());
277
278 BoxList bl(std::move(bv));
279
280 return BoxArray{std::move(bl)};

442void
443MLCellLinOp::applyBC (int amrlev, int mglev, MultiFab& in, BCMode bc_mode, StateMode,
444 const MLMGBndry* bndry, bool skip_fillboundary) const
445{
446 BL_PROFILE("MLCellLinOp::applyBC()");
447 // No coarsened boundary values, cannot apply inhomog at mglev>0.
448 BL_ASSERT(mglev == 0 || bc_mode == BCMode::Homogeneous);
449 BL_ASSERT(bndry != nullptr || bc_mode == BCMode::Homogeneous);
450
451 const int ncomp = getNComp();
452 const int cross = isCrossStencil();
453 const int tensorop = isTensorOp();
454 if (!skip_fillboundary) {
455 in.FillBoundary(0, ncomp, m_geom[amrlev][mglev].periodicity(),cross);
456 }
457
458 int flagbc = bc_mode == BCMode::Inhomogeneous;
459 const int imaxorder = maxorder;
460
461 const Real* dxinv = m_geom[amrlev][mglev].InvCellSize();
462 const Real dxi = dxinv[0];
463 const Real dyi = (AMREX_SPACEDIM >= 2) ? dxinv[1] : 1.0;
464 const Real dzi = (AMREX_SPACEDIM == 3) ? dxinv[2] : 1.0;
465
466 const auto& maskvals = m_maskvals[amrlev][mglev];
467 const auto& bcondloc = *m_bcondloc[amrlev][mglev];
468
469 FArrayBox foofab(Box::TheUnitBox(),ncomp);
470 const auto& foo = foofab.array();
471
472 MFItInfo mfi_info;
473 if (Gpu::notInLaunchRegion()) mfi_info.SetDynamic(true);
474
475 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(cross || tensorop || Gpu::notInLaunchRegion(),
476 "non-cross stencil not support for gpu");
477
478#ifdef _OPENMP
479#pragma omp parallel if (Gpu::notInLaunchRegion())
480#endif
481 for (MFIter mfi(in, mfi_info); mfi.isValid(); ++mfi)
482 {
483 const Box& vbx = mfi.validbox();
484 const auto& iofab = in.array(mfi);
485
486 const auto & bdlv = bcondloc.bndryLocs(mfi);
487 const auto & bdcv = bcondloc.bndryConds(mfi);
488
489 if (cross || tensorop)
490 {
491 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
492 {
493 const Orientation olo(idim,Orientation::low);
494 const Orientation ohi(idim,Orientation::high);
495 const Box blo = amrex::adjCellLo(vbx, idim);
496 const Box bhi = amrex::adjCellHi(vbx, idim);
497 const int blen = vbx.length(idim);
498 const auto& mlo = maskvals[olo].array(mfi);
499 const auto& mhi = maskvals[ohi].array(mfi);
500 const auto& bvlo = (bndry != nullptr) ? bndry->bndryValues(olo).array(mfi) : foo;
501 const auto& bvhi = (bndry != nullptr) ? bndry->bndryValues(ohi).array(mfi) : foo;
502 for (int icomp = 0; icomp < ncomp; ++icomp) {
503 const BoundCond bctlo = bdcv[icomp][olo];
504 const BoundCond bcthi = bdcv[icomp][ohi];
505 const Real bcllo = bdlv[icomp][olo];
506 const Real bclhi = bdlv[icomp][ohi];
507 if (idim == 0) {
508 AMREX_LAUNCH_HOST_DEVICE_FUSIBLE_LAMBDA (
509 blo, tboxlo, {
510 mllinop_apply_bc_x(0, tboxlo, blen, iofab, mlo,
511 bctlo, bcllo, bvlo,
512 imaxorder, dxi, flagbc, icomp);
513 },
514 bhi, tboxhi, {
515 mllinop_apply_bc_x(1, tboxhi, blen, iofab, mhi,
516 bcthi, bclhi, bvhi,
517 imaxorder, dxi, flagbc, icomp);
518 });
519 } else if (idim == 1) {
520 AMREX_LAUNCH_HOST_DEVICE_FUSIBLE_LAMBDA (
521 blo, tboxlo, {
522 mllinop_apply_bc_y(0, tboxlo, blen, iofab, mlo,
523 bctlo, bcllo, bvlo,
524 imaxorder, dyi, flagbc, icomp);
525 },
526 bhi, tboxhi, {
527 mllinop_apply_bc_y(1, tboxhi, blen, iofab, mhi,
528 bcthi, bclhi, bvhi,
529 imaxorder, dyi, flagbc, icomp);
530 });
531 } else {
532 AMREX_LAUNCH_HOST_DEVICE_FUSIBLE_LAMBDA (
533 blo, tboxlo, {
534 mllinop_apply_bc_z(0, tboxlo, blen, iofab, mlo,
535 bctlo, bcllo, bvlo,
536 imaxorder, dzi, flagbc, icomp);
537 },
538 bhi, tboxhi, {
539 mllinop_apply_bc_z(1, tboxhi, blen, iofab, mhi,
540 bcthi, bclhi, bvhi,
541 imaxorder, dzi, flagbc, icomp);
542 });
543 }
544 }
545 }
546 }
547 else
548 {
549#ifndef BL_NO_FORT
550 const RealTuple & bdl = bdlv[0];
551 const BCTuple & bdc = bdcv[0];
552
553 for (OrientationIter oitr; oitr; ++oitr)
554 {
555 const Orientation ori = oitr();
556
557 int cdr = ori;
558 Real bcl = bdl[ori];
559 int bct = bdc[ori];
560
561 const FArrayBox& fsfab = (bndry != nullptr) ? bndry->bndryValues(ori)[mfi] : foofab;
562
