73TinyProfiler::start () noexcept
74{
75#ifdef _OPENMP
76#pragma omp master
77#endif
78 if (stats.empty() && !regionstack.empty())
79 {
80 double t;
81 if (!uCUPTI) {
82 t = amrex::second();
83 } else {
84#ifdef AMREX_USE_CUPTI
85 cudaDeviceSynchronize();
86 cuptiActivityFlushAll(0);
87 activityRecordUserdata.clear();
88 t = amrex::second();
89#endif
90 }
91
92 ttstack.emplace_back(std::make_tuple(t, 0.0, &fname));
93 global_depth = ttstack.size();
94
95#ifdef AMREX_USE_CUDA
96 if (device_synchronize_around_region) {
97 amrex::Gpu::Device::synchronize();
98 }
99 nvtxRangePush(fname.c_str());
100#endif
101
102 for (auto const& region : regionstack)
103 {
104 Stats& st = statsmap[region][fname];
105 ++st.depth;
106 stats.push_back(&st);
107 }
108 }

111void
112TinyProfiler::stop () noexcept
113{
114#ifdef _OPENMP
115#pragma omp master
116#endif
117 if (!stats.empty())
118 {
119 double t;
120 int nKernelCalls = 0;
121#ifdef AMREX_USE_CUPTI
122 if (uCUPTI) {
123 cudaDeviceSynchronize();
124 cuptiActivityFlushAll(0);
125 t = computeElapsedTimeUserdata(activityRecordUserdata);
126 nKernelCalls = activityRecordUserdata.size();
127 } else
128#endif
129 {
130 t = amrex::second();
131 }
132
133 while (static_cast<int>(ttstack.size()) > global_depth) {
134 ttstack.pop_back();
135 };
136
137 if (static_cast<int>(ttstack.size()) == global_depth)
138 {
139 const std::tuple<double,double,std::string*>& tt = ttstack.back();
140
141 // first: wall time when the pair is pushed into the stack
142 // second: accumulated dt of children
143 double dtin;
144 double dtex;
145 if (!uCUPTI) {
146 dtin = t - std::get<0>(tt); // elapsed time since start() is called.
147 dtex = dtin - std::get<1>(tt);
148 } else {
149 dtin = t;
150 dtex = dtin - std::get<1>(tt);
151 }
152
153 for (Stats* st : stats)
154 {
155 --(st->depth);
156 ++(st->n);
157 if (st->depth == 0) {
158 st->dtin += dtin;
159 }
160 st->dtex += dtex;
161 st->usesCUPTI = uCUPTI;
162 if (uCUPTI) {
163 st->nk += nKernelCalls;
164 }
165 }
166
167 ttstack.pop_back();
168 if (!ttstack.empty()) {
169 std::tuple<double,double,std::string*>& parent = ttstack.back();
170 std::get<1>(parent) += dtin;
171 }
172
173#ifdef AMREX_USE_CUDA
174 if (device_synchronize_around_region) {
175 amrex::Gpu::Device::synchronize();
176 }
177 nvtxRangePop();
178#endif
179 } else {
180 improperly_nested_timers.insert(fname);
181 }
182
183 stats.clear();
184 }

274{
275 static bool finalized = false;
276 if (!bFlushing) { // If flushing, don't make this the last time!
277 if (finalized) {
278 return;
279 } else {

284 double t_final = amrex::second();
285
286 // make a local copy so that any functions call after this will not be recorded in the local copy.
287 auto lstatsmap = statsmap;
288
289 bool properly_nested = improperly_nested_timers.size() == 0;
290 ParallelDescriptor::ReduceBoolAnd(properly_nested);
291 if (!properly_nested) {
292 Vector<std::string> local_imp, sync_imp;
293 bool synced;
294 for (std::set<std::string>::const_iterator it = improperly_nested_timers.begin();
295 it != improperly_nested_timers.end(); ++it)
296 {
297 local_imp.push_back(*it);
298 }
299
300 amrex::SyncStrings(local_imp, sync_imp, synced);
301
302 if (ParallelDescriptor::IOProcessor()) {
303 amrex::Print() << "\nWARNING: TinyProfilers not properly nested!!!\n";
304 for (int i = 0; i < sync_imp.size(); ++i) {
305 amrex::Print() << " " << sync_imp[i] << "\n";
306 }
307 amrex::Print() << std::endl;
308 }

311 int nprocs = ParallelDescriptor::NProcs();
312 int ioproc = ParallelDescriptor::IOProcessorNumber();
313
314 double dt_max = t_final - t_init;
315 ParallelReduce::Max(dt_max, ioproc, ParallelDescriptor::Communicator());
316 double dt_min = t_final - t_init;
317 ParallelReduce::Min(dt_min, ioproc, ParallelDescriptor::Communicator());
318 double dt_avg = t_final - t_init;
319 ParallelReduce::Sum(dt_avg, ioproc, ParallelDescriptor::Communicator());
320 dt_avg /= double(nprocs);
321
322 if (ParallelDescriptor::IOProcessor())
323 {
324 amrex::Print() << "\n\n";
325 amrex::Print().SetPrecision(4)
326 <<"TinyProfiler total time across processes [min...avg...max]: "
327 << dt_min << " ... " << dt_avg << " ... " << dt_max << "\n";
328 }
329
330 // make sure the set of regions is the same on all processes.
331 {
332 Vector<std::string> localRegions, syncedRegions;
333 bool alreadySynced;
334
335 for (auto const& kv : lstatsmap) {
336 localRegions.push_back(kv.first);
337 }
338
339 amrex::SyncStrings(localRegions, syncedRegions, alreadySynced);
340
341 if (!alreadySynced) {
342 for (auto const& s : syncedRegions) {
343 if (lstatsmap.find(s) == lstatsmap.end()) {
344 lstatsmap.insert(std::make_pair(s,std::map<std::string,Stats>()));
345 }
346 }
347 }
348 }
349
350 PrintStats(lstatsmap[mainregion], dt_max);
351 for (auto& kv : lstatsmap) {
352 if (kv.first != mainregion) {
353 amrex::Print() << "\n\nBEGIN REGION " << kv.first << "\n";
354 PrintStats(kv.second, dt_max);
355 amrex::Print() << "END REGION " << kv.first << "\n";
356 }

361TinyProfiler::PrintStats (std::map<std::string,Stats>& regstats, double dt_max)
362{
363 // make sure the set of profiled functions is the same on all processes
364 {
365 Vector<std::string> localStrings, syncedStrings;
366 bool alreadySynced;
367
368 for(auto const& kv : regstats) {
369 localStrings.push_back(kv.first);
370 }
371
372 amrex::SyncStrings(localStrings, syncedStrings, alreadySynced);
373
374 if (! alreadySynced) { // add the new name
375 for (auto const& s : syncedStrings) {
376 if (regstats.find(s) == regstats.end()) {
377 regstats.insert(std::make_pair(s, Stats()));
378 }
379 }
380 }
381 }
382
383 if (regstats.empty()) return;
384
385 int nprocs = ParallelDescriptor::NProcs();
386 int ioproc = ParallelDescriptor::IOProcessorNumber();
387
388 std::vector<ProcStats> allprocstats;
389 int maxfnamelen = 0;
390 Long maxncalls = 0;
391
392 // now collect global data onto the ioproc
393 for (auto it = regstats.cbegin(); it != regstats.cend(); ++it)
394 {
395 Long n = it->second.n;
396 double dts[2] = {it->second.dtin, it->second.dtex};
397
398 std::vector<Long> ncalls(nprocs);
399 std::vector<double> dtdt(2*nprocs);
400
401 if (ParallelDescriptor::NProcs() == 1)
402 {
403 ncalls[0] = n;
404 dtdt[0] = dts[0];
405 dtdt[1] = dts[1];
406 } else
407 {
408 ParallelDescriptor::Gather(&n, 1, &ncalls[0], 1, ioproc);
409 ParallelDescriptor::Gather(dts, 2, &dtdt[0], 2, ioproc);
410 }
411
412 if (ParallelDescriptor::IOProcessor()) {
413 ProcStats pst;
414 for (int i = 0; i < nprocs; ++i) {
415 pst.nmin = std::min(pst.nmin, ncalls[i]);
416 pst.navg += ncalls[i];
417 pst.nmax = std::max(pst.nmax, ncalls[i]);
418 pst.dtinmin = std::min(pst.dtinmin, dtdt[2*i]);
419 pst.dtinavg += dtdt[2*i];
420 pst.dtinmax = std::max(pst.dtinmax, dtdt[2*i]);
421 pst.dtexmin = std::min(pst.dtexmin, dtdt[2*i+1]);
422 pst.dtexavg += dtdt[2*i+1];
423 pst.dtexmax = std::max(pst.dtexmax, dtdt[2*i+1]);
424 }
425 pst.navg /= nprocs;
426 pst.dtinavg /= nprocs;
427 pst.dtexavg /= nprocs;
428 pst.fname = it->first;
429#ifdef AMREX_USE_CUPTI
430 pst.usesCUPTI = it->second.usesCUPTI;
431#endif
432 allprocstats.push_back(pst);
433 maxfnamelen = std::max(maxfnamelen, int(pst.fname.size()));
434 maxncalls = std::max(maxncalls, pst.nmax);
435 }
436 }
437
438 if (ParallelDescriptor::IOProcessor())
439 {
440 amrex::OutStream() << std::setfill(' ') << std::setprecision(4);
441 int wt = 9;
442
443 int wnc = (int) std::log10 ((double) maxncalls) + 1;
444 wnc = std::max(wnc, int(std::string("NCalls").size()));
445 wt = std::max(wt, int(std::string("Excl. Min").size()));
446 int wp = 6;
447 wp = std::max(wp, int(std::string("Max %").size()));
448
449 const std::string hline(maxfnamelen+wnc+2+(wt+2)*3+wp+2,'-');
450#ifdef AMREX_USE_CUPTI
451 const std::string hlinehlf((maxfnamelen+wnc+2+(wt+2)*3+wp+2)/2-12,'-');
452#endif
453 // Exclusive time
454 std::sort(allprocstats.begin(), allprocstats.end(), ProcStats::compex);
455 amrex::OutStream() << "\n" << hline << "\n";
456 amrex::OutStream() << std::left
457 << std::setw(maxfnamelen) << "Name"
458 << std::right
459 << std::setw(wnc+2) << "NCalls"
460 << std::setw(wt+2) << "Excl. Min"
461 << std::setw(wt+2) << "Excl. Avg"
462 << std::setw(wt+2) << "Excl. Max"
463 << std::setw(wp+2) << "Max %"
464 << "\n" << hline << "\n";
465 for (auto it = allprocstats.cbegin(); it != allprocstats.cend(); ++it)
466 {
467#ifdef AMREX_USE_CUPTI
468 if (it->usesCUPTI)
469 {
470 amrex::OutStream() << hlinehlf << "START CUPTI Trace Stats-"
471 << hlinehlf << "\n";

476 << std::right

479 << std::setw(wt+2) << it->dtexavg
480 << std::setw(wt+2) << it->dtexmax
481 << std::setprecision(2) << std::setw(wp+1) << std::fixed
482 << it->dtexmax*(100.0/dt_max) << "%";
483 amrex::OutStream().unsetf(std::ios_base::fixed);
