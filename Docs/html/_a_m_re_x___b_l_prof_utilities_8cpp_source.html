45std::string amrex::SanitizeName(const std::string &sname) {
46 std::string s(sname);
47 std::size_t found;
48 Vector<char> replaceChars;
49 replaceChars.push_back(' ');
50 replaceChars.push_back(':');
51 replaceChars.push_back('(');
52 replaceChars.push_back(')');
53 replaceChars.push_back('[');
54 replaceChars.push_back(']');
55 replaceChars.push_back('}');
56 replaceChars.push_back('{');
57 replaceChars.push_back('<');
58 replaceChars.push_back('>');
59 replaceChars.push_back(',');
60 for(int c(0); c < replaceChars.size(); ++c) {
61 while((found = s.find(replaceChars[c])) != std::string::npos) {
62 s.replace(found, 1, "_");
63 }
64 }
65 return s;
66}

116amrex::Box amrex::FixCoarseBoxSize(const Box &fineBox, int rr) {
117 // -- we need this because these datasets are not always properly blocked
118 Box coarseBox(amrex::coarsen(fineBox, rr));
119 for(int dim(0); dim < BL_SPACEDIM; ++dim) {
120 while((coarseBox.smallEnd(dim) * rr) < fineBox.smallEnd(dim)) {
121 coarseBox.growLo(dim, -1);
122 }
123 while((coarseBox.bigEnd(dim) * rr) > (fineBox.bigEnd(dim) - rr)) {
124 coarseBox.growHi(dim, -1);
125 }
126 }
127 return coarseBox;
128}

254int amrex::NHops(const Box &tbox, const IntVect &ivfrom, const IntVect &ivto) {
255 int nhops(0);
256 for(int d(0); d < BL_SPACEDIM; ++d) {
257 int bl(tbox.length(d));
258 int ivl(std::min(ivfrom[d], ivto[d]));
259 int ivh(std::max(ivfrom[d], ivto[d]));
260 int dist(std::min(ivh - ivl, ivl + bl - ivh));
261 nhops += dist;
262 }
263 return nhops;
264}

315void amrex::WriteFab(const string &filenameprefix, const int xdim, const int ydim,
316 const double *data)
317{
318 if(xdim <= maxFabXDim) {
319
320 Write2DFab(filenameprefix, xdim, ydim, data);
321
322 } else { // wrap the 2d fab into a 3d one
323
324 int index(-1);
325 int xdim3d(maxFabXDim), ydim3d(ydim), zdim3d(xdim / maxFabXDim);
326 if(xdim % maxFabXDim) {
327 ++zdim3d;
328 }
329 Vector<double> data3d(xdim3d * ydim3d * zdim3d, (Real) BLProfiler::InvalidCFT);
330
331 for(int iy(0); iy < ydim; ++iy) {
332 for(int ix(0); ix < xdim; ++ix) {
333 index = (ix%xdim3d) + (iy*xdim3d) + ((ix/xdim3d)*xdim3d*ydim3d);
334 data3d[index] = data[ix + iy*xdim];
335 }
336 }
337 Write3DFab(filenameprefix, xdim3d, ydim3d, zdim3d, &data3d[0]);
338 }

400// ----------------------------------------------------------------------
401void amrex::CollectMProfStats(std::map<std::string, BLProfiler::ProfStats> &mProfStats,
402 const Vector<Vector<BLProfStats::FuncStat> > &funcStats,
403 const Vector<std::string> &fNames,
404 Real runTime, int whichProc)
405{
406 for(int fnum(0); fnum < funcStats.size(); ++fnum) {
407 BLProfiler::ProfStats ps;
408 ps.minTime = runTime;
409 ps.maxTime = 0.0;
410 for(int p(0); p < funcStats[fnum].size(); ++p) {
411 const BLProfStats::FuncStat &fs = funcStats[fnum][p];
412 if(p == whichProc) { // ---- store ncalls for whichProc only
413 ps.nCalls = fs.nCalls;
414 }
415 ps.totalTime += fs.totalTime;
416 ps.minTime = std::min(ps.minTime, fs.totalTime);
417 ps.maxTime = std::max(ps.maxTime, fs.totalTime);
418 }
419 Real numProcs(funcStats[fnum].size());
420 if(numProcs > 0.0) {
421 ps.avgTime = ps.totalTime / numProcs;
422 } else {
423 ps.avgTime = 0.0;
424 }
425 Real variance(0.0);
426 for(int p(0); p < funcStats[fnum].size(); ++p) {
427 const BLProfStats::FuncStat &fs = funcStats[fnum][p];
428 variance += (fs.totalTime - ps.avgTime) * (fs.totalTime - ps.avgTime);
429 }
430 if(funcStats[fnum].size() > 0) {
431 ps.variance = variance / static_cast<Real> (funcStats[fnum].size());
432 } else {
433 ps.variance = 0.0;
434 }
435
436 const std::string &fName(fNames[fnum]);
437 mProfStats.insert(std::pair<std::string, BLProfiler::ProfStats>(fName, ps));
438 }
439}
