46 //

65 Array4<Real> const& delm = delm_fab.array();

68 // Array "mask" is used to sever the link to ghost cells when the BCs
69 // are not periodic
70 // It is set to 1 when a cell can be used in computations, 0 otherwise
71 //

75 });

85 //
86 // Step 2: compute delta M (mass gain or loss) on (lo-1,lo+1)
87 //
88 AMREX_FOR_4D(grown1_bx, ncomp, i, j, k, n,
89 {
90 if(flags(i,j,k).isSingleValued())
91 {
92 Real divnc(0.0);
93 Real vtot(0.0);
94 Real wted_frac(0.0);
95 int ks = (AMREX_SPACEDIM == 3) ? -1 : 0;
96 int ke = (AMREX_SPACEDIM == 3) ? 1 : 0;
97
98 for (int kk(ks); kk <= ke; ++kk) {
99 for (int jj(-1); jj <= 1; ++jj) {
100 for (int ii(-1); ii <= 1; ++ii) {
101 if( (ii != 0 or jj != 0 or kk != 0) and
102 flags(i,j,k).isConnected(ii,jj,kk) and
103 dbox.contains(IntVect(AMREX_D_DECL(i+ii,j+jj,k+kk))))

107 vtot += wted_frac;
108 divnc += wted_frac * divc(i+ii,j+jj,k+kk,n);
109
110 }
111 }
112 }

254// Purpose: Given a collision between particle and EB surface, and
255// given that a neighbour cell owns the EB surface, a collision between
256// the particle and the EDGE of the EB facet might occur. This
257// function returns the coordinates of the closest point on the edge of
258// an EB facet. This function does not check of collisions.
259AMREX_GPU_DEVICE AMREX_FORCE_INLINE
260RealVect
261facets_nearest_pt (IntVect const& ind_pt, IntVect const& ind_loop, RealVect const& r_vec,
262 RealVect const& eb_normal, RealVect const& eb_p0,
263 GpuArray<Real,AMREX_SPACEDIM> const& dx)
264{
265 // Enumerate the possible EB facet edges invovlved.
266 int n_facets = 0;
267 IntVect ind_facets {AMREX_D_DECL(0, 0, 0)};
268 for (int d = 0; d < AMREX_SPACEDIM; ++d) {
269 if ( ind_pt[d] != ind_loop[d] ) {
270 ind_facets[n_facets++] = d;
271 }
272 }
273
274 // scalar characterizing EB facet position
275 Real eb_h = eb_normal.dotProduct(eb_p0);
276
277#if (__CUDACC_VER_MAJOR__ != 9) || (__CUDACC_VER_MINOR__ != 2)
278 Real min_dist = std::numeric_limits<Real>::max();
279#else
280 Real min_dist = 3.4e38_rt;
281#endif
282
283 RealVect c_vec;
284
285 // iterate over EB facet edges and find whichever has the closest nearest point
286 for (int i_facet=0; i_facet<n_facets; ++i_facet)
287 {
288 int tmp_facet = ind_facets[i_facet];
289
290 // determine the normal of the cell's facet (cube faces)
291 RealVect facet_normal {AMREX_D_DECL(0._rt, 0._rt, 0._rt)};
292 facet_normal[tmp_facet] = 1.; // whether facing inwards or outwards is not important here
293
294 // skip cases where cell faces conincide with the eb facets
295 if (AMREX_D_TERM(amrex::Math::abs(eb_normal[0]) == amrex::Math::abs(facet_normal[0]),
296 && amrex::Math::abs(eb_normal[1]) == amrex::Math::abs(facet_normal[1]),
297 && amrex::Math::abs(eb_normal[2]) == amrex::Math::abs(facet_normal[2])))
298 { continue; }
299
300 int ind_cell = ind_loop[tmp_facet];
301 int ind_nb = ind_pt[tmp_facet];
302
303 // determine position of the cell's facet
304 Real f_c;
305 if (ind_cell < ind_nb) {
306 f_c = ( ind_cell + 1 ) * dx[tmp_facet];
307 } else {
308 f_c = ind_cell * dx[tmp_facet];
309 }
310
311 RealVect facet_p0{AMREX_D_DECL((ind_loop[0] + 0.5_rt) * dx[0],
312 (ind_loop[1] + 0.5_rt) * dx[1],
313 (ind_loop[2] + 0.5_rt) * dx[2])};
314 facet_p0[tmp_facet] = f_c;
315
316 // scalar characterizing cell facet position
317 Real facet_h = facet_normal.dotProduct(facet_p0);
318
319 // compute EB facet edge by finding the intercept between EB surface (first plane)
320 // and the cell's facet (second plane)
321 //
322 // Purpose: Calculates the line (represented by a position and a
323 // direction vector) given by the intersection of two planes (defined
324 // by two normal (n1, n2) and two positions (h1 = n1.p1, h2 = n2.p2).
325 //
326 // When one plane is the EB surface, and the other is a face of the
327 // cell. Then this line represents the edge of the EB facet.
328 //
329 Real c_dp = eb_normal.dotProduct(facet_normal);
330 Real c_norm = 1._rt - c_dp*c_dp;
331 //
332 Real c1 = ( eb_h - facet_h * c_dp ) / c_norm;
333 Real c2 = ( facet_h - eb_h * c_dp ) / c_norm;
334 //
335 RealVect edge_p0{AMREX_D_DECL(c1*eb_normal[0] + c2*facet_normal[0],
336 c1*eb_normal[1] + c2*facet_normal[1],
337 c1*eb_normal[2] + c2*facet_normal[2])};
338 RealVect edge_v = eb_normal.crossProduct(facet_normal);
339
340 // this solution is a line representing the closest EB edge, now compute the point
341 // on the line which minimizes the distance to the particle
342 //
343 // Purpose: Given an a line an a point, this finds the point
344 // one the line which minimizes the cartesian distance. It also finds
345 // the corresponing distance along the line corresponding to this point
346 //
347 RealVect c = edge_p0 - r_vec;
348 Real lambda_tmp = - edge_v.dotProduct(c) / edge_v.dotProduct(edge_v);
349 RealVect c_vec_tmp{AMREX_D_DECL(edge_p0[0] + lambda_tmp*edge_v[0],
350 edge_p0[1] + lambda_tmp*edge_v[1],
351 edge_p0[2] + lambda_tmp*edge_v[2])};
352
353 // IMPORTANT: this point might be outside the cell
354 // -> in that case, it will be one of the cell's corners
355 //
356 // if closest point is outside cell, determine the furthest we can go along the
357 // EB edge line whilst staying within the cell.
358 //
359 // Purpose: Given a line which passes through a box in three dimensions
360 // (it can pass through the edges). Let lambda be a real value
361 // representing the coordinate along the line. This finds
362 // the min/max values of lambda, in order for the point described by
363 // lambda to be contained within the box.
364 //
365#if (__CUDACC_VER_MAJOR__ != 9) || (__CUDACC_VER_MINOR__ != 2)
366 Real cx_lo = -std::numeric_limits<Real>::max();
367 Real cy_lo = -std::numeric_limits<Real>::max();
368 Real cz_lo = -std::numeric_limits<Real>::max();
369 Real cx_hi = std::numeric_limits<Real>::max();
370 Real cy_hi = std::numeric_limits<Real>::max();
371 Real cz_hi = std::numeric_limits<Real>::max();
372 Real eps = std::numeric_limits<Real>::epsilon();
373#else
374 Real cx_lo = -3.4e38_rt;
375 Real cy_lo = -3.4e38_rt;
376 Real cz_lo = -3.4e38_rt;
377 Real cx_hi = 3.4e38_rt;
378 Real cy_hi = 3.4e38_rt;
379 Real cz_hi = 3.4e38_rt;
380 Real eps = 1.e-7_rt;
381#endif
382 // if the line runs parrallel to any of these dimensions (which is true for
383 // EB edges), then skip -> the min/max functions at the end will skip them
384 // due to the +/-huge(c...) defaults (above).
385 if ( amrex::Math::abs(edge_v[0]) > eps ) {
386 cx_lo = -( edge_p0[0] - ind_loop[0] * dx[0] ) / edge_v[0];
387 cx_hi = -( edge_p0[0] - ( ind_loop[0] + 1 ) * dx[0] ) / edge_v[0];
388 if ( edge_v[0] < 0._rt ) amrex::Swap(cx_lo, cx_hi);
389 }
390 //
391 if ( amrex::Math::abs(edge_v[1]) > eps ) {
392 cy_lo = -( edge_p0[1] - ind_loop[1] * dx[1] ) / edge_v[1];
393 cy_hi = -( edge_p0[1] - ( ind_loop[1] + 1 ) * dx[1] ) / edge_v[1];
394 if ( edge_v[1] < 0._rt ) amrex::Swap(cy_lo, cy_hi);
395 }
396 //
397 if ( amrex::Math::abs(edge_v[2]) > eps ) {
398 cz_lo = -( edge_p0[2] - ind_loop[2] * dx[2] ) / edge_v[2];
399 cz_hi = -( edge_p0[2] - ( ind_loop[2] + 1 ) * dx[2] ) / edge_v[2];
400 if ( edge_v[2] < 0._rt ) amrex::Swap(cz_lo, cz_hi);
401 }
402 //
403 Real lambda_min = amrex::max(cx_lo, cy_lo, cz_lo);
404 Real lambda_max = amrex::min(cx_hi, cy_hi, cz_hi);
405
406 if (lambda_tmp < lambda_min) {
407 lambda_tmp = lambda_min;
408 } else if ( lambda_tmp > lambda_max) {
409 lambda_tmp = lambda_max;
410 }
411
412 RealVect rc_vec;
413 for (int d=0; d<AMREX_SPACEDIM; ++d) {
414 c_vec_tmp[d] = edge_p0[d] + lambda_tmp*edge_v[d];
415 rc_vec[d] = c_vec_tmp[d] - r_vec[d];
416 }
417
418 // determine new distance to particle
419 Real min_dist_tmp = rc_vec.dotProduct(rc_vec);
420
421 // minimize distance
422 if (min_dist_tmp < min_dist) {
423 min_dist = min_dist_tmp;
424 c_vec = c_vec_tmp;
425 }
426 }
427
428 return c_vec;
429}

490 auto p_facets = facets.data();

492 AMREX_D_TERM(Array4<Real const> const& apx = areafrac[0]->const_array(mfi);,

501 fac[1] = (bcent(i,j,k,1)+Real(j)+0.5_rt) * dx_eb[1];,
502 fac[2] = (bcent(i,j,k,2)+Real(k)+0.5_rt) * dx_eb[2]);
503
504 Real axm = apx(i, j , k );
505 Real axp = apx(i+1,j , k );
506 Real aym = apy(i, j , k );
507 Real ayp = apy(i, j+1, k );
508#if (AMREX_SPACEDIM == 3)
509 Real azm = apz(i, j , k );
510 Real azp = apz(i, j , k+1);

512 (aym-ayp)*(aym-ayp) +

514#else

517#endif

525 fac[AMREX_SPACEDIM+1] = -anrmy;,

527 }

531 {
532 AMREX_D_TERM(Real dxinv = 1._rt/dx_eb[0];,
533 Real dyinv = 1._rt/dx_eb[1];,
534 Real dzinv = 1._rt/dx_eb[2]);
535 AMREX_D_TERM(Real x = i*dx_ls[0];,
536 Real y = j*dx_ls[1];,
537 Real z = k*dx_ls[2]);
538#if (__CUDACC_VER_MAJOR__ != 9) || (__CUDACC_VER_MINOR__ != 2)
539 Real min_dist2 = std::numeric_limits<Real>::max();
540#else
541 Real min_dist2 = 3.4e38_rt;
542#endif
543 int i_nearest = 0;
544 for (int ifac = 0; ifac < ncutcells; ++ifac) {
545 AMREX_D_TERM(Real cx = p_facets[ifac][0];,
546 Real cy = p_facets[ifac][1];,
547 Real cz = p_facets[ifac][2]);
548 Real dist2 = AMREX_D_TERM((x-cx)*(x-cx),+(y-cy)*(y-cy),+(z-cz)*(z-cz));
549 if (dist2 < min_dist2) {

552 }
553 }
