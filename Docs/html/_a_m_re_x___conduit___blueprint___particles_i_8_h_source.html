17//---------------------------------------------------------------------------//
18// Converts a AMReX Particle Tile into a Conduit Mesh Blueprint Hierarchy.
19//---------------------------------------------------------------------------//
20// Note:
21// This is a helper function, it's not part of the AMReX Blueprint Interface.
22//---------------------------------------------------------------------------//
23template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
24void
25ParticleTileToBlueprint(const ParticleTile<NStructReal,
26 NStructInt,
27 NArrayReal,
28 NArrayInt> &ptile,
29 const Vector<std::string> &real_comp_names,
30 const Vector<std::string> &int_comp_names,
31 conduit::Node &res,
32 const std::string &topology_name)
33{
34 int num_particles = ptile.GetArrayOfStructs().size();
35 int struct_size = sizeof(Particle<NStructReal, NStructInt>);
36
37 // knowing the above, we can zero copy the x,y,z positions + id, cpu
38 // and any user fields in the AOS
39
40 // get the first particle's struct
41 const auto &pstruct = ptile.GetArrayOfStructs();
42
43 // setup a blueprint description for the particle mesh
44 // create a coordinate set
45 std::string coordset_name = topology_name + "_coords";
46 conduit::Node &n_coords = res["coordsets"][coordset_name];
47 n_coords["type"] = "explicit";
48
49 // create an explicit points topology
50 conduit::Node &n_topo = res["topologies"][topology_name];
51 n_topo["coordset"] = coordset_name;
52 n_topo["type"] = "unstructured";
53 n_topo["elements/shape"] = "point";
54 n_topo["elements/connectivity"].set(
55 conduit::DataType::c_int(num_particles));
56 int *conn = n_topo["elements/connectivity"].value();
57
58 for(int i = 0; i < num_particles ; i++)
59 {
60 conn[i] = i;
61 }
62
63 //----------------------------------//
64 // point locations from from aos
65 //----------------------------------//
66
67 const char* pbuf_const = reinterpret_cast<const char*>(pstruct.data());
68 char* pbuf = const_cast<char*>(pbuf_const);
69
70 ParticleReal* xp = reinterpret_cast<ParticleReal*>(pbuf); pbuf += sizeof(ParticleReal);
71 n_coords["values/x"].set_external(xp,
72 num_particles,
73 0,
74 struct_size);
75#if AMREX_SPACEDIM > 1
76 ParticleReal* yp = reinterpret_cast<ParticleReal*>(pbuf); pbuf += sizeof(ParticleReal);
77 n_coords["values/y"].set_external(yp,
78 num_particles,
79 0,
80 struct_size);
81#endif
82#if AMREX_SPACEDIM > 2
83 ParticleReal* zp = reinterpret_cast<ParticleReal*>(pbuf); pbuf += sizeof(ParticleReal);
84 n_coords["values/z"].set_external(zp,
85 num_particles,
86 0,
87 struct_size);
88#endif
89
90 // fields
91 conduit::Node &n_fields = res["fields"];
92
93 // -----------------------------
94 // user defined, real aos fields
95 // -----------------------------
96
97 int vname_real_idx = 0;
98 // struct real fields, the first set are always the particle positions
99 // which we wrap above
100 for (int i = 0; i < NStructReal; i++)
101 {
102 ParticleReal* val = reinterpret_cast<ParticleReal*>(pbuf); pbuf += sizeof(ParticleReal);
103 conduit::Node &n_f = n_fields[real_comp_names.at(vname_real_idx)];
104 n_f["topology"] = topology_name;
105 n_f["association"] = "element";
106 n_f["values"].set_external(val,
107 num_particles,
108 0,
109 struct_size);
110
111 vname_real_idx++;
112 }
113
114 //----------------------------------//
115 // standard integer fields from aos
116 // (id, cpu)
117 //----------------------------------//
118
119 // id is the first int entry
120 int* id = reinterpret_cast<int*>(pbuf); pbuf += sizeof(int);
121 conduit::Node &n_f_id = n_fields[topology_name + "_id"];
122
123 n_f_id["topology"] = topology_name;
124 n_f_id["association"] = "element";
125 n_f_id["values"].set_external(id,
126 num_particles,
127 0,
128 struct_size);
129
130 // cpu is the second int entry
131 int* cpu = reinterpret_cast<int*>(pbuf); pbuf += sizeof(int);
132 conduit::Node &n_f_cpu = n_fields[topology_name + "_cpu"];
133
134 n_f_cpu["topology"] = topology_name;
135 n_f_cpu["association"] = "element";
136 n_f_cpu["values"].set_external(cpu,
137 num_particles,
138 0,
139 struct_size);
140
141 // --------------------------------
142 // user defined, integer aos fields
143 // --------------------------------
144
145 int vname_int_idx = 0;
146 for (int i = 0; i < NStructInt; i++)
147 {
148 int* val = reinterpret_cast<int*>(pbuf); pbuf += sizeof(int);
149 conduit::Node &n_f = n_fields[int_comp_names.at(vname_int_idx)];
150 n_f["topology"] = topology_name;
151 n_f["association"] = "element";
152 n_f["values"].set_external(val,
153 num_particles,
154 0,
155 struct_size);
156 vname_int_idx++;
157 }
158
159 // -------------------------
160 // user defined soa fields
161 // -------------------------
162
163 const auto &soa = ptile.GetStructOfArrays();
164
165 // for soa entries, we can use standard strides,
166 // since these are contiguous arrays
167
168 // array real fields
169 for (int i = 0; i < NArrayReal; i++)
170 {
171 conduit::Node &n_f = n_fields[real_comp_names.at(vname_real_idx)];
172 n_f["topology"] = topology_name;
173 n_f["association"] = "element";
174 n_f["values"].set_external(const_cast<ParticleReal*>(&soa.GetRealData(i)[0]),
175 num_particles);
176
177 vname_real_idx++;
178 }
179
180 // array int fields
181 for (int i = 0; i < NArrayInt; i++)
182 {
183 conduit::Node &n_f = n_fields[int_comp_names.at(vname_int_idx)];
184 n_f["topology"] = topology_name;
185 n_f["association"] = "element";
186 n_f["values"].set_external(const_cast<int*>(&soa.GetIntData(i)[0]),
187 num_particles);
188
189 vname_int_idx++;
190 }
191}

193//---------------------------------------------------------------------------//
194// Converts a AMReX Particle Container into a Conduit Mesh Blueprint Hierarchy.
195//---------------------------------------------------------------------------//
196template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
197void
198ParticleContainerToBlueprint(const ParticleContainer<NStructReal,
199 NStructInt,
200 NArrayReal,
201 NArrayInt> &pc,
202 const Vector<std::string> &real_comp_names,
203 const Vector<std::string> &int_comp_names,
204 conduit::Node &res,
205 const std::string &topology_name)
206{
207 BL_PROFILE("ParticleContainerToBlueprint()");
208
209 // validate varnames, which are used to provide field names
210 // for user defined aos and soa values.
211
212 BL_ASSERT(real_comp_names.size() == (NStructReal + NArrayReal) );
213 BL_ASSERT(int_comp_names.size() == (NStructInt + NArrayInt) );
214
215 int num_levels = pc.maxLevel() + 1;
216 int num_domains = 0;
217
218 // loop over levels
219
220 int rank = ParallelDescriptor::MyProc();
221 int nprocs = ParallelDescriptor::NProcs();
222
223 using MyParConstIter = ParConstIter<NStructReal, NStructInt, NArrayReal, NArrayInt>;
224
225 //
226 // blueprint expects unique ids for each domain published
227 // for particle data, this means we need a unique id for each
228 // tile across levels.
229 //
230 // to generate unique domain ids, we use a counting loop to
231 // calculate a per-level, per-rank offset
232 //
233
234 // this holds our unique offset for each level for the current rank
235 std::vector<int> my_lvl_offsets;
236 my_lvl_offsets.resize(num_levels);
237
238 // these hold the values we need to do a scan to calc offsets
239 std::vector<int> within_lvl_counts;
240 std::vector<int> within_lvl_offsets;
241 within_lvl_counts.resize(nprocs);
242 within_lvl_offsets.resize(nprocs);
243
244 int total_num_domains = 0;
245 for (int lev = 0; lev < num_levels; ++lev)
246 {
247 // count how many tiles this mpi task has
248 long level_num_domains = 0;
249 for (MyParConstIter pti(pc, lev); pti.isValid(); ++pti)
250 {
251 level_num_domains+=1;
252 }
253
254 // clear the within_lvl_counts and offsets data
255 for(int rank_idx = 0; rank_idx < nprocs ; rank_idx++)
256 {
257 within_lvl_counts[rank_idx] = 0;
258 within_lvl_offsets[rank_idx] = 0;
259 }
260
261 // set number of domains for current rank at current level
262 within_lvl_counts[rank] = level_num_domains;
263 // mpi sum reduce to get all rank's counts for this level
264 ParallelDescriptor::ReduceIntSum(&within_lvl_counts[0],
265 nprocs);
266
267 long level_total_num_domains = 0;
268 // scan to calc offset and total number of domains
269 for(int rank_idx = 0; rank_idx < nprocs ; rank_idx++)
270 {
271 within_lvl_offsets[rank_idx] = level_total_num_domains;
272 level_total_num_domains += within_lvl_counts[rank_idx];
273 }
274
275 // the offset for this rank at this level is the current
276 // total number of domains + the offset for this level
277 my_lvl_offsets[lev] = within_lvl_offsets[rank] + total_num_domains;
278
279 total_num_domains += level_num_domains;
280 }
281
282 for (int lev = 0; lev < num_levels; ++lev)
283 {
284 // get our unique global offset for this rank at this level
285 int lvl_offset = my_lvl_offsets[lev];
286 // keep a counter of local # of tiles
287 int num_lvl_local_tiles = 0;
288 for (MyParConstIter pti(pc, lev); pti.isValid(); ++pti)
289 {
290 // this will give us a unique id across levels for each tile
291 int domain_id = lvl_offset + num_lvl_local_tiles;
292
293 const auto& ptile = pti.GetParticleTile();
294 const std::string& patch_name = amrex::Concatenate("domain_",
295 domain_id,
296 6);
297 conduit::Node &patch = res[patch_name];
298 // add basic state info
299 patch["state/domain_id"] = domain_id;
300 ParticleTileToBlueprint(ptile,
301 real_comp_names,
302 int_comp_names,
303 patch,
304 topology_name);
305 num_lvl_local_tiles += 1;
306 }
307 }
308
309 conduit::Node info;
310 // blueprint verify makes sure we conform to whats expected
311 // for a multi-domain mesh
312 if(!conduit::blueprint::mesh::verify(res,info))
313 {
314 // ERROR -- doesn't conform to the mesh blueprint
315 // show what went wrong
316 amrex::Print() << "ERROR: Conduit Mesh Blueprint Verify Failed!\n"
317 << info.to_yaml();
318 }
319}
