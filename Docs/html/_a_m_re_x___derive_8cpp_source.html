225void
226DeriveRec::addRange (const DescriptorList& d_list,
227 int state_indx,
228 int src_comp,
229 int num_comp)
230{
231 StateRange* r = new StateRange;
232
233 r->typ = state_indx;
234 r->sc = src_comp;
235 r->nc = num_comp;
236 r->next = 0;
237 //
238 // Add to end of list.
239 //
240 if (rng == 0)
241 {
242 rng = r;
243 }
244 else
245 {
246 StateRange* prev = rng;
247 while (prev->next != 0)
248 prev = prev->next;
249 prev->next = r;
250 }
251 nsr++;
252 n_state += num_comp;
253
254 buildBC(d_list);
255 buildBC3D(d_list);
256}

274void
275DeriveRec::buildBC (const DescriptorList& d_list)
276{
277 BL_ASSERT(nsr > 0);
278 delete [] bcr;
279 bcr = new int[2*AMREX_SPACEDIM*n_state];
280 int* bci = bcr;
281 for (DeriveRec::StateRange* r = rng; r != 0; r = r->next)
282 {
283 const StateDescriptor& d = d_list[r->typ];
284
285 for (int k = 0; k < r->nc; k++)
286 {
287 const int* bc = d.getBC(r->sc + k).vect();
288
289 for (int j = 0; j < 2*AMREX_SPACEDIM; j++)
290 {
291 bci[j] = bc[j];
292 }
293 bci += 2*AMREX_SPACEDIM;
294 }
295 }
296}

298void
299DeriveRec::buildBC3D (const DescriptorList& d_list)
300{
301 BL_ASSERT(nsr > 0);
302 delete [] bcr3D;
303 bcr3D = new int[2*3*n_state]();
304 int* bci = bcr3D;
305 for (DeriveRec::StateRange* r = rng; r != 0; r = r->next)
306 {
307 const StateDescriptor& d = d_list[r->typ];
308
309 for (int k = 0; k < r->nc; k++)
310 {
311 const int* bc = d.getBC(r->sc + k).vect();
312
313 for (int j = 0; j < AMREX_SPACEDIM; j++)
314 {
315 bci[j] = bc[j];
316 }
317 bci += 3;
318 for (int j = 0; j < AMREX_SPACEDIM; j++)
319 {
320 bci[j] = bc[AMREX_SPACEDIM+j];
321 }
