45static
46bool
47amrDatasHaveSameDerives(const AmrData& amrd1,
48 const AmrData& amrd2)
49{
50 const Vector<std::string>& derives1 = amrd1.PlotVarNames();
51 const Vector<std::string>& derives2 = amrd2.PlotVarNames();
52 int length = derives1.size();
53 if (length != derives2.size())
54 return false;
55 for (int i=0; i<length; ++i)
56 if (derives1[i] != derives2[i])
57 return false;
58 return true;
59}
