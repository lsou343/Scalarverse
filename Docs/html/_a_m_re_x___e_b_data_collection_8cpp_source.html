10EBDataCollection::EBDataCollection (const EB2::Level& a_level,
11 const Geometry& a_geom,
12 const BoxArray& a_ba_in,
13 const DistributionMapping& a_dm,
14 const Vector<int>& a_ngrow, EBSupport a_support)
15 : m_ngrow(a_ngrow),
16 m_support(a_support),
17 m_geom(a_geom)
18{
19 // The BoxArray argument may not be cell-centered BoxArray.
20 const BoxArray& a_ba = amrex::convert(a_ba_in, IntVect::TheZeroVector());
21
22 if (m_support >= EBSupport::basic)
23 {
24 m_cellflags = new FabArray<EBCellFlagFab>(a_ba, a_dm, 1, m_ngrow[0], MFInfo(),
25 DefaultFabFactory<EBCellFlagFab>());
26 a_level.fillEBCellFlag(*m_cellflags, m_geom);
27 }
28
29 if (m_support >= EBSupport::volume)
30 {
31 m_volfrac = new MultiFab(a_ba, a_dm, 1, m_ngrow[1], MFInfo(), FArrayBoxFactory());
32 a_level.fillVolFrac(*m_volfrac, m_geom);
33
34 m_centroid = new MultiCutFab(a_ba, a_dm, AMREX_SPACEDIM, m_ngrow[1], *m_cellflags);
35 a_level.fillCentroid(*m_centroid, m_geom);
36 }
37
38 if (m_support == EBSupport::full)
39 {
40 const int ng = m_ngrow[2];
41
42 m_bndrycent = new MultiCutFab(a_ba, a_dm, AMREX_SPACEDIM, ng, *m_cellflags);
43 a_level.fillBndryCent(*m_bndrycent, m_geom);
44
45 m_bndryarea = new MultiCutFab(a_ba, a_dm, 1, ng, *m_cellflags);
46 a_level.fillBndryArea(*m_bndryarea, m_geom);
47
48 m_bndrynorm = new MultiCutFab(a_ba, a_dm, AMREX_SPACEDIM, ng, *m_cellflags);
49 a_level.fillBndryNorm(*m_bndrynorm, m_geom);
50
51 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
52 const BoxArray& faceba = amrex::convert(a_ba, IntVect::TheDimensionVector(idim));
53 m_areafrac[idim] = new MultiCutFab(faceba, a_dm, 1, ng, *m_cellflags);
54 m_facecent[idim] = new MultiCutFab(faceba, a_dm, AMREX_SPACEDIM-1, ng, *m_cellflags);
55 }
56
57 a_level.fillAreaFrac(m_areafrac, m_geom);
58 a_level.fillFaceCent(m_facecent, m_geom);
59 }
