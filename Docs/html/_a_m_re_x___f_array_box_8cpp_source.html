335void
336FArrayBox::Initialize ()
337{
338 if (initialized) return;
339 initialized = true;
340
341 BL_ASSERT(fabio == 0);
342
343 ParmParse pp("fab");
344
345 std::string fmt;
346 //
347 // This block can legitimately set FAB output format.
348 //
349 if (pp.query("format", fmt))
350 {
351 FABio* fio = 0;
352
353 if (fmt == "ASCII")
354 {
355 FArrayBox::format = FABio::FAB_ASCII;
356 fio = new FABio_ascii;
357 }
358 else if (fmt == "8BIT")
359 {
360 FArrayBox::format = FABio::FAB_8BIT;
361 fio = new FABio_8bit;
362 }
363 else if (fmt == "NATIVE")
364 {
365 FArrayBox::format = FABio::FAB_NATIVE;
366 fio = new FABio_binary(FPC::NativeRealDescriptor().clone());
367 }
368 else if (fmt == "NATIVE_32")
369 {
370 FArrayBox::format = FABio::FAB_NATIVE_32;
371 fio = new FABio_binary(FPC::Native32RealDescriptor().clone());
372 }
373 else if (fmt == "IEEE" || fmt == "IEEE32")
374 {
375 if (fmt == "IEEE")
376 {
377 FArrayBox::format = FABio::FAB_IEEE;
378 //amrex::Warning("IEEE fmt in ParmParse files is deprecated");
379 }
380 else
381 {
382 FArrayBox::format = FABio::FAB_IEEE_32;
383 }
384 fio = new FABio_binary(FPC::Ieee32NormalRealDescriptor().clone());
385 }
386 else
387 {
388 amrex::ErrorStream() << "FArrayBox::init(): Bad FABio::Format = " << fmt;
389 amrex::Abort();
390 }
391
392 setFABio(fio);
393 }
394 else
395 {
396 //
397 // We default to "NATIVE" if nothing is specified.
398 //
399 FArrayBox::format = FABio::FAB_NATIVE;
400
401 setFABio(new FABio_binary(FPC::NativeRealDescriptor().clone()));
402 }
403 //
404 // This block sets ordering which doesn't affect output format.
405 // It is only used when reading in an old FAB.
406 //
407 std::string ord;
408
409 if (pp.query("ordering", ord))
410 {
411 if (ord == "NORMAL_ORDER")
412 FArrayBox::setOrdering(FABio::FAB_NORMAL_ORDER);
413 else if (ord == "REVERSE_ORDER")
414 FArrayBox::setOrdering(FABio::FAB_REVERSE_ORDER);
415 else if (ord == "REVERSE_ORDER_2")
416 FArrayBox::setOrdering(FABio::FAB_REVERSE_ORDER_2);
417 else
418 {
419 amrex::ErrorStream() << "FArrayBox::init(): Bad FABio::Ordering = " << ord;
420 amrex::Abort();
421 }
422 }
423
424 initval = std::numeric_limits<Real>::has_quiet_NaN
425 ? std::numeric_limits<Real>::quiet_NaN()
426 : std::numeric_limits<Real>::max();
427
428 pp.query("initval", initval);
429 pp.query("do_initval", do_initval);
430 pp.query("init_snan", init_snan);
431
432 amrex::ExecOnFinalize(FArrayBox::Finalize);
433}

452FABio*
453FABio::read_header (std::istream& is,
454 FArrayBox& f)
455{
456// BL_PROFILE("FArrayBox::read_header_is");
457 int nvar;
458 Box bx;
459 FABio* fio = 0;
460 RealDescriptor* rd = 0;
461 char c;
462
463 is >> c;
464 if(c != 'F') amrex::Error("FABio::read_header(): expected \'F\'");
465 is >> c;
466 if(c != 'A') amrex::Error("FABio::read_header(): expected \'A\'");
467 is >> c;
468 if(c != 'B') amrex::Error("FABio::read_header(): expected \'B\'");
469
470 is >> c;
471 if(c == ':') { // ---- The "old" FAB format.
472 int typ_in, wrd_in;
473 is >> typ_in;
474 is >> wrd_in;
475
476 char machine[128];
477 is >> machine;
478 is >> bx;
479 is >> nvar;
480 //
481 // Set the FArrayBox to the appropriate size.
482 //
483 if (f.box() != bx || f.nComp() != nvar) {
484 f.resize(bx,nvar);
485 }
486 is.ignore(BL_IGNORE_MAX, '\n');
487 switch (typ_in)
488 {
489 case FABio::FAB_ASCII: fio = new FABio_ascii; break;
490 case FABio::FAB_8BIT: fio = new FABio_8bit; break;
491 case FABio::FAB_NATIVE:
492 case FABio::FAB_NATIVE_32:
493 case FABio::FAB_IEEE:
494 rd = RealDescriptor::newRealDescriptor(typ_in,
495 wrd_in,
496 machine,
497 FArrayBox::ordering);
498 fio = new FABio_binary(rd);
499 break;
500 default:
501 amrex::Error("FABio::read_header(): Unrecognized FABio header");
502 }
503 } else { // ---- The "new" FAB format.
504 is.putback(c);
505 rd = new RealDescriptor;
506 is >> *rd;
507 is >> bx;
508 is >> nvar;
509 //
510 // Set the FArrayBox to the appropriate size.
511 //
512 if (f.box() != bx || f.nComp() != nvar) {
513 f.resize(bx,nvar);
514 }
515 is.ignore(BL_IGNORE_MAX, '\n');
516 fio = new FABio_binary(rd);
517 }
518
519 if(is.fail()) {
520 amrex::Error("FABio::read_header() failed");
521 }
522
523 return fio;
524}

629{
630// BL_PROFILE("FArrayBox::readFrom_is_i");

633 BL_ASSERT(compIndex >= 0 && compIndex < nCompAvailable);
634
635 fabrd->skip(is, *this, compIndex); // skip data up to the component we want

662{

666
667 IntVect sm = bx.smallEnd();
668 IntVect bg = bx.bigEnd();
669
670 for(IntVect p(sm); p <= bg; bx.next(p)) {
671 os << p;
672 for(int k(0); k < num_comp; ++k) {
673 os << " " << f(p,k+comp);

675 os << '\n';
676 }
677 os << '\n';
678

680 amrex::Error("FABio_ascii::write() failed");

743void
744FABio_8bit::write (std::ostream& os,
745 const FArrayBox& f,
746 int comp,
747 int num_comp) const
748{
749 BL_ASSERT(comp >= 0 && num_comp >= 1 && (comp+num_comp) <= f.nComp());
750
751 const Real eps = 1.0e-8; // FIXME - whats a better value?
752 const Long siz = f.box().numPts();
753
754 unsigned char *c = new unsigned char[siz];
755
756 for(int k(0); k < num_comp; ++k) {
757 const Real mn = f.min<RunOn::Host>(k+comp);
758 const Real mx = f.max<RunOn::Host>(k+comp);
759 const Real* dat = f.dataPtr(k+comp);
760 Real rng = std::fabs(mx-mn);
761 rng = (rng < eps) ? 0.0 : 255.0/(mx-mn);
762 for(Long i(0); i < siz; ++i) {
763 Real v = rng*(dat[i]-mn);
764 int iv = (int) v;
765 c[i] = (unsigned char) iv;
766 }
767 os << mn << " " << mx << '\n' << siz << '\n';
768 os.write((char*)c,siz);
769 }
770
771 delete [] c;
772
773 if(os.fail()) {
774 amrex::Error("FABio_8bit::write() failed");
775 }

778void
779FABio_8bit::read (std::istream& is,
780 FArrayBox& f) const
781{
782 Long siz = f.box().numPts();
783 unsigned char* c = new unsigned char[siz];
784
785 Real mn, mx;
786 for(int nbytes, k = 0; k < f.nComp(); ++k) {
787 is >> mn >> mx >> nbytes;
788 BL_ASSERT(nbytes == siz);
789 while (is.get() != '\n') {
790 ; // ---- do nothing
791 }
792 is.read((char*)c,siz);
793 Real* dat = f.dataPtr(k);
794 const Real rng = (mx-mn)/255.0;
795 for (Long i = 0; i < siz; i++)
796 {
797 int iv = (int) c[i];
798 Real v = (Real) iv;
799 dat[i] = mn + rng*v;
800 }
801 }
802 if(is.fail()) {
803 amrex::Error("FABio_8bit::read() failed");
804 }
805
806 delete [] c;
807}
