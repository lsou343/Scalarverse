25 void amrex_fi_init_octree ()
26 {
27 ParmParse pp("amr");
28 int cnt = pp.countval("max_grid_size");
29 int max_grid_size = 0;
30 if (cnt == 0) {
31 max_grid_size = 8;
32 pp.add("max_grid_size", max_grid_size);
33 } else if (cnt == 1) {
34 pp.get("max_grid_size", max_grid_size);
35 } else {
36 amrex::Abort("amrex_fi_init_octree: must use the same max_grid_size for all levels");
37 }
38
39 int blocking_factor = 2*max_grid_size;
40 pp.add("blocking_factor", blocking_factor);
41
42 int max_grid_size_x = max_grid_size;
43 pp.query("max_grid_size_x", max_grid_size_x);
44 int blocking_factor_x = 2*max_grid_size_x;
45 pp.add("blocking_factor_x", blocking_factor_x);
46
47 int max_grid_size_y = max_grid_size;
48 pp.query("max_grid_size_y", max_grid_size_y);
49 int blocking_factor_y = 2*max_grid_size_y;
50 pp.add("blocking_factor_y", blocking_factor_y);
51
52 int max_grid_size_z = max_grid_size;
53 pp.query("max_grid_size_z", max_grid_size_z);
54 int blocking_factor_z = 2*max_grid_size_z;
55 pp.add("blocking_factor_z", blocking_factor_z);
56
57 pp.add("grid_eff", 1.0);
58
59 int max_level;
60 pp.get("max_level", max_level);
61
62 Vector<int> ref_ratio(max_level, 2);
63 pp.addarr("ref_ratio", ref_ratio);

66 void amrex_fi_build_octree_leaves (AmrCore* const amrcore, int* n, Vector<treenode>*& leaves,
67 int* level_offset)
68 {
69 leaves = new Vector<treenode>;
70 const int finest_level = amrcore->finestLevel();
71 const int myproc = ParallelDescriptor::MyProc();
72
73 FAmrCore* famrcore = dynamic_cast<FAmrCore*>(amrcore);
74
75 famrcore->octree_leaf_grids.resize(finest_level+1);
76 famrcore->octree_leaf_dmap.resize(finest_level+1);
77 famrcore->octree_leaf_dummy_mf.resize(finest_level+1);
78 famrcore->octree_li_full_to_leaf.resize(finest_level+1);
79 famrcore->octree_li_leaf_to_full.resize(finest_level+1);
80
81 BoxList bl;
82 Vector<int> iproc;
83
84 for (int lev = 0; lev <= finest_level; ++lev)
85 {
86 level_offset[lev] = leaves->size();
87
88 famrcore->octree_li_full_to_leaf[lev].clear();
89 famrcore->octree_li_leaf_to_full[lev].clear();
90
91 const BoxArray& ba = amrcore->boxArray(lev);
92 const DistributionMapping& dm = amrcore->DistributionMap(lev);
93 const int ngrids = ba.size();
94 BL_ASSERT(ba.size() < std::numeric_limits<int>::max());
95
96 if (lev == finest_level)
97 {
98 famrcore->octree_leaf_grids[lev] = ba;
99 famrcore->octree_leaf_dmap[lev] = dm;
100 famrcore->octree_leaf_dummy_mf[lev].reset
101 (new MultiFab(ba,dm,1,0,MFInfo().SetAlloc(false)));
102
103 int ilocal = 0;
104 for (int i = 0; i < ngrids; ++i) {
105 if (dm[i] == myproc) {
106 leaves->push_back({lev, i});
107 famrcore->octree_li_full_to_leaf[lev].push_back(ilocal++);
108 }
109 }
110 famrcore->octree_li_leaf_to_full[lev] = famrcore->octree_li_full_to_leaf[lev];
111 }
112 else
113 {
114 const BoxArray& fba = amrcore->boxArray(lev+1);
115 const IntVect& rr = amrcore->refRatio(lev);
116 bl.clear();
117 iproc.clear();
118 int ilocal_full = 0;
119 int ilocal_leaf = 0;
120 for (int i = 0; i < ngrids; ++i) {
121 Box bx = ba[i];
122 Box fbx = amrex::refine(bx,rr);
123 if (fba.intersects(fbx)) { // non-leaf
124 if (dm[i] == myproc) {
125 famrcore->octree_li_full_to_leaf[lev].push_back(-1);
126 ++ilocal_full;
127 }
128 } else { // leaves
129 bl.push_back(bx);
130 iproc.push_back(dm[i]);
131 if (dm[i] == myproc) {
132 leaves->push_back({lev, i});
133 famrcore->octree_li_full_to_leaf[lev].push_back(ilocal_leaf++);
134 famrcore->octree_li_leaf_to_full[lev].push_back(ilocal_full++);
135 }
136 }
137 }
138
139 if (bl.size() == 0) {
140 famrcore->octree_leaf_grids[lev] = BoxArray();
141 famrcore->octree_leaf_dmap[lev] = DistributionMapping();
142 famrcore->octree_leaf_dummy_mf[lev].reset(new MultiFab());
143 } else {
144 bool update_dummy_mf = false;
145 if (famrcore->octree_leaf_grids[lev] != bl.data()) {
146 famrcore->octree_leaf_grids[lev] = BoxArray(bl);
147 update_dummy_mf = true;
148 }
149 if (famrcore->octree_leaf_dmap[lev].ProcessorMap() != iproc) {
150 famrcore->octree_leaf_dmap[lev] = DistributionMapping(iproc);
151 update_dummy_mf = true;
152 }
153 if (update_dummy_mf) {
154 famrcore->octree_leaf_dummy_mf[lev].reset
155 (new MultiFab(famrcore->octree_leaf_grids[lev],
156 famrcore->octree_leaf_dmap[lev],
157 1,0,MFInfo().SetAlloc(false)));
158 }
159 }
160 }
161 }
162 level_offset[finest_level+1] = leaves->size();
163 *n = leaves->size();
164 }

180 void amrex_fi_octree_average_down_level (AmrCore* const amrcore, int flev,
181 MultiFab const * const fine,
182 MultiFab * const crse,
183 int scomp, int ncomp)
184 {
185 FAmrCore* famrcore = dynamic_cast<FAmrCore*>(amrcore);
186 const BoxArray& lba = famrcore->octree_leaf_grids[flev];
187 const DistributionMapping& ldm = famrcore->octree_leaf_dmap[flev];
188 const Vector<int>& li_leaf_to_full = famrcore->octree_li_leaf_to_full[flev];
189 const IntVect& rr = famrcore->refRatio(flev-1);
190
191 MultiFab lmf(amrex::coarsen(lba,rr),ldm,ncomp,0);
192
193 MultiFab fvolume;
194 const Geometry& fgeom = famrcore->Geom(flev);
195 if (!fgeom.IsCartesian())
196 {
197 fgeom.GetVolume(fvolume, lba, ldm, 0);
198 }
199
200#ifdef _OPENMP
201#pragma omp parallel if (Gpu::notInLaunchRegion())
202#endif
203 for (MFIter mfi(lmf,TilingIfNotGPU()); mfi.isValid(); ++mfi)
204 {
205 const Box& bx = mfi.tilebox();
206 Array4<Real> const& crsearr = lmf.array(mfi);
207 const int li = li_leaf_to_full[mfi.LocalIndex()];
208 Array4<Real const> const& finearr = fine->atLocalIdx(li).const_array();
209 if (fgeom.IsCartesian()) {
210 AMREX_LAUNCH_HOST_DEVICE_LAMBDA ( bx, tbx,
211 {
212 amrex_avgdown(tbx,crsearr,finearr,0,scomp,ncomp,rr);
213 });
214 } else {
215 Array4<Real const> const& finevolarr = fvolume.const_array(mfi);
216 AMREX_LAUNCH_HOST_DEVICE_LAMBDA ( bx, tbx,
217 {
218 amrex_avgdown_with_vol(tbx,crsearr,finearr,finevolarr,
219 0,scomp,ncomp,rr);
220 });
221 }
222 }
223
224 crse->ParallelCopy(lmf,0,scomp,ncomp);
225 }
