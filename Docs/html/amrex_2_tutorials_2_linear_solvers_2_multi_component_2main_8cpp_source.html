13// Description: This tutorial demonstrates how to implement a multi-component
14// nodal linear operator. This tutorial demonstrates the
15// "CFStrategy::ghostnodes" method for computing the reflux at
16// the coarse/fine boundary.
17//
18// See also: ./MCNodalLinOp.H, ./MCNodalLinOp.cpp
19// for implementation of the MC linear operator.

22#include <AMReX.H>
23#include <AMReX_ParmParse.H>
24#include <AMReX_MultiFab.H>
25#include <AMReX_Machine.H>
26#include <AMReX_MLMG.H>

28#include "MCNodalLinOp.H"

30using namespace amrex;
31
32int main (int argc, char* argv[])
33{
34 Initialize(argc, argv);
35 {
36
37 //
38 // Read in mesh structure (# amr levels, # nodes)
39 // Geometry will be a nnodes x nnodes (x nnodes) grid
40 // refined in the center up to nlevels.
41 //
42 struct {
43 int nlevels = 3;
44 int nnodes = 32;
45 int max_grid_size = 10000000;
46 } mesh;
47 {
48 ParmParse pp("mesh");
49 pp.query("nlevels",mesh.nlevels);
50 pp.query("nnodes",mesh.nnodes);
51 pp.query("max_grid_size",mesh.max_grid_size);
52 }
53
54 //
55 // Read in linear operator parameters:
56 // ncomp = number of components for a MC solve
57 // coeff = ncomp x ncomp list of coefficients
58 //
59 struct {
60 int ncomp=1;
61 Vector<Real> coeff = {1.0};
62 } op;
63 {
64 ParmParse pp("op");
65 pp.query("ncomp",op.ncomp);
66 pp.queryarr("coeff",op.coeff);
67 }
68
69 //
70 // Read in MLMG solver parameters
71 //
72 struct {
73 int verbose = -1;
74 int bottom_verbose = -1;
75 int max_iter = -1;
76 int fixed_iter = -1;
77 int max_fmg_iter = -1;
78 int linop_maxorder = -1;
79 int agglomeration = -1;
80 int consolidation = -1;
81 int max_coarsening_level = -1 ;
82 } mlmg;
83 {
84 ParmParse pp("mlmg");
85 pp.query("verbose",mlmg.verbose);
86 pp.query("bottom_verbose",mlmg.bottom_verbose );
87 pp.query("max_iter",mlmg.max_iter);
88 pp.query("max_fmg_iter",mlmg.max_fmg_iter);
89 pp.query("agglomeration",mlmg.agglomeration);
90 pp.query("consolidation",mlmg.consolidation);
91 pp.query("max_coarsening_level",mlmg.max_coarsening_level);
92 pp.query("fixed_iter",mlmg.fixed_iter);
93 }
94
95
96 //
97 // Initialize geometry and grids
98 //
99 Vector<Geometry> geom;
100 Vector<BoxArray> cgrids, ngrids;
101 Vector<DistributionMapping> dmap;
102 Vector<MultiFab> solution, rhs;
103 geom.resize(mesh.nlevels);
104 cgrids.resize(mesh.nlevels);
105 ngrids.resize(mesh.nlevels);
106 dmap.resize(mesh.nlevels);
107 solution.resize(mesh.nlevels);
108 rhs.resize(mesh.nlevels);
109 RealBox rb({AMREX_D_DECL(-0.5,-0.5,-0.5)},
110 {AMREX_D_DECL(0.5,0.5,0.5)});
111 Geometry::Setup(&rb, 0);
112 Box NDomain(IntVect{AMREX_D_DECL(0,0,0)},
113 IntVect{AMREX_D_DECL(mesh.nnodes,mesh.nnodes,mesh.nnodes)},
114 IntVect::TheNodeVector());
115 Box CDomain = convert(NDomain, IntVect::TheCellVector());
116
117 //
118 // Refine the grid
119 //
120 Box domain = CDomain;
121 for (int ilev = 0; ilev < mesh.nlevels; ++ilev)
122 {
123 geom[ilev].define(domain);
124 domain.refine(2);
125 }
126 Box cdomain = CDomain;
127 for (int ilev = 0; ilev < mesh.nlevels; ++ilev)
128 {
129 cgrids[ilev].define(cdomain);
130 cgrids[ilev].maxSize(mesh.max_grid_size); // TODO
131 cdomain.grow(-mesh.nnodes/4);
132 cdomain.refine(2);
133 ngrids[ilev] = cgrids[ilev];
134 ngrids[ilev].convert(IntVect::TheNodeVector());
135 }
136
137 //
138 // Initialize the solution and rhs fabs.
139 // Initialize the RHS fab to the function:
140 // RHS[0] = x1*(1-x1) * x2(1-x2) * x3(1-x3)
141 // RHS[1] = 0
142 // RHS[2] = 0 ... etc
143 //
144 int nghost = 2;
145 for (int ilev = 0; ilev < mesh.nlevels; ++ilev)
146 {
147 dmap [ilev].define(cgrids[ilev]);
148 solution[ilev].define(ngrids[ilev], dmap[ilev], op.ncomp, nghost);
149 solution[ilev].setVal(0.0);
150 solution[ilev].setMultiGhost(true);
151 rhs [ilev].define(ngrids[ilev], dmap[ilev], op.ncomp, nghost);
152 rhs [ilev].setVal(0.0);
153 rhs [ilev].setMultiGhost(true);
154
155 Box domain(geom[ilev].Domain());
156 const Real AMREX_D_DECL( dx = geom[ilev].CellSize()[0],
157 dy = geom[ilev].CellSize()[1],
158 dz = geom[ilev].CellSize()[2]);
159 const Real AMREX_D_DECL( minx = geom[ilev].ProbLo()[0],
160 miny = geom[ilev].ProbLo()[1],
161 minz = geom[ilev].ProbLo()[2]);
162 domain.convert(IntVect::TheNodeVector());
163 domain.grow(-1); // Shrink domain so we don't operate on any boundaries
164 for (MFIter mfi(solution[ilev], TilingIfNotGPU()); mfi.isValid(); ++mfi)
165 {
166 Box bx = mfi.tilebox();
167 bx.grow(1); // Expand to cover first layer of ghost nodes
168 bx = bx & domain; // Take intersection of box and the problem domain
169
170 Array4<Real> const& RHS = rhs[ilev].array(mfi);
171 for (int n = 0; n < op.ncomp; n++)
172 ParallelFor (bx,[=] AMREX_GPU_DEVICE(int i, int j, int k) {
173
174 Real AMREX_D_DECL(x1 = i*dx + minx,
175 x2 = j*dy + miny,
176 x3 = k*dz + minz);
177
178 if (n==0) RHS(i,j,k,n) = AMREX_D_TERM( (x1-0.5)*(x1+0.5),
179 * (x2-0.5)*(x2+0.5),
180 * (x3-0.5)*(x3+0.5));
181 else RHS(i,j,k,n) = 0.0;
182 });
183 }
184 rhs[ilev].FillBoundary(false,true);
185 }
186
187 //
188 // Set params to be passed to MLMG solver
189 //
190 LPInfo info;
191 if (mlmg.agglomeration >= 0) info.setAgglomeration(mlmg.agglomeration);
192 if (mlmg.consolidation >= 0) info.setConsolidation(mlmg.consolidation);
193 if (mlmg.max_coarsening_level >= 0) info.setMaxCoarseningLevel(mlmg.max_coarsening_level);
194
195 //
196 // Initialize the MCNodalLinOp linear operator
197 // (see ./MCNodalLinOp.cpp, ./MCNodalLinOp.H for implementation)
198 //
199 MCNodalLinOp linop;
200 linop.setNComp(op.ncomp);
201 linop.setCoeff(op.coeff);
202 linop.define(geom,cgrids,dmap,info);
203 linop.setDomainBC({AMREX_D_DECL(amrex::MLLinOp::BCType::Dirichlet,amrex::MLLinOp::BCType::Dirichlet,amrex::MLLinOp::BCType::Dirichlet)},
204 {AMREX_D_DECL(amrex::MLLinOp::BCType::Dirichlet,amrex::MLLinOp::BCType::Dirichlet,amrex::MLLinOp::BCType::Dirichlet)});
205 for (int ilev = 0; ilev < mesh.nlevels; ++ilev) linop.setLevelBC(ilev,&solution[ilev]);
206
207 //
208 // Initialize the MLMG solver
209 //
210 MLMG solver(linop);
211 if (mlmg.verbose >= 0) solver.setVerbose(mlmg.verbose);
212 if (mlmg.bottom_verbose >= 0) solver.setBottomVerbose(mlmg.bottom_verbose);
213 if (mlmg.fixed_iter >= 0) solver.setFixedIter(mlmg.fixed_iter);
214 if (mlmg.max_iter >= 0) solver.setMaxIter(mlmg.max_iter);
215 if (mlmg.max_fmg_iter >= 0)solver.setMaxFmgIter(mlmg.max_fmg_iter);
216 // IMPORTANT! Use the "CFStrategy::ghostnodes" strategy to avoid
217 // having to implement a complicated "reflux" routine!
218 solver.setCFStrategy(MLMG::CFStrategy::ghostnodes);
219
220 //
221 // Perform the solve
222 //
223 Real tol_rel = 1E-8, tol_abs = 1E-8;
224 solver.solve(GetVecOfPtrs(solution),GetVecOfConstPtrs(rhs),tol_rel,tol_abs);
225
226 //
227 // Write the output to ./solution
228 //
229 WriteMLMF ("solution",GetVecOfConstPtrs(solution),geom);
230
231 }
232 Finalize();
233}
