33void*
34CArena::alloc (std::size_t nbytes)
35{
36 std::lock_guard<std::mutex> lock(carena_mutex);
37
38 nbytes = Arena::align(nbytes == 0 ? 1 : nbytes);
39 //
40 // Find node in freelist at lowest memory address that'll satisfy request.
41 //
42 NL::iterator free_it = m_freelist.begin();
43
44 for ( ; free_it != m_freelist.end(); ++free_it) {
45 if ((*free_it).size() >= nbytes) {
46 break;
47 }
48 }
49
50 void* vp = 0;
51
52 if (free_it == m_freelist.end())
53 {
54 const std::size_t N = nbytes < m_hunk ? m_hunk : nbytes;
55
56 vp = allocate_system(N);
57
58 m_used += N;
59
60 m_alloc.push_back(std::make_pair(vp,N));
61
62 if (nbytes < m_hunk)
63 {
64 //
65 // Add leftover chunk to free list.
66 //
67 // Insert with a hint -- should be largest block in the set.
68 //
69 void* block = static_cast<char*>(vp) + nbytes;
70
71 m_freelist.insert(m_freelist.end(), Node(block, vp, m_hunk-nbytes));
72 }
73
74 m_busylist.insert(Node(vp, vp, nbytes));
75 }
76 else
77 {
78 BL_ASSERT((*free_it).size() >= nbytes);
79 BL_ASSERT(m_busylist.find(*free_it) == m_busylist.end());
80
81 vp = (*free_it).block();
82 m_busylist.insert(Node(vp, free_it->owner(), nbytes));
83
84 if ((*free_it).size() > nbytes)
85 {
86 //
87 // Insert remainder of free block back into freelist.
88 //
89 // Insert with a hint -- right after the current block being split.
90 //
91 Node freeblock = *free_it;
92
93 freeblock.size(freeblock.size() - nbytes);
94
95 freeblock.block(static_cast<char*>(vp) + nbytes);
96
97 m_freelist.insert(free_it, freeblock);
98 }
99
100 m_freelist.erase(free_it);
101 }
102
103 m_actually_used += nbytes;
104
105 BL_ASSERT(!(vp == 0));
106
107 return vp;
108}

110void
111CArena::free (void* vp)
112{
113 std::lock_guard<std::mutex> lock(carena_mutex);
114
115 if (vp == 0)
116 //
117 // Allow calls with NULL as allowed by C++ delete.
118 //
119 return;
120 //
121 // `vp' had better be in the busy list.
122 //
123 auto busy_it = m_busylist.find(Node(vp,0,0));
124 if (busy_it == m_busylist.end()) {
125 amrex::Abort("CArena::free: unknown pointer");
126 return;
127 }
128 BL_ASSERT(m_freelist.find(*busy_it) == m_freelist.end());
129
130 m_actually_used -= busy_it->size();
131
132 //
133 // Put free'd block on free list and save iterator to insert()ed position.
134 //
135 std::pair<NL::iterator,bool> pair_it = m_freelist.insert(*busy_it);
136
137 BL_ASSERT(pair_it.second == true);
138
139 NL::iterator free_it = pair_it.first;
140
141 BL_ASSERT(free_it != m_freelist.end() && (*free_it).block() == (*busy_it).block());
142 //
143 // And remove from busy list.
144 //
145 m_busylist.erase(busy_it);
146 //
147 // Coalesce freeblock(s) on lo and hi side of this block.
148 //
149 if (!(free_it == m_freelist.begin()))
150 {
151 NL::iterator lo_it = free_it;
152
153 --lo_it;
154
155 void* addr = static_cast<char*>((*lo_it).block()) + (*lo_it).size();
156
157 if (addr == (*free_it).block() && lo_it->coalescable(*free_it))
158 {
159 //
160 // This cast is needed as iterators to set return const values;
161 // i.e. we can't legally change an element of a set.
162 // In this case I want to change the size() of a block
163 // in the freelist. Since size() is not used in the ordering
164 // relations in the set, this won't effect the order;
165 // i.e. it won't muck up the ordering of elements in the set.
166 // I don't want to have to remove the element from the set and
167 // then reinsert it with a different size() as it'll just go
168 // back into the same place in the set.
169 //
170 Node* node = const_cast<Node*>(&(*lo_it));
171 BL_ASSERT(!(node == 0));
172 node->size((*lo_it).size() + (*free_it).size());
173 m_freelist.erase(free_it);
174 free_it = lo_it;
175 }
176 }
177
178 NL::iterator hi_it = free_it;
179
180 void* addr = static_cast<char*>((*free_it).block()) + (*free_it).size();
181
182 if (++hi_it != m_freelist.end() && addr == (*hi_it).block() && hi_it->coalescable(*free_it))
183 {
184 //
185 // Ditto the above comment.
186 //
187 Node* node = const_cast<Node*>(&(*free_it));
188 BL_ASSERT(!(node == 0));
189 node->size((*free_it).size() + (*hi_it).size());
190 m_freelist.erase(hi_it);
191 }
192}
