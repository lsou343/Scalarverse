151 // Compute the time step
152 // Implicit time step is imFactor*(explicit time step)
153 const Real* dx = geom.CellSize();
154 Real cfl = 0.9;
155 Real coeff = AMREX_D_TERM( 1./(dx[0]*dx[0]),
156 + 1./(dx[1]*dx[1]),
157 + 1./(dx[2]*dx[2]) );

171 {
172 MultiFab::Copy(phi_old, phi_new, 0, 0, 1, 0);
173
174 // new_phi = (I-dt)^{-1} * old_phi + dt
175 advance(phi_old, phi_new, dt, geom, ba, dm, bc);
176 time = time + dt;
177
178 // Tell the I/O Processor to write out which step we're doing
179 amrex::Print() << "Advanced step " << n << "\n";
180
181 // Write a plotfile of the current data (plot_int was defined in the inputs file)
182 if (plot_int > 0 && n%plot_int == 0)
183 {
184 const std::string& pltfile = amrex::Concatenate("plt",n,5);
185 WriteSingleLevelPlotfile(pltfile, phi_new, {"phi"}, geom, time, n);
186 }
187 }
188
189 // Call the timer again and compute the maximum difference between the start time and stop time
190 // over all processors

192 const int IOProc = ParallelDescriptor::IOProcessorNumber();
193 ParallelDescriptor::ReduceRealMax(stop_time,IOProc);
194
195 // Tell the I/O Processor to write out the "run time"
196 amrex::Print() << "Run time = " << stop_time << std::endl;
