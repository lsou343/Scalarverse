1135Real
1136Nyx::est_time_step (Real dt_old)
1137{
1138 BL_PROFILE("Nyx::est_time_step()");
1139 if (fixed_dt > 0)
1140 return fixed_dt;
1141
1142 // This is just a dummy value to start with
1143 Real est_dt = 1.0e+200;
1144
1145#ifndef NO_HYDRO
1146 const MultiFab& stateMF = get_new_data(State_Type);
1147#endif
1148
1149 Real cur_time = state[State_for_Time].curTime();
1150
1151#ifndef NO_HYDRO
1152 if (do_hydro)
1153 {
1154 Real a = get_comoving_a(cur_time);
1155 const auto dx = geom.CellSizeArray();
1156
1157 {
1158 Real dt = 1.e200;
1159 //Doesn't seem to be used:
1160 // Real grid_scl = std::cbrt(dx[0]*dx[1]*dx[2]);
1161 Real sound_speed_factor=sqrt(gamma*(gamma-1));
1162 Real dummy_small_dens=small_dens;
1163 int dummy_max_temp_dt=max_temp_dt;
1164
1165 amrex::Gpu::LaunchSafeGuard lsg(true);
1166 dt = amrex::ReduceMin(stateMF, 0,
1167 [=] AMREX_GPU_HOST_DEVICE (Box const& bx, Array4<Real const> const& u) noexcept -> Real
1168 {
1169 const auto lo = amrex::lbound(bx);
1170 const auto hi = amrex::ubound(bx);
1171#if !defined(__CUDACC__) || (__CUDACC_VER_MAJOR__ != 9) || (__CUDACC_VER_MINOR__ != 2)
1172 amrex::Real dt_gpu = std::numeric_limits<amrex::Real>::max();
1173#else
1174 amrex::Real dt_gpu = 1.e37;
1175#endif
1176
1177 for (int k = lo.z; k <= hi.z; ++k) {
1178 for (int j = lo.y; j <= hi.y; ++j) {
1179 for (int i = lo.x; i <= hi.x; ++i) {
1180 if(u(i,j,k,Density)<=1.1*dummy_small_dens && dummy_max_temp_dt==1)
1181 continue;
1182
1183 Real rhoInv = 1.0 / u(i,j,k,Density);
1184 Real ux = u(i,j,k,Xmom)*rhoInv;
1185 Real uy = u(i,j,k,Ymom)*rhoInv;
1186 Real uz = u(i,j,k,Zmom)*rhoInv;
1187
1188 // Use internal energy for calculating dt
1189 Real e = u(i,j,k,Eint)*rhoInv;
1190
1191 Real c;
1192 // Protect against negative e
1193#ifdef HEATCOOL
1194 if (e > 0.0)
1195 c=sound_speed_factor*std::sqrt(e);
1196#else
1197 if (e > 0.0)
1198 c=sound_speed_factor*std::sqrt(u(i,j,k,Density)*e/u(i,j,k,Density));
1199#endif
1200 else
1201 c = 0.0;
1202
1203 Real dt1 = dx[0]/(c + amrex::Math::abs(ux));
1204 Real dt2 = dx[1]/(c + amrex::Math::abs(uy));
1205 Real dt3 = dx[2]/(c + amrex::Math::abs(uz));
1206 dt_gpu = amrex::min(dt_gpu,amrex::min(dt1,amrex::min(dt2,dt3)));
1207
1208 }
1209 }
1210 }
1211 return dt_gpu;
1212 });
1213
1214 est_dt = std::min(est_dt, dt);
1215
1216 }
1217
1218 // If in comoving coordinates, then scale dt (based on u and c) by a
1219 est_dt *= a;
1220
1221 ParallelDescriptor::ReduceRealMin(est_dt);
1222 est_dt *= cfl;
1223 if (verbose && ParallelDescriptor::IOProcessor())
1224 std::cout << "...estdt from hydro at level "
1225 << level << ": "
1226 << est_dt << '\n';
1227 }
1228#endif
1229
1230#ifdef GRAVITY
1231 particle_est_time_step(est_dt);
1232#endif
1233
1234#ifdef AXCOMPLEX
1235 {
1236 const Real* dx = geom.CellSize();
1237 Real dt = string_dt*1.5*dx[0]/std::sqrt(msa*msa+12);
1238 if (verbose && ParallelDescriptor::IOProcessor())
1239 std::cout << "...estdt from strings : "<< dt <<'\n';
1240 est_dt = std::min(est_dt, dt);
1241 }
1242#endif
1243#ifdef AXREAL
1244 {
1245 const Real* dx = geom.CellSize();
1246 Real dt = ax_dt*dx[0];
1247 if (verbose && ParallelDescriptor::IOProcessor())
1248 std::cout << "...estdt from axion only : "<< dt <<'\n';
1249 est_dt = std::min(est_dt, dt);
1250 }
1251#endif
1252#ifdef FDM
1253 if (vonNeumann_dt >0 && (levelmethod[level]==FDlevel || levelmethod[level]==PSlevel) ){
1254 Real a = get_comoving_a(cur_time);
1255 const Real* dx = geom.CellSize();
1256 Real dt = dx[0]*dx[0]*a*a/2.0/hbaroverm;
1257#ifdef GRAVITY
1258 const MultiFab& phi = get_new_data(PhiGrav_Type);
1259 Real phi_max = std::abs(phi.max(0)-phi.min(0));
1260 dt = std::min(dt,hbaroverm/phi_max);
1261#endif
1262 if (levelmethod[level]==PSlevel)
1263 dt *= vonNeumann_dt;
1264 else if (vonNeumann_dt<1.0)
1265 dt *= vonNeumann_dt;
1266 if (verbose && ParallelDescriptor::IOProcessor())
1267 std::cout << "...estdt from von Neumann stability : "<< dt <<'\n';
1268 est_dt = std::min(est_dt, dt);
1269 }
1270#endif
1271
1272 if (level==0)
1273 comoving_est_time_step(cur_time,est_dt);
1274
1275 if (verbose && ParallelDescriptor::IOProcessor())
1276 std::cout << "Nyx::est_time_step at level "
1277 << level
1278 << ": estdt = "
1279 << est_dt << '\n';
1280

1284void
1285Nyx::computeNewDt (int finest_level,
1286 int sub_cycle,
1287 Vector<int>& n_cycle,
1288 const Vector<IntVect>& ref_ratio,
1289 Vector<Real>& dt_min,
1290 Vector<Real>& dt_level,
1291 Real stop_time,
1292 int post_regrid_flag)
1293{
1294 BL_PROFILE("Nyx::computeNewDt()");
1295 //
1296 // We are at the start of a coarse grid timecycle.
1297 // Compute the timesteps for the next iteration.
1298 //
1299 if (level > 0)
1300 return;
1301
1302 int i;
1303
1304 Real dt_0 = 1.0e+100;
1305 int n_factor = 1;
1306 for (i = 0; i <= finest_level; i++)
1307 {
1308 Nyx& adv_level = get_level(i);
1309 dt_min[i] = adv_level.est_time_step(dt_level[i]);
1310 }
1311
1312 if (fixed_dt <= 0.0)
1313 {
1314 if (post_regrid_flag == 1)
1315 {
1316 //
1317 // Limit dt's by pre-regrid dt
1318 //
1319 for (i = 0; i <= finest_level; i++)
1320 {
1321 dt_min[i] = std::min(dt_min[i], dt_level[i]);
1322 }
1323 //
1324 // Find the minimum over all levels
1325 //
1326 for (i = 0; i <= finest_level; i++)
1327 {
1328 n_factor *= n_cycle[i];
1329 dt_0 = std::min(dt_0, n_factor * dt_min[i]);
1330 }
1331 }
1332 else
1333 {
1334 bool sub_unchanged=true;
1335 if ((parent->maxLevel() > 0) && (level == 0) &&
1336 (parent->subcyclingMode() == "Optimal") &&
1337 (parent->okToRegrid(level) || parent->levelSteps(0) == 0) )
1338 {
1339 int new_cycle[finest_level+1];
1340 for (i = 0; i <= finest_level; i++)
1341 new_cycle[i] = n_cycle[i];
1342 // The max allowable dt
1343 Real dt_max[finest_level+1];
1344 for (i = 0; i <= finest_level; i++)
1345 {
1346 dt_max[i] = dt_min[i];
1347 }
1348 // find the maximum number of cycles allowed.
1349 int cycle_max[finest_level+1];
1350 cycle_max[0] = 1;
1351 for (i = 1; i <= finest_level; i++)
1352 {
1353 cycle_max[i] = parent->MaxRefRatio(i-1);
1354 }
1355 // estimate the amout of work to advance each level.
1356 Real est_work[finest_level+1];
1357 for (i = 0; i <= finest_level; i++)
1358 {
1359 est_work[i] = parent->getLevel(i).estimateWork();
1360 }
1361 // this value will be used only if the subcycling pattern is changed.
1362 dt_0 = parent->computeOptimalSubcycling(finest_level+1, new_cycle, dt_max, est_work, cycle_max);
1363 for (i = 0; i <= finest_level; i++)
1364 {
1365 if (n_cycle[i] != new_cycle[i])
1366 {
1367 sub_unchanged = false;
1368 n_cycle[i] = new_cycle[i];
1369 }
1370 }
1371
1372 }
1373
1374 if (sub_unchanged)
1375 //
1376 // Limit dt's by change_max * old dt
1377 //
1378 {
1379 for (i = 0; i <= finest_level; i++)
1380 {
1381 if (verbose && ParallelDescriptor::IOProcessor())
1382 {
1383 if (dt_min[i] > change_max*dt_level[i])
1384 {
1385 std::cout << "Nyx::compute_new_dt : limiting dt at level "
1386 << i << '\n';
1387 std::cout << " ... new dt computed: " << dt_min[i]
1388 << '\n';
1389 std::cout << " ... but limiting to: "
1390 << change_max * dt_level[i] << " = " << change_max
1391 << " * " << dt_level[i] << '\n';
1392 }
1393 }
1394
1395 dt_min[i] = std::min(dt_min[i], change_max * dt_level[i]);
1396 }
1397 //
1398 // Find the minimum over all levels
1399 //
1400 for (i = 0; i <= finest_level; i++)
1401 {
1402 n_factor *= n_cycle[i];
1403 dt_0 = std::min(dt_0, n_factor * dt_min[i]);
1404 }
1405 }
1406 else
1407 {
1408 if (verbose && ParallelDescriptor::IOProcessor())
1409 {
1410 std::cout << "Nyx: Changing subcycling pattern. New pattern:\n";
1411 for (i = 1; i <= finest_level; i++)
1412 std::cout << " Lev / n_cycle: " << i << " " << n_cycle[i] << '\n';
1413 }
1414 }
1415 }
1416 }
1417 else
1418 {
1419 dt_0 = fixed_dt;
1420 }
1421
1422 //
1423 // Limit dt's by the value of stop_time.
1424 //
1425 const Real eps = 0.001 * dt_0;
1426 Real cur_time = state[State_for_Time].curTime();
1427 if (stop_time >= 0.0)
1428 {
1429 if ((cur_time + dt_0) > (stop_time - eps))
1430 dt_0 = stop_time - cur_time;
1431 }
1432
1433 // Shrink the time step if necessary in order to hit the next plot_z_value
1434 if (level == 0 && ( plot_z_values.size() > 0 || analysis_z_values.size() > 0 ) )
1435 {
1436 bool dt_changed_plot = false;
1437 bool dt_changed_analysis = false;
1438
1439 if (plot_z_values.size() > 0)
1440 plot_z_est_time_step(dt_0,dt_changed_plot);
1441
1442 if (analysis_z_values.size() > 0)
1443 analysis_z_est_time_step(dt_0,dt_changed_analysis);
1444
1445 // Update the value of a if we didn't change dt in the call to plot_z_est_time_step or analysis_z_est_time_step.
1446 // If we didn't change dt there, then we have already done the integration.
1447 // If we did change dt there, then we need to re-integrate here.
1448 if (dt_changed_plot || dt_changed_analysis)
1449 integrate_comoving_a(cur_time,dt_0);
1450 }
1451 else
1452 {
1453 integrate_comoving_a(cur_time,dt_0);
1454 }
1455
1456 n_factor = 1;
1457 for (i = 0; i <= finest_level; i++)
1458 {
1459 n_factor *= n_cycle[i];
1460 dt_level[i] = dt_0 / n_factor;
1461 }
1462}

1484 int new_cycle[finest_level+1];
1485 for (i = 0; i <= finest_level; i++)
1486 new_cycle[i] = n_cycle[i];

1527 // Limit dt's by the value of stop_time.

1538 for (i = 0; i <= finest_level; i++)
1539 {
1540 n_factor *= n_cycle[i];
1541 dt_level[i] = dt_0 / n_factor;
1542 }
1543
1544 integrate_comoving_a(cur_time,dt_0);
1545}

1797 AMREX_GPU_DEVICE (int i, int j, int k) noexcept

1824#endif
