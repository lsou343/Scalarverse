238template <class FAB>
239void
240FabArrayCopyDescriptor<FAB>::AddBoxDoIt (FabArrayId fabarrayid,
241 const Box& destFabBox,
242 BoxList* returnedUnfilledBoxes,
243 int faindex,
244 int srccomp,
245 int destcomp,
246 int numcomp,
247 bool bUseValidBox,
248 BoxDomain& unfilledBoxDomain)
249{
250 const int myProc = ParallelDescriptor::MyProc();
251
252 FabArray<FAB>* fabArray = fabArrays[fabarrayid.Id()];
253
254 BL_ASSERT(faindex >= 0 && faindex < fabArray->size());
255
256 Box intersect = destFabBox;
257
258 if (bUseValidBox)
259 {
260 intersect &= fabArray->box(faindex);
261 }
262 else
263 {
264 intersect &= fabArray->fabbox(faindex);
265 }
266
267 if (intersect.ok())
268 {
269 FabCopyDescriptor<FAB>* fcd = new FabCopyDescriptor<FAB>;
270
271 int remoteProc = fabArray->DistributionMap()[faindex];
272 if(remoteProc >= ParallelDescriptor::NProcs()) {
273 std::ostringstream ss;
274 ss << ParallelDescriptor::MyProc() << ":: _in AddBoxDoIt: nProcs remoteProc = "
275 << ParallelDescriptor::NProcs() << " " << remoteProc << "\n";
276 amrex::Abort("Bad remoteProc: "+ss.str());
277 }
278 fcd->fillBoxId = nextFillBoxId;
279 fcd->subBox = intersect;
280 fcd->myProc = myProc;
281 fcd->copyFromProc = remoteProc;
282 fcd->copyFromIndex = faindex;
283 fcd->srcComp = srccomp;
284 fcd->destComp = destcomp;
285 fcd->nComp = numcomp;
286
287 if (ParallelDescriptor::sameTeam(remoteProc))
288 {
289 //
290 // Data is local.
291 //
292 fcd->fillType = FillLocally;
293 fcd->localFabSource = &(*fabArray)[faindex];
294 }
295 else
296 {
297 //
298 // Data is remote.
299 //
300 FabArrayBase::FabComTag fabComTag;
301
302 dataAvailable = false;
303 fcd->fillType = FillRemotely;
304 fcd->localFabSource = new FAB(intersect, numcomp);
305 fcd->cacheDataAllocated = true;
306 fabComTag.fabArrayId = fabarrayid.Id();
307 fabComTag.fillBoxId = nextFillBoxId;
308 fabComTag.fabIndex = faindex;
309 fabComTag.procThatNeedsData = myProc;
310 fabComTag.procThatHasData = remoteProc;
311 fabComTag.box = intersect;
312 fabComTag.srcComp = srccomp;
313 fabComTag.destComp = destcomp;
314 fabComTag.nComp = numcomp;
315 //
316 // Do not send the data yet.
317 //
318 fabComTagList.push_back(fabComTag);
319 }
320
321 fabCopyDescList[fabarrayid.Id()].insert(FCDMapValueType(fcd->fillBoxId,fcd));
322
323 if (returnedUnfilledBoxes != 0)
324 {
325 unfilledBoxDomain.rmBox(intersect);
326 }
327 }
328}

330template <class FAB>
331FillBoxId
332FabArrayCopyDescriptor<FAB>::AddBox (FabArrayId fabarrayid,
333 const Box& destFabBox,
334 BoxList* returnedUnfilledBoxes,
335 int srccomp,
336 int destcomp,
337 int numcomp)
338{
339 BoxDomain unfilledBoxDomain(destFabBox.ixType());

341 if (returnedUnfilledBoxes != 0)

358 numcomp,

365 returnedUnfilledBoxes->clear();
366 (*returnedUnfilledBoxes) = unfilledBoxDomain.boxList();
367 }

369 return FillBoxId(nextFillBoxId++, destFabBox);

452template <class FAB>
453void
454FabArrayCopyDescriptor<FAB>::CollectData ()
455{
456 dataAvailable = true;
457
458 if (ParallelDescriptor::NProcs() == 1) return;
459
460#ifdef BL_USE_MPI
461 typedef typename FAB::value_type value_type;
462
463 BL_PROFILE("FabArrayCopyDescriptor::CollectData()");
464
465 const int MyProc = ParallelDescriptor::MyProc();
466 amrex::ignore_unused(MyProc);
467
468 int Total_Rcvs_Size = 0;
469 //
470 // We use this to make finding matching FabComTags more efficient.
471 //
472 std::map< int,FabComTagIterContainer > RcvTags;
473
474 std::map<int,int> Snds, Rcvs, Npts;
475 //
476 // Set Rcvs[i] to # of blocks needed from CPU i
477 //
478 for (FabComTagContainer::const_iterator it = fabComTagList.begin(),
479 End = fabComTagList.end();
480 it != End;
481 ++it)
482 {
483 BL_ASSERT(it->box.ok());
484 BL_ASSERT(it->procThatNeedsData == MyProc);
485 BL_ASSERT(it->procThatHasData != MyProc);
486
487 const int Who = it->procThatHasData;
488 const int Cnt = (it->box.numPts())*(it->nComp);
489
490 RcvTags[Who].push_back(it);
491
492 Total_Rcvs_Size += Cnt;
493
494 if (Rcvs.count(Who) > 0)
495 {
496 Rcvs[Who] += 1;
497 }
498 else
499 {
500 Rcvs[Who] = 1;
501 }
502
503 if (Npts.count(Who) > 0)
504 {
505 Npts[Who] += Cnt;
506 }
507 else
508 {
509 Npts[Who] = Cnt;
510 }
511 }
512 BL_ASSERT(Rcvs.count(MyProc) == 0);
513
514 BL_ASSERT((Total_Rcvs_Size*sizeof(value_type)) < std::numeric_limits<int>::max());
515
516 const int NProcs = ParallelDescriptor::NProcs();
517
518 {
519 Vector<int> SndsArray(NProcs,0), RcvsArray(NProcs,0);
520
521 for (std::map<int,int>::const_iterator it = Rcvs.begin(), End = Rcvs.end(); it != End; ++it)
522 {
523 RcvsArray[it->first] = it->second;
524 }
525
526 {
527 BL_PROFILE_VAR("CollectData_Alltoall()", blpvCDATA);
528 BL_COMM_PROFILE(BLProfiler::Alltoall, sizeof(int), ParallelDescriptor::MyProc(),
529 BLProfiler::BeforeCall());
530
531 BL_MPI_REQUIRE( MPI_Alltoall(RcvsArray.dataPtr(),
532 1,
533 ParallelDescriptor::Mpi_typemap<int>::type(),
534 SndsArray.dataPtr(),
535 1,
536 ParallelDescriptor::Mpi_typemap<int>::type(),
537 ParallelDescriptor::Communicator()) );
538
539 BL_COMM_PROFILE(BLProfiler::Alltoall, sizeof(int), ParallelDescriptor::MyProc(),
540 BLProfiler::AfterCall());
541
542 BL_PROFILE_VAR_STOP(blpvCDATA);
543 }
544 BL_ASSERT(SndsArray[MyProc] == 0);
545
546 for (int i = 0; i < NProcs; i++)
547 if (SndsArray[i] > 0)
548 Snds[i] = SndsArray[i];
549 }
550
551 // There are two rounds of send and recv.
552 // First, the data receivers need to send the data senders meta-data (e.g., boxes).
553 // Then, the senders know what data to send and perform send.
554 const int SeqNum_md = ParallelDescriptor::SeqNum();
555 const int SeqNum_data = ParallelDescriptor::SeqNum();
556
557 const int N_snds = Snds.size();
558 const int N_rcvs = Rcvs.size();
559
560 if ( N_snds == 0 && N_rcvs == 0 ) return;
561
562 const int Nints = 4 + 3*AMREX_SPACEDIM; // # of ints in a meta-data
563
564 // for meta-data
565 Vector<int> md_sender, md_offset, md_icnts, md_bcnts;
566 int* md_recv_data = nullptr;
567 Vector<int*> md_send_data;
568 Vector<MPI_Request> md_recv_reqs, md_send_reqs;
569
570 // for data
571 Vector<int> data_sender, data_offset;
572 value_type* recv_data = nullptr;
573 Vector<value_type*> send_data;
574 Vector<MPI_Request> data_recv_reqs, data_send_reqs;
575
576 if (N_snds > 0)
577 {
578 // Recv meta-data
579
580 int N = 0;
581 for (std::map<int,int>::const_iterator it = Snds.begin(), End = Snds.end(); it != End; ++it)
582 {
583 md_sender.push_back(it->first);
584 md_bcnts.push_back(it->second);
585 int cnt = it->second * Nints;
586 md_icnts.push_back(cnt);
587 md_offset.push_back(N);
588 N += cnt;
589 }
590
591 md_recv_data = static_cast<int*>(amrex::The_Arena()->alloc(N*sizeof(int)));
592
593 for (int i = 0; i < N_snds; ++i)
594 {
595 md_recv_reqs.push_back(ParallelDescriptor::Arecv(&md_recv_data[md_offset[i]],
596 md_icnts[i], md_sender[i],
597 SeqNum_md).req());
598 }
599 }
600
601 if (N_rcvs > 0)
602 {
603 // Send meta-data
604 for (std::map<int,int>::const_iterator it = Rcvs.begin(), End = Rcvs.end(); it != End; ++it)
605 {
606 int rank = it->first;
607 int Nmds = it->second;
608 int cnt = Nmds * Nints;
609
610 int* p = static_cast<int*>(amrex::The_Arena()->alloc(cnt*sizeof(int)));
611 md_send_data.push_back(p);
612
613 const FabComTagIterContainer& tags = RcvTags[rank];
614
615 // initialized the data
616 int * md = p;
617 for (int i = 0; i < Nmds; ++i, md += Nints)
618 {
619 md[0] = tags[i]->fabArrayId;
620 md[1] = tags[i]->fabIndex;
621 md[2] = tags[i]->srcComp;
622 md[3] = tags[i]->nComp;
623 const int* lo = tags[i]->box.loVect();
624 const int* hi = tags[i]->box.hiVect();
625 const IntVect& bxtyp = tags[i]->box.type();
626 const int* tp = bxtyp.getVect();
627 AMREX_D_EXPR(md[4] = lo[0],
628 md[5] = lo[1],
629 md[6] = lo[2]);
630 AMREX_D_EXPR(md[4+ AMREX_SPACEDIM] = hi[0],
631 md[5+ AMREX_SPACEDIM] = hi[1],
632 md[6+ AMREX_SPACEDIM] = hi[2]);
633 AMREX_D_EXPR(md[4+2*AMREX_SPACEDIM] = tp[0],
634 md[5+2*AMREX_SPACEDIM] = tp[1],
635 md[6+2*AMREX_SPACEDIM] = tp[2]);
636 }
637
638 md_send_reqs.push_back(ParallelDescriptor::Asend(p,cnt,rank,SeqNum_md).req());
639 }
640 }
641
642 if (N_rcvs > 0)
643 {
644 recv_data = static_cast<value_type*>(amrex::The_Arena()->alloc(Total_Rcvs_Size*sizeof(value_type)));
645
646 // Post receives for data
647 int Idx = 0;
648 for (std::map<int,int>::const_iterator it = Npts.begin(); it != Npts.end(); ++it)
649 {
650 int Who = it->first;
651 int Cnt = it->second;
652 BL_ASSERT(Cnt > 0);
653 BL_ASSERT(Cnt < std::numeric_limits<int>::max());
654 data_sender.push_back(Who);
655 data_recv_reqs.push_back(ParallelDescriptor::Arecv(&recv_data[Idx],
656 Cnt,Who,SeqNum_data).req());
657 data_offset.push_back(Idx);
658 Idx += Cnt;
659 }
660 }
661
662 // Wait on meta-data and do send
663 if (N_snds > 0)
664 {
665 int send_counter = 0;
666 while (send_counter++ < N_snds)
667 {
668 MPI_Status status;
669 int index;
670 ParallelDescriptor::Waitany(md_recv_reqs, index, status);
671
672 int rank = status.MPI_SOURCE;
673 BL_ASSERT(status.MPI_TAG == SeqNum_md);
674 BL_ASSERT(rank == md_sender[index]);
675
676 const int* p = &md_recv_data[md_offset[index]];
677 int numboxes = md_bcnts[index];
678 Vector<int> faid(numboxes);
679 Vector<int> fidx(numboxes);
680 Vector<int> scomp(numboxes);
681 Vector<int> ncomp(numboxes);
682 Vector<int> npts(numboxes);
683 Vector<Box> bxs;
684 int N = 0;
685 const int * md = p;
686 for (int i = 0; i < numboxes; ++i, md += Nints)
687 {
688 faid[i] = md[0];
689 fidx[i] = md[1];
690 scomp[i] = md[2];
691 ncomp[i] = md[3];
692 bxs.push_back(Box(IntVect(&md[4]),
693 IntVect(&md[4+AMREX_SPACEDIM]),
694 IntVect(&md[4+AMREX_SPACEDIM*2])));
695 npts[i] = bxs.back().numPts()*ncomp[i];
696 N += npts[i];
697 }
698
699 BL_ASSERT(N < std::numeric_limits<int>::max());
700
701 value_type* data = static_cast<value_type*>(amrex::The_Arena()->alloc(N*sizeof(value_type)));
702 value_type* dptr = data;
703 send_data.push_back(data);
704
705 for (int i = 0; i < numboxes; ++i)
706 {
707 (*fabArrays[faid[i]])[fidx[i]].template copyToMem<RunOn::Host>(bxs[i],scomp[i],ncomp[i],dptr);
708 dptr += npts[i];
709 }
710
711 data_send_reqs.push_back(ParallelDescriptor::Asend(data,N,rank,SeqNum_data).req());
712 }
713
714 amrex::The_Arena()->free(md_recv_data);
715 }
716
717 // Wait and upack data
718 if (N_rcvs > 0)
719 {
720 Vector<MPI_Status> stats(N_rcvs);
721
722 ParallelDescriptor::Waitall(md_send_reqs, stats);
723 for (int i = 0; i < N_rcvs; ++i) {
724 amrex::The_Arena()->free(md_send_data[i]);
725 }
726
727 ParallelDescriptor::Waitall(data_recv_reqs, stats);
728
729 std::pair<FCDMapIter,FCDMapIter> match;
730 std::map< int,FabComTagIterContainer >::const_iterator found;
731
732 for (int k = 0; k < N_rcvs; k++)
733 {
734 const int Who = data_sender[k];
735 const value_type* dptr = &recv_data[data_offset[k]];
736
737 BL_ASSERT(dptr != 0);
738
739 found = RcvTags.find(Who);
740
741 BL_ASSERT(found != RcvTags.end());
742
743 const FabComTagIterContainer& tags = found->second;
744
745 for (FabComTagIterContainer::const_iterator it = tags.begin(), End = tags.end();
746 it != End;
747 ++it)
748 {
749 const FabArrayBase::FabComTag& tag = **it;
750
751 BL_ASSERT(tag.procThatHasData == Who);
752
753 match = fabCopyDescList[tag.fabArrayId].equal_range(tag.fillBoxId);
754
755 for (FCDMapIter fmi = match.first; fmi != match.second; ++fmi)
756 {
757 FabCopyDescriptor<FAB>* fcdp = (*fmi).second;
758
759 BL_ASSERT(fcdp->fillBoxId == tag.fillBoxId);
760
761 if (fcdp->subBox == tag.box)
762 {
763 BL_ASSERT(fcdp->localFabSource->dataPtr() != 0);
764 BL_ASSERT(fcdp->localFabSource->box() == tag.box);
765 const int Cnt = tag.box.numPts()*tag.nComp;
766 fcdp->localFabSource->template copyFromMem<RunOn::Host>(tag.box,0,tag.nComp,dptr);
767 dptr += Cnt;
768 break;
769 }
770 }
771 }
772 }
773
774 amrex::The_Arena()->free(recv_data);
775 }
776
777 // Finished send
778 if (N_snds > 0)
779 {
780 Vector<MPI_Status> stats(N_snds);
781 ParallelDescriptor::Waitall(data_send_reqs, stats);
782
783 for (int i = 0; i < N_snds; ++i) {
784 amrex::The_Arena()->free(send_data[i]);
785 }
786 }
787
788#endif /*BL_USE_MPI*/
789}

802 for (FCDMapIter fmi = match.first; fmi != match.second; ++fmi)

804 FabCopyDescriptor<FAB>* fcdp = (*fmi).second;
805
806 BL_ASSERT(fcdp->fillBoxId == fillboxid.Id());
807
808 destFab.template copy<RunOn::Host>
809 (*fcdp->localFabSource,
