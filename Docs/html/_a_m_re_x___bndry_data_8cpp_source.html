131void
132BndryData::define (const BoxArray& _grids,
133 const DistributionMapping& _dmap,
134 int _ncomp,
135 const Geometry& _geom)
136{
137 BL_PROFILE("BndryData::define()");
138
139 if (m_defined)
140 {
141 if (_grids == boxes() && m_ncomp == _ncomp && _geom.Domain() == geom.Domain())
142 //
143 // We want to allow reuse of BndryData objects that were define()d exactly as a previous call.
144 //
145 return;
146 //
147 // Otherwise we'll just abort. We could make this work but it's just as easy to start with a fresh Bndrydata object.
148 //
149 amrex::Abort("BndryData::define(): object already built");
150 }
151 geom = _geom;
152 m_ncomp = _ncomp;
153
154 BndryRegister::setBoxes(_grids);
155
156 masks.clear();
157 masks.resize(2*AMREX_SPACEDIM);
158
159 for (OrientationIter fi; fi; ++fi)
160 {
161 Orientation face = fi();
162 BndryRegister::define(face,IndexType::TheCellType(),0,1,1,_ncomp,_dmap);
163 masks[face].define(grids, _dmap, geom, face, 0, 2, NTangHalfWidth, 1, true);
164 }
165 //
166 // Define "bcond" and "bcloc".
167 //
168 // We note that all orientations of the FabSets have the same distribution.
169 // We'll use the low 0 side as the model.
170 //
171 //
172
173 bcloc.define(grids, _dmap);
174 bcond.define(grids, _dmap);
175
176 for (FabSetIter bfsi(bndry[Orientation(0,Orientation::low)]);
177 bfsi.isValid();
178 ++bfsi)
179 {
180 Vector< Vector<BoundCond> >& abc = bcond[bfsi];
181
182 abc.resize(2*AMREX_SPACEDIM);
183
184 for (OrientationIter fi; fi; ++fi)
185 {
186 abc[fi()].resize(_ncomp);
187 }
188 }
189
190 m_defined = true;
191}
