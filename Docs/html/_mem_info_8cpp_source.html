54void MemInfo::Init(const std::string& fname) {
55 if (class_initialized_)
56 return;
57 class_initialized_ = true;
58
59 // Get the hostname:
60 errno = 0;
61 if ((nlen_ = sysconf(_SC_HOST_NAME_MAX)) == -1) {
62 if (errno == 0) {
63 printf("Process %d: HOST_NAME_MAX not supported\n", my_rank_);
64 } else {
65 fprintf(stderr, "Process %d: ", my_rank_);
66 perror("MemInfo::Init(): problem with host name max");
67 }
68 nlen_ = 64;
69 }
70
71 hostname_ = new char[nlen_];
72 bool success = GetMyHostName();
73 if (!success)
74 printf("Process %d: could not get the hostname\n", my_rank_);
75
76 // Get page size and total number of pages on the node:
77 errno = 0;
78 if ((page_size_ = sysconf(_SC_PAGE_SIZE)) == -1) {
79 if (errno == 0) {
80 printf("Process %d: PAGE_SIZE not supported\n", my_rank_);
81 } else {
82 fprintf(stderr, "Process %d: ", my_rank_);
83 perror("MemInfo::Init(): problem with page size");
84 }
85 }
86 psize_in_gb_ = static_cast<float>(page_size_)/kGB;
87
88#if defined __APPLE__
89 int mib[2] = { CTL_HW, HW_MEMSIZE };
90 uint64_t memsize;
91 size_t len = sizeof(memsize);
92 if (sysctl(mib, 2, &memsize, &len, NULL, 0) == -1) {
93 fprintf(stderr, "Process %d: ", my_rank_);
94 perror("MemInfo::Init(): problem in sysctl call");
95 total_pages_ = 0;
96 } else {
97 total_pages_ = memsize/page_size_;
98 }
99#else
100 errno = 0;
101 if ((total_pages_ = sysconf(_SC_PHYS_PAGES)) == -1) {
102 if (errno == 0) {
103 printf("Process %d: PHYS_PAGES not supported\n", my_rank_);
104 } else {
105 fprintf(stderr, "Process %d: ", my_rank_);
106 perror("MemInfo::Init(): problem with phys pages");
107 }
108 total_pages_ = 0;
109 }
110#endif // __APPLE__
111
112 // Find min and max across all nodes:
113 long min_pag = 0, max_pag = 0;
114 BL_MPI_REQUIRE( MPI_Allreduce(&total_pages_, &min_pag, 1, MPI_LONG, MPI_MIN,
115 ParallelDescriptor::Communicator()) );
116 BL_MPI_REQUIRE( MPI_Allreduce(&total_pages_, &max_pag, 1, MPI_LONG, MPI_MAX,
117 ParallelDescriptor::Communicator()) );
118 global_min_ = static_cast<float>(min_pag) * psize_in_gb_;
119 global_max_ = static_cast<float>(max_pag) * psize_in_gb_;
120
121 // Open log file:
122 if (my_rank_ == master_rank_) {
123 std::string filename;
124 if (!fname.empty())
125 filename = fname;
126 else
127 filename = default_fname_;
128
129 if ((logfile_ = fopen(filename.c_str(), "w")) == NULL)
130 perror("MemInfo::Init(): Cannot open memory log file");
131 else
132 StampSysInfo(logfile_);
133 }
134
135 return;
136}

139bool MemInfo::MPIData() {
140 bool success = false;
141 int flag = 0;
142 BL_MPI_REQUIRE( MPI_Initialized(&flag) );
143
144 if (flag) {
145 BL_MPI_REQUIRE( MPI_Comm_size(ParallelDescriptor::Communicator(), &num_ranks_) );
146 BL_MPI_REQUIRE( MPI_Comm_rank(ParallelDescriptor::Communicator(), &my_rank_) );
147 master_rank_ = 0;
148 BL_MPI_REQUIRE( MPI_Get_version(&mpi_v_, &mpi_subv_) );
149 success = true;
150 } else {
151 my_rank_ = master_rank_ = 0;
152 num_ranks_ = 1;
153 mpi_v_ = -1;
154 mpi_subv_ = -1;
155 success = false;
156 }
157
158 return(success);
159}

174bool MemInfo::GetMyPages() {
175 bool success = true;
176
177#if defined __APPLE__
178 vm_statistics_data_t host_info;
179 mach_msg_type_number_t count = HOST_VM_INFO_COUNT;
180 if (host_statistics(mach_host_self(), HOST_VM_INFO,
181 reinterpret_cast<host_info_t>(&host_info), &count) != 0) {
182 fprintf(stderr, "Process %d: ", my_rank_);
183 perror("MemInfo::GetMyPages(): host_statistics error");
184 avail_pages_ = 0;
185 success = false;
186 } else {
187 avail_pages_ = host_info.free_count;
188 }
189#else
190 errno = 0;
191 if ((avail_pages_ = sysconf(_SC_AVPHYS_PAGES)) == -1) {
192 if (errno == 0) {
193 printf("Process %d: AVPHYS_PAGES not supported\n", my_rank_);
194 } else {
195 fprintf(stderr, "Process %d: ", my_rank_);
196 perror("MemInfo::GetMyPages() error");
197 }
198 avail_pages_ = 0;
199 success = false;
200 }
201#endif // __APPLE__
202
203 return(success);
204}

207void MemInfo::StampSysInfo(FILE* fname) {
208 struct utsname info;
209 if ((uname(&info) < 0) || (fname == NULL))
210 return;
211
212 fprintf(fname, "#################################");
213 fprintf(fname, "#################################\n");
214
215 char buf[kStrLen], t_stamp[64];
216 const time_t t_now = time(NULL);
217 struct tm* current_time = new tm;
218 current_time = localtime_r(&t_now, current_time);
219 asctime_r(current_time, t_stamp);
220
221 int len = 0;
222 len += snprintf(buf+len, kStrLen-len, "# Time: %s", t_stamp);
223 len += snprintf(buf+len, kStrLen-len, "# System: %s ", info.sysname);

225 len += snprintf(buf+len, kStrLen-len, "%s\n", info.machine);

239int MemInfo::BelowThreshold(float threshold) {
240 if (!class_initialized_)
241 Init(default_fname_);
242 bool success = GetMyPages();
243 if (!success)
244 return(-1);
245
246 int i = ((avail_pages_ * psize_in_gb_) >= threshold);
247 if (i == 0)
248 printf("Process %d: node %s is short on memory\n",
249 my_rank_, hostname_);
250
251 int sum = 0;
252 BL_MPI_REQUIRE( MPI_Allreduce(&i, &sum, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD) );
253
254 return(num_ranks_ - sum);
255}

258void MemInfo::LogSummary(const char* info) {
259 if ( ! class_initialized_) {
260 Init(default_fname_);
261 }
262 bool success = GetMyPages();
263 if ( ! success) {
264 return;
265 }
266
267 long avg_pag = 0, min_pag = 0, max_pag = 0;
268 BL_MPI_REQUIRE( MPI_Reduce(&avail_pages_, &min_pag, 1, MPI_LONG, MPI_MIN, master_rank_,
269 ParallelDescriptor::Communicator()) );
270 BL_MPI_REQUIRE( MPI_Reduce(&avail_pages_, &max_pag, 1, MPI_LONG, MPI_MAX, master_rank_,
271 ParallelDescriptor::Communicator()) );
272 BL_MPI_REQUIRE( MPI_Reduce(&avail_pages_, &avg_pag, 1, MPI_LONG, MPI_SUM, master_rank_,
273 ParallelDescriptor::Communicator()) );
274
275 if (my_rank_ == master_rank_) {
276 if (logfile_ == NULL) {
277 puts("Log file was not opened!");
278 } else {
279 float avg_gb = static_cast<float>(avg_pag)/num_ranks_ * psize_in_gb_;
280 float min_gb = static_cast<float>(min_pag) * psize_in_gb_;
281 float max_gb = static_cast<float>(max_pag) * psize_in_gb_;
282 fprintf(logfile_, "%s: Max = %f, Min = %f, Average = %f \n",
283 info, max_gb, min_gb, avg_gb);
284 fflush(logfile_);
285 }
286 }
287
288 return;

292void MemInfo::PrintAll(FILE* fout) {
293 if (fout == NULL) {
294 return;
295 }
296
297 if ( ! class_initialized_) {
298 Init(default_fname_);
299 }
300

303
304 char buf[kStrLen];
305 int len = 0;
306 len += snprintf(buf, kStrLen, "Rank %6d on node %s:",
307 my_rank_, hostname_);
308 if (success) {
309 len += snprintf(buf+len, kStrLen-len, "%6.2fGB free, %5.2f%% of total\n",
310 avail_mem, avail_mem/total_mem*100.0);

312 len += snprintf(buf+len, kStrLen-len, "could not get memory info");
313 }
314
315 if (my_rank_ == 0) {
316 StampSysInfo(fout);
317 fprintf(fout, "%s", buf);
318 for (int i = 1; i < num_ranks_; ++i) {
319 MPI_Status stat;
320 buf[0] = '\0';
321 BL_MPI_REQUIRE( MPI_Recv(buf, kStrLen, MPI_CHAR, i, i, MPI_COMM_WORLD, &stat) );
322 fprintf(fout, "%s", buf);
323 }
324 fflush(fout);
325 } else {
326 BL_MPI_REQUIRE( MPI_Send(buf, kStrLen, MPI_CHAR, 0, my_rank_, MPI_COMM_WORLD) );
327 }
328
329 return;
330}
