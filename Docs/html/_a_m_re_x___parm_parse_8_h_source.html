18//
19// ParmParse class implements a simple database for the storage and
20// retrieval of command-line and input-file arguments. The entries are
21// stored in a static table in (name,value_list) pairs.
22//
23// The format of the input file is a series of DEFINITIONS.
24//
25// A DEFINITION is of the form <name> = <value> <value> ...
26// The equal sign is important since the list of values can span multiple
27// lines.
28//
29// Comments in an input file include all text from a '#' character to the
30// end of the line. Here is an example input file:
31//
32// niter = 100 # niter is an integer
33// title = "Double Wammy" # example of a string with spaces
34// cell_size = 0.5 0.75 # cell spacing in each dimension
35// plot.var = Density 1 10 # a list of values
36// plot.var = Energy 5 12 # another list of values
37// bigarray = 1 2 3 4 5 6 7 8 # first part of array
38// 9 10 11 12 # continuation of bigarray
39// test = apple "boy blue" 10 20 30 40
40// FILE = prob_file # insert contents of this "prob_file" here
41//
42// The "FILE = <filename>" definition is special. Rather than just
43// adding this entry to the database, it reads the contents of <filename>
44// into the database.
45//
46// ParmParse stores all entries in a static table which is built the
47// first time a ParmParse object is constructed (usually in main()).
48// Subsequent invocations have access to this table.
49// A ParmParse constructor has an optional "prefix" argument that will
50// limit the searches to only those entries of the table with this prefix
51// in name. For example:
52// ParmParse pp("plot");
53// will find only those entries with name given by "plot.<string>".
54//
55// All values in the table are stored as strings. For example, the
56// values of "cell_size" in the above input file are stored as the
57// strings "0.5" and "0.75". These strings can be returned as either
58// strings or numeric values by the query functions.
59// Character strings with spaces must be delimited by double quotes
60// in the input file but the quotes are stripped before they are entered
61// into the table. For example, 'title' in the above input file has a
62// single value, the string 'Double Wammy' (without the quotes).
63// Each value in the list associated with a definition can be referred to
64// by its index number. The index numbers start at 0 just like an array
65// in the C programming language. Consider the definition of "test" in
66// the above input file. The first value 'apple'is a string with index
67// 0. The second value 'boy blue' is a string with index 1. The
68// remaining four values are integers indexed 2, 3, 4, and 5.
69//
70// For a string value to represent an integer or float it must fit the
71// following regular experssion:
72// Sign ::= '+' | '-'
73// Digit ::= '0' | '1' | ... | '9'
74// Integer ::= [Sign]Digit+
75// Exp ::= ('e'|'E')Integer
76// Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
77//
78// Where '+' indicates one or more occurences, '*' represents zero or
79// more occurences, '|' means one or the other and '[]' represents zero
80// or one occurence.
81//
82// Note that floats and doubles have the same string representation and
83// that the FORTRAN "double" exponent format is not supported.
84// That is, 1.0d+3 is not a valid representation of a floating point
85// number but that 1.0e+3 is acceptable.
86//
87// There are a host of functions allowing the user to query the database
88// and retrieve values. Here are some general rules about the names of
89// the member functions:
90//
91// * Functions with the string "get" in their names attempt to get a
92// value or an array of values from the table. They generate a
93// run-time error if they are not successful.
94//
95// * Functions with the string "query" in their names attempt to get a
96// value or an array of values from the table. They return the value 1
97// (true) if they are successful and 0 (false) if not.
98//
99// * Functions with the string "kth" in their names get values from
100// the Kth entry with the given name in the database. This is
101// necessary since there may be multiple definitions with the same
102// name in the database.
103//
104// * Functions without the string "kth" in their names get values from
105// the last entry with the given name in the database. Note that the
106// definitions from the command line are appended to the database table
107// and hence will be the last entries.
108//
109// * Functions with the string "arr" in their names get an Array of
110// values from the given entry in the table. The array argument is
111// resized (if necessary) to hold all the values requested.
112//
113// * Functions without the string "arr" in their names get single
114// values from the given entry in the table.
115//
116// The following is a code sample showing how to use ParmParse:
117//
118// main(int argc, char **argv)
119// {
120// char* in_file_name = argv[1];
121// ParmParse::Initialize(argc-2, argv+2, in_file_name);
122//
123// // Query table for value of "niter". If not in table
124// // then set to default value
125// if (!pp.query("niter",niter)) niter = 20;
126//
127// // read array of cell sizes if in table
128// Vector<float> dx;
129// if (nx=pp.countval("cell_size")) {
130// // get nx values starting at index 0 and store in dx.
131// // dx is automatically resized here.
132// pp.getarr("cell_size",dx,0,nx);
133// }
134// ParmParse::Finalize();
135// }
136//
137// void do_graphics()
138// {
139// //
140// // Will only query entries with the "plot" prefix:
141// //
142// ParmParse pp("plot");
143// //
144// // Read all variables with "plot.var" keyword.
145// //
146// std::string var_name;
147// Vector<int> range;
148// int num = pp.countname("var");
149// for (int k = 0; k < num; k++)
150// {
151// //
152// // Element 0 in list is a string.
153// //
154// pp.getkth("var",k,var_name,0);
155// //
156// // Elements 1 and 2 are integers.
157// // Note that "range" will be resized to hold 2 elements.
158// //
159// pp.getktharr("var",k,range,1,2);
160// cout << "variable = " << var_name << "lo, hi = ",
161// << range[0] << " " << range[1] << endl;
162// }
163// }
164// -----------------------------------------------------------------
165// ----------------------- END COMMENTS ---------------------------
166// -----------------------------------------------------------------

887 template <typename T, std::size_t N>
888 void get (const char* name, std::array<T,N>& ref) const {
889 std::vector<T> v;
890 this->getarr(name, v);
891 AMREX_ALWAYS_ASSERT(v.size() >= N);
892 for (std::size_t i = 0; i < N; ++i) {
893 ref[i] = v[i];
894 }
895 }
896
897 template <typename T, std::size_t N>
898 int query (const char* name, std::array<T,N>& ref) const {
899 std::vector<T> v;
900 int exist = this->queryarr(name, v);
901 if (exist) {
902 AMREX_ALWAYS_ASSERT(v.size() >= N);
903 for (std::size_t i = 0; i < N; ++i) {
904 ref[i] = v[i];
905 }
906 }
907 return exist;
908 }
