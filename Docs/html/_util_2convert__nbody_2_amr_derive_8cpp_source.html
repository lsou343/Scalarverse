63double parseByName(std::string file_name, std::string var_name) {
64 std::ifstream fin;
65 std::string line = "";
66
67 fin.open(file_name.c_str());
68 while(std::getline(fin, line)) {
69 if(line.find("//") == 0 || line.empty())
70 continue;
71 else {
72 int pos = line.find("=");
73 std::string name = line.substr(0,pos-1);
74 std::string svalue = line.substr(pos+1);
75 if(name == var_name) {
76 std::istringstream instr(svalue);
77 double value;
78 instr >> value;
79 return value;
80 }
81 }
82 }
83 return -9999.0;
84}

101/*
102Opens the output file and writes all of metadata attributes.
103Should be run only by the root process.
104*/
105void output_write_metadata(const std::string file_path,
106 const int nx, const int ny, const int nz, const double domain_size,
107 const double omega_b, const double omega_m, const double omega_l,
108 const double h, const double z) {
109
110 hid_t file = H5Fopen(file_path.c_str(), H5F_ACC_RDWR, H5P_DEFAULT);
111
112 //
113 // Output format root attribute
114 //
115
116 const std::string format_text = "nyx-lyaf";
117
118 hid_t str_type = H5Tcopy(H5T_C_S1);
119 hid_t scalar_space = H5Screate(H5S_SCALAR);
120
121 // Fix the str_type length for the format string.
122 H5Tset_size(str_type, strlen(format_text.c_str()));
123
124 hid_t attr = H5Acreate(file, "format", str_type, scalar_space, H5P_DEFAULT,
125 H5P_DEFAULT);
126 H5Awrite(attr, str_type, format_text.c_str());
127
128 //
129 // Domain metadata group
130 //
131
132 hid_t group = H5Gcreate(file, "domain", H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
133
134 // Dataspace for domain attributes
135 hsize_t num_dims[1] = {3};
136 hid_t grid_attr_space = H5Screate_simple(1, num_dims, NULL);
137
138 // Grid shape
139 int shape[3] = {nx, ny, nz};
140 attr = H5Acreate(group, "shape", H5T_STD_I32LE, grid_attr_space,
141 H5P_DEFAULT, H5P_DEFAULT);
142 H5Awrite(attr, H5T_NATIVE_INT, shape);
143
144 // Grid size
145 double size[3] = {domain_size, domain_size, domain_size};
146 attr = H5Acreate(group, "size", H5T_IEEE_F64LE, grid_attr_space,
147 H5P_DEFAULT, H5P_DEFAULT);
148 H5Awrite(attr, H5T_NATIVE_DOUBLE, size);
149
150 //
151 // Universe metadata group
152 //
153
154 group = H5Gcreate(file, "universe", H5P_DEFAULT, H5P_DEFAULT,
155 H5P_DEFAULT);
156
157 attr = H5Acreate(group, "omega_b", H5T_IEEE_F64LE, scalar_space,
158 H5P_DEFAULT, H5P_DEFAULT);
159 H5Awrite(attr, H5T_NATIVE_DOUBLE, &omega_b);
160 attr = H5Acreate(group, "omega_m", H5T_IEEE_F64LE, scalar_space,
161 H5P_DEFAULT, H5P_DEFAULT);
162 H5Awrite(attr, H5T_NATIVE_DOUBLE, &omega_m);
163 attr = H5Acreate(group, "omega_l", H5T_IEEE_F64LE, scalar_space,
164 H5P_DEFAULT, H5P_DEFAULT);
165 H5Awrite(attr, H5T_NATIVE_DOUBLE, &omega_l);
166 attr = H5Acreate(group, "hubble", H5T_IEEE_F64LE, scalar_space,
167 H5P_DEFAULT, H5P_DEFAULT);
168 H5Awrite(attr, H5T_NATIVE_DOUBLE, &h);
169 attr = H5Acreate(group, "redshift", H5T_IEEE_F64LE, scalar_space,
170 H5P_DEFAULT, H5P_DEFAULT);
171 H5Awrite(attr, H5T_NATIVE_DOUBLE, &z);
172
173 //
174 // Field groups
175 //
176
177 group = H5Gcreate(file, "native_fields", H5P_DEFAULT, H5P_DEFAULT,
178 H5P_DEFAULT);
179
180 group = H5Gcreate(file, "derived_fields", H5P_DEFAULT, H5P_DEFAULT,
181 H5P_DEFAULT);
182
183 // Close all resources.
184 H5Sclose(grid_attr_space);
185 H5Gclose(group);
186 H5Aclose(attr);
187 H5Sclose(scalar_space);
188 H5Tclose(str_type);
189 H5Fclose(file);
190 H5close();
191}

194/*
195Creates a dataset with the given cell dimensions, at the path
196"/native_fields/(field_name)".
197Should be run only by the master rank.
198*/
199void output_create_field(const std::string file_path, const std::string field_path,
200 const std::string units, const int nx, const int ny, const int nz) {
201 // Open the output.
202 hid_t file = H5Fopen(file_path.c_str(), H5F_ACC_RDWR, H5P_DEFAULT);
203 // Create a 3D, nx x ny x nz dataspace.
204 hsize_t dims[3] = {nx, ny, nz};
205 hid_t grid_space = H5Screate_simple(3, dims, NULL);
206 // Create the dataset.
207 hid_t dataset = H5Dcreate(file, field_path.c_str(), H5T_IEEE_F32LE,
208 grid_space, H5P_DEFAULT, H5P_DEFAULT,
209 H5P_DEFAULT);
210
211 if (dataset < 0) {
212 printf("Error: could not create dataset. H5 returned %i.\n", dataset);
213 exit(1);
214 }
215
216 // Don't forget to attach units attr.
217 hid_t scalar_space = H5Screate(H5S_SCALAR);
218 hid_t str_type = H5Tcopy(H5T_C_S1);
219 H5Tset_size(str_type, strlen(units.c_str()));
220 hid_t attr = H5Acreate(dataset, "units", str_type, scalar_space,
221 H5P_DEFAULT, H5P_DEFAULT);
222 H5Awrite(attr, str_type, units.c_str());
223
224 // Close resources.
225 H5Aclose(attr);
226 H5Tclose(str_type);
227 H5Sclose(scalar_space);
228 H5Dclose(dataset);
229 H5Sclose(grid_space);
230 H5Fclose(file);
231}

238void output_write_field(const std::string file_path, const std::string field_path,
239 MultiFab &mf) {
240
241 MPI_Comm comm = MPI_COMM_WORLD;
242 MPI_Info info = MPI_INFO_NULL;
243 int mpi_rank;
244 MPI_Comm_rank(comm, &mpi_rank);
245
246 // Create the file access prop list.
247 hid_t pa_plist = H5Pcreate(H5P_FILE_ACCESS);
248 H5Pset_fapl_mpio(pa_plist, comm, info);
249
250 // Open the file, and the group.
251 hid_t file = H5Fopen(file_path.c_str(), H5F_ACC_RDWR, pa_plist);
252 // Open the field dataset.
253 hid_t dataset = H5Dopen(file, field_path.c_str(), H5P_DEFAULT);
254
255 // Make sure the dataset is there.
256 if (dataset < 0) {
257 printf("Error on rank %i: Could not find dataset %s.\n", mpi_rank,
258 field_path.c_str());
259 exit(1);
260 }
261
262 // Grab the dataspace of the field dataset from file.
263 hid_t file_dataspace = H5Dget_space(dataset);
264
265 // Create collective io prop list.
266 hid_t collective_plist = H5Pcreate(H5P_DATASET_XFER);
267 H5Pset_dxpl_mpio(collective_plist, H5FD_MPIO_COLLECTIVE);
268
269 // Iterate over Fabs, select matching hyperslab and write.
270 hid_t status;
271 // slab lo index and shape.
272 hsize_t slab_offsets[3], slab_dims[3];
273 hid_t slab_dataspace;
274
275 int write_count = 0;
276
277 for (MFIter mfi(mf); mfi.isValid(); ++mfi) {
278 // Get fab, lo and hi vectors.
279 const Box& box = mfi.validbox();
280 const int *lo_vec = box.loVect();
281 const int *hi_vec = box.hiVect();
282
283 // this is in fortran order.
284 double *fab_data = mf[mfi].dataPtr();
285
286 // We must make a copy of the fab data in c-order
287 int nx = hi_vec[0] - lo_vec[0] + 1;
288 int ny = hi_vec[1] - lo_vec[1] + 1;
289 int nz = hi_vec[2] - lo_vec[2] + 1;
290 size_t num_cells = nx * ny * nz;
291
292 //size_t block_size = num_cells;
293 std::vector<double> h5_data_vec(num_cells);
294 double *h5_data = h5_data_vec.data();
295 //double* h5_data = (double*) malloc(block_size * sizeof(double));
296 if (h5_data == NULL) {
297 printf("Error allocating h5 data block.\n");
298 exit(1);
299 }
300
301 //int ix, iy, iz, c_index, f_index;
302 for (int ix = 0; ix < nx; ++ix) {
303 for (int iy = 0; iy < ny; ++iy) {
304 for (int iz = 0; iz < nz; ++iz) {
305 size_t c_index = (ix * ny + iy) * nz + iz;
306 size_t f_index = (iz * ny + iy) * nx + ix;
307 h5_data[c_index] = fab_data[f_index];
308 }
309 }
310 }
311
312 // Data is preped. Now write it.
313
314 // Set slab offset and shape.
315 slab_offsets[0] = lo_vec[0];
316 slab_offsets[1] = lo_vec[1];
317 slab_offsets[2] = lo_vec[2];
318 slab_dims[0] = nx;
319 slab_dims[1] = ny;
320 slab_dims[2] = nz;
321
322 // Create the slab space.
323 slab_dataspace = H5Screate_simple(3, slab_dims, NULL);
324
325 // Select the hyperslab matching this fab.
326 status = H5Sselect_hyperslab(file_dataspace, H5S_SELECT_SET,
327 slab_offsets, NULL, slab_dims, NULL);
328 if (status < 0) {
329 printf("Error on rank %i: could not select hyperslab.\n", mpi_rank);
330 exit(1);
331 }
332
333 // Write this pencil.
334 status = H5Dwrite(dataset, H5T_NATIVE_DOUBLE, slab_dataspace,
335 file_dataspace, collective_plist, h5_data);
336 if (status < 0) {
337 printf("Error on rank %i: could not write hyperslab.\n", mpi_rank);
338 exit(1);
339 }
340
341 H5Sclose(slab_dataspace);
342 write_count++;
343 }
344
345 // Close HDF5 resources.
346 H5Pclose(collective_plist);
347 H5Sclose(file_dataspace);
348 H5Dclose(dataset);
349 H5Fclose(file);
350 H5Pclose(pa_plist);
351}

354size_t proc_status_value(const std::string& field) {
355 std::ifstream in("/proc/self/status");
356 std::string line;
357 while (in) {
358 std::getline(in, line);
359 if (line.compare(0, field.length(), field) == 0) {
360 std::istringstream iss(line);
361 std::string f; size_t res;
362 iss >> f >> res;
363 return res;
364 }
365 }
366 return 0;

370void printHWM(const char* info, MPI_Comm comm) {
371 int io_rank=0, mpi_size, mpi_rank;
372 MPI_Comm_size(comm, &mpi_size);
373 MPI_Comm_rank(comm, &mpi_rank);
374
375 //size_t hwm = proc_status_value("VmHWM");
376 size_t hwm = proc_status_value("VmRSS");
377 size_t avg_hwm = 0, min_hwm = 0, max_hwm = 0;
378 BL_MPI_REQUIRE(MPI_Reduce(&hwm, &min_hwm, 1, MPI_LONG, MPI_MIN, io_rank, comm));
379 BL_MPI_REQUIRE(MPI_Reduce(&hwm, &max_hwm, 1, MPI_LONG, MPI_MAX, io_rank, comm));
380 BL_MPI_REQUIRE(MPI_Reduce(&hwm, &avg_hwm, 1, MPI_LONG, MPI_SUM, io_rank, comm));
381
382 if (mpi_rank == io_rank) {
383 printf("%s: Max = %zu, Min = %zu, Average = %zu\n", info, max_hwm, min_hwm, avg_hwm/mpi_size);
384 fflush(stdout);
385 }
386}

389int main(int argc, char **argv) {
390 amrex::Initialize(argc, argv);
391
392 {
393 // MPI info...
394 MPI_Comm comm = MPI_COMM_WORLD;
395 int mpi_size, mpi_rank;
396 MPI_Comm_size(comm, &mpi_size);
397 MPI_Comm_rank(comm, &mpi_rank);
398
399 CHK(printHWM("Start", comm));
400
401 //
402 // Parameter parsing
403 //
404
405 if (argc < 3) {
406 print_usage(argc, argv);
407 }
408
409 ParmParse pp;
410 if (pp.contains("help")) {
411 print_usage(argc, argv);
412 }
413 bool verbose = false;
414 if (pp.contains("verbose")) {
415 verbose = true;
416 AmrData::SetVerbose(true);
417 }
418 std::string input_path;
419 pp.get("input_path", input_path);
420 std::string output_path;
421 pp.get("output_path", output_path);
422
423 // -------------------
424 // Open input plotfile
425 // -------------------
426
427 if (ParallelDescriptor::IOProcessor()) {
428 std::cout << std::endl << "Opening " << input_path << " ...";
429 }
430
431 DataServices::SetBatchMode();
432 Amrvis::FileType fileType(Amrvis::NEWPLT);
433 DataServices dataServices(input_path, fileType);
434 if( ! dataServices.AmrDataOk()) {
435 DataServices::Dispatch(DataServices::ExitRequest, NULL);
436 }
437
438 if (ParallelDescriptor::IOProcessor()) {
439 std::cout << " done." << std::endl;
440 }
441
442 AmrData& amrData = dataServices.AmrDataRef();
443 int finestLevel = amrData.FinestLevel();
444 int Nlev = finestLevel + 1;
445
446 //
447 // Read metadata.
448 //
449
450 // Read comoving_a file
451 Real comoving_a;
452 if (ParallelDescriptor::IOProcessor()) {
453 std::string a_file_path = input_path + "/comoving_a";
454 std::ifstream a_file;
455 a_file.open(a_file_path.c_str(), std::ios::in);
456 if (!a_file.good()) {
457 amrex::FileOpenFailed(a_file_path);
458 }
459 a_file >> comoving_a;
460 }
461
462 double z = 1.0/comoving_a - 1.0;
463
464 // Read the_parameters
465 std::string params_file = input_path + "/the_parameters";
466 double h = parseByName(params_file, "nyx.comoving_h");
467 double omega_b = parseByName(params_file, "nyx.comoving_OmB");
468 double omega_m = parseByName(params_file, "nyx.comoving_OmM");
469 double omega_l = 1.0-omega_m;
470 double domain_size = parseByName(params_file, "geometry.prob_hi") * h;
471
472 if (h < 0.0 || omega_b < 0.0 || omega_m < 0.0 || domain_size < 0.0) {
473 Print() << "Error reading the_parameters file!" << std::endl;
474 exit(-1);
475 }
476
477 Print() << "Parameters: ";
478 Print() << "Om = " << omega_m << "; " << "Ob = " << omega_b << "; "
479 << "h_0 = " << h << "; " << "L = " << domain_size << std::endl;
480
481 // Parallel info... We might not need it.
482 int MyPE = ParallelDescriptor::MyProc();
483 int NumPEs = ParallelDescriptor::NProcs();
484
485 //
486 // Specify the data we need from the plotfile.
487 //
488
489 // For all lyaf analysis, we will need:
490 // - dark matter density
491 // - gas density
492 // - gas momemtum x, y, z
493 // - gas temperature
494
495 const int nGhost = 0;
496 int nComp = 1;
497 Vector<int> comps(nComp);
498 int i_density(amrData.StateNumber("particle_mass_density"));
499 comps[0] = i_density;
500
501 const Vector<string>& plotVarNames = amrData.PlotVarNames();
502 Vector<string> inVarNames(nComp);
503 Vector<int> destFillComps(nComp);
504 if (ParallelDescriptor::IOProcessor()) {
505 std::cout << std::endl << "Converting the following states: "
506 << std::endl;
507 }
508 for (int i = 0; i < nComp; ++i) {
509 inVarNames[i] = plotVarNames[comps[i]];
510 if (ParallelDescriptor::IOProcessor()) {
511 std::cout << " " << amrData.StateNumber(inVarNames[i])
512 << " (" << inVarNames[i] << ")" << std::endl;
513 }
514 destFillComps[i] = i;
515 }
516
517 //
518 // Make boxes and boxarray.
519 //
520
521 // Grab the number of cells in each direction (should be the same, but
522 // we're keeping things general).
523 Box pd(amrData.ProbDomain()[0]);
524 int64_t grid_nx = pd.bigEnd(0) - pd.smallEnd(0) + 1;
525 int64_t grid_ny = pd.bigEnd(1) - pd.smallEnd(1) + 1;
526 int64_t grid_nz = pd.bigEnd(2) - pd.smallEnd(2) + 1;
527 int64_t num_cells = grid_nx * grid_ny * grid_nz;
528
529 if (ParallelDescriptor::IOProcessor()) {
530 printf("The grid has a shape of (%i, %i, %i), %.3e cells total.\n\n",
531 (int)grid_nx, (int)grid_ny, (int)grid_nz, (double)num_cells);
532 printf("Making skewer chunks.\n");
533 fflush(stdout);
534 }
535
536 // Check if we can split along x evenly.
537 if (grid_nx % mpi_size != 0) {
538 if (ParallelDescriptor::IOProcessor()) {
539 printf("ERROR: domain decomposition.\n");
540 printf("The number of MPI ranks must fit evenly in the number of cells along x.\n");
541 }
542 MPI_Barrier(comm);
543 amrex::Finalize();
544 }
545
546 int chunk_size = grid_nx / mpi_size;
547
548 // The box for z-pencils
549 Box bx_pencil(amrData.ProbDomain()[0]);
550 // List of pencil boxes.
551 BoxList box_list;
552 // indexes
553 int i, ix_lo, ix_hi;
554 for (i = 0; i < mpi_size; ++i) {
555 ix_lo = chunk_size * i;
556 ix_hi = ix_lo + chunk_size - 1;
557
558 Box skewer;
559 skewer.setSmall(0, ix_lo);
560 skewer.setBig(0, ix_hi);
561 skewer.setSmall(1, 0);
562 skewer.setBig(1, grid_ny - 1);
563 skewer.setSmall(2, 0);
564 skewer.setBig(2, grid_nz - 1);
565
566 box_list.push_back(skewer);
567 }
568
569 // Make box array from the skewer box list.
570 BoxArray ba(box_list);
571
572 if (ParallelDescriptor::IOProcessor()) {
573 std::cout << "Creating multifabs." << std::endl;
574 }
575
576 // Make the multifabs.
577 int num_comps = 1; // keeping it simple...
578 int ng = 0; // don't add any ghost zones.
579 int level = 0;
580 int comp_start = 0;
581
582 DistributionMapping dmap(ba, ParallelDescriptor::NProcs());
583
584 MultiFab mf1(ba, dmap, num_comps, ng);
585 MultiFab mf2(ba, dmap, num_comps, ng);
586
587 //
588 // Start outputting.
589 //
590
591 if (ParallelDescriptor::IOProcessor()) {
592 std::cout << "Preparing output file." << std::endl;
593 }
594
595 // Create on master process.
596 if (ParallelDescriptor::IOProcessor()) {
597 // create the file.
598 output_create(output_path.c_str());
599
600 // write metadata.
601 output_write_metadata(output_path.c_str(),
602 grid_nx, grid_ny, grid_nz, domain_size,
603 omega_b, omega_m, omega_l, h, z);
604 }
605
606 ParallelDescriptor::Barrier();
607 Print() << "Before reads allocation in Fabs: " << TotalBytesAllocatedInFabs()/(1024*1024.) << " Mb." << std::endl;
608 CHK(printHWM("Before read/write", comm));
609
610 //
611 // Matter density
612 //
613
614 std::string field_path = "native_fields/matter_density";
615
616 if (ParallelDescriptor::IOProcessor()) {
617 std::cout << "\nReading particle_mass_density." << std::endl;
618 }
619
620 amrData.FillVar(mf1, level, "particle_mass_density", comp_start);
621 amrData.FlushGrids(amrData.StateNumber("particle_mass_density"));
622
623 if (ParallelDescriptor::IOProcessor()) {
624 std::cout << "Computing mean." << std::endl;
625 }
626
627 double mean_density = mf1.norm1() / num_cells;
628
629 if (ParallelDescriptor::IOProcessor()) {
630 printf("Mean matter density: %.14e Msun/Mpc**3\n", mean_density);
631 fflush(stdout);
632 }
633
634 // Convert to mean units.
635 double mean_density_inv = 1.0/mean_density;
636 mf1.mult(mean_density_inv);
637
638 if (ParallelDescriptor::IOProcessor()) {
639 std::cout << "Writing to file." << std::endl;
640 output_create_field(output_path, field_path, "(mean)",
641 grid_nx, grid_ny, grid_nz);
642 }
643 ParallelDescriptor::Barrier();
644 output_write_field(output_path, field_path, mf1);
645 ParallelDescriptor::Barrier();
646 CHK(printHWM("After read/write", comm));
647
648 Print() << "End allocation in Fabs: " << TotalBytesAllocatedInFabs()/(1024*1024.) << " Mb." << std::endl;
649 }
650 amrex::Finalize();
651 return 0;
652}
