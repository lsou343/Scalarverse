158void top_fork(MultiFab& soln, const MultiFab& rhs,
159 const MultiFab& alpha, const Vector<MultiFab*>& beta, const Geometry& geom)
160{
161 auto proc_n = ParallelContext::NProcsSub();
162 ForkJoin fj(Vector<int> {1, proc_n - 1});
163 fj.SetVerbose(fj_verbose);
164 fj.set_task_output_dir(task_output_dir);
165
166 // these multifabs go to task 0 only
167 fj.reg_mf (rhs , "rhs" , ForkJoin::Strategy::single, ForkJoin::Intent::in , 1);
168 fj.reg_mf (alpha, "alpha", ForkJoin::Strategy::single, ForkJoin::Intent::in , 1);
169 fj.reg_mf_vec(beta , "beta" , ForkJoin::Strategy::single, ForkJoin::Intent::in , 1);
170 fj.reg_mf (soln , "soln" , ForkJoin::Strategy::single, ForkJoin::Intent::out, 1);
171
172 // issue top-level fork-join
173 fj.fork_join(
174 [&geom] (ForkJoin &f) {
175 if (f.MyTask() == 0) {
176 // Do some non-MLMG tasks
177 amrex::Print() << "Pretending to do some chemistry ...\n";
178 } else {
179 // Do some MLMG solves
180 amrex::Print() << "Do some linear solves ...\n";
181 fork_solve(f.get_mf("soln"), f.get_mf("rhs"), f.get_mf("alpha"),
182 f.get_mf_vec("beta"), geom);

188void fork_solve(MultiFab& soln, const MultiFab& rhs,
189 const MultiFab& alpha, const Vector<MultiFab*>& beta, const Geometry& geom)
190{
191 // evenly split ranks among ntasks tasks
192 ForkJoin fj(ntasks);
193 fj.SetVerbose(fj_verbose);
194 fj.set_task_output_dir(task_output_dir);
195
196 // register how to copy multifabs to/from tasks
197 fj.reg_mf (rhs , "rhs" , ForkJoin::Strategy::split, ForkJoin::Intent::in);
198 fj.reg_mf (alpha, "alpha", ForkJoin::Strategy::split, ForkJoin::Intent::in);
199 fj.reg_mf_vec(beta , "beta" , ForkJoin::Strategy::split, ForkJoin::Intent::in);
200 fj.reg_mf (soln , "soln" , ForkJoin::Strategy::split, ForkJoin::Intent::out);
201
202 if (flag_modify_split) {
203 auto comp_n = soln.nComp();
204 if (comp_n > ntasks) {
205 amrex::Print() << "Doing a custom split where first component is ignored\n";
206 // test custom split, skip the first component
207 Vector<ForkJoin::ComponentSet> comp_split(ntasks);
208 for (int i = 0; i < ntasks; ++i) {
209 // split components across tasks
210 comp_split[i].lo = 1 + (comp_n-1) * i / ntasks;
211 comp_split[i].hi = 1 + (comp_n-1) * (i+1) / ntasks;
212 }
213
214 fj.modify_split("rhs", comp_split);
215 fj.modify_split("alpha", comp_split);
216 for (int i = 0; i < beta.size(); ++i) {
217 fj.modify_split("beta", i, comp_split);
218 }
219 fj.modify_split("soln", comp_split);
220 }
221 }
222
223 // can reuse ForkJoin object for multiple fork-join invocations
224 // creates forked multifabs only first time around, reuses them thereafter
225 for (int i = 0; i < 2; ++i) {
226 // issue fork-join
227 fj.fork_join(
228 [&geom, i] (ForkJoin &f) {
229 solve_all(f.get_mf("soln"), f.get_mf("rhs"), f.get_mf("alpha"),
230 f.get_mf_vec("beta"), geom);
231 }
232 );
233 }
234}

236void solve_all(MultiFab& soln, const MultiFab& rhs,
237 const MultiFab& alpha, const Vector<MultiFab*>& beta,
238 const Geometry& geom)
239{
240 const BoxArray& ba = soln.boxArray();
241 const DistributionMapping& dm = soln.DistributionMap();
242
243 if (rhs.nComp() == 1)
244 {
245 single_component_solve(soln, rhs, alpha, beta, geom);
246 }
247 else
248 {
249 for (int i = 0; i < soln.nComp(); ++i) {
250 MultiFab ssoln (ba, dm, 1, 1);
251 ssoln.setVal(0.0);
252
253 MultiFab srhs (rhs, amrex::make_alias, i, 1);
254 MultiFab salpha(alpha, amrex::make_alias, i, 1);
255 Vector<std::unique_ptr<MultiFab> > sbeta(AMREX_SPACEDIM);
256 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
257 sbeta[idim].reset(new MultiFab(*beta[idim], amrex::make_alias, i, 1));
258 }
259
260 single_component_solve(ssoln, srhs, salpha,
261 amrex::GetVecOfPtrs(sbeta), geom);
262
263 MultiFab::Copy(soln, ssoln, 0, i, 1, 0);
264 }
265 }
