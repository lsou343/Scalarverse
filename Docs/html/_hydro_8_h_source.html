108void
109normalize_species_fluxes(const int i, const int j, const int k, amrex::Array4<amrex::Real> const& flx, const int NumSpec)
110{
111 amrex::Real summ, fac;
112 summ = 0.0;
113 for (int n = 0; n < NumSpec; ++n)
114 summ += flx(i, j, k, FirstSpec + n);
115
116 // We skip the normalization if the sum is zero or within epsilon.
117 // There can be numerical problems here if the density flux is
118 // approximately zero at the interface but not exactly, resulting in
119 // division by a small number and/or resulting in one of the species
120 // fluxes being negative because of roundoff error. There are also other
121 // terms like artificial viscosity which can cause these problems.
122 // So checking that sum is sufficiently large helps avoid this.
123
124
125 fac = (std::abs(summ) > std::numeric_limits<amrex::Real>::epsilon() * std::abs(flx(i, j, k, URHO))) ? flx(i, j, k, URHO) / summ : 1.0;
126 for (int n = 0; n < NumSpec; ++n)
127 flx(i, j, k, FirstSpec + n) *= fac;
128}

170pc_update(
171 const int i,
172 const int j,
173 const int k,
174 amrex::Array4<const amrex::Real> const& u,
175 amrex::Array4<amrex::Real> const& update,
176 const amrex::GpuArray<const amrex::Array4<amrex::Real>, AMREX_SPACEDIM> flx,
177 amrex::Array4<const amrex::Real> const& vol,
178 amrex::Array4<const amrex::Real> const& pdivu,
179 amrex::Real const a_old,
180 amrex::Real const a_new,
181 amrex::Real const dt,
182 const amrex::Real gamma_minus_1)
183{
184 const amrex::Real volinv = 1 / vol(i, j, k);
185 const amrex::Real a_half = 0.5*(a_old + a_new);
186 const amrex::Real a_half_inv = 1.0 / a_half;
187 //Store all 8 factors
188 const amrex::Real a_fact[8] = {a_half_inv,1.0,1.0,1.0,a_half,a_half,a_half_inv,a_half_inv};
189 for (int n = 0; n < update.nComp(); ++n) {
190 update(i, j, k, n) += ( (flx[0](i, j, k, n) - flx[0](i + 1, j, k, n))
191 +(flx[1](i, j, k, n) - flx[1](i, j + 1, k, n))
192 +(flx[2](i, j, k, n) - flx[2](i, j, k + 1, n)) ) *
193 volinv * a_fact[n];
194 }
195
196 update(i, j, k, UEINT) += a_half * (a_new - a_old) * ( 2.0 - 3.0 * gamma_minus_1) * u(i,j,k,UEINT);
197 update(i, j, k, UEDEN) += a_half * (a_new - a_old) * ( 2.0 - 3.0 * gamma_minus_1) * u(i,j,k,UEINT);
198
199 update(i, j, k, UEINT) += pdivu(i, j, k)*u(i,j,k,UEINT)*gamma_minus_1*dt*a_half;
200}
