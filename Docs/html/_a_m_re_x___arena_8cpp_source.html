58void*
59Arena::allocate_system (std::size_t nbytes)
60{
61 void * p;
62#ifdef AMREX_USE_GPU
63 if (arena_info.use_cpu_memory)
64 {
65 p = std::malloc(nbytes);
66 if (p && arena_info.device_use_hostalloc) AMREX_MLOCK(p, nbytes);
67 }
68 else if (arena_info.device_use_hostalloc)
69 {
70 AMREX_HIP_OR_CUDA_OR_DPCPP(
71 AMREX_HIP_SAFE_CALL (hipHostMalloc(&p, nbytes, hipHostMallocMapped));,
72 AMREX_CUDA_SAFE_CALL(cudaHostAlloc(&p, nbytes, cudaHostAllocMapped));,
73 p = sycl::malloc_host(nbytes, Gpu::Device::syclContext()));
74 }
75 else
76 {
77 if (abort_on_out_of_gpu_memory) {
78 std::size_t free_mem_avail = Gpu::Device::freeMemAvailable();
79 if (nbytes >= free_mem_avail) {
80 amrex::Abort("Out of gpu memory. Free: " + std::to_string(free_mem_avail)
81 + " Asked: " + std::to_string(nbytes));
82 }
83 }
84
85 if (arena_info.device_use_managed_memory)
86 {
87 AMREX_HIP_OR_CUDA_OR_DPCPP
88 (AMREX_HIP_SAFE_CALL(hipMallocManaged(&p, nbytes));,
89 AMREX_CUDA_SAFE_CALL(cudaMallocManaged(&p, nbytes));,
90 p = sycl::malloc_shared(nbytes, Gpu::Device::syclDevice(), Gpu::Device::syclContext()));
91 if (arena_info.device_set_readonly)
92 {
93 Gpu::Device::mem_advise_set_readonly(p, nbytes);
94 }
95 if (arena_info.device_set_preferred)
96 {
97 const int device = Gpu::Device::deviceId();
98 Gpu::Device::mem_advise_set_preferred(p, nbytes, device);
99 }
100 }
101 else
102 {
103 AMREX_HIP_OR_CUDA_OR_DPCPP
104 (AMREX_HIP_SAFE_CALL ( hipMalloc(&p, nbytes));,
105 AMREX_CUDA_SAFE_CALL(cudaMalloc(&p, nbytes));,
106 p = sycl::malloc_device(nbytes, Gpu::Device::syclDevice(), Gpu::Device::syclContext()));
107 }
108 }
109#else
110 p = std::malloc(nbytes);
111 if (p && arena_info.device_use_hostalloc) AMREX_MLOCK(p, nbytes);
112#endif
113 if (p == nullptr) amrex::Abort("Sorry, malloc failed");
114 return p;
115}

117void
118Arena::deallocate_system (void* p, std::size_t nbytes)
119{
120#ifdef AMREX_USE_GPU
121 if (arena_info.use_cpu_memory)
122 {
123 if (p && arena_info.device_use_hostalloc) AMREX_MUNLOCK(p, nbytes);
124 std::free(p);
125 }
126 else if (arena_info.device_use_hostalloc)
127 {
128 AMREX_HIP_OR_CUDA_OR_DPCPP
129 (AMREX_HIP_SAFE_CALL ( hipHostFree(p));,
130 AMREX_CUDA_SAFE_CALL(cudaFreeHost(p));,
131 sycl::free(p,Gpu::Device::syclContext()));
132 }
133 else
134 {
135 AMREX_HIP_OR_CUDA_OR_DPCPP
136 (AMREX_HIP_SAFE_CALL ( hipFree(p));,
137 AMREX_CUDA_SAFE_CALL(cudaFree(p));,
138 sycl::free(p,Gpu::Device::syclContext()));
139 }
140#else
141 if (p && arena_info.device_use_hostalloc) AMREX_MUNLOCK(p, nbytes);
142 std::free(p);
143#endif

146void
147Arena::Initialize ()
148{
149 if (initialized) return;
150 initialized = true;
151
152 BL_ASSERT(the_arena == nullptr);
153 BL_ASSERT(the_device_arena == nullptr);
154 BL_ASSERT(the_managed_arena == nullptr);
155 BL_ASSERT(the_pinned_arena == nullptr);
156 BL_ASSERT(the_cpu_arena == nullptr);
157
158 ParmParse pp("amrex");
159 pp.query("use_buddy_allocator", use_buddy_allocator);
160 pp.query("buddy_allocator_size", buddy_allocator_size);
161 pp.query("the_arena_init_size", the_arena_init_size);
162 pp.query("the_arena_is_managed", the_arena_is_managed);
163 pp.query("abort_on_out_of_gpu_memory", abort_on_out_of_gpu_memory);
164
165#ifdef AMREX_USE_GPU
166 if (use_buddy_allocator)
167 {
168 if (buddy_allocator_size <= 0) {
169#ifdef AMREX_USE_DPCPP
170 // buddy_allocator_size = Gpu::Device::maxMemAllocSize() / 4L * 3L;
171 buddy_allocator_size = 1024L*1024L*1024L; // xxxxx DPCPP: todo
172#else
173 buddy_allocator_size = Gpu::Device::totalGlobalMem() / 4L * 3L;
174#endif
175 }
176 std::size_t chunk = 512*1024*1024;
177 buddy_allocator_size = (buddy_allocator_size/chunk) * chunk;
178 if (the_arena_is_managed) {
179 the_arena = new DArena(buddy_allocator_size, 512, ArenaInfo().SetPreferred());
180 } else {
181 the_arena = new DArena(buddy_allocator_size, 512, ArenaInfo().SetDeviceMemory());
182 }
183 }
184 else
185#endif
186 {
187#if defined(BL_COALESCE_FABS) || defined(AMREX_USE_GPU)
188 if (the_arena_is_managed) {
189 the_arena = new CArena(0, ArenaInfo().SetPreferred());
190 } else {
191 the_arena = new CArena(0, ArenaInfo().SetDeviceMemory());
192 }
193#ifdef AMREX_USE_GPU
194 if (the_arena_init_size <= 0) {
195#ifdef AMREX_USE_DPCPP
196// the_arena_init_size = Gpu::Device::maxMemAllocSize() / 4L * 3L;
197 the_arena_init_size = 1024L*1024L*1024L; // xxxxx DPCPP: todo
198#else
199 the_arena_init_size = Gpu::Device::totalGlobalMem() / 4L * 3L;
200#endif
201 }
202 void *p = the_arena->alloc(static_cast<std::size_t>(the_arena_init_size));
203 the_arena->free(p);
204#endif
205#else
206 the_arena = new BArena;
207#endif
208 }
209
210#ifdef AMREX_USE_GPU
211 the_device_arena = new CArena(0, ArenaInfo().SetDeviceMemory());
212#else
213 the_device_arena = new BArena;
214#endif
215
216#ifdef AMREX_USE_GPU
217 the_managed_arena = new CArena;
218#else
219 the_managed_arena = new BArena;
220#endif
221
222 // When USE_CUDA=FALSE, we call mlock to pin the cpu memory.
223 // When USE_CUDA=TRUE, we call cudaHostAlloc to pin the host memory.
224 the_pinned_arena = new CArena(0, ArenaInfo().SetHostAlloc());
225
226 std::size_t N = 1024UL*1024UL*8UL;
227
228 void *p = the_device_arena->alloc(N);
229 the_device_arena->free(p);
230
231 p = the_managed_arena->alloc(N);
232 the_managed_arena->free(p);
233
234 p = the_pinned_arena->alloc(N);
235 the_pinned_arena->free(p);
236
237 the_cpu_arena = new BArena;
238}

245 {
246 Long min_megabytes = Gpu::Device::totalGlobalMem() / (1024*1024);
