10 void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
11 {
12 int nx = nppc[0];
13 int ny = nppc[1];
14 int nz = nppc[2];
15
16 int ix_part = i_part/(ny * nz);
17 int iy_part = (i_part % (ny * nz)) % ny;
18 int iz_part = (i_part % (ny * nz)) / ny;
19
20 r[0] = (0.5+ix_part)/nx;
21 r[1] = (0.5+iy_part)/ny;
22 r[2] = (0.5+iz_part)/nz;
23 }

104void MDParticleContainer::computeForces()
105{
106 BL_PROFILE("MDParticleContainer::computeForces");
107
108 const int lev = 0;
109 const Geometry& geom = Geom(lev);
110 auto& plev = GetParticles(lev);
111
112 for(MFIter mfi = MakeMFIter(lev); mfi.isValid(); ++mfi)
113 {
114 int gid = mfi.index();
115 int tid = mfi.LocalTileIndex();
116 auto index = std::make_pair(gid, tid);
117
118 auto& ptile = plev[index];
119 auto& aos = ptile.GetArrayOfStructs();
120 const size_t np = aos.numParticles();
121
122 auto nbor_data = m_neighbor_list[lev][index].data();
123 ParticleType* pstruct = aos().dataPtr();
124
125 // now we loop over the neighbor list and compute the forces
126 AMREX_FOR_1D ( np, i,
127 {
128 ParticleType& p1 = pstruct[i];
129 p1.rdata(PIdx::ax) = 0.0;
130 p1.rdata(PIdx::ay) = 0.0;
131 p1.rdata(PIdx::az) = 0.0;
132
133 for (const auto& p2 : nbor_data.getNeighbors(i))
134 {
135 Real dx = p1.pos(0) - p2.pos(0);
136 Real dy = p1.pos(1) - p2.pos(1);
137 Real dz = p1.pos(2) - p2.pos(2);
138
139 Real r2 = dx*dx + dy*dy + dz*dz;
140 r2 = amrex::max(r2, Params::min_r*Params::min_r);
141
142 if (r2 > Params::cutoff*Params::cutoff) return;
143
144 Real r = sqrt(r2);
145
146 Real coef = (1.0 - Params::cutoff / r) / r2;
147 p1.rdata(PIdx::ax) += coef * dx;
148 p1.rdata(PIdx::ay) += coef * dy;
149 p1.rdata(PIdx::az) += coef * dz;
150 }
151 });
152 }

208void MDParticleContainer::moveParticles(const amrex::Real& dt)
209{
210 BL_PROFILE("MDParticleContainer::moveParticles");
211
212 const int lev = 0;
213 const Geometry& geom = Geom(lev);
214 const auto plo = Geom(lev).ProbLoArray();
215 const auto phi = Geom(lev).ProbHiArray();
216 const auto dx = Geom(lev).CellSizeArray();

218
219 for(MFIter mfi = MakeMFIter(lev); mfi.isValid(); ++mfi)
220 {
221 int gid = mfi.index();
222 int tid = mfi.LocalTileIndex();
223
224 auto& ptile = plev[std::make_pair(gid, tid)];
225 auto& aos = ptile.GetArrayOfStructs();
226 ParticleType* pstruct = &(aos[0]);
227
228 const size_t np = aos.numParticles();
229
230 // now we move the particles
231 AMREX_FOR_1D ( np, i,
232 {
233 ParticleType& p = pstruct[i];
234 p.rdata(PIdx::vx) += p.rdata(PIdx::ax) * dt;
235 p.rdata(PIdx::vy) += p.rdata(PIdx::ay) * dt;
236 p.rdata(PIdx::vz) += p.rdata(PIdx::az) * dt;
237
238 p.pos(0) += p.rdata(PIdx::vx) * dt;
239 p.pos(1) += p.rdata(PIdx::vy) * dt;
240 p.pos(2) += p.rdata(PIdx::vz) * dt;
241
242 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
243 while ( (p.pos(idim) < plo[idim]) or (p.pos(idim) > phi[idim]) ) {
244 if ( p.pos(idim) < plo[idim] ) {
245 p.pos(idim) = 2*plo[idim] - p.pos(idim);
246 } else {
247 p.pos(idim) = 2*phi[idim] - p.pos(idim);
248 }
249 p.rdata(idim) *= -1; // flip velocity
250 }
251 }
252 });
253 }
