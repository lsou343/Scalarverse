332 const MultiFab& drho_and_drhoU, const MultiFab& dphi,
333 const Vector<MultiFab*>& grad_delta_phi_cc)
334{
335 BL_PROFILE("Gravity::gravity_sync()");
336 BL_ASSERT(parent->finestLevel()>crse_level);
337
338 if (verbose)
339 {
340 amrex::Print() << " ... gravity_sync at crse_level " << crse_level << '\n';
341 amrex::Print() << " ... up to finest_level " << fine_level << '\n';
342 }

345 MultiFab crse_rhs(grids[crse_level], dmap[crse_level], 1, 0);
346 MultiFab::Copy(crse_rhs, drho_and_drhoU, 0, 0, 1, 0);

351 const Box& crse_domain = crse_geom.Domain();
352
353 // In the all-periodic case we enforce that CrseRhsSync sums to zero.
354 if (crse_geom.isAllPeriodic() && (grids[crse_level].numPts() == crse_domain.numPts()))
355 {
356 Real local_correction = 0;
357#ifdef _OPENMP
358#pragma omp parallel if (!system::regtest_reduction) reduction(+:local_correction)
359#endif
360 for (MFIter mfi(crse_rhs,true); mfi.isValid(); ++mfi)
361 local_correction += crse_rhs[mfi].sum<RunOn::Host>(mfi.tilebox(), 0, 1);
362 ParallelDescriptor::ReduceRealSum(local_correction);
363
364 local_correction /= grids[crse_level].numPts();
365
366 if (verbose)
367 amrex::Print() << "WARNING: Adjusting RHS in gravity_sync solve by " << local_correction << '\n';
368
369 crse_rhs.plus(-local_correction,0,1,0);
370 }
371
372 // delta_phi needs a ghost cell for the solve
373 Vector<std::unique_ptr<MultiFab> > delta_phi(fine_level - crse_level + 1);
374 for (int lev = crse_level; lev <= fine_level; lev++)
375 {
376 delta_phi[lev-crse_level].reset(new MultiFab(grids[lev], dmap[lev], 1, 1));
377 delta_phi[lev-crse_level]->setVal(0);
378 }
379
380 Vector <Vector<std::unique_ptr<MultiFab> > > ec_gdPhi(fine_level - crse_level + 1);
381 for (int lev = crse_level; lev <= fine_level; lev++) {
382 Nyx* Nyx_lev = dynamic_cast<Nyx*>(&parent->getLevel(lev));
383 ec_gdPhi[lev-crse_level].resize(BL_SPACEDIM);
384 for (int n = 0; n < BL_SPACEDIM; ++n)
385 ec_gdPhi[lev-crse_level][n].reset(new MultiFab(Nyx_lev->getEdgeBoxArray(n),
386 Nyx_lev->DistributionMap(),
387 1,0));
388 }
389
390 // Do multi-level solve for delta_phi
391 solve_for_delta_phi(crse_level, fine_level, crse_rhs,
392 amrex::GetVecOfPtrs(delta_phi),
393 amrex::GetVecOfVecOfPtrs(ec_gdPhi));
394
395 crse_rhs.clear();
396
397 // In the all-periodic case we enforce that delta_phi averages to zero.
398 if (crse_geom.isAllPeriodic() && (grids[crse_level].numPts() == crse_domain.numPts()) ) {
399 Real local_correction = 0.0;
400#ifdef _OPENMP
401#pragma omp parallel if (!system::regtest_reduction) reduction(+:local_correction)
402#endif
403 for (MFIter mfi(*delta_phi[0],true); mfi.isValid(); ++mfi) {
404 local_correction += (*delta_phi[0])[mfi].sum<RunOn::Host>(mfi.tilebox(),0,1);
405 }
406 ParallelDescriptor::ReduceRealSum(local_correction);
407
408 local_correction = local_correction / grids[crse_level].numPts();
409
410 for (int lev = crse_level; lev <= fine_level; lev++) {
411 delta_phi[lev-crse_level]->plus(-local_correction,0,1,1);
412 }
413 }
414
415 // Add delta_phi to phi_new, and grad(delta_phi) to grad(delta_phi_curr) on each level
416 for (int lev = crse_level; lev <= fine_level; lev++)
417 {
418 LevelData[lev]->get_new_data(PhiGrav_Type).plus(*delta_phi[lev-crse_level], 0, 1, 0);
419 for (int n = 0; n < BL_SPACEDIM; n++)
420 grad_phi_curr[lev][n]->plus(*ec_gdPhi[lev-crse_level][n], 0, 1, 0);
421 }
422
423 int is_new = 1;
424
425 // Average phi_new from fine to coarse level
426 for (int lev = fine_level - 1; lev >= crse_level; lev--)
427 {
428 const IntVect& ratio = parent->refRatio(lev);
429 amrex::average_down(LevelData[lev+1]->get_new_data(PhiGrav_Type),
430 LevelData[lev ]->get_new_data(PhiGrav_Type),
431 0, 1, ratio);
432 }
433
434 // Average the edge-based grad_phi from finer to coarser level
435 for (int lev = fine_level-1; lev >= crse_level; lev--)
436 average_fine_ec_onto_crse_ec(lev, is_new);
437
438 // Add the contribution of grad(delta_phi) to the flux register below if necessary.
439 if (crse_level > 0 && iteration == ncycle)
440 {
441 for (int n = 0; n < BL_SPACEDIM; ++n) {
442 phi_flux_reg[crse_level]->FineAdd(*ec_gdPhi[0][n], n, 0, 0, 1, 1.0);
443 }
444 }
445
446 for (int lev = crse_level; lev <= fine_level; lev++) {
447 grad_delta_phi_cc[lev-crse_level]->setVal(0.0);
448 const Geometry& geom = parent->Geom(lev);
449 amrex::average_face_to_cellcenter(*grad_delta_phi_cc[lev-crse_level],
450 amrex::GetVecOfConstPtrs(ec_gdPhi[lev-crse_level]),
451 geom);

456Gravity::get_crse_phi (int level,
457 MultiFab& phi_crse,
458 Real time)
459{
460 BL_PROFILE("Gravity::get_crse_phi()");
461 BL_ASSERT(level != 0);
462
463 const Real t_old = LevelData[level-1]->get_state_data(PhiGrav_Type).prevTime();
464 const Real t_new = LevelData[level-1]->get_state_data(PhiGrav_Type).curTime();
465 const Real alpha = (time - t_old) / (t_new - t_old);
466
467 phi_crse.clear();
468 phi_crse.define(grids[level-1], dmap[level-1], 1, 1);
469
470 // BUT NOTE we don't trust phi's ghost cells.
471 FArrayBox phi_crse_temp;
472
473 // Note that we must do these cases separately because it's possible to do a
474 // new solve after a regrid when the old data on the coarse grid may not yet
475 // be defined.
476 for (MFIter mfi(phi_crse,true); mfi.isValid(); ++mfi)
477 {
478 const Box& gtbx = mfi.growntilebox();
479
480 phi_crse_temp.resize(gtbx,1);
481 Elixir phi_crse_tmp_eli = phi_crse_temp.elixir();
482
483 if (fabs(alpha-1.0) < 1.e-15)
484 {
485 phi_crse[mfi].copy<RunOn::Device>(LevelData[level-1]->get_new_data(PhiGrav_Type)[mfi]);
486 }
487 else if (fabs(alpha) < 1.e-15)
488 {
489 phi_crse[mfi].copy<RunOn::Device>(LevelData[level-1]->get_old_data(PhiGrav_Type)[mfi]);
490 }
491 else
492 {
493 phi_crse_temp.copy<RunOn::Device>(LevelData[level-1]->get_old_data(PhiGrav_Type)[mfi]);
494 Real omalpha = 1.0 - alpha;
495 phi_crse_temp.mult<RunOn::Device>(omalpha);
496
497 phi_crse[mfi].copy<RunOn::Device>(LevelData[level-1]->get_new_data(PhiGrav_Type)[mfi],gtbx);
498 phi_crse[mfi].mult<RunOn::Device>(alpha,gtbx);
499 phi_crse[mfi].plus<RunOn::Device>(phi_crse_temp);
500 }
501 }
502
503 const Geometry& geom = parent->Geom(level-1);
504 phi_crse.FillBoundary(geom.periodicity());

634 Rhs_particles[lev]->setVal(0.);
635 }
636
637 const auto& rpp = amrex::GetVecOfPtrs(Rhs_particles);
638 AddParticlesToRhs(level,finest_level,ngrow_for_solve,rpp);
639 AddGhostParticlesToRhs(level,rpp);
640 AddVirtualParticlesToRhs(finest_level,rpp);
641 amrex::Gpu::Device::streamSynchronize();
642
643 Nyx* cs = dynamic_cast<Nyx*>(&parent->getLevel(level));
644
645 BL_ASSERT(cs != 0);
646
647 Real time = 0;
648
649 if (is_new == 1)
650 {
651 time = LevelData[level]->get_state_data(PhiGrav_Type).curTime();
652 }
653 else if (is_new == 0)
654 {
655 time = LevelData[level]->get_state_data(PhiGrav_Type).prevTime();
656 }
657
658 // Here we get comoving_a b/c the RHS should be 4 * pi * G * density / a
659 const Real a_inverse = 1. / (cs->get_comoving_a(time));
660
661// *****************************************************************************
662
663 for (int lev = 0; lev < num_levels; lev++)
664 {
665 if (is_new == 0)
666 {
667 // Working in result data structure directly
668 phi_p[lev] = &LevelData[level+lev]->get_old_data(PhiGrav_Type);
669 }
670 else
671 {
672 // Working in result data structure directly
673 phi_p[lev] = &LevelData[level+lev]->get_new_data(PhiGrav_Type);
674 }
675
676 if (!use_previous_phi_as_guess)
677 phi_p[lev]->setVal(0);
678
679 // Need to set the boundary values before "bndry" is defined so they get copied in
680 if (dirichlet_bcs) set_dirichlet_bcs(level+lev,phi_p[lev]);
681
682 Rhs_p[lev].reset(new MultiFab(grids[level+lev], dmap[level+lev], 1, 0));
683 Rhs_p[lev]->setVal(0.0);
684
685#ifndef NO_HYDRO
686 if (Nyx::Do_Hydro() == 1)
687 {
688 if (is_new == 1)
689 {
690 MultiFab::Copy(*Rhs_p[lev], LevelData[level+lev]->get_new_data(State_Type), density, 0, 1, 0);
691 }
692 else if (is_new == 0)
693 {
694 MultiFab::Copy(*Rhs_p[lev], LevelData[level+lev]->get_old_data(State_Type), density, 0, 1, 0);
695 }
696 }
697#endif
698
699#ifdef FDM
700 if (is_new == 1)
701 {
702 MultiFab::Add(*(Rhs_p[lev]), LevelData[level+lev]->get_new_data(Axion_Type), Nyx::AxDens, 0, 1, 0);
703 }
704 else if (is_new == 0)
705 {
706 MultiFab::Add(*(Rhs_p[lev]), LevelData[level+lev]->get_old_data(Axion_Type), Nyx::AxDens, 0, 1, 0);
707 }
708#endif
709
710 MultiFab::Add(*Rhs_p[lev], *Rhs_particles[lev], 0, 0, 1, 0);
711 }
712
713 // Average phi from fine to coarse level before the solve.
714 for (int lev = num_levels-1; lev > 0; lev--)
715 {
716 amrex::average_down(*phi_p[lev], *phi_p[lev-1],
717 0, 1, parent->refRatio(level+lev-1));
718 }
719
720// *****************************************************************************
721
722 // This correction is for fully periodic domains only.
723 if (parent->Geom(level).isAllPeriodic())
724 {
725 if (verbose)
726 amrex::Print() << " ... subtracting average density " << mass_offset
727 << " from RHS at each level " << '\n';
728
729 for (int lev = 0; lev < num_levels; lev++)
730 (*Rhs_p[lev]).plus(-mass_offset,0,1,0);
731
732 if(verbose>1)
733 amrex::Print() << "After mass correction "<<(*Rhs_p[0]).norm2(0) << std::endl;
734
735 // This is used to enforce solvability if appropriate.
736 if ( parent->Geom(level).Domain().numPts() == grids[level].numPts() )
737 {
738
739 Real sum = 0;
740 for (int lev = 0; lev < num_levels; lev++)
741 {
742 Nyx* nyx_level = dynamic_cast<Nyx*>(&(parent->getLevel(level+lev)));
743 sum += nyx_level->vol_weight_sum(*Rhs_p[lev],true);
744 }
745
746 // ParallelDescriptor::ReduceRealSum(sum);
747 sum /= parent->Geom(0).ProbSize();
748
749 const Real eps = 1.e-10 * std::abs(mass_offset);
750 if (std::abs(sum) > eps)
751 {
752 amrex::Print() << " ... current avg differs from mass_offset by " << sum << " " << '\n';
753 amrex::Print() << " ... Gravity::actual_multilevel_solve -- total mass has changed!" << '\n';;
754 }
755
756 if (verbose)
757 amrex::Print() << " ... subtracting " << sum << " to ensure solvability " << '\n';
758
759 for (int lev = 0; lev < num_levels; lev++)
760 (*Rhs_p[lev]).plus(-sum, 0, 1, 0);
761 }
762 }
764
765// *****************************************************************************
766
767 for (int lev = 0; lev < num_levels; lev++)
768 {
769 Rhs_p[lev]->mult(Ggravity, 0, 1);
770 Rhs_p[lev]->mult(a_inverse);
771 }
772
773// *****************************************************************************
774
775 const MultiFab* crse_bcdata = nullptr;
776 MultiFab CPhi;
777 if (level > 0) {
778 get_crse_phi(level, CPhi, time);
779 crse_bcdata = &CPhi;
780 }
781 Real rel_eps = ml_tol;
782 Real abs_eps = 0.;
783 Vector<std::array<MultiFab*,AMREX_SPACEDIM> > grad_phi_aa;
784 for (int amrlev = level; amrlev <= finest_level; ++amrlev) {
785 grad_phi_aa.push_back({AMREX_D_DECL(grad_phi[amrlev][0],
786 grad_phi[amrlev][1],
787 grad_phi[amrlev][2])});
788 }
789
790 solve_with_MLMG(level, finest_level, phi_p, amrex::GetVecOfConstPtrs(Rhs_p),
791 grad_phi_aa, crse_bcdata, rel_eps, abs_eps);
792
793 // Average grad_phi from fine to coarse level
794 for (int lev = finest_level; lev > level; lev--)
795 average_fine_ec_onto_crse_ec(lev-1,is_new);
796
797}

1244Gravity::CorrectRhsUsingOffset(int level, MultiFab& Rhs)
1245{
1246 BL_PROFILE("Gravity::CorrectRhsUsingOffset()");
1247
1248 amrex::Gpu::LaunchSafeGuard lsg(true);
1249 if (verbose)
1250 amrex::Print() << " ... subtracting average density from RHS in solve ... "

1252
1253 // for (MFIter mfi(Rhs); mfi.isValid(); ++mfi)
1254 // Rhs[mfi].plus(-mass_offset);

1256

1259
1260 // This checks if mass has been conserved--in particular if
1261 // virtual particles are correctly representing finer particles.
1262 if (level == 0)

1264 Nyx* nyx_level = dynamic_cast<Nyx*>(&(parent->getLevel(0)));
1265 Real sum = nyx_level->vol_weight_sum(Rhs,false);
1266
1267 const Real eps = 1.e-10 * std::abs(mass_offset);
