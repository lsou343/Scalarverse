39Real
40BrentRootFinder (GpuArray<Real,AMREX_SPACEDIM> const& lo,
41 GpuArray<Real,AMREX_SPACEDIM> const& hi,
42 int rangedir, F const& f) noexcept
43{
44 const Real tol = 1.e-12;
45 const int MAXITER = 100;
46 const Real EPS = 3.0e-15;
47
48 Real p, q, r, s;
49
50 GpuArray<Real,AMREX_SPACEDIM> aPt = lo;
51 GpuArray<Real,AMREX_SPACEDIM> bPt = hi;
52
53 Real fa = IF_f(f, aPt);
54 Real fb = IF_f(f, bPt);
55 Real c = bPt[rangedir];
56 Real fc = fb;
57
58 if (fb*fa > 0.0) {
59// amrex::AllPrint() << "fa " << fa << " fb " << fb << "\n";
60 amrex::Error("BrentRootFinder. Root must be bracketed, but instead the supplied end points have the same sign.");
61 return 0.0;
62 } else if (fa == 0.0) {
63 return aPt[rangedir];
64 } else if (fb == 0.0) {
65 return bPt[rangedir];
66 }
67
68 Real d = 0.0, e = 0.0;
69 int i;
70 for (i = 0; i < MAXITER; ++i)
71 {
72 if (fb*fc > 0)
73 {
74 // Rename a, b, c and adjust bounding interval d
75 c = aPt[rangedir];
76 fc = fa;
77 d = bPt[rangedir] - aPt[rangedir];
78 e = d;
79 }
80
81 if (amrex::Math::abs(fc) < amrex::Math::abs(fb))
82 {
83 aPt[rangedir] = bPt[rangedir];
84 bPt[rangedir] = c;
85 c = aPt[rangedir];
86 fa = fb;
87 fb = fc;
88 fc = fa;
89 }
90
91 // Convergence check
92 Real tol1 = 2.0 * EPS * amrex::Math::abs(bPt[rangedir]) + 0.5 * tol;
93 Real xm = 0.5 * (c - bPt[rangedir]);
94
95 if (amrex::Math::abs(xm) <= tol1 || fb == 0.0)
96 {
97 break;
98 }
99
100 if (amrex::Math::abs(e) >= tol1 && amrex::Math::abs(fa) > amrex::Math::abs(fb))
101 {
102 // Attempt inverse quadratic interpolation
103 s = fb / fa;
104 if (aPt[rangedir] == c)
105 {
106 p = 2.0 * xm * s;
107 q = 1.0 - s;
108 }
109 else
110 {
111 q = fa / fc;
112 r = fb / fc;
113 p = s * (2.0 * xm * q * (q-r) - (bPt[rangedir]-aPt[rangedir]) * (r-1.0));
114 q = (q-1.0) * (r-1.0) * (s-1.0);
115 }
116
117 // Check whether in bounds
118 if (p > 0) q = -q;
119
120 p = amrex::Math::abs(p);
121
122 if (2.0 * p < amrex::min(3.0*xm*q-amrex::Math::abs(tol1*q), 1.0*amrex::Math::abs(e*q)))
123 {
124 // Accept interpolation
125 e = d;
126 d = p / q;
127 }
128 else
129 {
130 // Interpolation failed, use bisection
131 d = xm;
132 e = d;
133 }
134 }
135 else
136 {
137 // Bounds decreasing too slowly, use bisection
138 d = xm;
139 e = d;
140 }
141
142 // Move last best guess to a
143 aPt[rangedir] = bPt[rangedir];
144 fa = fb;
145
146 // Evaluate new trial root
147 if (amrex::Math::abs(d) > tol1)
148 {
149 bPt[rangedir] = bPt[rangedir] + d;
150 }
151 else
152 {
153 if (xm < 0) bPt[rangedir] = bPt[rangedir] - tol1;
154 else bPt[rangedir] = bPt[rangedir] + tol1;
155 }
156
157 fb = IF_f(f, bPt);
158 }
159
160 if (i >= MAXITER)
161 {
162 amrex::Error("BrentRootFinder: exceeding maximum iterations.");
163 }
164
165 return bPt[rangedir];
166}
