6template <typename Interp>
7bool ProperlyNested (const IntVect& ratio, const IntVect& blocking_factor, int ngrow,
8 const IndexType& boxType, Interp* mapper)
9{
10 int ratio_max = ratio[0];
11#if (AMREX_SPACEDIM > 1)
12 ratio_max = std::max(ratio_max, ratio[1]);
13#endif
14#if (AMREX_SPACEDIM == 3)
15 ratio_max = std::max(ratio_max, ratio[2]);
16#endif
17 // There are at least this many coarse cells outside fine grids
18 // (except at physical boundaries).
19 const IntVect& nbuf = blocking_factor / ratio_max;
20
21 Box crse_box(IntVect(AMREX_D_DECL(0 ,0 ,0 )), IntVect(AMREX_D_DECL(4*nbuf[0]-1,
22 4*nbuf[1]-1,
23 4*nbuf[2]-1)));
24 crse_box.convert(boxType);
25 Box fine_box(nbuf, IntVect(AMREX_D_DECL(3*nbuf[0]-1,3*nbuf[1]-1,3*nbuf[2]-1)));
26 fine_box.convert(boxType);
27 fine_box.refine(ratio_max);
28 fine_box.grow(ngrow);
29 const Box& fine_box_coarsened = mapper->CoarseBox(fine_box, ratio_max);
30 return crse_box.contains(fine_box_coarsened);
31}

45template <typename MF, typename BC>
46EnableIf_t<IsFabArray<MF>::value>
47FillPatchSingleLevel (MF& mf, IntVect const& nghost, Real time,
48 const Vector<MF*>& smf, const Vector<Real>& stime,
49 int scomp, int dcomp, int ncomp,
50 const Geometry& geom,
51 BC& physbcf, int bcfcomp)
52{
53 BL_PROFILE("FillPatchSingleLevel");
54
55 AMREX_ASSERT(scomp+ncomp <= smf[0]->nComp());
56 AMREX_ASSERT(dcomp+ncomp <= mf.nComp());
57 AMREX_ASSERT(smf.size() == stime.size());
58 AMREX_ASSERT(smf.size() != 0);
59 AMREX_ASSERT(nghost.allLE(mf.nGrowVect()));
60
61 if (smf.size() == 1)
62 {
63 if (&mf == smf[0] and scomp == dcomp) {
64 mf.FillBoundary(dcomp, ncomp, nghost, geom.periodicity());
65 } else {
66 mf.ParallelCopy(*smf[0], scomp, dcomp, ncomp, IntVect{0}, nghost, geom.periodicity());
67 }
68 }
69 else if (smf.size() == 2)
70 {
71 BL_ASSERT(smf[0]->boxArray() == smf[1]->boxArray());
72 MF raii;
73 MF * dmf;
74 int destcomp;
75 bool sameba;
76 if (mf.boxArray() == smf[0]->boxArray() and
77 mf.DistributionMap() == smf[0]->DistributionMap())
78 {
79 dmf = &mf;
80 destcomp = dcomp;
81 sameba = true;
82 } else {
83 raii.define(smf[0]->boxArray(), smf[0]->DistributionMap(), ncomp, 0,
84 MFInfo(), smf[0]->Factory());
85
86 dmf = &raii;
87 destcomp = 0;
88 sameba = false;
89 }
90
91 if ((dmf != smf[0] and dmf != smf[1]) or scomp != dcomp)
92 {
93#ifdef _OPENMP
94#pragma omp parallel if (Gpu::notInLaunchRegion())
95#endif
96 for (MFIter mfi(*dmf,TilingIfNotGPU()); mfi.isValid(); ++mfi)
97 {
98 const Box& bx = mfi.tilebox();
99 const Real t0 = stime[0];
100 const Real t1 = stime[1];
101 auto const sfab0 = smf[0]->array(mfi);
102 auto const sfab1 = smf[1]->array(mfi);
103 auto dfab = dmf->array(mfi);
104
105 if (time == t0)
106 {
107 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
108 {
109 dfab(i,j,k,n+destcomp) = sfab0(i,j,k,n+scomp);
110 });
111 }
112 else if (time == t1)
113 {
114 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
115 {
116 dfab(i,j,k,n+destcomp) = sfab1(i,j,k,n+scomp);
117 });
118 }
119 else if (std::abs(t1-t0) > 1.e-16)
120 {
121 Real alpha = (t1-time)/(t1-t0);
122 Real beta = (time-t0)/(t1-t0);
123 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
124 {
125 dfab(i,j,k,n+destcomp) = alpha*sfab0(i,j,k,n+scomp)
126 + beta*sfab1(i,j,k,n+scomp);
127 });
128 }
129 else
130 {
131 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
132 {
133 dfab(i,j,k,n+destcomp) = sfab0(i,j,k,n+scomp);
134 });
135 }
136 }
137 }
138
139 if (sameba)
140 {
141 // Note that when sameba is true mf's BoxArray is nonoverlapping.
142 // So FillBoundary is safe.
143 mf.FillBoundary(dcomp, ncomp, nghost, geom.periodicity());
144 }
145 else
146 {
147 IntVect src_ngrow = IntVect::TheZeroVector();
148 IntVect dst_ngrow = nghost;
149
150 mf.ParallelCopy(*dmf, 0, dcomp, ncomp, src_ngrow, dst_ngrow, geom.periodicity());
151 }
152 }
153 else {
154 amrex::Abort("FillPatchSingleLevel: high-order interpolation in time not implemented yet");
155 }
156
157 physbcf(mf, dcomp, ncomp, nghost, time, bcfcomp);
158}

412 const MF& cmf, int scomp, int dcomp, int ncomp,
413 const Geometry& cgeom, const Geometry& fgeom,
414 BC& cbc, int cbccomp,
415 BC& fbc, int fbccomp,
416 const IntVect& ratio,
