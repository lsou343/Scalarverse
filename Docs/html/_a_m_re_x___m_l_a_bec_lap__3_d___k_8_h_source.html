378AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
379void tridiagonal_solve (Array1D<Real,0,31>& a_ls, Array1D<Real,0,31>& b_ls, Array1D<Real,0,31>& c_ls,
380 Array1D<Real,0,31>& r_ls, Array1D<Real,0,31>& u_ls, Array1D<Real,0,31>& gam,
381 int ilen ) noexcept
382{
383 Real bet = b_ls(0);
384 u_ls(0) = r_ls(0) / bet;
385
386 for (int i = 1; i <= ilen - 1; i++) {
387 gam(i) = c_ls(i-1) / bet;
388 bet = b_ls(i) - a_ls(i)*gam(i);
389 if (bet == 0) amrex::Abort(">>>TRIDIAG FAILED");
390 u_ls(i) = (r_ls(i)-a_ls(i)*u_ls(i-1)) / bet;
391 }
392 for (int i = ilen-2; i >= 0; i--) {
393 u_ls(i) = u_ls(i) - gam(i+1)*u_ls(i+1);
394 }
395}
