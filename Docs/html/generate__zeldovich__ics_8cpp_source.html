81// Redshifts
82double initial_z = 100.0;
83double caustic_z = 10.0;
84
85// Required cosmology
86double h = 0.675; // the weird little h used in the def. of Hubble rate.
87double H_0 = 100.0 * h; // the Hubble rate in the standard km Mpc^-1 s^-1.
88
89// The size of the box. Note that all lengths are in Mpc, not Mpc h^-1.
90double box_length = 1.0/0.675;
91
92// How many sheets should fit in the box. This determines the perturbation
93// wavelength (k mode).
94double num_sheets = 1;
95double offset = 0.75*box_length;//PI/2; // wave offset
96
97// Particle number (mass determined by critical density)
98long num_particles_1d = 128;
99
100// Normal vector to the sheets
101Point sheet_normal = get_unit_vector(Point(1.0, 0.0, 0.0));
102
103// =========== //
104// End params. You shouldn't have to adjust anything below this. I hope.
105// =========== //

107int main(int argc, char *argv[]) {
108 // Particle things.
109 long num_particles = pow(num_particles_1d, 3);
110 double crit_density = (3.0 * pow(H_0, 2) / (8.0 * PI * grav_const)
111 * rho_c_units_factor); // in M_sun / Mpc^3
112
113 // Lengths in the problem.
114 double half_cell = (box_length / num_particles_1d) / 2.0;
115 double spacing = (float) box_length / num_particles_1d; // distance between particles in one dimension
116 double initial_a = get_scale_factor(initial_z); // the current scale factor
117
118 // Fit an integer number of sheets in box
119 // The dumb way to do min of >2 floats, but it works fine for 3...
120 double normal_length = min(box_length / sheet_normal.x,
121 min(box_length / sheet_normal.y,
122 box_length / sheet_normal.z));
123 double wavelength = normal_length / num_sheets;
124 double k = 2 * PI / wavelength;
125
126 // overfill control
127 int p_fill_1d = 1.2 * num_particles_1d;
128 int p_fill = pow(p_fill_1d, 3);
129 double overflow_edge = box_length / 10.0;
130
131 // dummy stuff
132 double q, perturbation, x, y, z;
133 Point p;
134 Point * pp;
135
136 // the big vector with all positions
137 vector<Particle> particles, final_particles;
138
139 // Give the user that warm, fuzzy feeling.
140 std::cout << std::endl;
141 std::cout << "Zeldovich pancake IC generator" << std::endl;
142 std::cout << "==============================" << std::endl;
143 std::cout << std::endl;
144 std::cout << "Initial redshift:\t" << initial_z << std::endl;
145 std::cout << "Caustic redshift:\t" << caustic_z << std::endl;
146 std::cout << "Box length:\t\t" << box_length << " Mpc" << std::endl;
147 std::cout << "Perturbation length:\t" << wavelength << " Mpc" << std::endl;
148 std::cout << "Critical density:\t" << crit_density << " M_sun Mpc^-3"
149 << std::endl;
150 std::cout << "Number of particles:\t" << num_particles_1d << "^3 = "
151 << num_particles << std::endl << std::endl;
152
153 std::cout << "Filling grid." << std::endl;
154
155 // Fill particle_pos with regular grid. Note we haven't added elements to
156 // the vector yet, so use push_back.
157 for (int i = 0; i != p_fill_1d; ++i) {
158 x = -overflow_edge + (i * spacing) + half_cell;
159 for (int j = 0; j != p_fill_1d; ++j) {
160 y = -overflow_edge + (j * spacing) + half_cell;
161 for (int k = 0; k != p_fill_1d; ++k) {
162 z = -overflow_edge + (k * spacing) + half_cell;
163 particles.push_back(Particle(Point(x, y, z),
164 Point(0.0, 0.0, 0.0)));
165 }
166 }
167 }
168
169 std::cout << "Perturbing grid." << std::endl;
170
171 // Now "perturb" the x positions and velocities
172 for (int i = 0; i != particles.size(); ++i) {
173 pp = &(particles[i].position);
174
175 // zeldovich grid position
176 q = (pp->x * sheet_normal.x + pp->y * sheet_normal.y
177 + pp->z * sheet_normal.z) + offset;
178
179 // zeldovich position perturbation
180 perturbation = perturbed_x(q, initial_z, caustic_z, k) - q;
181
182 pp->x += perturbation * sheet_normal.x;
183 pp->y += perturbation * sheet_normal.y;
184 pp->z += perturbation * sheet_normal.z;
185
186 pp = &(particles[i].velocity);
187
188 // note perturbed_v_x will be in comoving km s^-1 and we need it in
189 // proper km s^-1, so multiply by ``initial_a``.
190 perturbation = perturbed_v_x(q, initial_z, caustic_z, k, H_0) * initial_a;
191
192 // don't need ``+=`` because velocity components are already 0.0
193 pp->x = perturbation * sheet_normal.x;
194 pp->y = perturbation * sheet_normal.y;
195 pp->z = perturbation * sheet_normal.z;
196 }
197
198 std::cout << "Filtering particles outside of domain." << std::endl;
199
200 //double fudged_box_length = box_length + 0.001;
201
202 // Run through the particles and add the particles inside the box to
203 // ``final_positions``. This is cheaper than Vector::erase, O(N).
204 for (int i = 0; i != particles.size(); ++i) {
205 p = particles[i].position;
206
207 if (p.x > 0.0 && p.x < box_length && p.y > 0.0 && p.y < box_length
208 && p.z > 0.0 && p.z < box_length)
209 {
210 // this particle is in the box, so add it to ``final_positions``
211 final_particles.push_back(particles[i]);
212 }
213 }
214
215 unsigned int final_num_particles = final_particles.size();
216 double particle_mass = crit_density * pow(box_length, 3) / final_num_particles;
217
218 std::cout << std::endl << "Saving " << final_num_particles
219 << " particles to file." << std::endl;
220 std::cout << "Mass density:\t\t"
221 << final_num_particles * particle_mass / pow(box_length, 3)
222 << " M_sun Mpc^-3" << std::endl;
223 std::cout << "Particle mass:\t\t" << particle_mass << " M_sun" << std::endl;
224
225 // params output
226 string params_filename = "zeldovich_params.csv";
227 ofstream params_file;
228 params_file.open(params_filename.c_str()); // need a char* here
229
230 // write headers
231 params_file << "initial_z,caustic_z,box_length,k,offset,H_0,"
232 << "normal_x,normal_y,normal_z" << std::endl;
233 // write param values
234 params_file << std::scientific;
235 params_file.precision(15);
236 params_file << initial_z << "," << caustic_z << "," << box_length << ","
237 << k << "," << offset << "," << H_0 << "," << sheet_normal.x
238 << "," << sheet_normal.y << "," << sheet_normal.z << ","
239 << std::endl;
240
241 // particles output
242 string p_filename = "zeldovich_particles.ascii";
243 ofstream particle_file;
244 particle_file.open(p_filename.c_str()); // need a char* here
245
246 // first line has to be total number of particles
247 particle_file << final_num_particles << std::endl;
248
249 particle_file << std::scientific;
250 particle_file.precision(15); // just in case
251
252 for (int i = 0; i != final_num_particles; ++i) {
253 write_particle(particle_file, final_particles[i], particle_mass);
254 }
255
256 std::cout << std::endl << "Done!" << std::endl;
257 std::cout << "The particle data is in `" << p_filename << "`" << std::endl;
258 std::cout << "The params are in `" << params_filename << "`"
259 << std::endl << std::endl;
260
261 // close it and exit
262 particle_file.close();
263
264 return 0;
265}

272// Zeldovich problem setup functions
273double perturbed_x(double q, double current_z, double final_z, double k) {
274 // Takes the unperturbed position of the particle on a regular grid, and
275 // returns the "perturbed" position, in comoving units.
276
277 return (q - (1.0 + final_z) / (1.0 + current_z) * sin(k * q) / k);
278}
279
280double perturbed_v_x(double q, double current_z, double final_z, double k,
281 double H_0) {
282 // Returns the "perturbed" velocity, in comoving units.
283 return (-H_0 * (1.0 + final_z) * sqrt(1.0 + current_z)
284 * sin(k * q) / k);
285}
286
287double perturbed_rho(double q, double average_rho, double current_z,
288 double final_z, double k) {
289 // Returns the "perturbed" density, in proper units.
290 return (average_rho / (1.0 - (1.0 + final_z) / (1.0 + current_z)
291 * cos(k * q)));
292}
293// end Zeldovich problem functions

295Point get_unit_vector(Point vec) {
296 // take `vec` and divide it by its norm, making it a unit vecotr in the same
297 // direction.
298 double norm = sqrt(pow(vec.x, 2) + pow(vec.y, 2) + pow(vec.z, 2));
299 vec.x /= norm;
300 vec.y /= norm;
301 vec.z /= norm;
302
303 return vec;
304}
305
306double get_scale_factor(double current_z) {
307 // Take in the redshift, `z`, return the scale factor, `a`.
308 return 1.0 / (1.0 + current_z);
309}
