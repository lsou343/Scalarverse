77void
78InterpBndryData::setBndryValues (const MultiFab& mf,
79 int mf_start,
80 int bnd_start,
81 int num_comp,
82 const IntVect& ref_ratio,
83 const BCRec& bc)
84{
85 //
86 // Check that boxarrays are identical.
87 //
88 BL_ASSERT(grids.size());
89 BL_ASSERT(grids == mf.boxArray());
90
91 for (int n = bnd_start; n < bnd_start+num_comp; ++n) {
92 setBndryConds(bc, ref_ratio, n);
93 }
94
95#ifdef _OPENMP
96#pragma omp parallel if (Gpu::notInLaunchRegion())
97#endif
98 for (MFIter mfi(mf,MFItInfo().SetDynamic(true)); mfi.isValid(); ++mfi)
99 {
100 const Box& bx = mfi.validbox();
101
102 for (OrientationIter fi; fi; ++fi)
103 {
104 const Orientation face = fi();
105
106 if (bx[face]==geom.Domain()[face] && !geom.isPeriodic(face.coordDir()))
107 {
108 //
109 // Physical bndry, copy from grid.
110 //
111 auto& bnd_fab = bndry[face][mfi];
112 auto const& src_fab = mf[mfi];
113 auto bnd_array = bnd_fab.array();
114 auto const src_array = src_fab.array();
115 const Box& b = src_fab.box() & bnd_fab.box();
116 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( b, num_comp, i, j, k, n,
117 {
118 bnd_array(i,j,k,n+bnd_start) = src_array(i,j,k,n+mf_start);
119 });
120 }
121 }
122 }
123}
124
125//
126// (1) set bndry type and location of bndry value on each face of
127// each grid
128// (2) set actual bndry value by:
129// (A) Interpolate from crse bndryRegister at crse/fine interface
130// (B) Copy from ghost region of MultiFab at physical bndry
131//

147void
148InterpBndryData::BndryValuesDoIt (BndryRegister& crse,
149 int c_start,
150 const MultiFab* fine,
151 int f_start,
152 int bnd_start,
153 int num_comp,
154 const IntVect& ratio,
155 const BCRec* bc,
156 int max_order)
157{
158 //
159 // Check that boxarrays are identical.
160 //
161 BL_ASSERT(grids.size());
162 BL_ASSERT(fine == nullptr || grids == fine->boxArray());
163 //
164 // Set bndry types and bclocs.
165 //
166 if (bc != nullptr) {
167 for (int n = bnd_start; n < bnd_start+num_comp; ++n) {
168 setBndryConds(*bc, ratio, n);
169 }
170 }
171 //
172 // First interpolate from coarse to fine on bndry.
173 //
174 const Box& fine_domain = geom.Domain();
175
176 //
177 // Mask turned off if covered by fine grid.
178 //
179 if (max_order==3 || max_order==1)
180 {
181 MultiFab foo(grids,bndry[0].DistributionMap(), 1, 0, MFInfo().SetAlloc(false), FArrayBoxFactory());
182
183 MFItInfo info;
184 if (Gpu::notInLaunchRegion()) info.SetDynamic(true);
185#ifdef _OPENMP
186#pragma omp parallel if (Gpu::notInLaunchRegion())
187#endif
188 for (MFIter mfi(foo,info); mfi.isValid(); ++mfi)
189 {
190 BL_ASSERT(grids[mfi.index()] == mfi.validbox());
191
192 const Box& fine_bx = mfi.validbox();
193
194 for (int i = 0; i < 2*AMREX_SPACEDIM; i++)
195 {
196 const int dir = ((i<AMREX_SPACEDIM) ? i : (i-AMREX_SPACEDIM));
197 const Orientation::Side side = ((i<AMREX_SPACEDIM) ? Orientation::low : Orientation::high);
198
199 const Orientation face(dir,side);
200
201 if (fine_bx[face] != fine_domain[face] || geom.isPeriodic(dir))
202 {
203 auto const crse_array = crse[face].array(mfi);
204 auto bdry_array = bndry[face].array(mfi);
205 Box const& b = bndry[face][mfi].box();
206 const auto rr = ratio.dim3();
207 //
208 // Internal or periodic edge, interpolate from crse data.
209 //
210 if (max_order == 1)
211 {
212 AMREX_HOST_DEVICE_FOR_4D ( b, num_comp, it, jt, kt, n,
213 {
214 interpbndrydata_o1(it,jt,kt,n,bdry_array,bnd_start,
215 crse_array,c_start,rr);
216 });
217 }
218 else
219 {
220 auto const mask_array = masks[face].array(mfi);
221 int is_not_covered = BndryData::not_covered;
222 switch (dir) {
223 case 0:
224 {
225 AMREX_HOST_DEVICE_FOR_4D ( b, num_comp, it, jt, kt, n,
226 {
227 interpbndrydata_x_o3(it,jt,kt,n,bdry_array,bnd_start,
228 crse_array,c_start,rr,
229 mask_array, is_not_covered);
230 });
231 break;
232 }
233#if (AMREX_SPACEDIM >= 2)
234 case 1:
235 {
236 AMREX_HOST_DEVICE_FOR_4D ( b, num_comp, it, jt, kt, n,
237 {
238 interpbndrydata_y_o3(it,jt,kt,n,bdry_array,bnd_start,
239 crse_array,c_start,rr,
240 mask_array, is_not_covered);
241 });
242 break;
243 }
244#if (AMREX_SPACEDIM == 3)
245 case 2:
246 {
247 AMREX_HOST_DEVICE_FOR_4D ( b, num_comp, it, jt, kt, n,
248 {
249 interpbndrydata_z_o3(it,jt,kt,n,bdry_array,bnd_start,
250 crse_array,c_start,rr,
251 mask_array, is_not_covered);
252 });
253 break;
254 }
255#endif
256#endif
257 default: {}
258 }
259 }
260 }
261 else if (fine != nullptr)
262 {
263 //
264 // Physical bndry, copy from ghost region of corresponding grid
265 //
266 auto& bnd_fab = bndry[face][mfi];
267 auto const& src_fab = (*fine)[mfi];
268 auto bnd_array = bnd_fab.array();
269 auto const src_array = src_fab.array();
270 const Box& b = bnd_fab.box() & src_fab.box();
271 AMREX_HOST_DEVICE_FOR_4D ( b, num_comp, ii, jj, kk, nn,
272 {
273 bnd_array(ii,jj,kk,nn+bnd_start) = src_array(ii,jj,kk,nn+f_start);
274 });
275 }
276 }
277 }
278 }
279 else
280 {
281 amrex::Abort("InterpBndryData::setBndryValues supports only max_order=1 or 3");
282 }
283}
