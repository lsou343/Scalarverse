1/*
2 Read a plotfile containing velocity fields and generate a file suitable for time-dependent input into
3 IAMR or PeleLM through the boundary. The format of such files is a folder containing an ascii file, HDR,
4 and a binary file, DAT. The data is uniformly spaced, grow-extended and suitable for componentwise parabolic
5 interpolation. In 3D, each ij plane will be parallel to the inflow face of the intended application,
6 and k will serve as a time coordinate. The HDR file includes the size and grid-spacing of the data,
7 followed by a list of offsets into the DATA file for each of the catenated planar fabs of velocity data.
8 In 2D the j axis represents time, and the data is written as a single plane.
9 */

38Extend (FArrayBox& xfab,
39 FArrayBox& vfab,
40 const Box& domain)
41{
42 Box tbx = vfab.box();
43
44 tbx.setBig(0, domain.bigEnd(0) + 3);
45
46 const int ygrow = BL_SPACEDIM==3 ? 3 : 1;
47
48 tbx.setBig(1, domain.bigEnd(1) + ygrow);
49
50 xfab.resize(tbx,1);
51
52 xfab.copy(vfab);
53 vfab.shift(0, domain.length(0));
54 xfab.copy(vfab);
55 vfab.shift(1, domain.length(1));
56 xfab.copy(vfab);
57 vfab.shift(0, -domain.length(0));
58 xfab.copy(vfab);
