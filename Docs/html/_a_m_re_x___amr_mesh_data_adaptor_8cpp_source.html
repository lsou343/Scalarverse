44// --------------------------------------------------------------------------
45int MeshStateMap::Initialize(
46 const std::vector<amrex::Vector<amrex::MultiFab> *> &states,
47 const std::vector<std::vector<std::string>> &names)
48{
49 int nStates = states.size();
50 for (int i = 0; i < nStates; ++i)
51 {
52 amrex::MultiFab& state = states[i]->at(0);
53 int nComp = state.nComp();
54
55 for (int j = 0; j < nComp; ++j)
56 {
57 const std::string &arrayName = names[i][j];
58
59 if (state.is_cell_centered())
60 {
61 this->Map[vtkDataObject::CELL][arrayName] = std::make_pair(i,j);
62 }
63 else if (state.is_nodal())
64 {
65 this->Map[vtkDataObject::POINT][arrayName] = std::make_pair(i,j);
66 }
67 }
68 }
69
70 return 0;
71}

476//-----------------------------------------------------------------------------
477int AmrMeshDataAdaptor::GetMesh(const std::string &meshName,
478 bool structureOnly, vtkDataObject *&mesh)
479{
480 mesh = nullptr;
481
482 if (meshName != "mesh")
483 {
484 SENSEI_ERROR("No mesh named \"" << meshName << "\"")
485 return -1;
486 }
487
488 if (this->Internals->States.size() < 1)
489 {
490 SENSEI_ERROR("No simualtion data, missing call to SetDataSource?")
491 return -1;
492 }
493
494 int rank = 0;
495 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
496
497 unsigned int nLevels = this->Internals->Mesh->finestLevel() + 1;
498
499 // initialize new vtk datasets
500 vtkOverlappingAMR *amrMesh = vtkOverlappingAMR::New();
501#if SENSEI_VERSION_MAJOR < 3
502 Internals->ManagedObjects.push_back(amrMesh);
503#endif
504 mesh = amrMesh;
505
506 // num levels and blocks per level
507 std::vector<int> nBlocks(nLevels);
508 for (unsigned int i = 0; i < nLevels; ++i)
509 nBlocks[i] = this->Internals->Mesh->boxArray(i).size();
510
511 amrMesh->Initialize(nLevels, nBlocks.data());
512
513 // origin
514 const amrex::RealBox& pd = this->Internals->Mesh->Geom(0).ProbDomain();
515 double origin[3] = {AMREX_ARLIM(pd.lo())};
516
517 // PinMesh works around a bug in VisIt 2.13.2.
518 // force the origin to 0,0,0
519 if (this->Internals->PinMesh)
520 {
521 for (int i = 0; i < 3; ++i)
522 origin[i] = 0.0;
523 }
524
525 amrMesh->SetOrigin(origin);
526
527 long gid = 0;
528 for (unsigned int i = 0; i < nLevels; ++i)
529 {
530 // domain decomp
531 const amrex::DistributionMapping &dmap = this->Internals->Mesh->DistributionMap(i);
532
533 // ghost zones
534 amrex::MultiFab& state = this->Internals->States[0]->at(i);
535 unsigned int ng = state.nGrow();
536
537 // spacing
538 const amrex::Geometry &geom = this->Internals->Mesh->Geom(i);
539 double spacing [3] = {AMREX_ARLIM(geom.CellSize())};
540 amrMesh->SetSpacing(i, spacing);
541
542 // refinement ratio
543 int rr = 1;
544 if (i != this->Internals->Mesh->maxLevel())
545 rr = this->Internals->Mesh->refRatio(i)[0];
546 amrMesh->SetRefinementRatio(i, rr);
547
548 // loop over boxes
549 const amrex::BoxArray& ba = this->Internals->Mesh->boxArray(i);
550 unsigned int nBoxes = ba.size();
551
552 for (unsigned int j = 0; j < nBoxes; ++j)
553 {
554 // cell centered box
555 amrex::Box cbox = ba[j];
556
557 // cell centered dimensions
558 int cboxLo[3] = {AMREX_ARLIM(cbox.loVect())};
559 int cboxHi[3] = {AMREX_ARLIM(cbox.hiVect())};
560
561 // vtk's representation of box metadata
562 vtkAMRBox block(cboxLo, cboxHi);
563 amrMesh->SetAMRBox(i, j, block);
564 amrMesh->SetAMRBlockSourceIndex(i, j, gid++);
565
566 // skip building a vtk amrMesh for the non local boxes
567 if (dmap[j] != rank)
568 continue;
569
570 // add ghost zones
571 for (int q = 0; q < AMREX_SPACEDIM; ++q)
572 cbox.grow(q, ng);
573
574 // node centered box
575 amrex::Box nbox = surroundingNodes(cbox);
576
577 // node centered dimensions
578 int nboxLo[3] = {AMREX_ARLIM(nbox.loVect())};
579 int nboxHi[3] = {AMREX_ARLIM(nbox.hiVect())};
580
581 // new vtk uniform amrMesh, node centered
582 vtkUniformGrid *ug = vtkUniformGrid::New();
583 ug->SetOrigin(origin);
584 ug->SetSpacing(spacing);
585 ug->SetExtent(nboxLo[0], nboxHi[0],
586 nboxLo[1], nboxHi[1], nboxLo[2], nboxHi[2]);
587
588 // pass the block into vtk
589 amrMesh->SetDataSet(i, j, ug);
590 ug->Delete();
591 }
592 }
593
594 return 0;
595}

610//-----------------------------------------------------------------------------
611int AmrMeshDataAdaptor::AddGhostCellsArray(vtkDataObject* mesh,
612 const std::string &meshName)
613{
614 if (meshName != "mesh")
615 {
616 SENSEI_ERROR("no mesh named \"" << meshName << "\"")
617 return -1;
618 }
619
620 vtkOverlappingAMR *amrMesh = dynamic_cast<vtkOverlappingAMR*>(mesh);
621 if (!amrMesh)
622 {
623 SENSEI_ERROR("Invalid mesh type "
624 << (mesh ? mesh->GetClassName() : "nullptr"))
625 }
626
627 if (this->Internals->States.size() < 1)
628 {
629 SENSEI_ERROR("No simualtion data, missing call to SetDataSource?")
630 return -1;
631 }
632
633 int rank = 0;
634 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
635
636 // loop over levels
637 unsigned int nLevels = this->Internals->Mesh->finestLevel() + 1;
638
639 std::vector<std::vector<unsigned char*>> masks(nLevels);
640 for (unsigned int i = 0; i < nLevels; ++i)
641 {
642 // allocate mask arrays
643 const amrex::BoxArray &boxes = this->Internals->Mesh->boxArray(i);
644 const amrex::DistributionMapping &dmap = this->Internals->Mesh->DistributionMap(i);
645 const amrex::Box &pdom = this->Internals->Mesh->Geom(i).Domain();
646
647 amrex::MultiFab& state = this->Internals->States[0]->at(i);
648 unsigned int ng = state.nGrow();
649
650 std::vector<unsigned char*> mask;
651 InSituUtils::AllocateBoxArray<unsigned char>(
652 pdom, boxes, dmap, ng, mask);
653
654 // mask ghost cells
655 InSituUtils::MaskGhostCells<unsigned char>(
656 pdom, boxes, dmap, ng, mask);
657
658 // store mask array
659 masks[i] = mask;
660 }
661
662 // loop over coarse levels
663 unsigned int nCoarseLevels = nLevels - 1;
664 for (unsigned int i = 0; i < nCoarseLevels; ++i)
665 {
666 int ii = i + 1;
667
668 // mask regions covered by refinement
669 amrex::MultiFab& state = this->Internals->States[0]->at(i);
670 unsigned int ng = state.nGrow();
671
672 const amrex::Box &pdom = this->Internals->Mesh->Geom(i).Domain();
673 const amrex::BoxArray &cBoxes = this->Internals->Mesh->boxArray(i);
674 const amrex::DistributionMapping &cMap = this->Internals->Mesh->DistributionMap(i);
675 const amrex::BoxArray &fBoxes = this->Internals->Mesh->boxArray(ii);
676 amrex::IntVect cRefRatio = this->Internals->Mesh->refRatio(i);
677
678 InSituUtils::MaskCoveredCells<unsigned char>(
679 pdom, cBoxes, cMap, fBoxes, cRefRatio, ng, masks[i]);
680 }
681
682 // loop over levels
683 for (unsigned int i = 0; i < nLevels; ++i)
684 {
685 const amrex::DistributionMapping &dmap = this->Internals->Mesh->DistributionMap(i);
686
687 // mask arrays for this level
688 std::vector<unsigned char*> &mask = masks[i];
689
690 // loop over boxes
691 const amrex::BoxArray& ba = this->Internals->Mesh->boxArray(i);
692 unsigned int nBoxes = ba.size();
693
694 for (unsigned int j = 0; j < nBoxes; ++j)
695 {
696 if (dmap[j] != rank)
697 continue;
698
699 vtkUniformGrid *blockMesh = amrMesh->GetDataSet(i, j);
700
701 if (!blockMesh)
702 {
703 SENSEI_ERROR("Empty block " << i << ", " << j)
704 return -1;
705 }
706
707 long nCells = blockMesh->GetNumberOfCells();
708
709 // transfer mask array into vtk
710 vtkUnsignedCharArray *ga = vtkUnsignedCharArray::New();
711 ga->SetName("vtkGhostType");
712 ga->SetArray(mask[j], nCells, 0);
713 blockMesh->GetCellData()->AddArray(ga);
714 ga->Delete();
715
716 // for debug can visualize the ghost cells
717 // FIXME -- a bug in Catalyst ignores internal ghost zones
718 // when using the VTK writrer. Until that bug gets fixed, one
719 // can manually inject this copy using a PV Python filter
720 ga = vtkUnsignedCharArray::New();
721 ga->SetName("GhostType");
722 ga->SetArray(mask[j], nCells, 1);
723 blockMesh->GetCellData()->AddArray(ga);
724 ga->Delete();
725 }
726 }
727
728 return 0;
729}

760 {
761 SENSEI_ERROR("Invalid association " << association)
762 return -1;
763 }
764
765 // find the indices of the multifab and component within for
766 // the named array
767 int fab = 0;
768 int comp = 0;
769 if (this->Internals->StateMetadata.GetIndex(arrayName, association, fab, comp))
770 {
771 SENSEI_ERROR("Failed to locate descriptor for "
772 << sensei::VTKUtils::GetAttributesName(association)
773 << " data array \"" << arrayName << "\"")
774 return -1;
775 }
776
777 // loop over levels
778 unsigned int nLevels = this->Internals->Mesh->finestLevel() + 1;
779
780 for (unsigned int i = 0; i < nLevels; ++i)
781 {
782 // domain decomp
783 const amrex::DistributionMapping &dmap = this->Internals->Mesh->DistributionMap(i);
784
785 // ghost zones
786 amrex::MultiFab& state = this->Internals->States[fab]->at(i);
787 unsigned int ng = state.nGrow();
788
789 // check centering
790 if (!((association == vtkDataObject::CELL) && state.is_cell_centered()) &&
791 !((association == vtkDataObject::POINT) && state.is_nodal()))
792 {
793 SENSEI_ERROR("association does not match MultiFab centering")
794 return -1;
795 }
796
797 // check component id
798 int nComp = state.nComp();
799 if (comp >= nComp)
800 {
801 SENSEI_ERROR("Component " << comp << " out of bounds")
802 return -1;
803 }
804
805 // loop over boxes
806 const amrex::BoxArray& ba = this->Internals->Mesh->boxArray(i);
807 unsigned int nBoxes = ba.size();
808
809 for (unsigned int j = 0; j < nBoxes; ++j)
810 {
811 // cell centered box
812 amrex::Box cbox = ba[j];
813
814 // add ghost zones
815 for (int q = 0; q < AMREX_SPACEDIM; ++q)
816 cbox.grow(q, ng);
817
818 // cell centered dimensions
819 int cboxLo[3] = {AMREX_ARLIM(cbox.loVect())};
820 int cboxHi[3] = {AMREX_ARLIM(cbox.hiVect())};
821
822 // skip building a vtk mesh for the non local boxes
823 if (dmap[j] != rank)
824 continue;
825
826 // node centered box
827 amrex::Box nbox = surroundingNodes(cbox);
828
829 // node centered dimensions
830 int nboxLo[3] = {AMREX_ARLIM(nbox.loVect())};
831 int nboxHi[3] = {AMREX_ARLIM(nbox.hiVect())};
832
833 // get the block mesh
834 vtkUniformGrid *ug = amrMesh->GetDataSet(i, j);
835
836 // node centered size
837 long nlen = 1;
838 for (int p = 0; p < 3; ++p)
839 nlen *= nboxHi[p] - nboxLo[p] + 1;
840
841 // cell centered size
842 long clen = 1;
843 for (int p = 0; p < 3; ++p)
844 clen *= cboxHi[p] - cboxLo[p] + 1;
845
846 // pointer to the data
847 amrex_real *pcd = state[j].dataPtr(comp);
848
849 // allocate vtk array
850 InSituUtils::amrex_tt<amrex_real>::vtk_type *da =
851 InSituUtils::amrex_tt<amrex_real>::vtk_type::New();
852
853 // set component name
854 da->SetName(arrayName.c_str());
855
856 if (state[j].box().ixType() == amrex::IndexType::TheCellType())
857 {
858 // zero copy cell centered
859 da->SetArray(pcd, clen, 1);
860 ug->GetCellData()->AddArray(da);
861 }
862 else if (state[j].box().ixType() == amrex::IndexType::TheNodeType())
863 {
864 // zero copy point centered
865 da->SetArray(pcd, nlen, 1);
866 ug->GetPointData()->AddArray(da);
867 }
868 else
869 {
870 SENSEI_WARNING("Face or edge centered component " << comp << " skipped")
871 }
872
873 da->Delete();
874
875#if defined(SENSEI_DEBUG)
876 // mark level id
877 vtkFloatArray *la = vtkFloatArray::New();
878 la->SetName("amrex_level_id");
879 la->SetNumberOfTuples(clen);
880 la->Fill(i);
881 ug->GetCellData()->AddArray(la);
882 la->Delete();
883
884 // mark mpi rank
885 vtkFloatArray *ra = vtkFloatArray::New();
886 ra->SetName("amrex_mpi_rank");
887 ra->SetNumberOfTuples(clen);
888 ra->Fill(rank);
889 ug->GetCellData()->AddArray(ra);
890 ra->Delete();
891#endif
892 }
893 }
894
895 return 0;
896}

898//-----------------------------------------------------------------------------
899int AmrMeshDataAdaptor::ReleaseData()
900{
901 this->Internals->Mesh = nullptr;
902 this->Internals->States.clear();
903 this->Internals->Names.clear();
904 this->Internals->StateMetadata.Clear();
905
906#if SENSEI_VERSION_MAJOR < 3
907 // free up mesh objects we allocated
908 size_t n = this->Internals->ManagedObjects.size();
909 for (size_t i = 0; i < n; ++i)
910 this->Internals->ManagedObjects[i]->Delete();
911 this->Internals->ManagedObjects.clear();
912#endif
913
914 return 0;
915}
