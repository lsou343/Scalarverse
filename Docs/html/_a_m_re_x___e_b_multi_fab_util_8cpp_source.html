681void EB_computeDivergence (MultiFab& divu, const Array<MultiFab const*,AMREX_SPACEDIM>& umac,
682 const Geometry& geom, bool already_on_centroids)
683{
684 AMREX_ASSERT(divu.nComp()==umac[0]->nComp());
685 AMREX_ASSERT(divu.nComp()==umac[1]->nComp());
686#if (AMREX_SPACEDIM == 3)
687 AMREX_ASSERT(divu.nComp()==umac[2]->nComp());
688#endif
689
690 if (!divu.hasEBFabFactory())
691 {
692 amrex::computeDivergence(divu, umac, geom);
693 }
694 else
695 {
696 const auto& factory = dynamic_cast<EBFArrayBoxFactory const&>(divu.Factory());
697 const auto& flags = factory.getMultiEBCellFlagFab();
698 const auto& vfrac = factory.getVolFrac();
699 const auto& area = factory.getAreaFrac();
700 const auto& fcent = factory.getFaceCent();
701
702 iMultiFab cc_mask;
703 if (!already_on_centroids) {
704 cc_mask.define(divu.boxArray(), divu.DistributionMap(), 1, 1);
705 cc_mask.BuildMask(geom.Domain(), geom.periodicity(), 1, 0, 0, 1);
706 }
707
708 const GpuArray<Real,AMREX_SPACEDIM> dxinv = geom.InvCellSizeArray();
709 MFItInfo info;
710 if (Gpu::notInLaunchRegion()) info.EnableTiling().SetDynamic(true);
711#ifdef _OPENMP
712#pragma omp parallel if (Gpu::notInLaunchRegion())
713#endif
714 for (MFIter mfi(divu,info); mfi.isValid(); ++mfi)
715 {
716 const Box& bx = mfi.tilebox();
717 const auto& flagfab = flags[mfi];
718 Array4<Real> const& divuarr = divu.array(mfi);
719 AMREX_D_TERM(Array4<Real const> const& uarr = umac[0]->const_array(mfi);,
720 Array4<Real const> const& varr = umac[1]->const_array(mfi);,
721 Array4<Real const> const& warr = umac[2]->const_array(mfi));
722
723 const auto fabtyp = flagfab.getType(bx);
724 if (fabtyp == FabType::covered) {
725 AMREX_HOST_DEVICE_FOR_4D(bx,divu.nComp(),i,j,k,n,
726 {
727 divuarr(i,j,k,n) = 0.0;
728 });
729 } else if (fabtyp == FabType::regular) {
730 AMREX_LAUNCH_HOST_DEVICE_LAMBDA(bx, b,
731 {
732 amrex_compute_divergence(b,divuarr,AMREX_D_DECL(uarr,varr,warr),dxinv);
733 });
734 } else {
735 Array4<int const> const& ccm = (already_on_centroids) ?
736 Array4<int const>{} : cc_mask.const_array(mfi);
737 Array4<Real const> const& vol = vfrac.const_array(mfi);
738 AMREX_D_TERM(Array4<Real const> const& apx = area[0]->const_array(mfi);,
739 Array4<Real const> const& apy = area[1]->const_array(mfi);,
740 Array4<Real const> const& apz = area[2]->const_array(mfi));
741 AMREX_D_TERM(Array4<Real const> const& fcx = fcent[0]->const_array(mfi);,
742 Array4<Real const> const& fcy = fcent[1]->const_array(mfi);,
743 Array4<Real const> const& fcz = fcent[2]->const_array(mfi));
744 Array4<EBCellFlag const> const& flagarr = flagfab.const_array();
745 AMREX_HOST_DEVICE_FOR_4D(bx,divu.nComp(),i,j,k,n,
746 {
747 eb_compute_divergence(i,j,k,n,divuarr,AMREX_D_DECL(uarr,varr,warr),
748 ccm, flagarr, vol, AMREX_D_DECL(apx,apy,apz),
749 AMREX_D_DECL(fcx,fcy,fcz), dxinv, already_on_centroids);
750 });
751 }
752 }
