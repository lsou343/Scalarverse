303
304 Vector<int> rr(num_levels-1);

308 RealBox real_box;

311 real_box.setHi(n, 20.0e-6);
312 }
313
314 // This sets the boundary conditions to be doubly or triply periodic
315 for (int i = 0; i < BL_SPACEDIM; i++) {
316 is_periodic[i] = 0;
317 }
318
319 IntVect dom_lo(IntVect(D_DECL(0,0,0)));
320 IntVect dom_hi(IntVect(D_DECL(n_cell-1, n_cell-1, n_cell-1)));
321 Box domain(dom_lo, dom_hi);
322
323 // make Geometry for each level
324 Vector<Geometry> geom(num_levels);
325 geom[0].define(domain,&real_box,CoordSys::cartesian,is_periodic);
326 for (int lev = 1; lev < num_levels; ++lev) {
327 geom[lev].define(amrex::refine(geom[lev-1].Domain(), rr[lev-1]),
328 &real_box, CoordSys::cartesian, is_periodic);
329 }
330
331 // make grids for each level
332 Vector<BoxArray> grids(num_levels);
333 grids[0].define(domain);
334 if (num_levels > 1) {
335 int n_fine = n_cell*rr[0];
336 IntVect refined_lo(D_DECL(3*n_fine/8,3*n_fine/8,3*n_fine/8));
337 IntVect refined_hi(D_DECL(5*n_fine/8-1,5*n_fine/8-1,5*n_fine/8-1));
338
339 // Build a box for the level 1 domain
340 Box refined_patch(refined_lo, refined_hi);
341 grids[1].define(refined_patch);
342 }
343
344 for (int lev = 0; lev < num_levels; lev++) {
345 grids[lev].maxSize(max_grid_size);
346 }
347
348 int Ncomp = 1;
349
350 Vector<DistributionMapping> dm(num_levels);
351 Vector<std::unique_ptr<MultiFab> > phi(num_levels);
352 Vector<std::unique_ptr<MultiFab> > rhs(num_levels);
353 Vector<std::array<std::unique_ptr<MultiFab>, BL_SPACEDIM> > eField(num_levels);
354
355 for (int lev = 0; lev < num_levels; ++lev) {
356 BoxArray nba = grids[lev];
357 nba.surroundingNodes();
358 dm[lev].define(grids[lev]);
359
360 rhs[lev].reset(new MultiFab(nba, dm[lev], Ncomp, 1));
361 phi[lev].reset(new MultiFab(nba, dm[lev], Ncomp, 2));
362
363 for (int idim = 0; idim < BL_SPACEDIM; ++idim) {
364 eField[lev][idim].reset(new MultiFab(nba, dm[lev], Ncomp, 1));
365 eField[lev][idim]->setVal(0.0);
366 }
367
368 rhs[lev]->setVal(0.0);
369 phi[lev]->setVal(0.0);
370 }
371
372 Vector<std::unique_ptr<FabArray<BaseFab<int> > > > masks(num_levels);
373 getLevelMasks(masks, grids, dm, geom);
374
375 Vector<std::unique_ptr<FabArray<BaseFab<int> > > > gather_masks(num_levels);
376 getLevelMasks(gather_masks, grids, dm, geom, n_buffer + 1); // convert from num nodes to num cells
377
378 ElectrostaticParticleContainer myPC(geom, dm, grids, rr);
379
380 myPC.InitParticles();
381
382 for (int step = 0; step <= max_step; ++step) {
383
384 myPC.DepositCharge(rhs);
385
386 computePhi(rhs, phi, grids, dm, geom, masks);
387
388 computeE(eField, phi, geom);
389
390 myPC.FieldGather(eField, gather_masks);
391
392 if (step % particle_output_int == 0) myPC.writeParticles(step);
393
394 // WritePlotFile(rhs, phi, eField, myPC, geom, step);
395
396 myPC.Evolve(eField, rhs, dt);
397
398 myPC.Redistribute();
399 }
