76void set_wrap(const Box& domain, const Box& valid_box, const std::array<bool, AMREX_SPACEDIM>& is_periodic, diy::AMRLink* link)
77{
78 for(int dir_x : {-1, 0, 1}) {
79 if (!is_periodic[0] && dir_x) continue;
80 if (dir_x < 0 && valid_box.loVect()[0] != domain.loVect()[0]) continue;
81 if (dir_x > 0 && valid_box.hiVect()[0] != domain.hiVect()[0]) continue;
82
83 for(int dir_y : {-1, 0, 1}) {
84 if (!is_periodic[1] && dir_y) continue;
85 if (dir_y < 0 && valid_box.loVect()[1] != domain.loVect()[1]) continue;
86 if (dir_y > 0 && valid_box.hiVect()[1] != domain.hiVect()[1]) continue;
87 for(int dir_z : {-1, 0, 1}) {
88 if (dir_x == 0 && dir_y == 0 && dir_z == 0)
89 continue;
90
91 if (!is_periodic[2] && dir_z) continue;
92 if (dir_z < 0 && valid_box.loVect()[2] != domain.loVect()[2]) continue;
93 if (dir_z > 0 && valid_box.hiVect()[2] != domain.hiVect()[2]) continue;
94
95 link->add_wrap(diy::Direction{dir_x, dir_y, dir_z});
96 }
97 }
98 }

101void set_neighbors(int level, int finest_level, const std::vector<int>& gid_offsets, const std::vector<int>& refinements,
102 const Box& domain, Vector<std::unique_ptr<MultiFab> >& particle_mf, const Box& valid_box,
103 const std::array<bool, AMREX_SPACEDIM>& is_periodic, diy::AMRLink* link)
104{
105 for(int nbr_lev = std::max(0, level - 1); nbr_lev <= std::min(finest_level, level + 1); ++nbr_lev) {
106 // gotta do this yoga to work around AMReX's static variables
107
108 Box nbr_lev_domain = domain;
109 nbr_lev_domain.refine(refinements[level]);
110
111 Periodicity periodicity(IntVect(AMREX_D_DECL(nbr_lev_domain.length(0) * is_periodic[0],
112 nbr_lev_domain.length(1) * is_periodic[1],
113 nbr_lev_domain.length(2) * is_periodic[2])));
114
115 const std::vector<IntVect>& pshifts = periodicity.shiftIntVect();
116 int ng = 0;
117 const BoxArray& ba = particle_mf[nbr_lev]->boxArray();
118 // TODO: check!
119 int ratio;
120 if (nbr_lev < level) {
121 ratio = refinements[level] / refinements[nbr_lev];
122 } else {
123 ratio = refinements[nbr_lev] / refinements[level];
124 }
125
126 Box gbx = valid_box;
127 if (nbr_lev < level)
128 gbx.coarsen(ratio);
129 else if (nbr_lev > level)
130 gbx.refine(ratio);
131 gbx.grow(1);
132
133 std::vector<std::pair<int, Box>> isects;
134
135 for(const auto& piv : pshifts) {
136 ba.intersections(gbx + piv, isects);
137 for(const auto& is : isects) {
138 // is.first is the index of neighbor box
139 // ba[is.first] is the neighbor box
140 int nbr_gid = gid_offsets.at(nbr_lev) + is.first;
141 const Box& nbr_box = ba[is.first];
142 Box nbr_ghost_box = grow(nbr_box, ng);
143 link->add_neighbor(diy::BlockID{nbr_gid,
144 -1}); // we don't know the proc, but we'll figure it out later through DynamicAssigner
145 // we ignore ghost data, hence nbr_box in last two parameter
146 link->add_bounds(nbr_lev, refinements.at(nbr_lev), bounds(nbr_box), bounds(nbr_box));

148 }
