406void
407StateDescriptor::setUpMaps (int& use_default_map,
408 const Interpolater* default_map,
409 int start_comp,
410 int num_comp,
411 Interpolater**& maps,
412 int& nmaps,
413 int*& map_start_comp,
414 int*& map_num_comp,
415 int*& max_start_comp,
416 int*& min_end_comp) const
417{
418 BL_ASSERT(start_comp>=0 && start_comp+num_comp-1 < ncomp && num_comp>0);
419
420 maps = 0;
421 map_start_comp = 0;
422 map_num_comp = 0;
423 max_start_comp = 0;
424 min_end_comp = 0;
425 //
426 // First, count number of interpolaters needed and allocate.
427 //
428 Interpolater* map = mapper_comp[start_comp];
429 if (!map) map = (Interpolater*) default_map;
430 nmaps = 1;
431 int icomp = start_comp+1;
432
433 use_default_map = 1;
434 while (icomp < start_comp+num_comp)
435 {
436 Interpolater* mapper_icomp = mapper_comp[icomp];
437 if (!mapper_icomp)
438 {
439 mapper_icomp = (Interpolater *) default_map;
440 }
441 else
442 {
443 use_default_map = 0;
444 }
445 if (map != mapper_icomp)
446 {
447 map = mapper_icomp;
448 nmaps++;
449 }
450 icomp++;
451 }
452
453 if (use_default_map) return;
454
455 maps = new Interpolater*[nmaps];
456 map_start_comp = new int[nmaps];
457 map_num_comp = new int[nmaps];
458 min_end_comp = new int[nmaps];
459 max_start_comp = new int[nmaps];
460 //
461 // Now fill the slots.
462 //
463 int imap = 0;
464
465 if (mapper_comp[start_comp])
466 {
467 maps[imap] = mapper_comp[start_comp];
468 }
469 else
470 {
471 maps[imap] = (Interpolater *) default_map;
472 }
473
474 icomp = start_comp+1;
475 map_start_comp[imap] = start_comp;
476 map_num_comp[imap] = 1;
477 min_end_comp[imap] = min_map_end_comp[start_comp];
478 max_start_comp[imap] = max_map_start_comp[start_comp];
479
480 while (icomp < start_comp+num_comp)
481 {
482 Interpolater* mapper_icomp = mapper_comp[icomp];
483
484 if (!mapper_icomp)
485 mapper_icomp = (Interpolater *) default_map;
486
487 if (maps[imap] != mapper_icomp)
488 {
489 imap++;
490
491 BL_ASSERT (imap < nmaps);
492
493 maps[imap] = mapper_icomp;
494 map_start_comp[imap] = icomp;
495 map_num_comp[imap] = 1;
496 min_end_comp[imap] = min_map_end_comp[icomp];
497 max_start_comp[imap] = max_map_start_comp[icomp];
498
499 }
500 else
501 {
502 map_num_comp[imap]++;
503 min_end_comp[imap] = std::max(min_end_comp[imap],min_map_end_comp[icomp]);
504 max_start_comp[imap] = std::min(max_start_comp[imap],max_map_start_comp[icomp]);
505 }
506 icomp++;
507 }
508}

541StateDescriptor::sameInterps (int a_scomp,

543{

548
