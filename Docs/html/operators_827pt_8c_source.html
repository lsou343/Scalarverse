98void rebuild_operator(level_type * level, level_type *fromLevel, double a, double b){
99 // form restriction of alpha[], beta_*[] coefficients from fromLevel
100 if(fromLevel != NULL){
101 restriction(level,VECTOR_ALPHA ,fromLevel,VECTOR_ALPHA ,RESTRICT_CELL );
102 restriction(level,VECTOR_BETA_I,fromLevel,VECTOR_BETA_I,RESTRICT_FACE_I);
103 restriction(level,VECTOR_BETA_J,fromLevel,VECTOR_BETA_J,RESTRICT_FACE_J);
104 restriction(level,VECTOR_BETA_K,fromLevel,VECTOR_BETA_K,RESTRICT_FACE_K);
105 } // else case assumes alpha/beta have been set
106
107 // exchange alpha/beta/... (must be done before calculating Dinv)
108 exchange_boundary(level,VECTOR_ALPHA ,STENCIL_SHAPE_BOX); // safe
109 exchange_boundary(level,VECTOR_BETA_I,STENCIL_SHAPE_BOX);
110 exchange_boundary(level,VECTOR_BETA_J,STENCIL_SHAPE_BOX);
111 exchange_boundary(level,VECTOR_BETA_K,STENCIL_SHAPE_BOX);
112
113 // black box rebuild of D^{-1}, l1^{-1}, dominant eigenvalue, ...
114 rebuild_operator_blackbox(level,a,b,2);
115
116 // exchange Dinv/L1inv/...
117 exchange_boundary(level,VECTOR_DINV ,STENCIL_SHAPE_BOX); // safe
118 exchange_boundary(level,VECTOR_L1INV,STENCIL_SHAPE_BOX);
119}
