608void mlebabeclap_normalize (Box const& box, Array4<Real> const& phi,
609 Real alpha, Array4<Real const> const& a,
610 Real dhx, Real dhy,
611 Array4<Real const> const& bX, Array4<Real const> const& bY,
612 Array4<const int> const& ccm, Array4<EBCellFlag const> const& flag,
613 Array4<Real const> const& vfrc,
614 Array4<Real const> const& apx, Array4<Real const> const& apy,
615 Array4<Real const> const& fcx, Array4<Real const> const& fcy,
616 Array4<Real const> const& ba, Array4<Real const> const& bc,
617 Array4<Real const> const& beb,
618 bool is_dirichlet, bool beta_on_centroid, int ncomp) noexcept
619{
620 amrex::Loop(box, ncomp, [=] (int i, int j, int k, int n) noexcept
621 {
622 if (flag(i,j,k).isRegular())
623 {
624 phi(i,j,k,n) /= alpha*a(i,j,k) + dhx*(bX(i,j,k,n) + bX(i+1,j,k,n))
625 + dhy*(bY(i,j,k,n) + bY(i,j+1,k,n));
626 }
627 else if (flag(i,j,k).isSingleValued())
628 {
629 Real kappa = vfrc(i,j,k);
630 Real apxm = apx(i,j,k);
631 Real apxp = apx(i+1,j,k);
632 Real apym = apy(i,j,k);
633 Real apyp = apy(i,j+1,k);
634
635 Real sxm = bX(i,j,k,n);
636 if (apxm != 0.0 and apxm != 1.0 && !beta_on_centroid) {
637 int jj = j + static_cast<int>(amrex::Math::copysign(1.0,fcx(i,j,k)));
638 Real fracy = (ccm(i-1,jj,k) || ccm(i,jj,k))
639 ? amrex::Math::abs(fcx(i,j,k)) : 0.0;
640 sxm = (1.0-fracy)*sxm;
641 }
642
643 Real sxp = -bX(i+1,j,k,n);
644 if (apxp != 0.0 and apxp != 1.0 && !beta_on_centroid) {
645 int jj = j + static_cast<int>(amrex::Math::copysign(1.0,fcx(i+1,j,k)));
646 Real fracy = (ccm(i,jj,k) || ccm(i+1,jj,k))
647 ? amrex::Math::abs(fcx(i+1,j,k)) : 0.0;
648 sxp = (1.0-fracy)*sxp;
649 }
650
651 Real sym = bY(i,j,k,n);
652 if (apym != 0.0 and apym != 1.0 && !beta_on_centroid) {
653 int ii = i + static_cast<int>(amrex::Math::copysign(1.0,fcy(i,j,k)));
654 Real fracx = (ccm(ii,j-1,k) || ccm(ii,j,k))
655 ? amrex::Math::abs(fcy(i,j,k)) : 0.0;
656 sym = (1.0-fracx)*sym;
657 }
658
659 Real syp = -bY(i,j+1,k,n);
660 if (apyp != 0.0 and apyp != 1.0 && !beta_on_centroid) {
661 int ii = i + static_cast<int>(amrex::Math::copysign(1.0,fcy(i,j+1,k)));
662 Real fracx = (ccm(ii,j,k) || ccm(ii,j+1,k))
663 ? amrex::Math::abs(fcy(i,j+1,k)) : 0.0;
664 syp = (1.0-fracx)*syp;
665 }
666
667 Real vfrcinv = (1.0/kappa);
668 Real gamma = alpha*a(i,j,k) + vfrcinv *
669 (dhx*(apxm*sxm-apxp*sxp) +
670 dhy*(apym*sym-apyp*syp));
671
672 if (is_dirichlet) {
673 Real dapx = apxm-apxp;
674 Real dapy = apym-apyp;
675 Real anorm = std::hypot(dapx,dapy);
676 Real anorminv = 1.0/anorm;
677 Real anrmx = dapx * anorminv;
678 Real anrmy = dapy * anorminv;
679
680 Real bctx = bc(i,j,k,0);
681 Real bcty = bc(i,j,k,1);
682 Real dx_eb = get_dx_eb(vfrc(i,j,k));
683
684 Real dg, gx, gy, sx, sy;
685 if (amrex::Math::abs(anrmx) > amrex::Math::abs(anrmy)) {
686 dg = dx_eb / amrex::Math::abs(anrmx);
687 gx = bctx - dg*anrmx;
688 gy = bcty - dg*anrmy;
689 sx = amrex::Math::copysign(1.0,anrmx);
690 sy = amrex::Math::copysign(1.0,anrmy);
691 } else {
692 dg = dx_eb / amrex::Math::abs(anrmy);
693 gx = bctx - dg*anrmx;
694 gy = bcty - dg*anrmy;
695 sx = amrex::Math::copysign(1.0,anrmx);
696 sy = amrex::Math::copysign(1.0,anrmy);
697 }
698
699 Real phig_gamma = (1.0 + gx*sx + gy*sy + gx*gy*sx*sy);
700 Real feb_gamma = -phig_gamma/dg * ba(i,j,k) * beb(i,j,k,n);
701 gamma += vfrcinv*(-dhx)*feb_gamma;
702 }
703
704 phi(i,j,k,n) /= gamma;
705 }
706 });
