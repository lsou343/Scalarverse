28// AmrLevel::writePlotFile(dir, os, how);
29
30 //
31 // The list of indices of State to write to plotfile.
32 // first component of pair is state_type,
33 // second component of pair is component # within the state_type
34 //
35 std::vector<std::pair<int,int> > plot_var_map;
36 for (int typ = 0; typ < desc_lst.size(); typ++) {
37 for (int comp = 0; comp < desc_lst[typ].nComp();comp++) {
38 if (parent->isStatePlotVar(desc_lst[typ].name(comp)) &&
39 desc_lst[typ].getType() == IndexType::TheCellType()) {
40 plot_var_map.push_back(std::pair<int,int>(typ,comp));
41 }
42 }
43 }
44
45 int num_derive = 0;
46 std::list<std::string> derive_names;
47 const std::list<DeriveRec>& dlist = derive_lst.dlist();
48
49 for (std::list<DeriveRec>::const_iterator it = dlist.begin();
50 it != dlist.end();
51 ++it)
52 {
53 if (parent->isDerivePlotVar(it->name()))
54 {
55 derive_names.push_back(it->name());
56 num_derive++;
57 }
58 }
59
60 int n_data_items = plot_var_map.size() + num_derive + 1;
61
62 Real cur_time = state[State_Type].curTime();
63
64 if (level == 0 && ParallelDescriptor::IOProcessor())
65 {
66 //
67 // The first thing we write out is the plotfile type.
68 //
69 os << thePlotFileType() << '\n';
70
71 if (n_data_items == 0)
72 amrex::Error("Must specify at least one valid data item to plot");
73
74 os << n_data_items << '\n';
75
76 //
77 // Names of variables -- first state, then derived
78 //
79 for (int i = 0; i < plot_var_map.size(); i++)
80 {
81 int typ = plot_var_map[i].first;
82 int comp = plot_var_map[i].second;
83 os << desc_lst[typ].name(comp) << '\n';
84 }
85
86 for ( std::list<std::string>::iterator it = derive_names.begin();
87 it != derive_names.end(); ++it)
88 {
89 const DeriveRec* rec = derive_lst.get(*it);
90 os << rec->variableName(0) << '\n';
91 }
92
93 // volfrac
94 os << "vfrac\n";
95
96 os << BL_SPACEDIM << '\n';
97 os << parent->cumTime() << '\n';
98 int f_lev = parent->finestLevel();
99 os << f_lev << '\n';
100 for (int i = 0; i < BL_SPACEDIM; i++)
101 os << Geom().ProbLo(i) << ' ';
102 os << '\n';
103 for (int i = 0; i < BL_SPACEDIM; i++)
104 os << Geom().ProbHi(i) << ' ';
105 os << '\n';
106 for (int i = 0; i < f_lev; i++)
107 os << parent->refRatio(i)[0] << ' ';
108 os << '\n';
109 for (int i = 0; i <= f_lev; i++)
110 os << parent->Geom(i).Domain() << ' ';
111 os << '\n';
112 for (int i = 0; i <= f_lev; i++)
113 os << parent->levelSteps(i) << ' ';
114 os << '\n';
115 for (int i = 0; i <= f_lev; i++)
116 {
117 for (int k = 0; k < BL_SPACEDIM; k++)
118 os << parent->Geom(i).CellSize()[k] << ' ';
119 os << '\n';
120 }
121 os << (int) Geom().Coord() << '\n';
122 os << "0\n"; // Write bndry data.
123
124 }
125 // Build the directory to hold the MultiFab at this level.
126 // The name is relative to the directory containing the Header file.
127 //
128 static const std::string BaseName = "/Cell";
129 char buf[64];
130 sprintf(buf, "Level_%d", level);
131 std::string sLevel = buf;
132 //
133 // Now for the full pathname of that directory.
134 //
135 std::string FullPath = dir;
136 if (!FullPath.empty() && FullPath[FullPath.size()-1] != '/')
137 FullPath += '/';
138 FullPath += sLevel;
139 //
140 // Only the I/O processor makes the directory if it doesn't already exist.
141 //
142 if (ParallelDescriptor::IOProcessor())
143 if (!amrex::UtilCreateDirectory(FullPath, 0755))
144 amrex::CreateDirectoryFailed(FullPath);
145 //
146 // Force other processors to wait till directory is built.
147 //
148 ParallelDescriptor::Barrier();
149
150 if (ParallelDescriptor::IOProcessor())
151 {
152 os << level << ' ' << grids.size() << ' ' << cur_time << '\n';
153 os << parent->levelSteps(level) << '\n';
154
155 for (int i = 0; i < grids.size(); ++i)
156 {
157 RealBox gridloc = RealBox(grids[i],geom.CellSize(),geom.ProbLo());
158 for (int n = 0; n < BL_SPACEDIM; n++)
159 os << gridloc.lo(n) << ' ' << gridloc.hi(n) << '\n';
160 }
161 //
162 // The full relative pathname of the MultiFabs at this level.
163 // The name is relative to the Header file containing this name.
164 // It's the name that gets written into the Header.
165 //
166 if (n_data_items > 0)
167 {
168 std::string PathNameInHeader = sLevel;
169 PathNameInHeader += BaseName;
170 os << PathNameInHeader << '\n';
171 }
172
173 // volfrac threshhold for amrvis
174 if (level == parent->finestLevel()) {
175 for (int lev = 0; lev <= parent->finestLevel(); ++lev) {
176 os << "1.0e-6\n";
177 }
178 }
179 }
180 //
181 // We combine all of the multifabs -- state, derived, etc -- into one
182 // multifab -- plotMF.
183 // NOTE: we are assuming that each state variable has one component,
184 // but a derived variable is allowed to have multiple components.
185 int cnt = 0;
186 const int nGrow = 0;
187 MultiFab plotMF(grids,dmap,n_data_items,nGrow, MFInfo(), Factory());
188 MultiFab* this_dat = 0;
189 //
190 // Cull data from state variables -- use no ghost cells.
191 //
192 for (int i = 0; i < plot_var_map.size(); i++)
193 {
194 int typ = plot_var_map[i].first;
195 int comp = plot_var_map[i].second;
196 this_dat = &state[typ].newData();
197 MultiFab::Copy(plotMF,*this_dat,comp,cnt,1,nGrow);
198#ifdef AMREX_TESTING
199 // to avoid fcompare failure
200 if (typ == Cost_Type) {
201 plotMF.setVal(0.0, cnt, 1, nGrow);
202 }
203#endif
204 cnt++;
205 }
206 //
207 // Cull data from derived variables.
208 //
209 if (derive_names.size() > 0)
210 {
211 for (std::list<std::string>::iterator it = derive_names.begin();
212 it != derive_names.end(); ++it)
213 {
214 auto derive_dat = derive(*it,cur_time,nGrow);
215 MultiFab::Copy(plotMF,*derive_dat,0,cnt,1,nGrow);
216 cnt++;
217 }
218 }
219
220 plotMF.setVal(0.0, cnt, 1, nGrow);
221
222 MultiFab::Copy(plotMF,volFrac(),0,cnt,1,nGrow);
223
224 //
225 // Use the Full pathname when naming the MultiFab.
226 //
227 std::string TheFullPath = FullPath;
228 TheFullPath += BaseName;
229 VisMF::Write(plotMF,TheFullPath,how,true);
230}
