195 template <class T, class H>
196 std::size_t removeDupDoit (Vector<T>& vec, std::size_t start, std::size_t stop)
197 {
198 std::size_t N = stop-start;
199 if (N < 2) return stop;
200
201 T* const data = vec.data() + start;
202 T const sentinel = data[0]; // duplicates will be set to sentinel and removed later
203 H const hasher;
204 for (std::size_t i = 1; i < N; ) {
205 if (data[i] == sentinel) {
206 ++i;
207 continue;
208 }
209
210 std::size_t const hash = hasher(data[i]) % N;
211 if (i == hash) { // data[i] in correct hash position
212 ++i;
213 continue;
214 }
215
216 if (data[i] == data[hash]) {
217 data[i] = sentinel; // because it's a duplicate
218 ++i;
219 continue;
220 }
221
222 if (data[hash] == sentinel) {
223 std::swap(data[hash], data[i]);
224 // after swap, new data[i] holds sentinel
225 // newdata[hash] in correct hash poitiion
226 ++i;
227 continue;
228 }
229
230 std::size_t const hashhash = hasher(data[hash]) % N;
231 if (hashhash != hash) { // data[hash] not in correct has poision, thus will yield it's poistion
232 std::swap(data[i], data[hash]);
233 // after swap, new data[hash] in correct hash position
234 // new data[i] not sure
235 if (hash < i) { // we have seen new data[i]
236 ++i;
237 } // else next iteration we will work on data[i]
238 } else { // data[hash] in correct hash position, but data[i] is not because of hash collision
239 ++i;
240 }
241 }
242
243 // Now there are three types for data[i]
244 // (1) sentinel
245 // (2) data[i] != sentinel and hash(data[i]) == i
246 // (3) data[i] != sentinel and hash(data[i]) != i because of hash collision
247 // All type 2s are unique, and all sentinels except one can be removed.
248 // We will move all type 2s to the beginning, all sentinels to the end.
249 // This will leave all type 3s in the middle. Then we will work on the middle
250 // part plus one sentinel.
251
252 std::size_t swapPos = 0;
253 for (std::size_t i = 0; i < N; ++i) {
254 // move type 2 to the beginning pointed to by swapPos
255 if (data[i] != sentinel and i == hasher(data[i]) % N) {
256 std::swap(data[i], data[swapPos++]);
257 }
258 }
259
260 // Now we have moved all type 2 elements to the beginning, [0,swapPos)
261
262 std::size_t sentinelPos = N;
263 for (std::size_t i = swapPos; i < sentinelPos; ) {
264 // move type 1 to the end
265 if(data[i] == sentinel) {
266 std::swap(data[i], data[--sentinelPos]);
267 } else {
268 ++i;
269 }
270 }
271
272 // recursively work on the middle part
273 return detail::removeDupDoit<T,H>(vec, start+swapPos, start+sentinelPos+1);
274 }
