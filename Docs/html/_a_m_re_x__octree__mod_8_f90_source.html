95 subroutine amrex_octree_iter_build (oti, level)
96 type(amrex_octree_iter) :: oti
97 integer, intent(in), optional :: level
98 integer :: nnodes, lev_offset, tid, nthreads, n_per_thread, nlft
99 if (present(level)) then
100 lev_offset = leaf_nodes_level_offset(level)
101 nnodes = leaf_nodes_level_offset(level+1) - lev_offset
102 else
103 lev_offset = 0
104 nnodes = size(leaf_nodes)
105 end if
106 tid = omp_get_thread_num()
107 nthreads = omp_get_num_threads()
108 if (nnodes < nthreads) then ! there are more threads than nodes
109 if (tid < nnodes) then
110 oti%begin_index = tid
111 oti%end_index = tid+1
112 else
113 oti%begin_index = 0
114 oti%end_index = 0
115 end if
116 else
117 n_per_thread = nnodes / nthreads
118 nlft = nnodes - n_per_thread*nthreads
119 if (tid < nlft) then
120 oti%begin_index = tid*(n_per_thread+1)
121 oti%end_index = oti%begin_index + n_per_thread+1
122 else
123 oti%begin_index = tid*n_per_thread + nlft
124 oti%end_index = oti%begin_index + n_per_thread
125 end if
126 end if
127 oti%begin_index = oti%begin_index + lev_offset + 1 ! because this is Fortran
128 oti%end_index = oti%end_index + lev_offset + 1
129 oti%current_index = oti%begin_index - 1
130 end subroutine amrex_octree_iter_build
