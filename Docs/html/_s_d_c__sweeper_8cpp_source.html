7void SDC_advance(MultiFab& phi_old,
8 MultiFab& phi_new,
9 std::array<MultiFab, AMREX_SPACEDIM>& flux,
10 Real dt,
11 const Geometry& geom,
12 const Vector<BCRec>& bc,
13 MLMG& mlmg,
14 MLABecLaplacian& mlabec,
15 SDCstruct &SDC, Real a, Real d, Real r)
16{
17
18 /* This is a multi-implicit SDC example time step for an
19 advection-diffusion-reaction equation of the form
20
21 phi_t = A(phi)+D(phi)+R(phi)
22
23 The advection is treated explicilty and the diffusion and reaction implicitly
24 and uncoupled
25
26 the constants a,d, and r control the strength of each term
27 */
28 Real qij;
29
30 const BoxArray &ba=phi_old.boxArray();
31 const DistributionMapping &dm=phi_old.DistributionMap();
32
33 // Copy old phi into first SDC node
34 MultiFab::Copy(SDC.sol[0],phi_old, 0, 0, 1, 0);
35
36 // Fill the ghost cells of each grid from the other grids
37 // includes periodic domain boundaries
38 SDC.sol[0].FillBoundary(geom.periodicity());
39
40 // Fill non-periodic physical boundaries
41 FillDomainBoundary(SDC.sol[0], geom, bc);
42
43 // Compute the first function value
44 int sdc_m=0;
45 SDC_feval(flux,geom,bc,SDC,a,d,r,sdc_m,-1);
46
47 // Copy first function value to all nodes
48 for (int sdc_n = 1; sdc_n < SDC.Nnodes; sdc_n++)
49 {
50 MultiFab::Copy(SDC.f[0][sdc_n],SDC.f[0][0], 0, 0, 1, 0);
51 MultiFab::Copy(SDC.f[1][sdc_n],SDC.f[1][0], 0, 0, 1, 0);
52 if (SDC.Npieces==3)
53 MultiFab::Copy(SDC.f[2][sdc_n],SDC.f[2][0], 0, 0, 1, 0);
54 }
55
56
57 // Now do the actual sweeps
58 for (int k=1; k <= SDC.Nsweeps; ++k)
59 {
60 amrex::Print() << "sweep " << k << "\n";
61
62 // Compute RHS integrals
63 SDC.SDC_rhs_integrals(dt);
64
65 // Substep over SDC nodes
66 for (int sdc_m = 0; sdc_m < SDC.Nnodes-1; sdc_m++)
67 {
68 // use phi_new as rhs and fill it with terms at this iteration
69 SDC.SDC_rhs_k_plus_one(phi_new,dt,sdc_m);
70
71 // get the best initial guess for implicit solve
72 MultiFab::Copy(SDC.sol[sdc_m+1],phi_new, 0, 0, 1, 0);
73 for ( MFIter mfi(SDC.sol[sdc_m+1]); mfi.isValid(); ++mfi )
74 {
75 // const Box& bx = mfi.validbox();
76 qij = dt*SDC.Qimp[sdc_m][sdc_m+1];
77 SDC.sol[sdc_m+1][mfi].saxpy(qij,SDC.f[1][sdc_m+1][mfi]);
78 }
79
80 // Solve for the first implicit part
81 SDC_fcomp(phi_new, geom, bc, SDC, mlmg, mlabec,dt,d,r,sdc_m+1,1);
82
83 if (SDC.Npieces==3)
84 {
85 // Build rhs for 2nd solve
86 MultiFab::Copy(phi_new, SDC.sol[sdc_m+1],0, 0, 1, 0);
87
88 // Add in the part for the 2nd implicit term to rhs
89 SDC.SDC_rhs_misdc(phi_new,dt,sdc_m);
90
91 // Solve for the second implicit part
92 SDC_fcomp(phi_new, geom, bc, SDC, mlmg, mlabec,dt,d,r,sdc_m+1,2);
93 }
94 // Compute the function values at node sdc_m+1
95 SDC_feval(flux,geom,bc,SDC,a,d,r,sdc_m+1,-1);
96
97 } // end SDC substep loop
98 } // end sweeps loop
99
100 // Return the last node in phi_new
101 MultiFab::Copy(phi_new, SDC.sol[SDC.Nnodes-1], 0, 0, 1, 0);
102
103}

117

121 if (npiece < 0)
122 {
123 nlo=0;

125 }

128 nlo=npiece;
129 nhi=npiece+1;
130 }

135 const Box& bx = mfi.validbox();

152void SDC_fcomp(MultiFab& rhs,
153 const Geometry& geom,
154 const Vector<BCRec>& bc,
155 SDCstruct &SDC,
156 MLMG &mlmg,
157 MLABecLaplacian& mlabec,
158 Real dt,Real d,Real r,
159 int sdc_m,int npiece)
160{
161 /* Solve implicitly for the implicit terms of the equation at the SDC node "sdc_m".
162 The input parameter "npiece" describes which term to do. */
163
164 const BoxArray &ba=SDC.sol[0].boxArray();
165 const DistributionMapping &dm=SDC.sol[0].DistributionMap();
166
167 const Box& domain_bx = geom.Domain();
168 const Real* dx = geom.CellSize();
169 Real qij;
170
171 // relative and absolute tolerances for linear solve
172 const Real tol_rel = 1.e-10;
173 const Real tol_abs = 0.0;
174
175 if (npiece == 1)
176 {
177 // Do diffusion solve
178
179 // Fill the ghost cells of each grid from the other grids
180 // includes periodic domain boundaries
181 rhs.FillBoundary(geom.periodicity());
182
183 // Fill non-periodic physical boundaries
184 FillDomainBoundary(rhs, geom, bc);
185
186 // Set diffusion scalar in solve
187 qij = d*dt*SDC.Qimp[sdc_m-1][sdc_m];
188 Real ascalar = 1.0;
189 mlabec.setScalars(ascalar, qij);
190
191 // set the boundary conditions
192 mlabec.setLevelBC(0, &rhs);
193 mlabec.setLevelBC(0, &SDC.sol[sdc_m]);
194 // Do the multigrid solve
195 mlmg.solve({&SDC.sol[sdc_m]}, {&rhs}, tol_rel, tol_abs);
196
197 }
198 else
199 { // Do reaction solve y - qij*y*(1-y)*(y-1/2) = rhs
200
201 // make a flag to change how the reaction is done
202 int nflag=1; // Lazy approximation
203
204 qij = r*dt*SDC.Qimp[sdc_m-1][sdc_m];
205 for ( MFIter mfi(SDC.sol[sdc_m]); mfi.isValid(); ++mfi )
206 {
207 const Box& bx = mfi.validbox();
208 SDC_fcomp_reaction_F(BL_TO_FORTRAN_BOX(bx),
209 BL_TO_FORTRAN_BOX(domain_bx),
210 BL_TO_FORTRAN_ANYD(SDC.sol[sdc_m][mfi]),
211 BL_TO_FORTRAN_ANYD(rhs[mfi]),
212 BL_TO_FORTRAN_ANYD(SDC.f[2][sdc_m][mfi]),
213 &qij,&nflag);
214 }
215
216 }
