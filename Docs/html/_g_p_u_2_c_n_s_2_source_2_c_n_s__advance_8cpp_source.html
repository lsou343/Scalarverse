8Real
9CNS::advance (Real time, Real dt, int /*iteration*/, int /*ncycle*/)
10{
11 BL_PROFILE("CNS::advance()");
12
13 for (int i = 0; i < num_state_data_types; ++i) {
14 state[i].allocOldData();
15 state[i].swapTimeLevels(dt);
16 }
17
18 MultiFab& S_new = get_new_data(State_Type);
19 MultiFab& S_old = get_old_data(State_Type);
20 MultiFab dSdt(grids,dmap,NUM_STATE,0,MFInfo(),Factory());
21 MultiFab Sborder(grids,dmap,NUM_STATE,NUM_GROW,MFInfo(),Factory());
22
23 FluxRegister* fr_as_crse = nullptr;
24 if (do_reflux && level < parent->finestLevel()) {
25 CNS& fine_level = getLevel(level+1);
26 fr_as_crse = fine_level.flux_reg.get();
27 }
28
29 FluxRegister* fr_as_fine = nullptr;
30 if (do_reflux && level > 0) {
31 fr_as_fine = flux_reg.get();
32 }
33
34 if (fr_as_crse) {
35 fr_as_crse->setVal(Real(0.0));
36 }
37
38 // RK2 stage 1
39 FillPatch(*this, Sborder, NUM_GROW, time, State_Type, 0, NUM_STATE);
40 compute_dSdt(Sborder, dSdt, Real(0.5)*dt, fr_as_crse, fr_as_fine);
41 // U^* = U^n + dt*dUdt^n
42 MultiFab::LinComb(S_new, Real(1.0), Sborder, 0, dt, dSdt, 0, 0, NUM_STATE, 0);
43 computeTemp(S_new,0);
44
45 // RK2 stage 2
46 // After fillpatch Sborder = U^n+dt*dUdt^n
47 FillPatch(*this, Sborder, NUM_GROW, time+dt, State_Type, 0, NUM_STATE);
48 compute_dSdt(Sborder, dSdt, Real(0.5)*dt, fr_as_crse, fr_as_fine);
49 // S_new = 0.5*(Sborder+S_old) = U^n + 0.5*dt*dUdt^n
50 MultiFab::LinComb(S_new, Real(0.5), Sborder, 0, Real(0.5), S_old, 0, 0, NUM_STATE, 0);
51 // S_new += 0.5*dt*dSdt
52 MultiFab::Saxpy(S_new, Real(0.5)*dt, dSdt, 0, 0, NUM_STATE, 0);
53 // We now have S_new = U^{n+1} = (U^n+0.5*dt*dUdt^n) + 0.5*dt*dUdt^*
54 computeTemp(S_new,0);
55
56 return dt;
57}

59void
60CNS::compute_dSdt (const MultiFab& S, MultiFab& dSdt, Real dt,
61 FluxRegister* fr_as_crse, FluxRegister* fr_as_fine)
62{
63 BL_PROFILE("CNS::compute_dSdt()");
64
65 const auto dx = geom.CellSizeArray();
66 const auto dxinv = geom.InvCellSizeArray();
67 const int ncomp = NUM_STATE;
68 const int neqns = 5;
69 const int ncons = 7;
70 const int nprim = 8;
71
72 Array<MultiFab,AMREX_SPACEDIM> fluxes;
73 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
74 fluxes[idim].define(amrex::convert(S.boxArray(),IntVect::TheDimensionVector(idim)),
75 S.DistributionMap(), ncomp, 0);
76 }
77
78 Parm const* lparm = parm.get();
79
80 FArrayBox qtmp, slopetmp;
81 for (MFIter mfi(S); mfi.isValid(); ++mfi)
82 {
83 const Box& bx = mfi.tilebox();
84
85 auto const& sfab = S.array(mfi);
86 auto const& dsdtfab = dSdt.array(mfi);
87 AMREX_D_TERM(auto const& fxfab = fluxes[0].array(mfi);,
88 auto const& fyfab = fluxes[1].array(mfi);,
89 auto const& fzfab = fluxes[2].array(mfi););
90
91 const Box& bxg2 = amrex::grow(bx,2);
92 qtmp.resize(bxg2, nprim);
93 Elixir qeli = qtmp.elixir();
94 auto const& q = qtmp.array();
95
96 amrex::ParallelFor(bxg2,
97 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
98 {
99 cns_ctoprim(i, j, k, sfab, q, *lparm);
100 });
101
102 const Box& bxg1 = amrex::grow(bx,1);
103 slopetmp.resize(bxg1,neqns);
104 Elixir slopeeli = slopetmp.elixir();
105 auto const& slope = slopetmp.array();
106
107 // x-direction
108 int cdir = 0;
109 const Box& xslpbx = amrex::grow(bx, cdir, 1);
110 amrex::ParallelFor(xslpbx,
111 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
112 {
113 cns_slope_x(i, j, k, slope, q);
114 });
115 const Box& xflxbx = amrex::surroundingNodes(bx,cdir);
116 amrex::ParallelFor(xflxbx,
117 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
118 {
119 cns_riemann_x(i, j, k, fxfab, slope, q, *lparm);
120 for (int n = neqns; n < ncons; ++n) fxfab(i,j,k,n) = Real(0.0);
121 });
122
123 // y-direction
124 cdir = 1;
125 const Box& yslpbx = amrex::grow(bx, cdir, 1);
126 amrex::ParallelFor(yslpbx,
127 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
128 {
129 cns_slope_y(i, j, k, slope, q);
130 });
131 const Box& yflxbx = amrex::surroundingNodes(bx,cdir);
132 amrex::ParallelFor(yflxbx,
133 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
134 {
135 cns_riemann_y(i, j, k, fyfab, slope, q, *lparm);
136 for (int n = neqns; n < ncons; ++n) fyfab(i,j,k,n) = Real(0.0);
137 });
138
139 // z-direction
140 cdir = 2;
141 const Box& zslpbx = amrex::grow(bx, cdir, 1);
142 amrex::ParallelFor(zslpbx,
143 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
144 {
145 cns_slope_z(i, j, k, slope, q);
146 });
147 const Box& zflxbx = amrex::surroundingNodes(bx,cdir);
148 amrex::ParallelFor(zflxbx,
149 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
150 {
151 cns_riemann_z(i, j, k, fzfab, slope, q, *lparm);
152 for (int n = neqns; n < ncons; ++n) fzfab(i,j,k,n) = Real(0.0);
153 });
154
155 // don't have to do this, but we could
156 qeli.clear(); // don't need them anymore
157 slopeeli.clear();
158
159 amrex::ParallelFor(bx, ncons,
160 [=] AMREX_GPU_DEVICE (int i, int j, int k, int n) noexcept
161 {
162 cns_flux_to_dudt(i, j, k, n, dsdtfab, AMREX_D_DECL(fxfab,fyfab,fzfab), dxinv);
163 });
164
165 if (gravity != Real(0.0)) {
166 const Real g = gravity;
167 const int irho = Density;
168 const int imz = Zmom;
169 const int irhoE = Eden;
170 amrex::ParallelFor(bx,
171 [=] AMREX_GPU_DEVICE (int i, int j, int k) noexcept
172 {
173 dsdtfab(i,j,k,imz) += g * sfab(i,j,k,irho);
174 dsdtfab(i,j,k,irhoE) += g * sfab(i,j,k,imz);
175 });
176 }
177 }
178
179 if (fr_as_crse) {
180 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
181 const Real dA = (idim == 0) ? dx[1]*dx[2] : ((idim == 1) ? dx[0]*dx[2] : dx[0]*dx[1]);
182 const Real scale = -dt*dA;
183 fr_as_crse->CrseInit(fluxes[idim], idim, 0, 0, NUM_STATE, scale, FluxRegister::ADD);
184 }
185 }
186
187 if (fr_as_fine) {
188 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
189 const Real dA = (idim == 0) ? dx[1]*dx[2] : ((idim == 1) ? dx[0]*dx[2] : dx[0]*dx[1]);
190 const Real scale = dt*dA;
191 fr_as_fine->FineAdd(fluxes[idim], idim, 0, 0, NUM_STATE, scale);
192 }
193 }
194}
