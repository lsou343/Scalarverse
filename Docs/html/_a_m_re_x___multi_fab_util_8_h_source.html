275 const IntVect& ratio, int ngcrse, bool mfiter_is_definitely_safe)
276{
277 AMREX_ASSERT(fine.is_nodal());
278 AMREX_ASSERT(crse.is_nodal());
279 AMREX_ASSERT(crse.nComp() == fine.nComp());
280
281 int ncomp = crse.nComp();
282 using value_type = typename FAB::value_type;
283
284 if (mfiter_is_definitely_safe || isMFIterSafe(fine, crse))
285 {
286#ifdef _OPENMP

288#endif
289 for (MFIter mfi(crse,TilingIfNotGPU()); mfi.isValid(); ++mfi)
290 {
291 const Box& bx = mfi.growntilebox(ngcrse);
292 Array4<value_type> const& crsearr = crse.array(mfi);
293 Array4<value_type const> const& finearr = fine.const_array(mfi);
294
295 AMREX_LAUNCH_HOST_DEVICE_LAMBDA ( bx, tbx,

298 });
299 }
300 }
