477void mlndlap_gauss_seidel_aa (Box const& bx, Array4<Real> const& sol,
478 Array4<Real const> const& rhs, Array4<Real const> const& sig,
479 Array4<int const> const& msk,
480 GpuArray<Real,AMREX_SPACEDIM> const& dxinv,
481 bool is_rz) noexcept
482{
483 Real facx = (1.0/6.0)*dxinv[0]*dxinv[0];
484 Real facy = (1.0/6.0)*dxinv[1]*dxinv[1];
485 Real fxy = facx + facy;
486 Real f2xmy = 2.0*facx - facy;
487 Real fmx2y = 2.0*facy - facx;
488
489 amrex::Loop(bx, [=] (int i, int j, int k) noexcept
490 {
491 if (msk(i,j,k)) {
492 sol(i,j,k) = 0.0;
493 } else {
494 Real s0 = (-2.0)*fxy*(sig(i-1,j-1,k)+sig(i,j-1,k)+sig(i-1,j,k)+sig(i,j,k));
495 Real Ax = sol(i-1,j-1,k)*fxy*sig(i-1,j-1,k)
496 + sol(i+1,j-1,k)*fxy*sig(i ,j-1,k)
497 + sol(i-1,j+1,k)*fxy*sig(i-1,j ,k)
498 + sol(i+1,j+1,k)*fxy*sig(i ,j ,k)
499 + sol(i-1,j,k)*f2xmy*(sig(i-1,j-1,k)+sig(i-1,j,k))
500 + sol(i+1,j,k)*f2xmy*(sig(i ,j-1,k)+sig(i ,j,k))
501 + sol(i,j-1,k)*fmx2y*(sig(i-1,j-1,k)+sig(i,j-1,k))
502 + sol(i,j+1,k)*fmx2y*(sig(i-1,j ,k)+sig(i,j ,k))
503 + sol(i,j,k)*s0;
504
505 if (is_rz) {
506 Real fp = facy / static_cast<Real>(2*i+1);
507 Real fm = facy / static_cast<Real>(2*i-1);
508 Real frzlo = fm*sig(i-1,j-1,k)-fp*sig(i,j-1,k);
509 Real frzhi = fm*sig(i-1,j ,k)-fp*sig(i,j ,k);
510 s0 += - frzhi - frzlo;
511 Ax += frzhi*(sol(i,j+1,k)-sol(i,j,k))
512 + frzlo*(sol(i,j-1,k)-sol(i,j,k));
513 }
514
515 sol(i,j,k) += (rhs(i,j,k) - Ax) / s0;
516 }
517 });
518}

520AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
521void tridiagonal_solve (Array1D<Real,0,31>& a_ls, Array1D<Real,0,31>& b_ls, Array1D<Real,0,31>& c_ls,
522 Array1D<Real,0,31>& r_ls, Array1D<Real,0,31>& u_ls, Array1D<Real,0,31>& gam,
523 int ilen ) noexcept
524{
525 Real bet = b_ls(0);
526 u_ls(0) = r_ls(0) / bet;
527
528 for (int i = 1; i <= ilen - 1; i++) {
529 gam(i) = c_ls(i-1) / bet;
530 bet = b_ls(i) - a_ls(i)*gam(i);
531 if (bet == 0) amrex::Abort(">>>TRIDIAG FAILED");
532 u_ls(i) = (r_ls(i)-a_ls(i)*u_ls(i-1)) / bet;
533 }
534 for (int i = ilen-2; i >= 0; i--) {
535 u_ls(i) = u_ls(i) - gam(i+1)*u_ls(i+1);
536 }
537}

539AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
540void mlndlap_gauss_seidel_with_line_solve_aa (Box const& bx, Array4<Real> const& sol,
541 Array4<Real const> const& rhs, Array4<Real const> const& sig,
542 Array4<int const> const& msk,
543 GpuArray<Real,AMREX_SPACEDIM> const& dxinv,
544 bool is_rz) noexcept
545{
546 Real facx = (1.0/6.0)*dxinv[0]*dxinv[0];
547 Real facy = (1.0/6.0)*dxinv[1]*dxinv[1];
548 Real fxy = facx + facy;
549 Real f2xmy = 2.0*facx - facy;
550 Real fmx2y = 2.0*facy - facx;
551
552 if (is_rz) {
553 amrex::Abort("mlndlap_gauss_seidel_with_line_solve_aa is not implemented in r-z 2D ");
554 }
555
556 const auto lo = amrex::lbound(bx);
557 const auto hi = amrex::ubound(bx);
558
559 int idir = -1;
560 int ilen = 33;
561 int k = 0;
562 if (dxinv[0] <= dxinv[1]) {
563 idir = 1;
564 ilen = hi.y - lo.y + 1;
565 }
566 if (dxinv[1] <= dxinv[0]) {
567 idir = 0;
568 ilen = hi.x - lo.x + 1;
569 }
570
571 if (ilen > 32) amrex::Abort("mlndlap_gauss_seidel_with_line_solve_aa is hard-wired to be no longer than 32");
572
573 Array1D<Real,0,31> a_ls,b_ls,c_ls,u_ls,r_ls,gam;
574
575 if (idir == 1) {
576 for (int i = lo.x; i <= hi.x; ++i)
577 {
578 for (int j = lo.y; j <= hi.y; ++j)
579 {
580 if (msk(i,j,k)) {
581 a_ls(j-lo.y) = 0.;
582 b_ls(j-lo.y) = 1.;
583 c_ls(j-lo.y) = 0.;
584 u_ls(j-lo.y) = 0.;
585 r_ls(j-lo.y) = 0.;
586 }
587 else
588 {
589 Real s0 = (-2.0)*fxy*(sig(i-1,j-1,k)+sig(i,j-1,k)+sig(i-1,j,k)+sig(i,j,k));
590
591 Real Ax = sol(i-1,j-1,k)*fxy*sig(i-1,j-1,k)
592 + sol(i+1,j-1,k)*fxy*sig(i ,j-1,k)
593 + sol(i-1,j+1,k)*fxy*sig(i-1,j ,k)
594 + sol(i+1,j+1,k)*fxy*sig(i ,j ,k)
595 + sol(i-1,j,k)*f2xmy*(sig(i-1,j-1,k)+sig(i-1,j,k))
596 + sol(i+1,j,k)*f2xmy*(sig(i ,j-1,k)+sig(i ,j,k));
597
598 a_ls(j-lo.y) = fmx2y*(sig(i-1,j-1,k)+sig(i,j-1,k));
599 b_ls(j-lo.y) = s0;
600 c_ls(j-lo.y) = fmx2y*(sig(i-1,j ,k)+sig(i,j ,k));
601 u_ls(j-lo.y) = 0.;
602 r_ls(j-lo.y) = rhs(i,j,k) - Ax;
603 }
604 }
605 tridiagonal_solve(a_ls, b_ls, c_ls, r_ls, u_ls, gam, ilen);
606
607 for (int j = lo.y; j <= hi.y; ++j)
608 {
609 sol(i,j,k) = u_ls(j-lo.y);
610 }
611 }
612 } else if (idir == 0) {
613 for (int j = lo.y ;j <= hi.y; ++j)
614 {
615 for (int i = lo.x; i <= hi.x; ++i)
616 {
617 if (msk(i,j,k)) {
618 a_ls(i-lo.x) = 0.;
619 b_ls(i-lo.x) = 1.;
620 c_ls(i-lo.x) = 0.;
621 u_ls(i-lo.x) = 0.;
622 r_ls(i-lo.x) = 0.;
623 }
624 else
625 {
626 Real s0 = (-2.0)*fxy*(sig(i-1,j-1,k)+sig(i,j-1,k)+sig(i-1,j,k)+sig(i,j,k));
627
628 Real Ax = sol(i-1,j-1,k)*fxy*sig(i-1,j-1,k)
629 + sol(i+1,j-1,k)*fxy*sig(i ,j-1,k)
630 + sol(i-1,j+1,k)*fxy*sig(i-1,j ,k)
631 + sol(i+1,j+1,k)*fxy*sig(i ,j ,k)
632 + sol(i,j-1,k)*fmx2y*(sig(i-1,j-1,k)+sig(i,j-1,k))
633 + sol(i,j+1,k)*fmx2y*(sig(i-1,j ,k)+sig(i,j ,k));
634
635 a_ls(i-lo.x) = f2xmy*(sig(i-1,j-1,k)+sig(i-1,j,k));
636 b_ls(i-lo.x) = s0;
637 c_ls(i-lo.x) = f2xmy*(sig(i ,j-1,k)+sig(i ,j,k));
638 u_ls(i-lo.x) = 0.;
639 r_ls(i-lo.x) = rhs(i,j,k) - Ax;
640
641 }
642 }
643 tridiagonal_solve(a_ls, b_ls, c_ls, r_ls, u_ls, gam, ilen);
644
645 for (int i = lo.x; i <= hi.x; ++i)
646 {
647 sol(i,j,k) = u_ls(i-lo.x);
648 }
649 }
650 } else {
651 amrex::Abort("mlndlap_gauss_seidel_with_line_solve_aa is wrong direction.");
652 }

847AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
848void mlndlap_divu (int i, int j, int k, Array4<Real> const& rhs, Array4<Real const> const& vel,
849 Array4<int const> const& msk, GpuArray<Real,AMREX_SPACEDIM> const& dxinv,
850 Box const& nodal_domain,
851 GpuArray<LinOpBCType, AMREX_SPACEDIM> const& bclo,
852 GpuArray<LinOpBCType, AMREX_SPACEDIM> const& bchi,
853 bool is_rz) noexcept
854{
855 Real facx = 0.5*dxinv[0];
856 Real facy = 0.5*dxinv[1];
857
858 const auto domlo = amrex::lbound(nodal_domain);
859 const auto domhi = amrex::ubound(nodal_domain);
860
861 if (msk(i,j,k)) {
862 rhs(i,j,k) = 0.0;
863 } else {
864
865 Real zero_ilo = 1.0;
866 Real zero_ihi = 1.0;
867 Real zero_jlo = 1.0;
868 Real zero_jhi = 1.0;
869
870 // The nodal divergence operator should not see the tangential velocity
871 // at an inflow face
872 if ((bclo[0] == LinOpBCType::Neumann or bclo[0] == LinOpBCType::inflow)
873 and i == domlo.x) zero_ilo = 0.0;
874 if ((bchi[0] == LinOpBCType::Neumann or bchi[0] == LinOpBCType::inflow)
875 and i == domhi.x) zero_ihi = 0.0;
876 if ((bclo[1] == LinOpBCType::Neumann or bclo[1] == LinOpBCType::inflow)
877 and j == domlo.y) zero_jlo = 0.0;
878 if ((bchi[1] == LinOpBCType::Neumann or bchi[1] == LinOpBCType::inflow)
879 and j == domhi.y) zero_jhi = 0.0;
880
881 rhs(i,j,k) = facx*(-vel(i-1,j-1,k,0)*zero_jlo + vel(i,j-1,k,0)*zero_jlo
882 -vel(i-1,j ,k,0)*zero_jhi + vel(i,j ,k,0)*zero_jhi)
883 + facy*(-vel(i-1,j-1,k,1)*zero_ilo - vel(i,j-1,k,1)*zero_ihi
884 +vel(i-1,j ,k,1)*zero_ilo + vel(i,j ,k,1)*zero_ihi);
885 if (is_rz) {
886 // Here we assume we can't have inflow in the radial direction
887 Real fm = facy / static_cast<Real>(6*i-3);
888 Real fp = facy / static_cast<Real>(6*i+3);
889 rhs(i,j,k) += fm*(vel(i-1,j,k,1)-vel(i-1,j-1,k,1))
890 - fp*(vel(i ,j,k,1)-vel(i ,j-1,k,1));
891 }
892 }

923AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
924void mlndlap_divu_compute_fine_contrib (int i, int j, int, Box const& fvbx,
925 Array4<Real> const& frh, Array4<Real const> const& vel,
926 GpuArray<Real,AMREX_SPACEDIM> const& dxinv,
927 Box const& nodal_domain,
928 GpuArray<LinOpBCType, AMREX_SPACEDIM> const& bclo,
929 GpuArray<LinOpBCType, AMREX_SPACEDIM> const& bchi,
930 bool is_rz) noexcept
931{
932 const auto domlo = amrex::lbound(nodal_domain);
933 const auto domhi = amrex::ubound(nodal_domain);
934
935 IntVect iv(i,j);
936 if (fvbx.contains(iv) and !fvbx.strictly_contains(iv))
937 {
938 Real zero_ilo = 1.0;
939 Real zero_ihi = 1.0;
940 Real zero_jlo = 1.0;

942
943 // The nodal divergence operator should not see the tangential velocity
944 // at an inflow face
945 if ((bclo[0] == LinOpBCType::Neumann or bclo[0] == LinOpBCType::inflow)
