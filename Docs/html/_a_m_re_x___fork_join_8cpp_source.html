51ForkJoin::ForkJoin (const Vector<double> &task_rank_pct)
52{
53 auto rank_n = ParallelContext::NProcsSub(); // number of ranks in current frame
54 auto ntasks = task_rank_pct.size();
55 Vector<int> task_rank_n(ntasks);
56 int prev = 0;
57 double accum = 0;
58 for (int i = 0; i < ntasks; ++i) {
59 accum += task_rank_pct[i];
60 int cur = std::lround(rank_n * accum);
61 task_rank_n[i] = cur - prev;
62 prev = cur;
63 }
64
65 init(task_rank_n);
66}

68void
69ForkJoin::init(const Vector<int> &task_rank_n)
70{
71 ParmParse pp("forkjoin");
72 pp.query("verbose", flag_verbose);
73
74 const auto task_n = task_rank_n.size();
75 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(task_n > 0,
76 "ForkJoin must have at least 1 task");
77 int min_task_rank_n = task_rank_n[0];
78 for (int i = 1; i < task_n; ++i) {
79 min_task_rank_n = std::min(min_task_rank_n, task_rank_n[i]);
80 }
81 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(min_task_rank_n > 0,
82 "All tasks must have at least one rank");
83 auto rank_n = ParallelContext::NProcsSub(); // number of ranks in current frame
84 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(std::accumulate(task_rank_n.begin(),task_rank_n.end(),0) == rank_n,
85 "Sum of ranks assigned to tasks must sum to parent number of ranks");
86
87 // split ranks into contiguous chunks
88 // task i has ranks over the interval [split_bounds[i], split_bounds[i+1])
89 split_bounds.resize(task_n + 1);
90 split_bounds[0] = 0;
91 for (int i = 0; i < task_n; ++i) {
92 split_bounds[i + 1] = split_bounds[i] + task_rank_n[i];
93 }
94
95 if (flag_verbose) {
96 amrex::Print() << "Initialized ForkJoin:\n";
97 for (int i = 0; i < task_n; ++i) {
98 int glo_rank_lo = ParallelContext::local_to_global_rank(split_bounds[i]);
99 int glo_rank_hi = ParallelContext::local_to_global_rank(split_bounds[i+1]-1);
100 amrex::Print() << " Task " << i << " has " << NProcsTask(i)
101 << " Ranks: [" << glo_rank_lo << ", " << glo_rank_hi << "]\n";
102 }
103 }

106void
107ForkJoin::reg_mf (MultiFab &mf, const std::string &name, int idx,
108 Strategy strategy, Intent intent, int owner)
109{
110 if (idx >= data[name].size()) {
111 data[name].resize(idx + 1);
112 }
113 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(data[name][idx].empty(),
114 "Can only register to a (name, index) pair once");
115
116 IntVect ngrow = mf.nGrowVect(); // default use original MultiFab's grow cells
117
118 // compute how components are copied to tasks
119 // default evenly split components across tasks
120 int comp_n = mf.nComp(); // number of components in original
121 Vector<ComponentSet> comp_split;
122 comp_split.resize(NTasks());
123 for (int i = 0; i < NTasks(); ++i) {
124 if (strategy == Strategy::split) {
125 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(NTasks() <= comp_n,
126 "Number of tasks cannot be larger than number of components!");
127 // split components across tasks
128 comp_split[i].lo = comp_n * i / NTasks();
129 comp_split[i].hi = comp_n * (i+1) / NTasks();
130 } else {
131 // copy all components to task
132 comp_split[i].lo = 0;
133 comp_split[i].hi = comp_n;
134 }
135 }
136
137 data[name][idx] = MFFork(&mf, strategy, intent, owner, ngrow, std::move(comp_split));
138}

178void
179ForkJoin::copy_data_to_tasks ()
180{
181 BL_PROFILE("ForkJoin::copy_data_to_tasks()");
182 if (flag_verbose) {
183 amrex::Print() << "Copying data into fork-join tasks ...\n";
184 }
185 for (auto &p : data) { // for each name
186 const auto &mf_name = p.first;
187 for (int idx = 0; idx < p.second.size(); ++idx) { // for each index
188 auto &mff = p.second[idx];
189 const auto &orig = *mff.orig;
190 const auto &ba = orig.boxArray();
191 const auto &comp_split = mff.comp_split;
192 auto &forked = mff.forked;
193
194 forked.reserve(NTasks()); // does nothing if forked MFs already created
195 for (int i = 0; i < NTasks(); ++i) {
196 // check if this task needs this MF
197 if (mff.strategy != Strategy::single || i == mff.owner_task) {
198 int task_comp_n = comp_split[i].hi - comp_split[i].lo;
199
200 // create task's MF if first time through
201 if (forked.size() <= i) {
202 if (flag_verbose) {
203 amrex::Print() << " Creating forked " << mf_name << "[" << idx << "] for task " << i
204 << (mff.strategy == Strategy::split ? " (split)" : " (whole)") << std::endl;
205 }
206 // look up the distribution mapping for this (box array, task) pair
207 const DistributionMapping &dm = get_dm(ba, i, orig.DistributionMap());
208 forked.emplace_back(ba, dm, task_comp_n, mff.ngrow);
209 } else if (flag_verbose) {
210 amrex::Print() << " Forked " << mf_name << "[" << idx << "] for task " << i
211 << " already created" << std::endl;
212 }
213 AMREX_ASSERT(i < forked.size());
214
215 // copy data if needed
216 if (mff.intent == Intent::in || mff.intent == Intent::inout) {
217 if (flag_verbose) {
218 amrex::Print() << " Copying " << mf_name << "[" << idx << "] components ["
219 << comp_split[i].lo << ", " << comp_split[i].hi << ") into to task " << i << std::endl;
220 }
221 // parallel copy data into forked MF
222 forked[i].Redistribute(orig, comp_split[i].lo, 0, task_comp_n, mff.ngrow);
223 }
224
225 } else {
226 // this task doesn't use the MultiFab
227 if (forked.size() <= i) {
228 // first time through, push empty placeholder (not used)
229 forked.push_back(MultiFab());
230 }
231 }
232 }
233 AMREX_ASSERT(forked.size() == NTasks());
234 }

282const DistributionMapping &
283ForkJoin::get_dm (const BoxArray& ba, int task_idx, const DistributionMapping& dm_orig)
284{
285 AMREX_ASSERT(task_idx < NTasks());
286
287 auto &dm_vec = dms[ba.getRefID()];
288 if (dm_vec.size() == 0) {
289 // new entry
290 dm_vec.resize(NTasks());
291 }
292 AMREX_ASSERT(dm_vec.size() == NTasks());
293
294 if (dm_vec[task_idx] == nullptr) {
295 // create DM of current box array over current task's ranks
296 int rank_lo = split_bounds[task_idx]; // note that these ranks are not necessarily global
297 int nprocs_task = NProcsTask(task_idx);
298
299 Vector<int> pmap = dm_orig.ProcessorMap(); // DistributionMapping stores global ranks
300 for (auto& r : pmap) {
301 int lr = ParallelContext::global_to_local_rank(r);
302 lr = lr%nprocs_task + rank_lo;
303 r = ParallelContext::local_to_global_rank(lr);
304 }
305
306 dm_vec[task_idx].reset(new DistributionMapping(std::move(pmap)));
307
308 if (flag_verbose) {

312

314
315 } else {
