68 m_lo_inhomog_neumann[n].end(), 1);

70 m_hi_inhomog_neumann[n].end(), 1);
71 if (itlo != m_lo_inhomog_neumann[n].end() or
72 ithi != m_hi_inhomog_neumann[n].end())

86 const Real dyi = (AMREX_SPACEDIM >= 2) ? m_geom[amrlev][mglev].InvCellSize(1) : 1.0;
87 const Real dzi = (AMREX_SPACEDIM == 3) ? m_geom[amrlev][mglev].InvCellSize(2) : 1.0;
88 const Real xlo = problo[0];

90 const Box& domain = m_geom[amrlev][mglev].Domain();

92 const Real beta = getBScalar();

94 FArrayBox foo(Box(IntVect(0),IntVect(1)));
95 bool has_bcoef = (bcoef[0] != nullptr);
96
97 const auto& maskvals = m_maskvals[amrlev][mglev];
98 const auto& bcondloc = *m_bcondloc[amrlev][mglev];
99 const auto& bndry = *m_bndry_sol[amrlev];
100

103
104#ifdef _OPENMP
105#pragma omp parallel if (Gpu::notInLaunchRegion())

257 hypre_solver->setACoeffs(alpha);

262 {
263 hypre_solver->setBCoeffs(bc);
