9double evaluateBeta(double x, double y, double z, double h, int add_Bxx, int add_Byy, int add_Bzz){
10 double b = 0.25;
11 double a = 2.0*M_PI; // one period on [0,1]^3
12
13 double B = 1.0 + b*sin(a*x)*sin(a*y)*sin(a*z);
14//double Bx = a*b*cos(a*x)*sin(a*y)*sin(a*z);
15//double By = a*b*sin(a*x)*cos(a*y)*sin(a*z);
16//double Bz = a*b*sin(a*x)*sin(a*y)*cos(a*z);
17 double Bxx = -a*a*b*sin(a*x)*sin(a*y)*sin(a*z);
18 double Byy = -a*a*b*sin(a*x)*sin(a*y)*sin(a*z);
19 double Bzz = -a*a*b*sin(a*x)*sin(a*y)*sin(a*z);
20
21 // 4th order correction to approximate the conversion of cell-centered values to cell-averaged...
22 if(add_Bxx)B+=(h*h/24.0)*Bxx;
23 if(add_Byy)B+=(h*h/24.0)*Byy;
24 if(add_Bzz)B+=(h*h/24.0)*Bzz;
25 return(B);
26}
