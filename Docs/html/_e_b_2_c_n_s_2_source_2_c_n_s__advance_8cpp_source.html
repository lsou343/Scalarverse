21 MultiFab& S_old = get_old_data(State_Type);

41 }
42
43 // RK2 stage 1
44 FillPatch(*this, Sborder, NUM_GROW, time, State_Type, 0, NUM_STATE);
45 compute_dSdt(Sborder, dSdt, 0.5*dt, fr_as_crse, fr_as_fine);
46 // U^* = U^n + dt*dUdt^n
47 MultiFab::LinComb(S_new, 1.0, Sborder, 0, dt, dSdt, 0, 0, NUM_STATE, 0);
48 computeTemp(S_new,0);
49
50 // RK2 stage 2
51 // After fillpatch Sborder = U^n+dt*dUdt^n
52 FillPatch(*this, Sborder, NUM_GROW, time+dt, State_Type, 0, NUM_STATE);
53 compute_dSdt(Sborder, dSdt, 0.5*dt, fr_as_crse, fr_as_fine);
54 // S_new = 0.5*(Sborder+S_old) = U^n + 0.5*dt*dUdt^n
55 MultiFab::LinComb(S_new, 0.5, Sborder, 0, 0.5, S_old, 0, 0, NUM_STATE, 0);
56 // S_new += 0.5*dt*dSdt
57 MultiFab::Saxpy(S_new, 0.5*dt, dSdt, 0, 0, NUM_STATE, 0);
58 // We now have S_new = U^{n+1} = (U^n+0.5*dt*dUdt^n) + 0.5*dt*dUdt^*
59 computeTemp(S_new,0);
60
61 return dt;
62}
