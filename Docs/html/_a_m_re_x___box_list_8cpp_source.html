20static void chop_boxes (Box* bxv, const Box& bx, int nboxes)
21{
22 if (nboxes == 1)
23 {
24 *bxv = bx;
25 }
26 else
27 {
28 int longdir;
29 int longlen = bx.longside(longdir);
30 int chop_pnt = bx.smallEnd(longdir) + longlen/2;
31 Box bxleft = bx;
32 Box bxright = bxleft.chop(longdir, chop_pnt);
33
34 int nleft = nboxes / 2;
35 chop_boxes(bxv, bxleft, nleft);
36
37 int nright = nboxes - nleft;
38 chop_boxes(bxv+nleft, bxright, nright);
39 }
40}

183BoxList::BoxList(const Box& bx, const IntVect& tilesize)
184 : btype(bx.ixType())
185{
186 int ntiles = 1;
187 IntVect nt;
188 for (int d=0; d<AMREX_SPACEDIM; d++) {
189 nt[d] = (bx.length(d)+tilesize[d]-1)/tilesize[d];
190 ntiles *= nt[d];
191 }
192
193 IntVect small, big, ijk; // note that the initial values are all zero.
194 ijk[0] = -1;
195 for (int t=0; t<ntiles; ++t) {
196 for (int d=0; d<AMREX_SPACEDIM; d++) {
197 if (ijk[d]<nt[d]-1) {
198 ijk[d]++;
199 break;
200 } else {
201 ijk[d] = 0;
202 }
203 }
204
205 for (int d=0; d<AMREX_SPACEDIM; d++) {
206 small[d] = ijk[d]*tilesize[d];
207 big[d] = std::min(small[d]+tilesize[d]-1, bx.length(d)-1);
208 }
209
210 Box tbx(small, big, btype);
211 tbx.shift(bx.smallEnd());
212 push_back(tbx);
213 }

612void
613boxDiff (BoxList& bl_diff, const Box& b1in, const Box& b2)
614{
615 AMREX_ASSERT(b1in.sameType(b2));
616
617 bl_diff.clear();
618 bl_diff.set(b2.ixType());
619
620 if ( !b2.contains(b1in) )
621 {
622 Box b1(b1in);
623 if ( !b1.intersects(b2) )
624 {
625 bl_diff.push_back(b1);
626 }
627 else
628 {
629 const int* b2lo = b2.loVect();
630 const int* b2hi = b2.hiVect();
631
632 for (int i = AMREX_SPACEDIM-1; i >= 0; i--)
633 {
634 const int* b1lo = b1.loVect();
635 const int* b1hi = b1.hiVect();
636
637 if ((b1lo[i] < b2lo[i]) && (b2lo[i] <= b1hi[i]))
638 {
639 Box bn(b1);
640 bn.setSmall(i,b1lo[i]);
641 bn.setBig(i,b2lo[i]-1);
642 bl_diff.push_back(bn);
643 b1.setSmall(i,b2lo[i]);
644 }
645 if ((b1lo[i] <= b2hi[i]) && (b2hi[i] < b1hi[i]))
646 {
647 Box bn(b1);
648 bn.setSmall(i,b2hi[i]+1);
649 bn.setBig(i,b1hi[i]);
650 bl_diff.push_back(bn);
651 b1.setBig(i,b2hi[i]);
652 }
653 }
654 }
655 }

658int
659BoxList::simplify (bool best)
660{
661 std::sort(m_lbox.begin(), m_lbox.end(), [](const Box& l, const Box& r) {
662 return l.smallEnd() < r.smallEnd(); });
663
664 //
665 // If we're not looking for the "best" we can do in one pass, we
666 // limit how far afield we look for abutting boxes. This greatly
667 // speeds up this routine for large numbers of boxes. It does not
668 // do quite as good a job though as full brute force.
669 //

684int
685BoxList::simplify_doit (int depth)
686{
687 //
688 // Try to merge adjacent boxes.
689 //
690 int count = 0, lo[AMREX_SPACEDIM], hi[AMREX_SPACEDIM];
691
692 for (iterator bla = begin(), End = end(); bla != End; ++bla)
693 {
694 const int* alo = bla->loVect();
695 const int* ahi = bla->hiVect();
696
697 iterator blb = bla + 1;
698 for (int cnt = 0; blb != End && cnt < depth; ++cnt, ++blb)
699 {
700 const int* blo = blb->loVect();
701 const int* bhi = blb->hiVect();
702 //
703 // Determine if a and b can be coalesced.
704 // They must have equal extents in all index directions
705 // except possibly one, and must abutt in that direction.
706 //
707 bool canjoin = true;
708 int joincnt = 0;
709 for (int i = 0; i < AMREX_SPACEDIM; i++)
710 {
711 if (alo[i]==blo[i] && ahi[i]==bhi[i])
712 {
713 lo[i] = alo[i];
714 hi[i] = ahi[i];
715 }
716 else if (alo[i]<=blo[i] && blo[i]<=ahi[i]+1)
717 {
718 lo[i] = alo[i];
719 hi[i] = std::max(ahi[i],bhi[i]);
720 joincnt++;
721 }
722 else if (blo[i]<=alo[i] && alo[i]<=bhi[i]+1)
723 {
724 lo[i] = blo[i];
725 hi[i] = std::max(ahi[i],bhi[i]);
726 joincnt++;
727 }
728 else
729 {
730 canjoin = false;
731 break;
732 }
733 }
734 if (canjoin && (joincnt <= 1))
735 {
736 //
737 // Modify b and set a to empty
738 //
739 blb->setSmall(IntVect(lo));
740 blb->setBig(IntVect(hi));
741 *bla = Box();
742 ++count;
743 break;
744 }
745 }
746 }
747
748 removeEmpty();
749
750 return count;
751}

769BoxList&
770BoxList::maxSize (const IntVect& chunk)
771{
772 Vector<Box> new_boxes;
773 for (auto const& bx : m_lbox) {
774 const IntVect boxlen = amrex::enclosedCells(bx).size();
775 const IntVect boxlo = bx.smallEnd();
776 IntVect ratio{1}, numblk{1}, extra{0};
777 IntVect sz = boxlen;
778 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
779 if (boxlen[idim] > chunk[idim]) {
780 int bs = chunk[idim];
781 int nlen = boxlen[idim];
782 while ((bs%2 == 0) && (nlen%2 == 0)) {
783 ratio[idim] *= 2;
784 bs /= 2;
785 nlen /= 2;
786 }
787 numblk[idim] = (nlen+bs-1)/bs;
788 sz[idim] = nlen/numblk[idim];
789 extra[idim] = nlen - sz[idim]*numblk[idim];
790 }
791 }
792 if (numblk == 1) {
793 new_boxes.push_back(bx);
794 } else {
795#if (AMREX_SPACEDIM == 3)
796 for (int k = 0; k < numblk[2]; ++k) {
797 int klo = (k < extra[2]) ? k*(sz[2]+1)*ratio[2] : (k*sz[2]+extra[2])*ratio[2];
798 int khi = (k < extra[2]) ? klo+(sz[2]+1)*ratio[2]-1 : klo+sz[2]*ratio[2]-1;
799 klo += boxlo[2];
800 khi += boxlo[2];
801#endif
802#if (AMREX_SPACEDIM >= 2)
803 for (int j = 0; j < numblk[1]; ++j) {
804 int jlo = (j < extra[1]) ? j*(sz[1]+1)*ratio[1] : (j*sz[1]+extra[1])*ratio[1];
805 int jhi = (j < extra[1]) ? jlo+(sz[1]+1)*ratio[1]-1 : jlo+sz[1]*ratio[1]-1;
806 jlo += boxlo[1];
807 jhi += boxlo[1];
808#endif
