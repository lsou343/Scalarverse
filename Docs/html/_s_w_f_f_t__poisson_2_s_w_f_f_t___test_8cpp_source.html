99SWFFT_Test::init_rhs ()
100{
101 const Real* dx = geom.CellSize();
102 Box domain(geom.Domain());
103
104 for (MFIter mfi(rhs,true); mfi.isValid(); ++mfi)
105 {
106 const Box& tbx = mfi.tilebox();
107 fort_init_rhs(BL_TO_FORTRAN_BOX(tbx),
108 BL_TO_FORTRAN_ANYD(rhs[mfi]),
109 BL_TO_FORTRAN_BOX(domain),
110 geom.ProbLo(), geom.ProbHi(), dx);
111 }
112
113 Real sum_rhs = rhs.sum();
114 amrex::Print() << "Sum of rhs over the domain was " << sum_rhs << std::endl;
115
116 sum_rhs = sum_rhs / domain.numPts();
117 rhs.plus(-sum_rhs,0,1,0);
118
119 sum_rhs = rhs.sum();
120 amrex::Print() << "Sum of rhs over the domain is now " << sum_rhs << std::endl;
121}
