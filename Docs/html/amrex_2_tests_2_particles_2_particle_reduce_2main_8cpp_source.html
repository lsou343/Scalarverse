13void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
14{
15 int nx = nppc[0];
16 int ny = nppc[1];
17 int nz = nppc[2];
18
19 int ix_part = i_part/(ny * nz);
20 int iy_part = (i_part % (ny * nz)) % ny;
21 int iz_part = (i_part % (ny * nz)) / ny;
22
23 r[0] = (0.5+ix_part)/nx;
24 r[1] = (0.5+iy_part)/ny;
25 r[2] = (0.5+iz_part)/nz;
26}
