10
11 PRAGMA_THREAD_ACROSS_BOXES(level,box)

18 double * __restrict__ grid = level->my_boxes[box].vectors[component_id] + ghosts*(1+jStride+kStride);

192 }}}

194 }

197 #ifdef USE_MPI
