26AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
27void mlebabeclap_apply_bc_x (int side, Box const& box, int blen,
28 Array4<Real> const& phi,
29 Array4<int const> const& mask,
30 Array4<Real const> const& area,
31 BoundCond bct, Real bcl,
32 Array4<Real const> const& bcval,
33 int maxorder, Real dxinv, int inhomog, int icomp) noexcept
34{
35 const auto lo = amrex::lbound(box);
36 const auto hi = amrex::ubound(box);
37 const int i = lo.x; // boundary cell
38 const int s = 1-2*side; // +1 for lo and -1 for hi
39 switch (bct) {
40 case AMREX_LO_NEUMANN:
41 {
42 for (int k = lo.z; k <= hi.z; ++k) {
43 for (int j = lo.y; j <= hi.y; ++j) {
44 if (mask(i,j,k) == 0 and mask(i+s,j,k) == 1) {
45 phi(i,j,k,icomp) = phi(i+s,j,k,icomp);
46 }
47 }
48 }
49 break;
50 }
51 case AMREX_LO_REFLECT_ODD:
52 {
53 for (int k = lo.z; k <= hi.z; ++k) {
54 for (int j = lo.y; j <= hi.y; ++j) {
55 if (mask(i,j,k) == 0 and mask(i+s,j,k) == 1) {
56 phi(i,j,k,icomp) = -phi(i+s,j,k,icomp);
57 }
58 }
59 }
60 break;
61 }
62 case AMREX_LO_DIRICHLET:
63 {
64 const int NX = amrex::min(blen+1, maxorder);
65 GpuArray<Real,4> x{-bcl * dxinv, Real(0.5), Real(1.5), Real(2.5)};
66 Array2D<Real, 0, 3, 0, 2> coef{};
67 for (int r = 0; r <= maxorder-2; ++r) {
68 poly_interp_coeff(-Real(0.5), &x[0], r+2, &(coef(0,r)));
69 }
70 for (int k = lo.z; k <= hi.z; ++k) {
71 for (int j = lo.y; j <= hi.y; ++j) {
72 if (mask(i,j,k) == 0 and mask(i+s,j,k) == 1) {
73 int order = 1;
74 bool has_cutfaces = false;
75 for (int r = 0; r <= NX-2; ++r) {
76 Real a = area(i+(1-side)+s*r,j,k);
77 if (a > Real(0.0)) {
78 ++order;
79 if (a < Real(1.0)) {
80 has_cutfaces = true;
81 }
82 } else {
83 break;
84 }
85 }
86 if (has_cutfaces) order = amrex::min(2,order);
87 if (order == 1) {
88 if (inhomog) {
89 phi(i,j,k,icomp) = bcval(i,j,k,icomp);
90 } else {
91 phi(i,j,k,icomp) = Real(0.0);
92 }
93 } else {
94 Real tmp = Real(0.0);
95 for (int m = 1; m < order; ++m) {
96 tmp += phi(i+m*s,j,k,icomp) * coef(m,order-2);
97 }
98 phi(i,j,k,icomp) = tmp;
99 if (inhomog) {
100 phi(i,j,k,icomp) += bcval(i,j,k,icomp)*coef(0,order-2);
101 }
102 }
103 }
104 }
105 }
106 break;
107 }
108 default: {}
109 }
