62DArena::alloc (std::size_t nbytes)
63{
64 if (nbytes == 0) return nullptr; // behavior different from the standard
65
66 // We need to allocate this many blocks
67 unsigned int nblocks = (nbytes+m_block_size-1)/m_block_size;
68
69 int order = 0;
70 unsigned int ntmp = nblocks;
71 while (ntmp >>= 1) {
72 ++order;
73 }
74 if ((nblocks & (nblocks - 1)) != 0) { // nblocks is NOT power of 2
75 ++order;
76 }
77 // We have found the miminal order that satisfies 2^order >= nblocks
78
79 std::lock_guard<std::mutex> lock(m_mutex);
80
81 std::ptrdiff_t offset = allocate_order(order);
82 if (offset >= 0) {
83 offset *= m_block_size; // # of order 0 blocks -> # of bytes
84 m_used.insert({offset,order});
85 return m_baseptr + offset;
86 } else {

104void
105DArena::free (void* p)
106{
107 std::lock_guard<std::mutex> lock(m_mutex);
108
109 std::ptrdiff_t offset = (char*)p - m_baseptr;
110 auto r = m_used.find(offset);
111 if (r != m_used.end()) {
112 int order = r->second;
113 m_used.erase(r);
114 offset /= m_block_size; // # of bytes -> # of order 0 blocks
115 deallocate_order(order, offset);
116 } else {
117 auto r2 = m_system.find(p);
118 if (r2 != m_system.end()) {
119 deallocate_system(r2->first, r2->second);
120 m_system.erase(r2);
121 } else {
122 amrex::Abort("DArena::free failed. Unknown pointer");
123 }

127std::ptrdiff_t
128DArena::allocate_order (int order)
129{
130 std::ptrdiff_t offset = -1;
131 if (!m_free[order].empty()) {
132 offset = *m_free[order].cbegin();
133 m_free[order].erase(m_free[order].cbegin());
134 } else if (order < m_max_order) {
135 // We have to allocate from higher order that has larger blocks
136 offset = allocate_order(order+1);
137 if (offset >= 0) {
138 // We have succeed. We need to split and put our buddy into free buckets
139 m_free[order].insert(make_buddy(order,offset));
140 }
141 }
142
143 return offset;
144}

146void
147DArena::deallocate_order (int order, std::ptrdiff_t offset)
148{
149 std::ptrdiff_t buddy = make_buddy(order,offset);
150 auto r = m_free[order].find(buddy);
151 if (r != m_free[order].end()) {
152 m_free[order].erase(r);
153 deallocate_order(order+1,std::min(offset,buddy));
154 } else {
155 m_free[order].insert(offset);
