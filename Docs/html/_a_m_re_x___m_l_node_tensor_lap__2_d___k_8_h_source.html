31void mlndtslap_interpadd (int i, int j, int, Array4<Real> const& fine,
32 Array4<Real const> const& crse, Array4<int const> const& msk) noexcept
33{
34 if (!msk(i,j,0)) {
35 int ic = amrex::coarsen(i,2);
36 int jc = amrex::coarsen(j,2);
37 bool i_is_odd = (ic*2 != i);
38 bool j_is_odd = (jc*2 != j);
39 if (i_is_odd and j_is_odd) {
40 // Node on a X-Y face
41 fine(i,j,0) += ts_interp_face_xy(crse,ic,jc);
42 } else if (i_is_odd) {
43 // Node on X line
44 fine(i,j,0) += ts_interp_line_x(crse,ic,jc);
45 } else if (j_is_odd) {
46 // Node on Y line
47 fine(i,j,0) += ts_interp_line_y(crse,ic,jc);
48 } else {
49 // Node coincident with coarse node
50 fine(i,j,0) += crse(ic,jc,0);
