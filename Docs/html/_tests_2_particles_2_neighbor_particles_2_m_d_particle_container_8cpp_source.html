10 void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
11 {
12 int nx = nppc[0];
13 int ny = nppc[1];
14 int nz = nppc[2];
15
16 int ix_part = i_part/(ny * nz);
17 int iy_part = (i_part % (ny * nz)) % ny;
18 int iz_part = (i_part % (ny * nz)) / ny;
19
20 r[0] = (0.5+ix_part)/nx;
21 r[1] = (0.5+iy_part)/ny;
22 r[2] = (0.5+iz_part)/nz;
23 }
