231 inline
232 void
233 AMRErrorTag_GRAD(const Box& bx,
234 Array4<const Real> const& dat,
235 Array4<char> const& tag,
236 Real threshold,
237 char tagval)
238 {
239 amrex::ParallelFor(bx,
240 [=] AMREX_GPU_HOST_DEVICE (int i, int j, int k)
241 {
242 auto ax = amrex::Math::abs(dat(i+1,j,k) - dat(i,j,k));
243 ax = amrex::max(ax,amrex::Math::abs(dat(i,j,k) - dat(i-1,j,k)));
244#if AMREX_SPACEDIM == 1
245 if (ax >= threshold) tag(i,j,k) = tagval;
246#else
247 auto ay = amrex::Math::abs(dat(i,j+1,k) - dat(i,j,k));
248 ay = amrex::max(ay,amrex::Math::abs(dat(i,j,k) - dat(i,j-1,k)));
249#if AMREX_SPACEDIM > 2
250 auto az = amrex::Math::abs(dat(i,j,k+1) - dat(i,j,k));
251 az = amrex::max(az,amrex::Math::abs(dat(i,j,k) - dat(i,j,k-1)));
252#endif
253 if (amrex::max(AMREX_D_DECL(ax,ay,az)) >= threshold) {
254 tag(i,j,k) = tagval;
255 }
256#endif
257 });
258 }

304 AMRErrorTag_BOX(const Box& bx,
305 Array4<char> const& tag,
306 const RealBox& tag_rb,
307 const Geometry& geom,
308 char tagval) noexcept
309 {
310 auto plo = geom.ProbLoArray();
311 auto dx = geom.CellSizeArray();
312 class RealBox trb(bx,dx.data(),plo.data());
313 if (tag_rb.intersects(trb))
314 {
315 amrex::ParallelFor(bx,
316 [=] AMREX_GPU_HOST_DEVICE (int i, int j, int k)
317 {
318 GpuArray<Real,AMREX_SPACEDIM> pt = {{AMREX_D_DECL(plo[0]+(Real(i)+Real(0.5))*dx[0],
319 plo[1]+(Real(j)+Real(0.5))*dx[1],
320 plo[2]+(Real(k)+Real(0.5))*dx[2])}};
321 if (tag_rb.contains(pt.data())) {
322 tag(i,j,k) = tagval;
323 }
324 });
