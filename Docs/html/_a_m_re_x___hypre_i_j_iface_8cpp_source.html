93void HypreIJIface::solve(
94 const HypreRealType rel_tol, const HypreRealType abs_tol, const HypreIntType max_iter)
95{
96 // Assuming that Matrix/rhs etc. has been assembled by calling code
97
98 HYPRE_IJMatrixGetObject(m_mat, (void**)&m_parA);
99 HYPRE_IJVectorGetObject(m_rhs, (void**)&m_parRhs);
100 HYPRE_IJVectorGetObject(m_sln, (void**)&m_parSln);
101
102 if (m_write_files) {
103 const std::string matfile = amrex::Concatenate(
104 m_file_prefix + "_A", m_write_counter) + ".out";
105 const std::string rhsfile = amrex::Concatenate(
106 m_file_prefix + "_b", m_write_counter) + ".out";
107 HYPRE_IJMatrixPrint(m_mat, matfile.c_str());
108 HYPRE_IJVectorPrint(m_rhs, rhsfile.c_str());
109 }
110
111 m_solverSetTolPtr(m_solver, rel_tol);
112 m_solverSetMaxIterPtr(m_solver, max_iter);
113 if ((abs_tol > 0.0) && (m_solverSetAbsTolPtr != nullptr))
114 m_solverSetAbsTolPtr(m_solver, abs_tol);
115
116 // setup
117 if (m_need_setup || m_recompute_preconditioner) {
118 if (m_has_preconditioner)
119 m_solverPrecondPtr(
120 m_solver, m_precondSolvePtr, m_precondSetupPtr, m_precond);
121
122 m_solverSetupPtr(m_solver, m_parA, m_parRhs, m_parSln);
123 m_need_setup = false;
124 }
125
126 // solve
127 m_solverSolvePtr(m_solver, m_parA, m_parRhs, m_parSln);
128
129 // diagnostics
130 m_solverNumItersPtr(m_solver, &m_num_iterations);
131 m_solverFinalResidualNormPtr(m_solver, &m_final_res_norm);
132
133 if (m_write_files) {
134 const std::string slnfile = amrex::Concatenate(
135 m_file_prefix + "_x", m_write_counter) + ".out";
136 HYPRE_IJVectorPrint(m_sln, slnfile.c_str());
137
138 // Increment counter if the user has requested output of multiple solves
139 if (!m_overwrite_files) ++m_write_counter;
140 }
141
142 if (m_verbose > 1)
143 amrex::Print() << "HYPRE " << m_solver_name
144 << ": Num. iterations = " << m_num_iterations
145 << "; Relative residual = " << m_final_res_norm
146 << std::endl;
147}
