119void push_momentum_boris(amrex::Real& uxp, amrex::Real& uyp, amrex::Real& uzp, amrex::Real& gaminv,
120 amrex::Real Exp, amrex::Real Eyp, amrex::Real Ezp,
121 amrex::Real Bxp, amrex::Real Byp, amrex::Real Bzp,
122 amrex::Real q, amrex::Real m, amrex::Real dt)
123{
124 amrex::Real cst = q*dt*0.5/m;
125 constexpr amrex::Real clight = 2.99792458e8;
126 constexpr amrex::Real clightisq = 1./(clight*clight);
127
128 uxp += Exp * cst;
129 uyp += Eyp * cst;
130 uzp += Ezp * cst;
131
132 amrex::Real usq = (uxp*uxp + uyp*uyp + uzp*uzp) * clightisq;
133 amrex::Real gaminvtmp = 1.0/std::sqrt(1.0+usq);
134
135 amrex::Real tx = gaminvtmp * Bxp * cst;
136 amrex::Real ty = gaminvtmp * Byp * cst;
137 amrex::Real tz = gaminvtmp * Bzp * cst;
138 amrex::Real tsqi = 2.0/(1.0 + tx*tx + ty*ty + tz*tz);
139 amrex::Real sx = tx*tsqi;
140 amrex::Real sy = ty*tsqi;
141 amrex::Real sz = tz*tsqi;
142 amrex::Real uxppr = uxp + uyp*tz - uzp*ty;
143 amrex::Real uyppr = uyp + uzp*tx - uxp*tz;
144 amrex::Real uzppr = uzp + uxp*ty - uyp*tx;
145 uxp += uyppr*sz - uzppr*sy;
146 uyp += uzppr*sx - uxppr*sz;
147 uzp += uxppr*sy - uyppr*sx;
148
149 uxp += Exp*cst;
150 uyp += Eyp*cst;
151 uzp += Ezp*cst;
152
153 usq = (uxp*uxp + uyp*uyp + uzp*uzp) * clightisq;
154 gaminv = 1.0/std::sqrt(1.0+usq);
155}
