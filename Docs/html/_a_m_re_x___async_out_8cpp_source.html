79 const int nfiles = s_noutfiles;
80 const int nprocs = ParallelDescriptor::NProcs();
81 const int nmaxspots = (nprocs + (nfiles-1)) / nfiles; // max spots per file
82 const int nfull = nfiles + nprocs - nmaxspots*nfiles; // the first nfull files are full
83
84 int ifile, ispot, nspots;
85 if (rank < nfull*nmaxspots) {
86 ifile = rank / nmaxspots;
87 ispot = rank - ifile*nmaxspots;
88 nspots = nmaxspots;
89 } else {
90 int tmpproc = rank-nfull*nmaxspots;
91 ifile = tmpproc/(nmaxspots-1);
92 ispot = tmpproc - ifile*(nmaxspots-1);
93 ifile += nfull;
94 nspots = nmaxspots - 1;
95 }
