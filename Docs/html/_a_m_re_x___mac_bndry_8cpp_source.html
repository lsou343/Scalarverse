32void
33MacBndry::setBndryConds (const BCRec& phys_bc,
34 const IntVect& ratio,
35 int comp)
36{
37 m_phys_bc = phys_bc;
38
39 //
40 // ALL BCLOC VALUES ARE NOW DEFINED AS A LENGTH IN PHYSICAL
41 // DIMENSIONS *RELATIVE* TO THE FACE, NOT IN ABSOLUTE PHYSICAL SPACE
42 //
43 const BoxArray& ba = boxes();
44 const Real* dx = geom.CellSize();
45 const Box& domain = geom.Domain();
46 //
47 // We note that all orientations of the FabSets have the same distribution.
48 // We'll use the low 0 side as the model.
49 //
50 for (FabSetIter fsi(bndry[Orientation(0,Orientation::low)]); fsi.isValid(); ++fsi)
51 {
52 const int i = fsi.index();
53 const Box& grd = ba[i];
54 RealTuple& bloc = bcloc[fsi];
55 Vector< Vector<BoundCond> >& bctag = bcond[fsi];
56
57 for (OrientationIter fi; fi; ++fi)
58 {
59 const Orientation face = fi();
60 const int dir = face.coordDir();
61
62 if (domain[face] == grd[face] && !geom.isPeriodic(dir))
63 {
64 //
65 // All physical bc values are located on face.
66 //
67 const int p_bc = (face.isLow() ? phys_bc.lo(dir) : phys_bc.hi(dir));
68
69 bctag[face][comp] = (p_bc == PhysBCType::outflow)
70 ? AMREX_LO_DIRICHLET : AMREX_LO_NEUMANN;
71 bloc[face] = 0;
72 }
73 else
74 {
75 //
76 // Internal bndry.
77 //
78 const Real delta = dx[dir]*ratio[dir];
79
80 bctag[face][comp] = AMREX_LO_DIRICHLET;
81 bloc[face] = 0.5*delta;
82 }
83 }
84 }
85}
