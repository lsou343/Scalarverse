176void
177Nyx::compute_average_density ()
178{
179 amrex::Gpu::LaunchSafeGuard lsg(true);
180 int finest_level = parent->finestLevel();
181 Real time = state[State_Type].curTime();
182 const Geometry& crse_geom = parent->Geom(0);
183
184 // This is a static in the Nyx class
185 average_gas_density = 0;
186 average_dm_density = 0;
187 average_neutr_density = 0;
188 average_total_density = 0;
189#ifdef FDM
190 average_ax_density = 0;
191#endif
192 // Add up the baryonic density
193 if (do_hydro == 1)
194 {
195 for (int lev = 0; lev <= finest_level; lev++)
196 {
197 Nyx& nyx_lev = get_level(lev);
198 average_gas_density += nyx_lev.vol_weight_sum("density", time, true);
199 }
200 }
201
202#ifdef GRAVITY
203#ifdef FDM
204 for (int lev = 0; lev <= finest_level; lev++)
205 {
206 Nyx& nyx_lev = get_level(lev);
207 average_ax_density += nyx_lev.vol_weight_sum("AxDens", time,true);
208 }
209#endif
210 // Define the dark matter density on all levels.
211 if (Nyx::theDMPC())
212 {
213 Vector<std::unique_ptr<MultiFab> > particle_mf;
214 Nyx::theDMPC()->AssignDensity(particle_mf);
215
216 // Note that we don't need to call the average_down routine because the
217 // vol_weight_sum routine zeroes out the data under the finer grids
218
219 // Add up the dark matter density; here masked is true because we don't want
220 // to double count the coarse values under fine grids.
221 for (int lev = 0; lev <= finest_level; lev++)
222 {
223 Nyx& nyx_lev = get_level(lev);
224 average_dm_density += nyx_lev.vol_weight_sum(*particle_mf[lev],true);
225 }
226 }
227#ifdef NEUTRINO_PARTICLES
228 if (Nyx::theNPC())
229 {
230 Vector<std::unique_ptr<MultiFab> > particle_mf;
231 Nyx::theNPC()->AssignDensity(particle_mf);
232
233 // Note that we don't need to call the average_down routine because the
234 // vol_weight_sum routine zeroes out the data under the finer grids
235
236 // Add up the neutrino density
237 for (int lev = 0; lev <= finest_level; lev++)
238 {
239 Nyx& nyx_lev = get_level(lev);
240 average_neutr_density += nyx_lev.vol_weight_sum(*particle_mf[lev],true);
241 }
242 }
243#endif
244#endif
245
246 // Divide by physical volume of domain.
247 if (do_hydro == 1)
248 average_gas_density /= crse_geom.ProbSize();
249 average_dm_density /= crse_geom.ProbSize();
250 average_neutr_density /= crse_geom.ProbSize();
251
252 // Define the total density = gas density + dark matter density
253 if (do_hydro == 1)
254 {
255 average_total_density = average_gas_density + average_dm_density + average_neutr_density;
256 }
257 else
258 {
259 average_total_density = average_dm_density + average_neutr_density;
260 }
261
262#ifdef FDM
263 average_ax_density /= crse_geom.ProbSize();
264 average_total_density += average_ax_density;
265#endif
266
267 if (verbose > 0 && ParallelDescriptor::IOProcessor())
268 {
269 if (do_hydro == 1)
270 {
271 std::cout << "Average gas density " << average_gas_density << '\n';
272 }
273 std::cout << "Average dm density " << average_dm_density << '\n';
274#ifdef NEUTRINO_PARTICLES
275 std::cout << "Average neutrino density " << average_neutr_density << '\n';
276#endif
277#ifdef FDM
278 std::cout << "Average fdm density " << average_ax_density << '\n';
279#endif
280 std::cout << "Average total density " << average_total_density << '\n';
281 }
282 amrex::Gpu::synchronize();
