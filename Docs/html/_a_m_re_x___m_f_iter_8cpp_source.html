536Real*
537MFIter::add_reduce_value(Real* val, MFReducer r)
538{
539
540 if (Gpu::inLaunchRegion()) {
541
542 reducer = r;
543
544 // For the reduce lists, the outer vector is length
545 // (endIndex - beginIndex) and has a contribution from
546 // every tile. The inner vector is all of the individual
547 // variables to reduce over. While the former is a known
548 // quantity, the latter is not, so we'll push_back for
549 // each new quantity to reduce. Since the elements will
550 // be added in the same order in every MFIter iteration,
551 // we can access them in the same order in each entry
552 // of the reduce_list.
553
554 if (real_reduce_list.empty()) {
555 real_reduce_list.resize(length());
556 real_device_reduce_list.resize(length());
557 }
558
559 // Store the current value of the data.
560
561 Real reduce_val = *val;
562 real_reduce_list[currentIndex].push_back(reduce_val);
563
564 // Create a device copy of the data to update within
565 // the kernel.
566
567 Real* dval = static_cast<Real*>(amrex::The_MFIter_Arena()->alloc(sizeof(Real)));
568 real_device_reduce_list[currentIndex].push_back(dval);
569
570 // Queue up a host to device copy of the input data,
571 // so that we start from the correct value.
572
573 const int list_idx = real_reduce_list[currentIndex].size() - 1;
574
575 Gpu::htod_memcpy_async(real_device_reduce_list[currentIndex][list_idx],
576 &real_reduce_list[currentIndex][list_idx],
577 sizeof(Real));
578
579 // If we haven't already, store the address to the variable
580 // we will update at the end.
581
582 if (real_reduce_val.size() < real_reduce_list[currentIndex].size())
583 real_reduce_val.push_back(val);
584
585 return dval;
586
587 }
588 else {
589
590 return val;
591
592 }

598// Reduce over the values in the list.
599void
600MFIter::reduce()
601{
602
603 // Do nothing if we're not currently executing on the device.
604
605 if (Gpu::notInLaunchRegion()) return;
606
607 // Do nothing if we don't have enough values to reduce on.
608
609 if (real_reduce_list.empty()) return;
610
611 // Assume that the number of reductions we want is fixed
612 // in each vector, and just grab the number from the first
613 // entry.
614
615 const int num_reductions = real_reduce_list[0].size();
616
617 BL_ASSERT(real_reduce_list.size() == length());
618
619 for (int j = 0; j < num_reductions; ++j) {
620
621 Real result;
622
623 if (reducer == MFReducer::SUM) result = 0.0e0;
624 if (reducer == MFReducer::MIN) result = 1.e200;
625 if (reducer == MFReducer::MAX) result = -1.e200;
626
627 for (int i = 0; i < real_reduce_list.size(); ++i) {
628
629 // Double check our assumption from above.
630
631 BL_ASSERT(real_reduce_list[i].size() == num_reductions);
632
633 if (reducer == MFReducer::SUM) {
634 result += real_reduce_list[i][j];
635 }
636 else if (reducer == MFReducer::MIN) {
637 result = std::min(result, real_reduce_list[i][j]);
638 }
639 else if (reducer == MFReducer::MAX) {
640 result = std::max(result, real_reduce_list[i][j]);
641 }
642
643 }
644
645 *(real_reduce_val[j]) = result;
646
647 }
648
649}

659void
660MFGhostIter::Initialize ()
661{
662 int rit = 0;
663 int nworkers = 1;
664#ifdef BL_USE_TEAM
665 if (ParallelDescriptor::TeamSize() > 1) {
666 rit = ParallelDescriptor::MyRankInTeam();
667 nworkers = ParallelDescriptor::TeamSize();
668 }
669#endif
670
671 int tid = OpenMP::get_thread_num();
672 int nthreads = OpenMP::get_num_threads();
673
674 int npes = nworkers*nthreads;
675 int pid = rit*nthreads+tid;
676
677 BoxList alltiles;
678 Vector<int> allindex;
679 Vector<int> alllocalindex;
680
681 for (int i=0; i < fabArray.IndexArray().size(); ++i) {
682 int K = fabArray.IndexArray()[i];
683 const Box& vbx = fabArray.box(K);
684 const Box& fbx = fabArray.fabbox(K);
685
686 const BoxList& diff = amrex::boxDiff(fbx, vbx);
687
688 for (BoxList::const_iterator bli = diff.begin(); bli != diff.end(); ++bli) {
689 BoxList tiles(*bli, FabArrayBase::mfghostiter_tile_size);
690 int nt = tiles.size();
691 for (int it=0; it<nt; ++it) {
692 allindex.push_back(K);
693 alllocalindex.push_back(i);
694 }
695 alltiles.catenate(tiles);
696 }
697 }
698
699 int n_tot_tiles = alltiles.size();
700 int navg = n_tot_tiles / npes;
701 int nleft = n_tot_tiles - navg*npes;
702 int ntiles = navg;
703 if (pid < nleft) ntiles++;
704
705 // how many tiles should we skip?
706 int nskip = pid*navg + std::min(pid,nleft);
707 BoxList::const_iterator bli = alltiles.begin();
708 for (int i=0; i<nskip; ++i) ++bli;
709
710 lta.indexMap.reserve(ntiles);
711 lta.localIndexMap.reserve(ntiles);
712 lta.tileArray.reserve(ntiles);
713
714 for (int i=0; i<ntiles; ++i) {
715 lta.indexMap.push_back(allindex[i+nskip]);
716 lta.localIndexMap.push_back(alllocalindex[i+nskip]);
717 lta.tileArray.push_back(*bli++);
718 }
719
720 currentIndex = beginIndex = 0;
721 endIndex = lta.indexMap.size();
722
723 lta.nuse = 0;
724 index_map = &(lta.indexMap);
725 local_index_map = &(lta.localIndexMap);
726 tile_array = &(lta.tileArray);
727}
