3"""
4Search the Fortran source code for procedures marked with !$gpu in their specification part:
5
6 subroutine sub(a)
7
8 integer :: a
9
10 !$gpu
11
12 ...
13
14 end subroutine sub
15
16For each one, generate a device copy by prepending attributes(device) prior to the procedure statement:
17
18 attributes(device) subroutine sub_device(a)
19
20 integer :: a
21
22 !$gpu
23
24 ...
25
26 end subroutine sub_device
27
28The host copy is retained with the same name, unless no_host_version = True. The suffix can be modified
29using the argument device_suffix. Combining these options (no_host_version = True, device_suffix = "")
30can be used to implement a device-only implementation.
31
32Limitations:
33
34- At present, this detects function imports in use statements if there is a use/only
35 statement that imports only functions and ends in "! function". Cannot span multiple lines.
36
37- In general use/only statements that continue to multiple lines via "&" are not handled properly
38
39- Unrestricted private statements are disallowed, since we do not generate public statements for
40 the newly created device functions. Doing this generation would require us to keep track of which
41 ifdefs should be applied, since the IBM compiler won't allow you to generate a public statement
42 if the corresponding function is ifdef'd out.
43"""

64def case_insensitive_replace(s, old, new, device_suffix, count=-1, on_condition=None):
65 # Does a case-insensitive replacement of string old
66 # by string new in the string s.
67 # Does count number of replacements.
68 if not old or old==new:
69 # We were passed the empty string for old.
70 # or we're replacing old with itself.
71 return s
72 n_replaced = 0
73 ridx_start = 0
74 old = old.lower()
75 new = new.lower()
76 while(n_replaced < count or count == -1):
77 if on_condition:
78 istart, iend, found = on_condition(s[ridx_start:])
79 if found:
80 istart += ridx_start
81 iend += ridx_start
82 else:
83 break
84 else:
85 try:
86 istart = s.lower().index(old, ridx_start)
87 except ValueError:
88 break
89 else:
90 iend = istart + len(old)
91 if new.lower() == old.lower() + device_suffix:
92 actual_new = s[istart:iend] + device_suffix
93 else:
94 actual_new = new
95 s = s[:istart] + actual_new + s[iend:]
96 ridx_start = istart + len(actual_new)
97 n_replaced = n_replaced + 1
98 return s

129def append_device_to_line(line, subs, device_suffix):
130
131 new_line = line
132
133 # Look for statements of the form use module, only: In this case
134 # we want to replace everything after the : but not before it.
135
136 if 'use ' in line.lower() and 'only' in line.lower() and ':' in line:
137
138 # Strip off any comments at the end of the line
139 mcomment = re_detect_endcomment.search(line)
140 comment = ''
141 if mcomment:
142 comment = line[mcomment.start():]
143 line = line[:mcomment.start()]
144
145 sline = line.split(':')
146
147 new_line = sline[0] + ': '
148
149 for var in sline[1].split(','):
150
151 found_sub = False
152 for sub_name in subs:
153 if sub_name.lower() == var.strip().lower():
154 new_line += var.strip().lower() + device_suffix
155 found_sub = True
156 break
157
158 if not found_sub:
159 new_line += var
160
161 if var != sline[1].split(',')[-1]:
162 new_line += ', '
163
164 new_line = " ".join([new_line, comment])
165
166 else:
167
168 for sub_name in subs:
169 re_subroutine_call = re.compile('(call ' + sub_name.lower() + '\s|\‍()')
170 re_subroutine_device_call = re.compile('call ' + sub_name.lower() + device_suffix + '\s|\‍(')
171 m_subroutine_call = re_subroutine_call.search(new_line.lower())
172 m_subroutine_device_call = re_subroutine_device_call.search(new_line.lower())
173 if m_subroutine_call and not m_subroutine_device_call:
174 actual_new_line = new_line[:m_subroutine_call.start()]
175 actual_new_line = actual_new_line + m_subroutine_call.group(1).replace(sub_name.lower(), sub_name + device_suffix)
176 actual_new_line = actual_new_line + new_line[m_subroutine_call.end():]
177 new_line = actual_new_line
178 break
179
180 if sub_name.lower() in line.lower().split() and sub_name.lower() + device_suffix not in line.lower().split():
181 new_line = case_insensitive_replace(new_line, sub_name, sub_name + device_suffix, device_suffix,
182 on_condition=lambda x: get_replace_procedure(x, sub_name.lower()))
183
184 elif sub_name.lower() in line.lower() and sub_name.lower() + device_suffix not in line.lower():
185 # Catch function calls here.
186 # Make sure "bar(" is not any of "foobar(" or "foo % bar(" or "foo_bar("
187 old_fun = sub_name.lower() + '('
188 new_fun = sub_name.lower() + device_suffix + '('
189 re_old_fun = re.compile(sub_name.lower() + " *\‍(")
190 re_not_fun = re.compile("((\w)|(% *))" + sub_name.lower() + " *\‍(")
191 actual_new_line = ''
192 while(new_line):
193 m_old_fun = re_old_fun.search(new_line.lower())
194 if m_old_fun:
195 m_not_fun = re_not_fun.search(new_line.lower())
196 if m_not_fun and m_old_fun.end() == m_not_fun.end():
197 # reject this replacement, it's not a function call
198 actual_new_line = actual_new_line + new_line[:m_not_fun.end()]
199 new_line = new_line[m_not_fun.end():]
200 else:
201 # do the replacement, it's a function call
202 substring = case_insensitive_replace(new_line[:m_old_fun.end()], old_fun, new_fun, device_suffix)
203 actual_new_line = actual_new_line + substring
204 new_line = new_line[m_old_fun.end():]
205 else:
206 actual_new_line = actual_new_line + new_line
207 break
208 new_line = actual_new_line
209 # continue because there could be multiple functions calls on the same line
210 # or functions that satisfy the elif but not necessarily the regex criteria.
211 continue
212
213 return new_line

216def get_function_uses(line, device_suffix):
217 # Device functions in another module will

234def append_device(procedure, device_suffix):
235
236 new_procedure = procedure
237
238 called_subs = []
239 for line in procedure.split('\n'):
240 called_subs = called_subs + get_function_uses(line, device_suffix)
241 if "call " in line.lower():
242 try:
243 ex_index = line.index('!')
244 except:
245 pass
246 else:
247 if ex_index > line.lower().index('call'):
248 pass
249 else:
250 continue
251 sub_name = line.lower().split('call ')[1].split('(')[0].strip()
252 if sub_name != "syncthreads":
253 if not device_suffix in sub_name and not sub_name.lower() in lowerlist(called_subs):
254 called_subs.append(sub_name)
255
256 new_procedure = [append_device_to_line(line, called_subs, device_suffix) for line in procedure.split('\n')]
257 new_procedure = '\n'.join(new_procedure)

267def get_replace_procedure(line, pname):
268 m = re.search('(?P<head>(\w|(% *))*)(?P<proc>' + pname.lower() + ')(?P<tail>((\w| *%| *=)*))', line.lower())
269 istart = -1
270 iend = -1
271 found = False
272 if m and m.group('proc') and not m.group('head') and not m.group('tail'):
273 found = True
274 istart = m.start('proc')
275 iend = m.end('proc')
276 return istart, iend, found

279def create_device_version(procedure, device_suffix):
280
281 procedure_name = get_procedure_name(procedure).lower()
282
283 procedure_type = get_procedure_type(procedure)
284
285 # Append _device to the name before writing it to the file.
286 # We want to cover this both in the subroutine / end subroutine
287 # statement and in the bind(c) statement.
288
289 device_procedure = []
290
291 re_subroutine_dec = re.compile(procedure_type + ' +' + procedure_name)
292
293 for line in procedure.split('\n'):
294 m_subroutine_dec = re_subroutine_dec.search(line.lower())
295 if (m_subroutine_dec or "bind" in line.lower()) and not line.strip().startswith('!'):
296 new_line = case_insensitive_replace(line, procedure_name, procedure_name + device_suffix, device_suffix, on_condition=lambda x: get_replace_procedure(x, procedure_name))
297 else:
298 new_line = line
299 device_procedure.append(new_line)
300
301 device_procedure = '\n'.join(device_procedure)
302
303 # Make sure any procedure calls append _device to the procedure we are
304 # calling. Note that this assumes that everything we are calling either
305 # already has _device appended to it, or is being generated through this
306 # script. We'll guard against known CUDA intrinsics.
307
308 device_procedure = append_device(device_procedure, device_suffix)
309
310 return device_procedure

314def create_host_version(device_procedure, device_suffix):
315
316 # Create a host version of the procedure as well.
317
318 host_procedure = device_procedure.replace(device_suffix,"").replace("attributes(device)", "attributes(host)")
319 assert(host_procedure.count(device_suffix) == 0)
320
321 # Some functions will have device code that is wrapped in #ifdef AMREX_USE_GPU_PRAGMA or
322 # #if defined(AMREX_USE_GPU_PRAGMA), and host code otherwise. To deal with this, we'll undefine
323 # AMREX_USE_GPU_PRAGMA inside the host version of the function.
324
325 host_procedure = "#undef AMREX_USE_GPU_PRAGMA\n" + host_procedure + "\n#define AMREX_USE_GPU_PRAGMA\n"
326
327 return host_procedure

331def update_fortran_procedures(ffile, device_suffix, no_host_version):
332 """For procedures marked up with !$gpu, generate a copy marked
333 with attributes(device), and create a host copy as well, to
334 avoid the CUF restriction on attributes(host, device) In order
335 to avoid name collisions, we'll name one with a _device suffix."""
336
337 gpu_targets = []
338
339 # open the input Fortran file
340 try:
341 fin = open(ffile, "r")
342 except IOError:
343 sys.exit("Cannot open Fortran file {} for reading".format(ffile))
344
345 # read in the file, close it, and then reopen it for writing
346 lines = fin.readlines()
347
348 fin.close()
349
350 try:
351 fout = open(ffile, "w")
352 except IOError:
353 sys.exit("Cannot open Fortran file {} for writing".format(ffile))
354
355 # loop through the file and look for all procedures
356
357 found_procedure = False
358 device_procedure = ""
359 host_procedure = ""
360 procedure = ""
361
362 for line in lines:
363
364 if ("subroutine" in line.lower() or "function" in line.lower()) and not found_procedure and not "module" in line.lower():
365
366 # Check to make sure we are not in a comment.
367
368 if '!' not in line:
369 found_procedure = True
370 else:
371 if 'subroutine' in line.lower() and line.index('!') > line.lower().index('subroutine'):
372 found_procedure = True
373 elif 'function' in line.lower() and line.index('!') > line.lower().index('function'):
374 found_procedure = True
375
376 if ("end subroutine" in line.lower() or "end function" in line.lower()):
377
378 procedure += line
379
380 # Now if the procedure contains !$gpu, prepend
381 # attributes(device) to the procedure.
382
383 if "!$gpu" in procedure:
384
385 # Some code may contain a legacy AMREX_DEVICE statement.
386
387 if "AMREX_DEVICE" in procedure:
388 procedure = procedure.replace("AMREX_DEVICE", "attributes(device)", 1)
389 else:
390 if get_procedure_type(procedure) == "subroutine":
391 procedure = case_insensitive_replace(procedure, "subroutine", "attributes(device) subroutine", device_suffix, 1)
392 elif get_procedure_type(procedure) == "function":
393 procedure = case_insensitive_replace(procedure, "function", "attributes(device) function", device_suffix, 1)
394 else:
395 sys.exit("Error - Procedure is not a subroutine or function!")
396
397 # Capture the name of the procedure and add it to our targets list.
398
399 if "subroutine" in line.lower() or "function" in line.lower():
400
401 procedure_type = get_procedure_type(procedure)
402
403 procedure_name = get_procedure_name(procedure)
404
405 if not procedure_name.lower() in gpu_targets:
406 gpu_targets.append(procedure_name)
407
408 # Create device and host versions of the procedure.
409 device_procedure = create_device_version(procedure, device_suffix)
410 if not no_host_version:
411 host_procedure = create_host_version(device_procedure, device_suffix)
412
413 # Blank out the original procedure; there is nothing left to do with it.
414
415 procedure = ""
416
417
418
419 elif "AMREX_LAUNCH" in procedure:
420
421 # Some code may contain a legacy AMREX_LAUNCH statement.
422 # In this case all we want to do is update the calls
423 # inside to append_device, but no other processing
424 # needs to be done.
425
426 procedure = append_device(procedure, device_suffix)
427
428
429
430 # Now write out whatever we have generated.
431
432 if host_procedure:
433 fout.write(host_procedure)
434
435 if device_procedure:
436 fout.write(device_procedure)
437
438 if procedure:
439 fout.write(procedure)
440
441 # Clear everything out for the next procedure.
442
443 device_procedure = ""
444 host_procedure = ""
445 procedure = ""
446 line = ""
447 found_procedure = False
448
449 if found_procedure:
450 procedure += line
451 else:
452 fout.write(line)
453
454 fout.close()
455
456
457
458 # Now handle any remaining logistics that required a full pass through.
459
460 fin = open(ffile, 'r')
461 lines = fin.readlines()
462 fin.close()
463
464 fout = open(ffile, 'w')
465
466 in_device_subroutine = False
467
468 subroutine_imported_functions = []
469
470 for line in lines:
471
472 if in_device_subroutine:
473 # Detect any used functions from other modules
474 subroutine_imported_functions = subroutine_imported_functions + get_function_uses(line, device_suffix)
475
476 # If there are any calls inside a device procedure to
477 # a known device procedure but it is missing the _device
478 # suffix, add that now. The most likely case will be
479 # calls to functions, where it wasn't possible in the initial
480 # processing to know what is a function (versus a variable).
481
482 for target in gpu_targets:
483 if "end function " + target.lower() + device_suffix in line.lower() or "end subroutine " + target.lower() + device_suffix in line.lower():
484 in_device_subroutine = False
485 subroutine_imported_functions = []
486 break
487
488 elif "function " + target.lower() + device_suffix in line.lower() or "subroutine " + target.lower() + device_suffix in line.lower():
489 in_device_subroutine = True
490 break
491
492 if in_device_subroutine:
493 line = append_device_to_line(line, set(gpu_targets + subroutine_imported_functions), device_suffix)
494
495 fout.write(line)
496
497 fout.close()
