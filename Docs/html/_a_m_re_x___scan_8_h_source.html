201#if (__SYCL_COMPILER_VERSION <= 20200827)
202 sycl::intel::sub_group const& sg = gh.item.get_sub_group();
203#else
204 sycl::ONEAPI::sub_group const& sg = gh.item.get_sub_group();
205#endif
206 int lane = sg.get_local_id()[0];
207 int warp = sg.get_group_id()[0];
208#if (__SYCL_COMPILER_VERSION <= 20200715)
209 int nwarps = sg.get_group_range();
210#else

212#endif

218 T* shared = (T*)(gh.local);
219 T* shared2 = shared + Gpu::Device::warp_size;

221 // First of all, get block virtual id. We must do this to
222 // avoid deadlock because blocks may be launched in any order.
223 // Anywhere in this function, we should not use blockIdx.
224 int virtual_block_id = 0;
225 if (gridDimx > 1) {
226 int& virtual_block_id_shared = *((int*)(shared2+nwarps));
227 if (threadIdxx == 0) {
228 unsigned int bid = Gpu::Atomic::Inc<sycl::access::address_space::global_space>
229 (virtual_block_id_p, gridDimx);
230 virtual_block_id_shared = bid;
231 }
232 gh.item.barrier(sycl::access::fence_space::local_space);
233 virtual_block_id = virtual_block_id_shared;
234 }
235
236 // Each block processes [ibegin,iend).
237 N ibegin = nelms_per_block * virtual_block_id;
238 N iend = amrex::min(static_cast<N>(ibegin+nelms_per_block), n);
239 BlockStatusT& block_status = block_status_p[virtual_block_id];
240
241 //
242 // The overall algorithm is based on "Single-pass Parallel
243 // Prefix Scan with Decoupled Look-back" by D. Merrill &
244 // M. Garland.
245 //
246
247 // Each block is responsible for nchunks chunks of data,
248 // where each chunk has blockDim.x elements, one for each
249 // thread in the block.
250 T sum_prev_chunk = 0; // inclusive sum from previous chunks.
251 T tmp_out[nchunks]; // block-wide inclusive sum for chunks
252 for (int ichunk = 0; ichunk < nchunks; ++ichunk) {
253 N offset = ibegin + ichunk*blockDimx;
254 if (offset >= iend) break;
255
256 offset += threadIdxx;
257 T x0 = (offset < iend) ? fin(offset) : 0;
258 if (type == Type::exclusive && offset == n-1) {
259 *totalsum_p += x0;
260 }
261 T x = x0;
262 // Scan within a warp
263 for (int i = 1; i <= Gpu::Device::warp_size; i *= 2) {
264 T s = sg.shuffle_up(x, i);
265 if (lane >= i) x += s;
266 }
267
268 // x now holds the inclusive sum within the warp. The
269 // last thread in each warp holds the inclusive sum of
270 // this warp. We will store it in shared memory.
271 if (lane == Gpu::Device::warp_size - 1) {
272 shared[warp] = x;
273 }
274
275 gh.item.barrier(sycl::access::fence_space::local_space);
276
277 // The first warp will do scan on the warp sums for the
278 // whole block.
279 if (warp == 0) {
280 T y = (lane < nwarps) ? shared[lane] : 0;
281 for (int i = 1; i <= Gpu::Device::warp_size; i *= 2) {
282 T s = sg.shuffle_up(y, i);
283 if (lane >= i) y += s;
284 }
285
286 if (lane < nwarps) shared2[lane] = y;
287 }
288
289 gh.item.barrier(sycl::access::fence_space::local_space);
290
291 // shared[0:nwarps) holds the inclusive sum of warp sums.
292
293 // Also note x still holds the inclusive sum within the
294 // warp. Given these two, we can compute the inclusive
295 // sum within this chunk.
296 T sum_prev_warp = (warp == 0) ? 0 : shared2[warp-1];
297 tmp_out[ichunk] = sum_prev_warp + sum_prev_chunk +
298 ((type == Type::inclusive) ? x : x-x0);
299 sum_prev_chunk += shared2[nwarps-1];
300 }
301
302 // sum_prev_chunk now holds the sum of the whole block.
303 if (threadIdxx == 0 && gridDimx > 1) {
304 block_status.write((virtual_block_id == 0) ? 'p' : 'a',
305 sum_prev_chunk);
306 }
307
308 if (virtual_block_id == 0) {
309 for (int ichunk = 0; ichunk < nchunks; ++ichunk) {
310 N offset = ibegin + ichunk*blockDimx + threadIdxx;
311 if (offset >= iend) break;
312 fout(offset, tmp_out[ichunk]);
313 if (offset == n-1) {
314 *totalsum_p += tmp_out[ichunk];
315 }
316 }
317 } else if (virtual_block_id > 0) {
318
319 if (warp == 0) {
320 T exclusive_prefix = 0;
321 BlockStatusT volatile* pbs = block_status_p;
322 for (int iblock0 = virtual_block_id-1; iblock0 >= 0; iblock0 -= Gpu::Device::warp_size)
323 {
324 int iblock = iblock0-lane;
325 detail::STVA<T> stva{'p', 0};
326 if (iblock >= 0) {
327 stva = pbs[iblock].wait(gh.item);
328 }
329
330 T x = stva.value;
331
332 // implement our own __ballot
333 unsigned status_bf = (stva.status == 'p') ? (0x1u << lane) : 0;
334 for (int i = 1; i < Gpu::Device::warp_size; i *= 2) {
335 status_bf |= sg.shuffle_xor(status_bf, i);
336 }
337
338 bool stop_lookback = status_bf & 0x1u;
339 if (stop_lookback == false) {
340 if (status_bf != 0) {
341 T y = x;
342 if (lane > 0) x = 0;
343 unsigned int bit_mask = 0x1u;
344 for (int i = 1; i < Gpu::Device::warp_size; ++i) {
345 bit_mask <<= 1;
346 if (i == lane) x = y;
347 if (status_bf & bit_mask) {
348 stop_lookback = true;
349 break;
350 }
351 }
352 }
353
354 for (int i = Gpu::Device::warp_size/2; i > 0; i /= 2) {
355 x += sg.shuffle_down(x,i);
356 }
357 }
358
359 if (lane == 0) { exclusive_prefix += x; }
360 if (stop_lookback) break;
361 }
362
363 if (lane == 0) {
364 block_status.write('p', block_status.get_aggregate() + exclusive_prefix);
365 shared[0] = exclusive_prefix;
366 }
367 }
368
369 gh.item.barrier(sycl::access::fence_space::local_space);
370
371 T exclusive_prefix = shared[0];
372
373 for (int ichunk = 0; ichunk < nchunks; ++ichunk) {
374 N offset = ibegin + ichunk*blockDimx + threadIdxx;
375 if (offset >= iend) break;
376 T t = tmp_out[ichunk] + exclusive_prefix;
377 fout(offset, t);
378 if (offset == n-1) {
379 *totalsum_p += t;
380 }
381 }
382 }
383 });
384
385 T totalsum;
386 Gpu::dtoh_memcpy_async(&totalsum, totalsum_p, sizeof(T));
387 Gpu::streamSynchronize();
388
389 The_Device_Arena()->free(dp);
390
391 AMREX_GPU_ERROR_CHECK();
392
393 return totalsum;
394}
