21 We use an MLABecLaplacian operator:
22
23 (ascalar*acoef - bscalar div bcoef grad) phi = RHS
24
25 for an implicit discretization of the heat equation
26
27 (I - div dt grad) phi^{n+1} = phi^n

54 for (int n = 0; n < phi_old.nComp(); ++n)

58 // lo-side BCs
59 if (bc[n].lo(idim) == BCType::int_dir) {
60 bc_lo[idim] = LinOpBCType::Periodic;
61 }
62 else if (bc[n].lo(idim) == BCType::foextrap) {

107 // then we will in face_bcoef MultiFabs and load them into the solver.

112 IntVect::TheDimensionVector(idim));
113 face_bcoef[idim].define(ba, acoef.DistributionMap(), 1, 0);
114 face_bcoef[idim].setVal(dt);
115 }
116 mlabec.setBCoeffs(0, amrex::GetArrOfConstPtrs(face_bcoef));

131 // relative and absolute tolerances for linear solve
132 const Real tol_rel = 1.e-10;
133 const Real tol_abs = 0.0;
134
135 // Solve linear system
136 mlmg.solve({&phi_new}, {&phi_old}, tol_rel, tol_abs);
137}
