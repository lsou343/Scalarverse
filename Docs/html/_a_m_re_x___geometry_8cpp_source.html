109void
110Geometry::Setup (const RealBox* rb, int coord, int const* isper) noexcept
111{
112 Geometry* gg = AMReX::top()->getDefaultGeometry();
113
114 if (gg->ok) return;
115
116 BL_ASSERT(!OpenMP::in_parallel());
117
118 ParmParse pp("geometry");
119
120 if (coord >=0 && coord <= 2) {
121 gg->SetCoord( (CoordType) coord );
122 } else {
123 coord = 0; // default is Cartesian coordinates
124 pp.query("coord_sys",coord);
125 gg->SetCoord( (CoordType) coord );
126 }
127
128 if (rb == nullptr) {
129 Vector<Real> prob_lo(AMREX_SPACEDIM);
130 Vector<Real> prob_hi(AMREX_SPACEDIM);
131 pp.getarr("prob_lo",prob_lo,0,AMREX_SPACEDIM);
132 BL_ASSERT(prob_lo.size() == AMREX_SPACEDIM);
133 pp.getarr("prob_hi",prob_hi,0,AMREX_SPACEDIM);
134 BL_ASSERT(prob_hi.size() == AMREX_SPACEDIM);
135 gg->prob_domain.setLo(prob_lo);
136 gg->prob_domain.setHi(prob_hi);
137 gg->SetOffset(prob_lo.data());
138 } else {
139 gg->prob_domain.setLo(rb->lo());
140 gg->prob_domain.setHi(rb->hi());
141 gg->SetOffset(rb->lo());
142 }
143
144 //
145 // Now get periodicity info.
146 //
147 if (isper == nullptr)
148 {
149 Vector<int> is_per(AMREX_SPACEDIM,0);
150 pp.queryarr("is_periodic",is_per,0,AMREX_SPACEDIM);
151 for (int n = 0; n < AMREX_SPACEDIM; n++) {
152 gg->is_periodic[n] = is_per[n];
153 }
154 }
155 else
156 {
157 for (int n = 0; n < AMREX_SPACEDIM; n++) {
158 gg->is_periodic[n] = isper[n];
159 }
160 }
161
162 gg->ok = true;
163}

276void
277Geometry::periodicShift (const Box& target,
278 const Box& src,
279 Vector<IntVect>& out) const noexcept
280{
281 out.resize(0);
282
283 Box locsrc(src);
284
285 int nist,njst,nkst;
286 int niend,njend,nkend;
287 nist = njst = nkst = 0;
288 niend = njend = nkend = 0;
289 AMREX_D_TERM( nist , =njst , =nkst ) = -1;
290 AMREX_D_TERM( niend , =njend , =nkend ) = +1;
291
292 int ri,rj,rk;
293 for (ri = nist; ri <= niend; ri++)
294 {
295 if (ri != 0 && !is_periodic[0])
296 continue;
297 if (ri != 0 && is_periodic[0])
298 locsrc.shift(0,ri*domain.length(0));
299
300 for (rj = njst; rj <= njend; rj++)
301 {
302 if (rj != 0
303#if (AMREX_SPACEDIM > 1)
304 && !is_periodic[1]
305#endif
306 )
307 {
308 continue;
309 }
310 if (rj != 0
311#if (AMREX_SPACEDIM > 1)
312 && is_periodic[1]
313#endif
314 )
315 {
316 locsrc.shift(1,rj*domain.length(1));
317 }
318
319 for (rk = nkst; rk <= nkend; rk++)
320 {
321 if (rk!=0
322#if (AMREX_SPACEDIM == 3)
323 && !is_periodic[2]
324#endif
325 )
326 {
327 continue;
328 }
329 if (rk!=0
330#if (AMREX_SPACEDIM == 3)
331 && is_periodic[2]
332#endif
333 )
334 {
335 locsrc.shift(2,rk*domain.length(2));
336 }
337
338 if (ri == 0 && rj == 0 && rk == 0)
339 continue;
340 //
341 // If losrc intersects target, then add to "out".
342 //
343 if (target.intersects(locsrc))
344 {
345 out.push_back(IntVect(AMREX_D_DECL(ri*domain.length(0),
346 rj*domain.length(1),
347 rk*domain.length(2))));
348 }
349 if (rk != 0
350#if (AMREX_SPACEDIM == 3)
351 && is_periodic[2]
352#endif
353 )
354 {
355 locsrc.shift(2,-rk*domain.length(2));
356 }
357 }
358 if (rj != 0
359#if (AMREX_SPACEDIM > 1)
360 && is_periodic[1]
361#endif
362 )
363 {
364 locsrc.shift(1,-rj*domain.length(1));
365 }
366 }
367 if (ri != 0 && is_periodic[0])
368 locsrc.shift(0,-ri*domain.length(0));
369 }

396void
397Geometry::computeRoundoffDomain ()
398{
399 for (int k = 0; k < AMREX_SPACEDIM; k++)
400 {
401 offset[k] = prob_domain.lo(k);
402 dx[k] = prob_domain.length(k)/(Real(domain.length(k)));
403 inv_dx[k] = 1.0/dx[k];
404 }
405
406 roundoff_domain = prob_domain;
407 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
408 {
409 int ilo = Domain().smallEnd(idim);
410 int ihi = Domain().bigEnd(idim);
411 Real plo = ProbLo(idim);
412 Real phi = ProbHi(idim);
413 Real idx = InvCellSize(idim);
414 Real deltax = CellSize(idim);
415
416#ifdef AMREX_SINGLE_PRECISION_PARTICLES
417 Real tolerance = std::max(1.e-4*deltax, 1.e-10*phi);
418#else
419 Real tolerance = std::max(1.e-8*deltax, 1.e-14*phi);
420#endif
421 // bisect the point at which the cell no longer maps to inside the domain
422 Real lo = static_cast<Real>(phi) - Real(0.5)*static_cast<Real>(deltax);
423 Real hi = static_cast<Real>(phi) + Real(0.5)*static_cast<Real>(deltax);
424
425 Real mid = bisect(lo, hi,
426 [=] AMREX_GPU_HOST_DEVICE (Real x) -> Real
427 {
428 int i = int(Math::floor((x - plo)*idx)) + ilo;
429 bool inside = i >= ilo and i <= ihi;
430 return static_cast<Real>(inside) - Real(0.5);
431 }, tolerance);
432 roundoff_domain.setHi(idim, mid - tolerance);
433 }
434}
