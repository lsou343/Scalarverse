218//
219//Advance grids at this level in time.
220//
221Real
222AmrLevelAdv::advance (Real time,
223 Real dt,
224 int iteration,
225 int ncycle)
226{
227 MultiFab& S_mm = get_new_data(Phi_Type);
228 Real maxval = S_mm.max(0);
229 Real minval = S_mm.min(0);
230
231 amrex::Print() << "phi max = " << maxval << ", min = " << minval << std::endl;
232 for (int k = 0; k < NUM_STATE_TYPE; k++) {
233 state[k].allocOldData();
234 state[k].swapTimeLevels(dt);
235 }
236
237 MultiFab& S_new = get_new_data(Phi_Type);
238
239 const Real prev_time = state[Phi_Type].prevTime();
240 const Real cur_time = state[Phi_Type].curTime();
241 const Real ctr_time = 0.5*(prev_time + cur_time);
242
243 const Real* dx = geom.CellSize();
244 const Real* prob_lo = geom.ProbLo();
245
246 //
247 // Get pointers to Flux registers, or set pointer to zero if not there.
248 //
249 FluxRegister *fine = 0;
250 FluxRegister *current = 0;
251
252 int finest_level = parent->finestLevel();
253
254 if (do_reflux && level < finest_level) {
255 fine = &getFluxReg(level+1);
256 fine->setVal(0.0);
257 }
258
259 if (do_reflux && level > 0) {
260 current = &getFluxReg(level);
261 }
262
263 MultiFab fluxes[BL_SPACEDIM];
264
265 if (do_reflux)
266 {
267 for (int j = 0; j < BL_SPACEDIM; j++)
268 {
269 BoxArray ba = S_new.boxArray();
270 ba.surroundingNodes(j);
271 fluxes[j].define(ba, dmap, NUM_STATE, 0);
272 }
273 }
274
275 // State with ghost cells
276 MultiFab Sborder(grids, dmap, NUM_STATE, NUM_GROW);
277 FillPatch(*this, Sborder, NUM_GROW, time, Phi_Type, 0, NUM_STATE);
278
279 // MF to hold the mac velocity
280 MultiFab Umac[BL_SPACEDIM];
281 for (int i = 0; i < BL_SPACEDIM; i++) {
282 BoxArray ba = S_new.boxArray();
283 ba.surroundingNodes(i);
284 Umac[i].define(ba, dmap, 1, iteration);
285 }
286
287#ifdef _OPENMP
288#pragma omp parallel
289#endif
290 {
291 FArrayBox flux[BL_SPACEDIM], uface[BL_SPACEDIM];
292
293 for (MFIter mfi(S_new, true); mfi.isValid(); ++mfi)
294 {
295 const Box& bx = mfi.tilebox();
296
297 const FArrayBox& statein = Sborder[mfi];
298 FArrayBox& stateout = S_new[mfi];
299
300 // Allocate fabs for fluxes and Godunov velocities.
301 for (int i = 0; i < BL_SPACEDIM ; i++) {
302 const Box& bxtmp = amrex::surroundingNodes(bx,i);
303 flux[i].resize(bxtmp,NUM_STATE);
304 uface[i].resize(amrex::grow(bxtmp, iteration), 1);
305 }
306
307 get_face_velocity(&level, &ctr_time,
308 AMREX_D_DECL(BL_TO_FORTRAN(uface[0]),
309 BL_TO_FORTRAN(uface[1]),
310 BL_TO_FORTRAN(uface[2])),
311 dx, prob_lo);
312
313 for (int i = 0; i < BL_SPACEDIM ; i++) {
314 const Box& bxtmp = mfi.grownnodaltilebox(i, iteration);
315 Umac[i][mfi].copy<RunOn::Host>(uface[i], bxtmp);
316 }
317 advect(&time, bx.loVect(), bx.hiVect(),
318 BL_TO_FORTRAN_3D(statein),
319 BL_TO_FORTRAN_3D(stateout),
320 AMREX_D_DECL(BL_TO_FORTRAN_3D(uface[0]),
321 BL_TO_FORTRAN_3D(uface[1]),
322 BL_TO_FORTRAN_3D(uface[2])),
323 AMREX_D_DECL(BL_TO_FORTRAN_3D(flux[0]),
324 BL_TO_FORTRAN_3D(flux[1]),
325 BL_TO_FORTRAN_3D(flux[2])),
326 dx, &dt);
327
328 if (do_reflux) {
329 for (int i = 0; i < BL_SPACEDIM ; i++)
330 fluxes[i][mfi].copy<RunOn::Host>(flux[i],mfi.nodaltilebox(i));
331 }
332 }
333 }
334
335 if (do_reflux) {
336 if (current) {
337 for (int i = 0; i < BL_SPACEDIM ; i++)
338 current->FineAdd(fluxes[i],i,0,0,NUM_STATE,1.);
339 }
340 if (fine) {
341 for (int i = 0; i < BL_SPACEDIM ; i++)
342 fine->CrseInit(fluxes[i],i,0,0,NUM_STATE,-1.);
343 }
344 }
345
346#ifdef AMREX_PARTICLES
347 if (TracerPC) {
348 TracerPC->AdvectWithUmac(Umac, level, dt);
349 }
350#endif
351
352 return dt;
353}

355//
356//Estimate time step.
357//
358Real
359AmrLevelAdv::estTimeStep (Real)
360{
361 // This is just a dummy value to start with
362 Real dt_est = 1.0e+20;
363
364 const Real* dx = geom.CellSize();
365 const Real* prob_lo = geom.ProbLo();
366 const Real cur_time = state[Phi_Type].curTime();
367 const MultiFab& S_new = get_new_data(Phi_Type);
368
369#ifdef _OPENMP
370#pragma omp parallel reduction(min:dt_est)
371#endif
372 {
373 FArrayBox uface[BL_SPACEDIM];
374
375 for (MFIter mfi(S_new, true); mfi.isValid(); ++mfi)
376 {
377 for (int i = 0; i < BL_SPACEDIM ; i++) {
378 const Box& bx = mfi.nodaltilebox(i);
379 uface[i].resize(bx,1);
380 }
381
382 get_face_velocity(&level, &cur_time,
383 AMREX_D_DECL(BL_TO_FORTRAN(uface[0]),
384 BL_TO_FORTRAN(uface[1]),
385 BL_TO_FORTRAN(uface[2])),
386 dx, prob_lo);
387
388 for (int i = 0; i < BL_SPACEDIM; ++i) {
389 Real umax = uface[i].norm<RunOn::Host>(0);
390 if (umax > 1.e-100) {
391 dt_est = std::min(dt_est, dx[i] / umax);
392 }
393 }
394 }
395 }
396
397 ParallelDescriptor::ReduceRealMin(dt_est);
398 dt_est *= cfl;
399
400 if (verbose) {
401 amrex::Print() << "AmrLevelAdv::estTimeStep at level " << level
402 << ": dt_est = " << dt_est << std::endl;
403 }
404
405 return dt_est;
406}

417//
418//Compute initial `dt'.
419//
420void
421AmrLevelAdv::computeInitialDt (int finest_level,
422 int sub_cycle,
423 Vector<int>& n_cycle,
424 const Vector<IntVect>& ref_ratio,
425 Vector<Real>& dt_level,
426 Real stop_time)
427{
428 //
429 // Grids have been constructed, compute dt for all levels.
430 //
431 if (level > 0)
432 return;
433
434 Real dt_0 = 1.0e+100;
435 int n_factor = 1;
436 for (int i = 0; i <= finest_level; i++)
437 {
438 dt_level[i] = getLevel(i).initialTimeStep();
439 n_factor *= n_cycle[i];
440 dt_0 = std::min(dt_0,n_factor*dt_level[i]);
441 }
442
443 //
444 // Limit dt's by the value of stop_time.
445 //
446 const Real eps = 0.001*dt_0;
447 Real cur_time = state[Phi_Type].curTime();
448 if (stop_time >= 0.0) {
449 if ((cur_time + dt_0) > (stop_time - eps))
450 dt_0 = stop_time - cur_time;
451 }
452
453 n_factor = 1;
454 for (int i = 0; i <= finest_level; i++)
455 {
456 n_factor *= n_cycle[i];
457 dt_level[i] = dt_0/n_factor;
458 }
459}

461//
462//Compute new `dt'.
463//
464void
465AmrLevelAdv::computeNewDt (int finest_level,
466 int sub_cycle,
467 Vector<int>& n_cycle,
468 const Vector<IntVect>& ref_ratio,
469 Vector<Real>& dt_min,
470 Vector<Real>& dt_level,
471 Real stop_time,
472 int post_regrid_flag)
473{
474 //
475 // We are at the end of a coarse grid timecycle.
476 // Compute the timesteps for the next iteration.
477 //
478 if (level > 0)
479 return;
480
481 for (int i = 0; i <= finest_level; i++)
482 {
483 AmrLevelAdv& adv_level = getLevel(i);
484 dt_min[i] = adv_level.estTimeStep(dt_level[i]);
485 }
486
487 if (post_regrid_flag == 1)
488 {
489 //
490 // Limit dt's by pre-regrid dt
491 //
492 for (int i = 0; i <= finest_level; i++)
493 {
494 dt_min[i] = std::min(dt_min[i],dt_level[i]);
495 }
496 }
497 else
498 {
499 //
500 // Limit dt's by change_max * old dt
501 //
502 static Real change_max = 1.1;
503 for (int i = 0; i <= finest_level; i++)
504 {
505 dt_min[i] = std::min(dt_min[i],change_max*dt_level[i]);
506 }
507 }
508
509 //
510 // Find the minimum over all levels
511 //
512 Real dt_0 = 1.0e+100;
513 int n_factor = 1;
514 for (int i = 0; i <= finest_level; i++)
515 {
516 n_factor *= n_cycle[i];
517 dt_0 = std::min(dt_0,n_factor*dt_min[i]);
518 }
519
520 //
521 // Limit dt's by the value of stop_time.
522 //
523 const Real eps = 0.001*dt_0;
524 Real cur_time = state[Phi_Type].curTime();
525 if (stop_time >= 0.0) {
526 if ((cur_time + dt_0) > (stop_time - eps))
527 dt_0 = stop_time - cur_time;
528 }
529
530 n_factor = 1;
531 for (int i = 0; i <= finest_level; i++)
532 {
533 n_factor *= n_cycle[i];
534 dt_level[i] = dt_0/n_factor;
535 }
536}
