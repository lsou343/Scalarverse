168struct BATbndryReg
169{
170 BATbndryReg (Orientation a_face, IndexType a_typ,
171 int a_in_rad, int a_out_rad, int a_extent_rad)
172 : m_face(a_face), m_typ(a_typ), m_crse_ratio(1)
173 {
174 m_loshft = IntVect(-a_extent_rad);
175 m_hishft = IntVect( a_extent_rad);
176 IntVect nodal = a_typ.ixType();
177 m_hishft += nodal;
178 const int d = a_face.coordDir();
179 if (m_face.isLow()) {
180 m_loshft[d] = nodal[d] - a_out_rad;
181 m_hishft[d] = nodal[d] + a_in_rad - 1;
182 } else {
183 m_loshft[d] = 1 - a_in_rad;
184 m_hishft[d] = a_out_rad;
185 }
186 m_doilo = IntVect(a_extent_rad);
187 m_doihi = IntVect(a_extent_rad);
188 m_doihi += nodal;
189 if (m_face.isLow()) { // domain of influence in index space
190 m_doilo[d] = std::max(0, a_out_rad - nodal[d]);
191 m_doihi[d] = 0;
192 } else {
193 m_doilo[d] = 0;
194 m_doihi[d] = a_out_rad;
195 }
196 }
197
198 Box operator() (const Box& a_bx) const noexcept {
199 IntVect lo = amrex::coarsen(a_bx.smallEnd(), m_crse_ratio);
200 IntVect hi = amrex::coarsen(a_bx.bigEnd(), m_crse_ratio);
201 const int d = m_face.coordDir();
202 if (m_face.isLow()) {
203 hi[d] = lo[d];
204 } else {
205 lo[d] = hi[d];
206 }
207 lo += m_loshft;
208 hi += m_hishft;
209 return Box(lo,hi,m_typ);
210 }
211
212 Box coarsen (Box const& a_box) const noexcept { return amrex::coarsen(a_box,m_crse_ratio); }
213
214 IntVect doiLo () const noexcept { return m_doilo; }
215 IntVect doiHi () const noexcept { return m_doihi; }
216
217 IndexType index_type () const noexcept { return m_typ; }
218 IntVect coarsen_ratio () const noexcept { return m_crse_ratio; }
219
220 friend bool operator== (BATbndryReg const& a, BATbndryReg const& b) noexcept {
221 return a.m_face == b.m_face and a.m_typ == b.m_typ and a.m_crse_ratio == b.m_crse_ratio
222 and a.m_loshft == b.m_loshft and a.m_hishft == b.m_hishft
223 and a.m_doilo == b.m_doilo and a.m_doihi == b.m_doihi;
224 }
