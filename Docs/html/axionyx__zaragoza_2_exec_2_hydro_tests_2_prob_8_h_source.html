121 //Use middle for comparison, and assume lo[0]=0:
122 // prob_lo[0] + (i+0.5) * dx[0] != prob_lo[0] + (float(i - lo[0])+0.5) * dx[0]

128 amrex::Real ctr_x = prob_param[frac_comp] * (prob_lo[0] + prob_hi[0]);
129 , amrex::Real ctr_y = prob_param[frac_comp] * (prob_lo[1] + prob_hi[1]);
130 , amrex::Real ctr_z = prob_param[frac_comp] * (prob_lo[2] + prob_hi[2]);)

132 // Set the states
133 if (int(prob_param[idir_comp]) == 1) {
134 if (x <= ctr_x) {
135 state(i, j, k, URHO) = prob_param[rho_l_comp];
136 state(i, j, k, UMX) = prob_param[rho_l_comp] * prob_param[u_l_comp];
137 state(i, j, k, UMY) = 0.0;
138 state(i, j, k, UMZ) = 0.0;
139 state(i, j, k, UEDEN) = prob_param[rhoe_l_comp] + 0.5 * prob_param[rho_l_comp] *
140 prob_param[u_l_comp] *
141 prob_param[u_l_comp];
142 state(i, j, k, UEINT) = prob_param[rhoe_l_comp];
143 } else {
144 state(i, j, k, URHO) = prob_param[rho_r_comp];
145 state(i, j, k, UMX) = prob_param[rho_r_comp] * prob_param[u_r_comp];
146 state(i, j, k, UMY) = 0.0;
147 state(i, j, k, UMZ) = 0.0;
148 state(i, j, k, UEDEN) = prob_param[rhoe_r_comp] + 0.5 * prob_param[rho_r_comp] *
149 prob_param[u_r_comp] *
150 prob_param[u_r_comp];
151 state(i, j, k, UEINT) = prob_param[rhoe_r_comp];
152 }
153 } else if (int(prob_param[idir_comp]) == 2) {
154 if (y <= ctr_y) {
155 state(i, j, k, URHO) = prob_param[rho_l_comp];
156 state(i, j, k, UMX) = 0.0;
157 state(i, j, k, UMY) = prob_param[rho_l_comp] * prob_param[u_l_comp];
158 state(i, j, k, UMZ) = 0.0;
159 state(i, j, k, UEDEN) = prob_param[rhoe_l_comp] + 0.5 * prob_param[rho_l_comp] *
160 prob_param[u_l_comp] *
161 prob_param[u_l_comp];
162 state(i, j, k, UEINT) = prob_param[rhoe_l_comp];
163 } else {
164 state(i, j, k, URHO) = prob_param[rho_r_comp];
165 state(i, j, k, UMX) = 0.0;
166 state(i, j, k, UMY) = prob_param[rho_r_comp] * prob_param[u_r_comp];

171 state(i, j, k, UEINT) = prob_param[rhoe_r_comp];

174 if (z <= ctr_z) {
175 state(i, j, k, URHO) = prob_param[rho_l_comp];
176 state(i, j, k, UMX) = 0.0;
177 state(i, j, k, UMY) = 0.0;
178 state(i, j, k, UMZ) = prob_param[rho_l_comp] * prob_param[u_l_comp];
179 state(i, j, k, UEDEN) = prob_param[rhoe_l_comp] + 0.5 * prob_param[rho_l_comp] *
180 prob_param[u_l_comp] *
181 prob_param[u_l_comp];
182 state(i, j, k, UEINT) = prob_param[rhoe_l_comp];
183 } else {

186 state(i, j, k, UMY) = 0.0;

189 prob_param[u_r_comp] *
190 prob_param[u_r_comp];
191 state(i, j, k, UEINT) = prob_param[rhoe_r_comp];
192 }
193 } else {
194 amrex::Abort("invalid idir");
195 }
196 }
197 else
198 {
199 // Set explosion pressure -- we will convert the point-explosion energy into
200 // a corresponding pressure distributed throughout the perturbed volume
201 amrex::Real vctr =
202 4.0 / 3.0 * M_PI * (prob_param[r_init_comp] * prob_param[r_init_comp] * prob_param[r_init_comp]);
203 amrex::Real p_exp = (prob_param[gamma_comp] - 1.0) * prob_param[exp_energy_comp] / vctr;
204
205 //Use left edge for min, and assume lo[0]=0:
206 // prob_lo[0] + (i) * dx[0] != prob_lo[0] + (i + 0.5) * dx[0]
207 // prob_lo[0] + (i) * dx[0] != prob_lo[0] + (i - lo[0]) * dx[0]
208 AMREX_D_TERM(const amrex::Real xmin = prob_lo[0] + (i) * dx[0];
209 , const amrex::Real ymin = prob_lo[1] + (j) * dx[1];
210 , const amrex::Real zmin = prob_lo[2] + (k) * dx[2];)
211
212 AMREX_D_TERM(
213 amrex::Real dx_sub = dx[0] / static_cast<amrex::Real>(prob_param[nsub_comp]);
214 amrex::Real ctr_x = 0.5 * (prob_lo[0] + prob_hi[0]);
215 , amrex::Real dy_sub = dx[1] / static_cast<amrex::Real>(prob_param[nsub_comp]);
216 amrex::Real ctr_y = 0.5 * (prob_lo[1] + prob_hi[1]);
217 , amrex::Real dz_sub = dx[2] / static_cast<amrex::Real>(prob_param[nsub_comp]);
218 amrex::Real ctr_z = 0.5 * (prob_lo[2] + prob_hi[2]);)
219
220 // We initialize by summing over subvolumes of each cell
221 int npert = 0;
222 int n_amb = 0;
223
224#if AMREX_SPACEDIM > 2
225 for (int kk = 0; kk < prob_param[nsub_comp]; kk++) {
226 const amrex::Real z = zmin + (kk + 0.5) * dz_sub - ctr_z;
227#endif
228#if AMREX_SPACEDIM > 1
229 for (int jj = 0; jj < prob_param[nsub_comp]; jj++) {
230 const amrex::Real y = ymin + (jj + 0.5) * dy_sub - ctr_y;
231#endif
232 for (int ii = 0; ii < prob_param[nsub_comp]; ii++) {
233 const amrex::Real x = xmin + (ii + 0.5) * dx_sub - ctr_x;
234 amrex::Real dist = AMREX_D_TERM(x * x, +y * y, +z * z);
235
236 if (dist <= prob_param[r_init_comp] * prob_param[r_init_comp]) {
237 npert += 1;
238 } else {
239 n_amb += 1;
240 }
241 }
242#if AMREX_SPACEDIM > 1
243 }
244#endif
245#if AMREX_SPACEDIM > 2
246 }
247#endif
248
249 amrex::Real p_zone =
250 (static_cast<amrex::Real>(npert) * p_exp +
251 static_cast<amrex::Real>(n_amb) * prob_param[p_ambient_comp]) /
252 static_cast<amrex::Real>(prob_param[nsub_comp] * prob_param[nsub_comp] * prob_param[nsub_comp]);
253
254 amrex::Real eint = p_zone / (prob_param[gamma_comp] - 1.0);
255
256 state(i, j, k, URHO) = prob_param[dens_ambient_comp];
257 state(i, j, k, UMX) = 0.0;
258 state(i, j, k, UMY) = 0.0;
259 state(i, j, k, UMZ) = 0.0;
260
261 state(i, j, k, UEDEN) = eint + 0.5 *
262 (state(i, j, k, UMX) * state(i, j, k, UMX) +
263 state(i, j, k, UMY) * state(i, j, k, UMY) +
264 state(i, j, k, UMZ) * state(i, j, k, UMZ)) /
265 state(i, j, k, URHO);
266
267 state(i, j, k, UEINT) = eint;
268
269 }
270 //This uses 2 species, rather than just 1 (Doesn't set state(i, j, k, UFS) = state(i, j, k, URHO);)
271 amrex::Real FirstSpec = Eint + 1;
272 if (state.nComp() > FirstSpec)
273 {
274 //XHYDROGEN hardcoded to .76 init_managed
275 state(i,j,k,FirstSpec ) = .76 * state(i,j,k,URHO);
276 state(i,j,k,FirstSpec+1) = (1.0 - .76) * state(i,j,k,URHO);
277 }
278}
