86void XYPlotDataList::AddFArrayBox(FArrayBox &fab, int whichdir, int level) {
87 BL_ASSERT(level >= 0);
88 BL_ASSERT(level <= maxLevel);
89
90 whichDir = whichdir;
91 if(fabBoxLists[level].isEmpty()) { // set correct index type
92 fabBoxLists[level].convert(fab.box().ixType());
93 }
94 fabBoxLists[level].push_back(fab.box());
95 updatedQ = false;
96 int istartx(fab.smallEnd()[whichdir]);
97 XYPlotDataListLink *pdll = new XYPlotDataListLink(fab.dataPtr(), istartx,
98 fab.length()[whichdir]);
99 list<XYPlotDataListLink *>::iterator li = dataSets[level].begin();
100 if(li == dataSets[level].end()) {
101 dataSets[level].push_back(pdll);
102 } else {
103 while((*li)->StartXi() < istartx && li != dataSets[level].end()) {
104 ++li;
105 if(li == dataSets[level].end()) {
106 break;
107 }
108 }
109 dataSets[level].insert(li, pdll);
110 }
111}

115void XYPlotDataList::UpdateStats() {
116 if(updatedQ) {
117 return;
118 }
119 BL_ASSERT(dataSets[minLevel].empty() == false);
120
121 int ilev, startxi, endxi;
122 startxi = dataSets[minLevel].front()->StartXi();
123 endxi = dataSets[minLevel].back()->EndXi();
124 startX = offsetX + dX[minLevel] * (double) startxi;
125 endX = offsetX + dX[minLevel] * (double) endxi;
126
127for(int iCurLevel(maxLevel); iCurLevel >= minLevel; --iCurLevel) {
128
129 // back out the probDomain
130 Box probDomain = fabBoxLists[minLevel].minimalBox();
131 probDomain.refine(amrex::CRRBetweenLevels(minLevel, iCurLevel, xypdlRatios));
132 for(int isd(0); isd < BL_SPACEDIM; ++isd) {
133 if(isd != whichDir) { // squish the pd
134 if(fabBoxLists[iCurLevel].size() > 0) {
135 probDomain.setSmall(isd, (*(fabBoxLists[iCurLevel].begin())).smallEnd(isd));
136 probDomain.setBig(isd, (*(fabBoxLists[iCurLevel].begin())).bigEnd(isd));
137 }
138 }
139 }
140
141 Vector<BoxList> unfilledBoxLists(iCurLevel + 1);
142 if(unfilledBoxLists[iCurLevel].isEmpty()) { // convert to correct type
143 unfilledBoxLists[iCurLevel].convert(probDomain.ixType());
144 }
145 unfilledBoxLists[iCurLevel].push_back(probDomain);
146
147 for(ilev = iCurLevel; ilev >= minLevel; --ilev) {
148 if(fabBoxLists[ilev].isEmpty()) { // set correct index type
149 fabBoxLists[ilev].convert(probDomain.ixType());
150 }
151 fillBoxLists[ilev].clear();
152 fillBoxLists[ilev].convert(unfilledBoxLists[ilev].ixType());
153 fillBoxLists[ilev].join(unfilledBoxLists[ilev]);
154 fillBoxLists[ilev].intersect(fabBoxLists[ilev]);
155 BoxList tempUnfilled(probDomain.ixType());
156 for(BoxList::iterator bli = unfilledBoxLists[ilev].begin();
157 bli != unfilledBoxLists[ilev].end(); ++bli)
158 {
159 tempUnfilled.join(amrex::complementIn(*bli, fabBoxLists[ilev]));
160 }
161 unfilledBoxLists[ilev].clear();
162 unfilledBoxLists[ilev].join(tempUnfilled);
163
164 if(ilev > minLevel) {
165 unfilledBoxLists[ilev - 1].clear();
166 unfilledBoxLists[ilev - 1].convert(probDomain.ixType());
167 unfilledBoxLists[ilev - 1].join(unfilledBoxLists[ilev]);
168 unfilledBoxLists[ilev - 1].coarsen(amrex::CRRBetweenLevels(ilev - 1,
169 ilev, xypdlRatios));
170 for(int isd(0); isd < BL_SPACEDIM; ++isd) {
171 if(isd != whichDir) { // squish the boxlist
172 if(fabBoxLists[ilev - 1].size() > 0) {
173 for(BoxList::iterator bli = unfilledBoxLists[ilev - 1].begin();
174 bli != unfilledBoxLists[ilev - 1].end(); ++bli)
175 {
176 (*bli).setSmall(isd, (*(fabBoxLists[ilev - 1].begin())).smallEnd(isd));
177 (*bli).setBig(isd, (*(fabBoxLists[ilev - 1].begin())).bigEnd(isd));
178 }
179 }
180 }
181 }
182 }
183 }
184
185 numPoints[iCurLevel] = 0;
186 for(ilev = minLevel; ilev <= iCurLevel; ++ilev) {
187 for(BoxList::iterator bli = fillBoxLists[ilev].begin();
188 bli != fillBoxLists[ilev].end(); ++bli)
189 {
190 numPoints[iCurLevel] += (*bli).length(whichDir);
191 }
192 }
193 xypdlXVal[iCurLevel].resize(numPoints[iCurLevel]);
194 xypdlYVal[iCurLevel].resize(numPoints[iCurLevel]);
195
196 list<OrderedBoxes> orderedBoxes;
197 for(ilev = minLevel; ilev <= iCurLevel; ++ilev) {
198 for(BoxList::iterator bli = fillBoxLists[ilev].begin();
199 bli != fillBoxLists[ilev].end(); ++bli)
200 {
201 Box refinedBox(*bli);
202 refinedBox.refine(amrex::CRRBetweenLevels(ilev, iCurLevel, xypdlRatios));
203 orderedBoxes.push_back(OrderedBoxes(ilev, whichDir, *bli, refinedBox));
204 }
205 }
206 orderedBoxes.sort();
207 int xIndex(0);
208 for(list<OrderedBoxes>::iterator obli = orderedBoxes.begin();
209 obli != orderedBoxes.end(); ++obli)
210 {
211 for(int i((*obli).DataBox().smallEnd()[whichDir]);
212 i <= (*obli).DataBox().bigEnd()[whichDir]; ++i)
213 {
214 int obLev((*obli).ILevel());
215 Real xval((0.5 + i) * dX[obLev] + offsetX);
216 xypdlXVal[iCurLevel][xIndex] = xval;
217 for(list<XYPlotDataListLink *>::iterator li = (dataSets[obLev]).begin();
218 li != (dataSets[obLev]).end(); ++li)
219 {
220 XYPlotDataListLink *xypd = *li;
221 if(i >= xypd->StartXi() && i < xypd->EndXi()) {
222 Real yval(xypd->XYPDLLData()[i - xypd->StartXi()]);
223 xypdlYVal[iCurLevel][xIndex] = yval;
224 }
225 }
226 ++xIndex;
227 }
228 }
229
230 xypdlLoY[iCurLevel] = std::numeric_limits<Real>::max();
231 xypdlHiY[iCurLevel] = -std::numeric_limits<Real>::max();
232 for(ilev = minLevel; ilev <= iCurLevel; ++ilev) {
233 for(int ii(0); ii < xypdlYVal[ilev].size(); ++ii) {
234 xypdlLoY[iCurLevel] = std::min(xypdlLoY[iCurLevel], xypdlYVal[ilev][ii]);
235 xypdlHiY[iCurLevel] = std::max(xypdlHiY[iCurLevel], xypdlYVal[ilev][ii]);
236 }
237 }
238
239} // end for(iCurLevel...)
240
241 updatedQ = true;
242}
