94StateData::define (const Box& p_domain,
95 const BoxArray& grds,
96 const DistributionMapping& dm,
97 const StateDescriptor& d,
98 Real time,
99 Real dt,
100 const FabFactory<FArrayBox>& factory)
101{
102 BL_PROFILE("StateData::define()");
103 domain = p_domain;
104 desc = &d;
105 arena = nullptr;
106 grids = grds;
107 dmap = dm;
108 m_factory.reset(factory.clone());
109 //
110 // Convert to proper type.
111 //
112 IndexType typ(desc->getType());
113 StateDescriptor::TimeCenter t_typ(desc->timeType());
114 if (!typ.cellCentered())
115 {
116 domain.convert(typ);
117 grids.convert(typ);
118 }
119 if (t_typ == StateDescriptor::Point)
120 {
121 new_time.start = new_time.stop = time;
122 old_time.start = old_time.stop = time - dt;

125 {
126 new_time.start = time;
127 new_time.stop = time+dt;
128 old_time.start = time-dt;
129 old_time.stop = time;
130 }
131 int ncomp = desc->nComp();
132
133 new_data.reset(new MultiFab(grids,dmap,ncomp,desc->nExtra(),
134 MFInfo().SetTag("StateData").SetArena(arena),
135 *m_factory));
136 old_data.reset();

216void
217StateData::restartDoit (std::istream& is, const std::string& chkfile)
218{
219 BL_PROFILE("StateData::restartDoit()");
220
221 is >> old_time.start;
222 is >> old_time.stop;
223 is >> new_time.start;
224 is >> new_time.stop;
225
226 int nsets;
227 is >> nsets;
228
229 new_data.reset(new MultiFab(grids,dmap,desc->nComp(),desc->nExtra(),
230 MFInfo().SetTag("StateData").SetArena(arena),
231 *m_factory));
232 old_data.reset();
233 if (nsets == 2) {
234#ifndef FDM_SLIM_STATE
235 old_data.reset(new MultiFab(grids,dmap,desc->nComp(),desc->nExtra(),
236 MFInfo().SetTag("StateData").SetArena(arena),
237 *m_factory));
238#endif
239 }
240 //
241 // If no data is written then we just allocate the MF instead of reading it in.
242 // This assumes that the application will do something with it.
243 // We set it to zero in case a compiler complains about uninitialized data.
244 //
245 if (nsets == 0) {
246 new_data->setVal(0.0);
247 }
248
249 std::string mf_name;
250 std::string FullPathName;
251
252 for(int ns(1); ns <= nsets; ++ns) {
253 MultiFab *whichMF = nullptr;
254 if(ns == 1) {
255 whichMF = new_data.get();
256 } else if(ns == 2) {
257#ifndef FDM_SLIM_STATE
258 whichMF = old_data.get();
259#else
260 whichMF = new_data.get();
261#endif
262 } else {
263 amrex::Abort("**** Error in StateData::restart: invalid nsets.");
264 }
265
266 is >> mf_name;
267 //
268 // Note that mf_name is relative to the Header file.
269 // We need to prepend the name of the chkfile directory.
270 //
271 FullPathName = chkfile;
272 if ( ! chkfile.empty() && chkfile[chkfile.length()-1] != '/') {
273 FullPathName += '/';
274 }
275 FullPathName += mf_name;
276
277 // ---- check for preread header
278 std::string FullHeaderPathName(FullPathName + "_H");
279 const char *faHeader = 0;
280 if(faHeaderMap != 0) {
281 std::map<std::string, Vector<char> >::iterator fahmIter;
282 fahmIter = faHeaderMap->find(FullHeaderPathName);
283 if(fahmIter != faHeaderMap->end()) {
284 faHeader = fahmIter->second.dataPtr();
285 }
286 }
287
288 VisMF::Read(*whichMF, FullPathName, faHeader);
289 }

450void
451StateData::FillBoundary (FArrayBox& dest,
452 Real time,
453 const Real* dx,
454 const RealBox& prob_domain,
455 int dest_comp,
456 int src_comp,
457 int num_comp)
458{
459 BL_PROFILE("StateData::FillBoundary(dx)");
460 BL_ASSERT(dest.box().ixType() == desc->getType());
461
462 if (domain.contains(dest.box())) return;
463
464 const Box& bx = dest.box();
465 const int* dlo = dest.loVect();
466 const int* dhi = dest.hiVect();
467 const int* plo = domain.loVect();
468 const int* phi = domain.hiVect();
469
470 Vector<int> bcrs;
471
472 Real xlo[AMREX_SPACEDIM];
473 BCRec bcr;
474 const Real* problo = prob_domain.lo();
475
476 for (int i = 0; i < AMREX_SPACEDIM; i++)
477 {
478 xlo[i] = problo[i] + dx[i]*(dlo[i]-plo[i]);
479 }
480 for (int i = 0; i < num_comp; )
481 {
482 const int dc = dest_comp+i;
483 const int sc = src_comp+i;
484 Real* dat = dest.dataPtr(dc);
485
486 if (desc->primary(sc))
487 {
488 const int groupsize = desc->groupsize(sc);
489
490 BL_ASSERT(groupsize != 0);
491
492 if (groupsize+i <= num_comp)
493 {
494 //
495 // Can do the whole group at once.
496 //
497 bcrs.resize(2*AMREX_SPACEDIM*groupsize);
498
499 int* bci = bcrs.dataPtr();
500
501 for (int j = 0; j < groupsize; j++)
502 {
503 amrex::setBC(bx,domain,desc->getBC(sc+j),bcr);
504
505 const int* bc = bcr.vect();
506
507 for (int k = 0; k < 2*AMREX_SPACEDIM; k++)
508 bci[k] = bc[k];
509
510 bci += 2*AMREX_SPACEDIM;
511 }
512 //
513 // Use the "group" boundary fill routine.
514 //
515 desc->bndryFill(sc)(dat,dlo,dhi,plo,phi,dx,xlo,&time,bcrs.dataPtr(),groupsize);
516 i += groupsize;
517 }
518 else
519 {
520 amrex::setBC(bx,domain,desc->getBC(sc),bcr);
521 desc->bndryFill(sc)(dat,dlo,dhi,plo,phi,dx,xlo,&time,bcr.vect());
522 i++;
523 }
524 }
525 else
526 {
527 amrex::setBC(bx,domain,desc->getBC(sc),bcr);
528 desc->bndryFill(sc)(dat,dlo,dhi,plo,phi,dx,xlo,&time,bcr.vect());
529 i++;
530 }
531 }
532
533#ifdef AMREX_USE_GPU
534 // Add a synchronize here in case the user code launched kernels
535 // to handle the boundary fills.
536 Gpu::synchronize();
537#endif
538}

540void
541StateData::FillBoundary (Box const& bx,
542 FArrayBox& dest,
543 Real time,
544 const Geometry& geom,
545 int dest_comp,
546 int src_comp,
547 int num_comp)
548{
549 BL_PROFILE("StateData::FillBoundary(geom)");
550 BL_ASSERT(bx.ixType() == desc->getType());
551
552 if (domain.contains(bx)) return;
553
554 Vector<BCRec> bcr(num_comp);
555
556 for (int i = 0; i < num_comp; )
557 {
558 const int dc = dest_comp+i;
559 const int sc = src_comp+i;
560
561 if (desc->primary(sc))
562 {
563 const int groupsize = desc->groupsize(sc);
564
565 BL_ASSERT(groupsize != 0);
566
567 if (groupsize+i <= num_comp)
568 {
569 for (int j = 0; j < groupsize; j++)
570 {
571 amrex::setBC(bx,domain,desc->getBC(sc+j),bcr[j]);
572 }
573 //
574 // Use the "group" boundary fill routine.
575 //
576 desc->bndryFill(sc)(bx,dest,dc,groupsize,geom,time,bcr,0,sc);
577 i += groupsize;
578 }
579 else
580 {
581 amrex::setBC(bx,domain,desc->getBC(sc),bcr[0]);
582 desc->bndryFill(sc)(bx,dest,dc,1,geom,time,bcr,0,sc);
583 i++;
584 }
585 }
586 else
587 {
588 amrex::setBC(bx,domain,desc->getBC(sc),bcr[0]);
589 desc->bndryFill(sc)(bx,dest,dc,1,geom,time,bcr,0,sc);
590 i++;
591 }
