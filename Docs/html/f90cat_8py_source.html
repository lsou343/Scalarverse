43def is_sorted(files):
44 """ this should check whether the files are properly sorted"""
45 have_sorted = True
46
47 for n, currfile in enumerate(files):
48 # nothing before the current file should depend on it
49 for q in range(n):
50 prevfile = files[q]
51 for d in prevfile.needs:
52 if d in currfile.defined:
53 have_sorted = False
54 break
55
56 return have_sorted
