115void
116BoxDomain::add (const Box& b)
117{
118 BL_ASSERT(b.ixType() == ixType());
119
120 Vector<Box> tmp, check;
121
122 check.push_back(b);
123
124 for (const auto& bx : *this)
125 {
126 tmp.clear();
127 for (auto& cbx : check)
128 {
129 if (cbx.intersects(bx))
130 {
131 const BoxList& tmpbl = amrex::boxDiff(cbx, bx);
132 tmp.insert(std::end(tmp), std::begin(tmpbl), std::end(tmpbl));
133 cbx = Box();
134 }
135 }
136 check.erase(std::remove_if(check.begin(), check.end(),
137 [](const Box& x) { return x.isEmpty(); }),
138 check.end());
139 check.insert(std::end(check), std::begin(tmp), std::end(tmp));
140 }
141 join(check);
142 BL_ASSERT(ok());
143}
