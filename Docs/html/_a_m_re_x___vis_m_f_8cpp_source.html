707VisMF::Header::CalculateMinMax (const FabArray<FArrayBox>& mf,
708 int procToWrite, MPI_Comm comm)
709{
710 amrex::ignore_unused(procToWrite,comm);
711
712// BL_PROFILE("VisMF::CalculateMinMax");
713
714 m_min.resize(m_ba.size());
715 m_max.resize(m_ba.size());
716
717 bool run_on_device = Gpu::inLaunchRegion()
718 and (mf.arena() == The_Arena() or
719 mf.arena() == The_Device_Arena() or
720 mf.arena() == The_Managed_Arena());

722#ifdef BL_USE_MPI
723 //
724 // Calculate m_min and m_max on the CPU owning the fab.
725 //
726 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
727 const int idx = mfi.index();
728
729 m_min[idx].resize(m_ncomp);
730 m_max[idx].resize(m_ncomp);
731
732 BL_ASSERT(mf[mfi].box().contains(m_ba[idx]));
733
734 for(Long j(0); j < m_ncomp; ++j) {
735 if (run_on_device) {
736 m_min[idx][j] = mf[mfi].min<RunOn::Device>(m_ba[idx],j);
737 m_max[idx][j] = mf[mfi].max<RunOn::Device>(m_ba[idx],j);
738 } else {
739 m_min[idx][j] = mf[mfi].min<RunOn::Host>(m_ba[idx],j);
740 m_max[idx][j] = mf[mfi].max<RunOn::Host>(m_ba[idx],j);
741 }
742 }
743 }
744
745 Vector<int> nmtags(ParallelDescriptor::NProcs(comm), 0);
746 Vector<int> offset(ParallelDescriptor::NProcs(comm), 0);
747
748 const Vector<int> &pmap = mf.DistributionMap().ProcessorMap();

751 ++nmtags[pmap[i]];
752 }
753
754 for(int i(0), N(nmtags.size()); i < N; ++i) {
755 //
756 // Each Fab corresponds to 2*m_ncomp Reals.
757 //
758 nmtags[i] *= 2*m_ncomp;
759 }
760
761 for(int i(1), N(offset.size()); i < N; ++i) {
762 offset[i] = offset[i-1] + nmtags[i-1];
763 }

766
767 if(senddata.empty()) {
768 //

770 //
771 senddata.resize(1);
772 }
773
774 int ioffset = 0;
775
776 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
777 const int idx = mfi.index();
778 for(int i(0); i < m_ncomp; ++i) {
779 senddata[ioffset+i] = m_min[idx][i];
780 senddata[ioffset+m_ncomp+i] = m_max[idx][i];
781 }
782 ioffset += 2*m_ncomp;
783 }
784
785 BL_ASSERT(ioffset == nmtags[ParallelDescriptor::MyProc(comm)]);
786
787 Vector<Real> recvdata(mf.size()*2*m_ncomp);
788
789 BL_COMM_PROFILE(BLProfiler::Gatherv, recvdata.size() * sizeof(Real),
790 ParallelDescriptor::MyProc(comm), BLProfiler::BeforeCall());
791
792 BL_MPI_REQUIRE( MPI_Gatherv(senddata.dataPtr(),
793 nmtags[ParallelDescriptor::MyProc(comm)],
794 ParallelDescriptor::Mpi_typemap<Real>::type(),
795 recvdata.dataPtr(),
796 nmtags.dataPtr(),
797 offset.dataPtr(),
798 ParallelDescriptor::Mpi_typemap<Real>::type(),
799 procToWrite,
800 comm) );
801
802 BL_COMM_PROFILE(BLProfiler::Gatherv, recvdata.size() * sizeof(Real),
803 ParallelDescriptor::MyProc(comm), BLProfiler::AfterCall());
804
805 if(ParallelDescriptor::MyProc(comm) == procToWrite) {
806 for(int i(0), N(mf.size()); i < N; ++i) {
807 if(pmap[i] != procToWrite) {
808 m_min[i].resize(m_ncomp);
809 m_max[i].resize(m_ncomp);
810 }
811 }
812
813 for(int j(0), N(mf.size()); j < N; ++j) {
814 if(pmap[j] != procToWrite) {
815 for(int k(0); k < m_ncomp; ++k) {
816 m_min[j][k] = recvdata[offset[pmap[j]]+k];
817 m_max[j][k] = recvdata[offset[pmap[j]]+k+m_ncomp];
818 }
819
820 offset[pmap[j]] += 2*m_ncomp;
821 }
822 }
823 }
824#else
825 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
826 const int idx = mfi.index();
827
828 m_min[idx].resize(m_ncomp);
829 m_max[idx].resize(m_ncomp);
830
831 BL_ASSERT(mf[mfi].box().contains(m_ba[idx]));
832
833 for(Long j(0); j < m_ncomp; ++j) {
834 if (run_on_device) {
835 m_min[idx][j] = mf[mfi].min<RunOn::Device>(m_ba[idx],j);
836 m_max[idx][j] = mf[mfi].max<RunOn::Device>(m_ba[idx],j);
837 } else {
838 m_min[idx][j] = mf[mfi].min<RunOn::Host>(m_ba[idx],j);
839 m_max[idx][j] = mf[mfi].max<RunOn::Host>(m_ba[idx],j);
840 }
841 }
842 }
843#endif /*BL_USE_MPI*/
844
845#ifdef BL_FIXHEADERDENORMS
846 if(ParallelDescriptor::MyProc(comm) == procToWrite) {
847 for(int i(0); i < m_min.size(); ++i) {
848 for(int j(0); j < m_min[i].size(); ++j) {
849 if(std::abs(m_min[i][j]) < 1.0e-300) {
850 m_min[i][j] = 0.0;
851 }
852 }
853 }
854 for(int i(0); i < m_max.size(); ++i) {
855 for(int j(0); j < m_max[i].size(); ++j) {
856 if(std::abs(m_max[i][j]) < 1.0e-300) {
857 m_max[i][j] = 0.0;
858 }
859 }
860 }
861 }
862#endif
863
864 // ---- calculate fabarray min max values
865 m_famin.resize(m_ncomp);
866 m_famax.resize(m_ncomp);
867 for(int comp(0); comp < m_ncomp; ++comp) {
868 m_famin[comp] = std::numeric_limits<Real>::max();
869 m_famax[comp] = -std::numeric_limits<Real>::max();
870 }
871
872 for(int ibox(0); ibox < m_min.size(); ++ibox) {
873 for(int comp(0); comp < m_min[ibox].size(); ++comp) {
874 m_famin[comp] = std::min(m_famin[comp], m_min[ibox][comp]);
875 }
876 }
877 for(int ibox(0); ibox < m_max.size(); ++ibox) {
878 for(int comp(0); comp < m_max[ibox].size(); ++comp) {
879 m_famax[comp] = std::max(m_famax[comp], m_max[ibox][comp]);
880 }
881 }

941Long
942VisMF::Write (const FabArray<FArrayBox>& mf,
943 const std::string& mf_name,
944 VisMF::How how,
945 bool set_ghost)
946{
947 BL_PROFILE("VisMF::Write(FabArray)");
948 BL_ASSERT(mf_name[mf_name.length() - 1] != '/');
949 BL_ASSERT(currentVersion != VisMF::Header::Undefined_v1);
950
951 // ---- add stream retry
952 // ---- add stream buffer (to nfiles)
953 RealDescriptor *whichRD = nullptr;
954 if(FArrayBox::getFormat() == FABio::FAB_NATIVE) {
955 whichRD = FPC::NativeRealDescriptor().clone();
956 } else if(FArrayBox::getFormat() == FABio::FAB_NATIVE_32) {
957 whichRD = FPC::Native32RealDescriptor().clone();
958 } else if(FArrayBox::getFormat() == FABio::FAB_IEEE_32) {
959 whichRD = FPC::Ieee32NormalRealDescriptor().clone();
960 } else {
961 whichRD = FPC::NativeRealDescriptor().clone(); // to quiet clang static analyzer
962 Abort("VisMF::Write unable to execute with the current fab.format setting. Use NATIVE, NATIVE_32 or IEEE_32");
963 }
964 bool doConvert(*whichRD != FPC::NativeRealDescriptor());
965
966 if(set_ghost and mf.nGrowVect() != 0) {
967 FabArray<FArrayBox>* the_mf = const_cast<FabArray<FArrayBox>*>(&mf);
968
969 bool run_on_device = Gpu::inLaunchRegion()
970 and (mf.arena() == The_Arena() or
971 mf.arena() == The_Device_Arena() or
972 mf.arena() == The_Managed_Arena());
973
974 for(MFIter mfi(*the_mf); mfi.isValid(); ++mfi) {
975 const int idx(mfi.index());
976
977 for(int j(0); j < mf.nComp(); ++j) {
978 if (run_on_device) {
979 const Real valMin(mf[mfi].min<RunOn::Device>(mf.box(idx), j));
980 const Real valMax(mf[mfi].max<RunOn::Device>(mf.box(idx), j));
981 const Real val((valMin + valMax) / 2.0);
982 the_mf->get(mfi).setComplement<RunOn::Device>(val, mf.box(idx), j, 1);
983 } else {
984 const Real valMin(mf[mfi].min<RunOn::Host>(mf.box(idx), j));
985 const Real valMax(mf[mfi].max<RunOn::Host>(mf.box(idx), j));
986 const Real val((valMin + valMax) / 2.0);
987 the_mf->get(mfi).setComplement<RunOn::Host>(val, mf.box(idx), j, 1);
988 }
989 }
990 }
991 }
992
993 amrex::prefetchToHost(mf);
994
995 // ---- check if mf has sparse data
996 bool useSparseFPP(false);
997 const Vector<int> &pmap = mf.DistributionMap().ProcessorMap();
998 std::set<int> procsWithData;
999 Vector<int> procsWithDataVector;
1000 for(int i(0); i < pmap.size(); ++i) {
1001 procsWithData.insert(pmap[i]);
1002 }
1003 if(allowSparseWrites && (procsWithData.size() < nOutFiles)) {
1004 useSparseFPP = true;
1005// amrex::Print() << "SSSSSSSS: in VisMF::Write: useSparseFPP for: " << mf_name << '\n';
1006 for(std::set<int>::iterator it = procsWithData.begin(); it != procsWithData.end(); ++it) {
1007 procsWithDataVector.push_back(*it);
1008 }
1009 }
1010
1011 int coordinatorProc(ParallelDescriptor::IOProcessorNumber());
1012 Long bytesWritten(0);
1013 bool calcMinMax(false);
1014 VisMF::Header hdr(mf, how, currentVersion, calcMinMax);
1015
1016 std::string filePrefix(mf_name + FabFileSuffix);
1017
1018 NFilesIter nfi(nOutFiles, filePrefix, groupSets, setBuf);
1019
1020 bool oldHeader(currentVersion == VisMF::Header::Version_v1);
1021
1022 if(useSparseFPP) {
1023 nfi.SetSparseFPP(procsWithDataVector);
1024 } else if(useDynamicSetSelection) {
1025 nfi.SetDynamic();
1026 }
1027 for( ; nfi.ReadyToWrite(); ++nfi) {
1028 // ---- find the total number of bytes including fab headers if needed
1029 const FABio &fio = FArrayBox::getFABio();
1030 int whichRDBytes(whichRD->numBytes()), nFABs(0);
1031 Long writeDataItems(0), writeDataSize(0);
1032 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
1033 const FArrayBox &fab = mf[mfi];
1034 if(oldHeader) {
1035 std::stringstream hss;
1036 fio.write_header(hss, fab, fab.nComp());
1037 bytesWritten += static_cast<std::streamoff>(hss.tellp());
1038 }
1039 bytesWritten += fab.box().numPts() * mf.nComp() * whichRDBytes;
1040 ++nFABs;
1041 }
1042 char *allFabData(nullptr);
1043 bool canCombineFABs(false);
1044 if((nFABs > 1 || doConvert) && VisMF::useSingleWrite) {
1045 allFabData = new(std::nothrow) char[bytesWritten];
1046 } // ---- else { no need to make a copy for one fab }
1047 if(allFabData == nullptr) {
1048 canCombineFABs = false;
1049 } else {
1050 canCombineFABs = true;
1051 }
1052
1053 if(canCombineFABs) {
1054 Long writePosition(0);
1055 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
1056 int hLength(0);
1057 const FArrayBox &fab = mf[mfi];
1058 writeDataItems = fab.box().numPts() * mf.nComp();
1059 writeDataSize = writeDataItems * whichRDBytes;
1060 char *afPtr = allFabData + writePosition;
1061 if(oldHeader) {
1062 std::stringstream hss;
1063 fio.write_header(hss, fab, fab.nComp());
1064 hLength = static_cast<std::streamoff>(hss.tellp());
1065 auto tstr = hss.str();
1066 memcpy(afPtr, tstr.c_str(), hLength); // ---- the fab header
1067 }
1068 if(doConvert) {
1069 RealDescriptor::convertFromNativeFormat(static_cast<void *> (afPtr + hLength),
1070 writeDataItems,
1071 fab.dataPtr(), *whichRD);
1072 } else { // ---- copy from the fab
1073 memcpy(afPtr + hLength, fab.dataPtr(), writeDataSize);
1074 }
1075 writePosition += hLength + writeDataSize;
1076 }
1077 nfi.Stream().write(allFabData, bytesWritten);
1078 nfi.Stream().flush();
1079 delete [] allFabData;
1080
1081 } else { // ---- write fabs individually
1082 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
1083 int hLength(0);
1084 const FArrayBox &fab = mf[mfi];
1085 writeDataItems = fab.box().numPts() * mf.nComp();
1086 writeDataSize = writeDataItems * whichRDBytes;
1087 if(oldHeader) {
1088 std::stringstream hss;
1089 fio.write_header(hss, fab, fab.nComp());
1090 hLength = static_cast<std::streamoff>(hss.tellp());
1091 auto tstr = hss.str();
1092 nfi.Stream().write(tstr.c_str(), hLength); // ---- the fab header
1093 nfi.Stream().flush();
1094 }
1095 if(doConvert) {
1096 char *cDataPtr = new char[writeDataSize];
1097 RealDescriptor::convertFromNativeFormat(static_cast<void *> (cDataPtr),
1098 writeDataItems,
1099 fab.dataPtr(), *whichRD);
1100 nfi.Stream().write(cDataPtr, writeDataSize);
1101 nfi.Stream().flush();
1102 delete [] cDataPtr;
1103 } else { // ---- copy from the fab
1104 nfi.Stream().write((char *) fab.dataPtr(), writeDataSize);
1105 nfi.Stream().flush();
1106 }
1107 }
1108 }
1109 }
1110
1111 if(nfi.GetDynamic()) {
1112 coordinatorProc = nfi.CoordinatorProc();
1113 }
1114
1115 if (Gpu::inLaunchRegion()) {
1116 amrex::prefetchToDevice(mf); // CalculateMinMax might do work on device
1117 }
1118
1119 if(currentVersion == VisMF::Header::Version_v1 ||
1120 currentVersion == VisMF::Header::NoFabHeaderMinMax_v1)
1121 {
1122 hdr.CalculateMinMax(mf, coordinatorProc);
1123 }
1124
1125 VisMF::FindOffsets(mf, filePrefix, hdr, currentVersion, nfi,
1126 ParallelDescriptor::Communicator());
1127
1128 bytesWritten += VisMF::WriteHeader(mf_name, hdr, coordinatorProc);
1129
1130 delete whichRD;
1131
1132 return bytesWritten;
1133}

1170void
1171VisMF::FindOffsets (const FabArray<FArrayBox> &mf,
1172 const std::string &filePrefix,
1173 VisMF::Header &hdr,
1174 VisMF::Header::Version /*whichVersion*/,
1175 NFilesIter &nfi, MPI_Comm comm)
1176{
1177// BL_PROFILE("VisMF::FindOffsets");
1178
1179 const int myProc(ParallelDescriptor::MyProc(comm));
1180 const int nProcs(ParallelDescriptor::NProcs(comm));
1181 int coordinatorProc(ParallelDescriptor::IOProcessorNumber(comm));
1182 if(nfi.GetDynamic()) {
1183 coordinatorProc = nfi.CoordinatorProc();
1184 }
1185
1186 if(FArrayBox::getFormat() == FABio::FAB_ASCII ||
1187 FArrayBox::getFormat() == FABio::FAB_8BIT)
1188 {
1189
1190#ifdef BL_USE_MPI
1191 Vector<int> nmtags(nProcs,0);
1192 Vector<int> offset(nProcs,0);
1193
1194 const Vector<int> &pmap = mf.DistributionMap().ProcessorMap();
1195
1196 for(int i(0), N(mf.size()); i < N; ++i) {
1197 ++nmtags[pmap[i]];
1198 }
1199
1200 for(int i(1), N(offset.size()); i < N; ++i) {
1201 offset[i] = offset[i-1] + nmtags[i-1];
1202 }
1203
1204 Vector<Long> senddata(nmtags[myProc]);
1205
1206 if(senddata.empty()) {
1207 // Can't let senddata be empty as senddata.dataPtr() will fail.
1208 senddata.resize(1);
1209 }
1210
1211 int ioffset(0);
1212
1213 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
1214 senddata[ioffset++] = hdr.m_fod[mfi.index()].m_head;
1215 }
1216
1217 BL_ASSERT(ioffset == nmtags[myProc]);
1218
1219 Vector<Long> recvdata(mf.size());
1220
1221 BL_COMM_PROFILE(BLProfiler::Gatherv, recvdata.size() * sizeof(Long),
1222 myProc, BLProfiler::BeforeCall());
1223
1224 BL_MPI_REQUIRE( MPI_Gatherv(senddata.dataPtr(),
1225 nmtags[myProc],
1226 ParallelDescriptor::Mpi_typemap<Long>::type(),
1227 recvdata.dataPtr(),
1228 nmtags.dataPtr(),
1229 offset.dataPtr(),
1230 ParallelDescriptor::Mpi_typemap<Long>::type(),
1231 coordinatorProc,
1232 comm) );
1233
1234 BL_COMM_PROFILE(BLProfiler::Gatherv, recvdata.size() * sizeof(Long),
1235 myProc, BLProfiler::AfterCall());
1236
1237 if(myProc == coordinatorProc) {
1238 Vector<int> cnt(nProcs,0);
1239
1240 for(int j(0), N(mf.size()); j < N; ++j) {
1241 const int i(pmap[j]);
1242 hdr.m_fod[j].m_head = recvdata[offset[i]+cnt[i]];
1243
1244 const std::string name(NFilesIter::FileName(nOutFiles, filePrefix, i, groupSets));
1245
1246 hdr.m_fod[j].m_name = VisMF::BaseName(name);
1247
1248 ++cnt[i];
1249 }
1250 }
1251#endif /*BL_USE_MPI*/
1252
1253
1254 } else { // ---- calculate offsets
1255
1256 RealDescriptor *whichRD = nullptr;
1257 if(FArrayBox::getFormat() == FABio::FAB_NATIVE) {
1258 whichRD = FPC::NativeRealDescriptor().clone();
1259 } else if(FArrayBox::getFormat() == FABio::FAB_NATIVE_32) {
1260 whichRD = FPC::Native32RealDescriptor().clone();
1261 } else if(FArrayBox::getFormat() == FABio::FAB_IEEE_32) {
1262 whichRD = FPC::Ieee32NormalRealDescriptor().clone();
1263 }
1264 const FABio &fio = FArrayBox::getFABio();
1265 int whichRDBytes(whichRD->numBytes());
1266 int nComps(mf.nComp());
1267
1268 if(myProc == coordinatorProc) { // ---- calculate offsets
1269 const BoxArray &mfBA = mf.boxArray();
1270 const DistributionMapping &mfDM = mf.DistributionMap();
1271 Vector<Long> fabHeaderBytes(mfBA.size(), 0);
1272 int nFiles(NFilesIter::ActualNFiles(nOutFiles));
1273 int whichFileNumber(-1);
1274 std::string whichFileName;
1275 Vector<Long> currentOffset(nProcs, 0L);
1276
1277 if(hdr.m_vers == VisMF::Header::Version_v1) {
1278 // ---- find the length of the fab header instead of asking the file system
1279 for(int i(0); i < mfBA.size(); ++i) {
1280 std::stringstream hss;
1281 FArrayBox tempFab(mf.fabbox(i), nComps, false); // ---- no alloc
1282 fio.write_header(hss, tempFab, tempFab.nComp());
1283 fabHeaderBytes[i] = static_cast<std::streamoff>(hss.tellp());
1284 }
1285 }
1286
1287 std::map<int, Vector<int> > rankBoxOrder; // ---- [rank, boxarray index array]
1288 for(int i(0); i < mfBA.size(); ++i) {
1289 rankBoxOrder[mfDM[i]].push_back(i);
1290 }
1291
1292 Vector<int> fileNumbers;
1293 if(nfi.GetDynamic()) {
1294 fileNumbers = nfi.FileNumbersWritten();
1295 }
1296 else if(nfi.GetSparseFPP()) { // if sparse, write to (file number = rank)
1297 fileNumbers.resize(nProcs);
1298 for(int i(0); i < nProcs; ++i) {
1299 fileNumbers[i] = i;
1300 }
1301 }
1302 else {
1303 fileNumbers.resize(nProcs);
1304 for(int i(0); i < nProcs; ++i) {
1305 fileNumbers[i] = NFilesIter::FileNumber(nFiles, i, groupSets);
1306 }
1307 }
1308
1309 const Vector< Vector<int> > &fileNumbersWriteOrder = nfi.FileNumbersWriteOrder();
1310
1311 for(int fn(0); fn < fileNumbersWriteOrder.size(); ++fn) {
1312 for(int ri(0); ri < fileNumbersWriteOrder[fn].size(); ++ri) {
1313 int rank(fileNumbersWriteOrder[fn][ri]);
1314 std::map<int, Vector<int> >::iterator rboIter = rankBoxOrder.find(rank);
1315
1316 if(rboIter != rankBoxOrder.end()) {
1317 Vector<int> &index = rboIter->second;
1318 whichFileNumber = fileNumbers[rank];
1319 whichFileName = VisMF::BaseName(NFilesIter::FileName(whichFileNumber, filePrefix));
1320
1321 for(int i(0); i < index.size(); ++i) {
1322 hdr.m_fod[index[i]].m_name = whichFileName;
1323 hdr.m_fod[index[i]].m_head = currentOffset[whichFileNumber];
1324 currentOffset[whichFileNumber] += mf.fabbox(index[i]).numPts() * nComps * whichRDBytes
1325 + fabHeaderBytes[index[i]];
1326 }
1327 }
1328 }
1329 }
1330 }
1331 delete whichRD;
1332 }

1490 int coordinatorProc,

1493 BL_PROFILE("VisMF::Read()");
1494
1495 VisMF::Header hdr;
1496 Real hEndTime, hStartTime, faCopyTime(0.0);
1497 Real startTime(amrex::second());
1498 static Real totalTime(0.0);
1499 int myProc(ParallelDescriptor::MyProc());
1500 int messTotal(0);
1501
1502 if(verbose && myProc == coordinatorProc) {
1503 amrex::AllPrint() << myProc << "::VisMF::Read: about to read: " << mf_name << std::endl;
1504 }
1505
1506 std::string FullHdrFileName(mf_name + TheMultiFabHdrFileSuffix);
1507
1508 {
1509 hStartTime = amrex::second();

1511 if(faHeader == nullptr) {

1513 ParallelDescriptor::ReadAndBcastFile(FullHdrFileName, fileCharPtr);

1516 fileCharPtrString = faHeader;
1517 }
1518 std::istringstream infs(fileCharPtrString, std::istringstream::in);
1519
1520 infs >> hdr;
1521
1522 hEndTime = amrex::second();
1523 }
1524
1525 // This allows us to read in an empty MultiFab without an error -- but only if explicitly told to
1526 if (allow_empty_mf > 0)
1527 {
1528 if (hdr.m_ba.size() == 0) return;
1529 } else {
1530 if (hdr.m_ba.size() == 0)
1531 {
1532 amrex::Print() << "In trying to read " << mf_name << std::endl;
1533 amrex::Error("Empty box array");
1534 }
1535 }
1536
1537
1538 if (mf.empty()) {
1539 DistributionMapping dm(hdr.m_ba);
1540 mf.define(hdr.m_ba, dm, hdr.m_ncomp, hdr.m_ngrow, MFInfo(), FArrayBoxFactory());
1541 } else {
1542 BL_ASSERT(amrex::match(hdr.m_ba,mf.boxArray()));
1543 }
1544
1545#ifdef BL_USE_MPI
1546
1547 // ---- This limits the number of concurrent readers per file.
1548 int nOpensPerFile(nMFFileInStreams);
1549 int nProcs(ParallelDescriptor::NProcs());
1550 bool noFabHeader(NoFabHeader(hdr));
1551
1552 if(noFabHeader && useSynchronousReads) {
1553
1554 // ---- This code is only for reading in file order
1555 bool doConvert(hdr.m_writtenRD != FPC::NativeRealDescriptor());
1556
1557 // ---- Create an ordered map of which processors read which
1558 // ---- Fabs in each file
1559
1560 std::map<std::string, Vector<FabReadLink> > FileReadChains; // ---- [filename, chain]
1561 std::map<std::string, std::set<int> > readFileRanks; // ---- [filename, ranks]
1562
1563 int nBoxes(hdr.m_ba.size());
1564 for(int i(0); i < nBoxes; ++i) { // ---- create the map
1565 int undefined(-1);
1566 std::string fname(hdr.m_fod[i].m_name);
1567 FileReadChains[fname].push_back(FabReadLink(undefined, undefined, hdr.m_fod[i].m_head, hdr.m_ba[i]));
1568 }
1569
1570 std::map<std::string, Vector<FabReadLink> >::iterator frcIter;
1571
1572 int indexFileOrder(0);
1573 int currentRank(0);
1574 FabArray<FArrayBox> fafabFileOrder;
1575 BoxArray baFileOrder(hdr.m_ba.size());
1576
1577 Vector<int> ranksFileOrder(mf.DistributionMap().size(), -1);
1578
1579 Vector<int> nRanksPerFile(FileReadChains.size());
1580 amrex::NItemsPerBin(nProcs, nRanksPerFile);
1581 int currentFileIndex(0);
1582
1583 for(frcIter = FileReadChains.begin(); frcIter != FileReadChains.end(); ++frcIter) {
1584 const std::string &fileName = frcIter->first;
1585 Vector<FabReadLink> &frc = frcIter->second;
1586 // ---- sort by offset
1587 std::sort(frc.begin(), frc.end(), [] (const FabReadLink &a, const FabReadLink &b)
1588 { return a.fileOffset < b.fileOffset; } );
1589
1590 Vector<int> nBoxesPerRank(nRanksPerFile[currentFileIndex]);
1591 amrex::NItemsPerBin(frc.size(), nBoxesPerRank);
1592 int frcIndex(0);
1593
1594 for(int nbpr(0); nbpr < nBoxesPerRank.size(); ++nbpr) {
1595 for(int nb(0); nb < nBoxesPerRank[nbpr]; ++nb) {
1596
1597 baFileOrder.set(indexFileOrder, frc[frcIndex].box);
1598 ranksFileOrder[indexFileOrder] = currentRank;
1599 frc[frcIndex].rankToRead = currentRank;
1600 frc[frcIndex].faIndex = indexFileOrder;
1601 readFileRanks[fileName].insert(currentRank);
1602
1603 ++frcIndex;
1604 ++indexFileOrder;
1605 }
1606 ++currentRank;
1607 currentRank = std::min(currentRank, nProcs - 1);
1608 }
1609 ++currentFileIndex;
1610 }
1611
1612 DistributionMapping dmFileOrder(std::move(ranksFileOrder));
1613
1614 bool inFileOrder(mf.DistributionMap() == dmFileOrder && mf.boxArray() == baFileOrder);
1615 if(inFileOrder) {
1616 if(myProc == coordinatorProc && verbose) {
1617 amrex::AllPrint() << "VisMF::Read: inFileOrder" << std::endl;
1618 }
1619 } else {
1620 if(myProc == coordinatorProc && verbose) {
1621 amrex::AllPrint() << "VisMF::Read: not inFileOrder" << std::endl;
1622 }
1623 // ---- make a temporary fabarray in file order
1624 fafabFileOrder.define(baFileOrder, dmFileOrder, hdr.m_ncomp, hdr.m_ngrow, MFInfo(), mf.Factory());
1625 }
1626
1627 FabArray<FArrayBox> &whichFA = inFileOrder ? mf : fafabFileOrder;
1628
1629 // ---- check that a rank only needs to read one file
1630 std::map<std::string, std::set<int> >::iterator rfrIter;
1631 std::set<int>::iterator setIter;
1632
1633 for(rfrIter = readFileRanks.begin(); rfrIter != readFileRanks.end(); ++rfrIter) {
1634 std::set<int> &rfrSplitSet = rfrIter->second;
1635 if(rfrSplitSet.size() == 0) {
1636 continue;
1637 }
1638 // ---- split the set into nstreams sets
1639 int ssSize(rfrSplitSet.size());
1640 int nStreams(std::min(ssSize, nOpensPerFile));
1641 int ranksPerStream(ssSize / nStreams); // ---- plus some remainder...
1642 Vector<std::set<int> > streamSets(nStreams);
1643 int sIndex(0), sCount(0);
1644 for(setIter = rfrSplitSet.begin(); setIter != rfrSplitSet.end(); ++setIter) {
1645 streamSets[sIndex].insert(*setIter);
1646 if(++sCount >= ranksPerStream) {
1647 sCount = 0;
1648 ++sIndex;
1649 sIndex = std::min<int>(sIndex, streamSets.size() - 1);
1650 }
1651 }
1652
1653 for(int iSet(0); iSet < streamSets.size(); ++iSet) {
1654 Vector<int> readRanks;
1655 std::set<int> &rfrSet = streamSets[iSet];
1656 for(setIter = rfrSet.begin(); setIter != rfrSet.end(); ++setIter) {
1657 readRanks.push_back(*setIter);
1658 }
1659
1660 if(rfrSet.find(myProc) != rfrSet.end()) { // ---- myProc needs to read this file
1661 const std::string &fileName = rfrIter->first;
1662 std::string fullFileName(VisMF::DirName(mf_name) + fileName);
1663 frcIter = FileReadChains.find(fileName);
1664 BL_ASSERT(frcIter != FileReadChains.end());
1665 Vector<FabReadLink> &frc = frcIter->second;
1666 for(NFilesIter nfi(fullFileName, readRanks); nfi.ReadyToRead(); ++nfi) {
1667
1668 // ---- confirm the data is contiguous in the stream
1669 Long firstOffset(-1);
1670 for(int i(0); i < frc.size(); ++i) {
1671 if(myProc == frc[i].rankToRead) {
1672 firstOffset = frc[i].fileOffset;
1673 break;
1674 }
1675 }
1676
1677 bool dataIsContiguous(true);
1678 Long currentOffset(firstOffset), bytesToRead(0);
1679 int nFABs(0);
1680
1681 for(int i(0); i < frc.size(); ++i) {
1682 if(myProc == frc[i].rankToRead) {
1683 if(currentOffset != frc[i].fileOffset) {
1684 dataIsContiguous = false;
1685 } else {
1686 FArrayBox &fab = whichFA[frc[i].faIndex];
1687 Long fabBytesToRead(fab.box().numPts() * fab.nComp() * hdr.m_writtenRD.numBytes());
1688 currentOffset += fabBytesToRead;
1689 bytesToRead += fabBytesToRead;
1690 ++nFABs;
1691 }
1692 }
1693 }
1694 char *allFabData;
1695 bool canCombineFABs(false);
1696 if(nFABs > 1 && dataIsContiguous && VisMF::useSingleRead) {
1697 allFabData = new(std::nothrow) char[bytesToRead];
1698 if(allFabData == nullptr) {
1699 canCombineFABs = false;
1700 } else {
1701 canCombineFABs = true;
1702 }
1703 }
1704 if(canCombineFABs) {
1705 nfi.Stream().seekp(firstOffset, std::ios::beg);
1706 nfi.Stream().read(allFabData, bytesToRead);
1707
1708 currentOffset = 0; // ---- this is now relative to allFabData
1709
1710 for(int i(0); i < frc.size(); ++i) {
1711 if(myProc == frc[i].rankToRead) {
1712 char *afPtr = allFabData + currentOffset;
1713 FArrayBox &fab = whichFA[frc[i].faIndex];
1714 Long readDataItems(fab.box().numPts() * fab.nComp());
1715 if(doConvert) {
1716 RealDescriptor::convertToNativeFormat(fab.dataPtr(), readDataItems,
1717 afPtr, hdr.m_writtenRD);
1718 } else {
1719 memcpy(fab.dataPtr(), afPtr, fab.nBytes());
1720 }
1721 currentOffset += readDataItems * hdr.m_writtenRD.numBytes();
1722 }
1723 }
1724 delete [] allFabData;
1725
1726 } else { // ---- cannot use one read
1727 for(int i(0); i < frc.size(); ++i) {
1728 if(myProc == frc[i].rankToRead) {
1729 if(static_cast<std::streamoff>(nfi.SeekPos()) != frc[i].fileOffset) {
1730 nfi.Stream().seekp(frc[i].fileOffset, std::ios::beg);
1731 }
1732 FArrayBox &fab = whichFA[frc[i].faIndex];
1733 Long readDataItems(fab.box().numPts() * fab.nComp());
1734 if(doConvert) {
1735 RealDescriptor::convertToNativeFormat(fab.dataPtr(), readDataItems,
1736 nfi.Stream(), hdr.m_writtenRD);
1737 } else {
1738 nfi.Stream().read((char *) fab.dataPtr(), fab.nBytes());
1739 }
1740 }
1741 }
1742 }
1743
1744 } // ---- end NFilesIter
1745 }
1746
1747 }
1748 }
1749
1750 if( ! inFileOrder) {
1751 faCopyTime = amrex::second();
1752 mf.copy(fafabFileOrder);
1753 faCopyTime = amrex::second() - faCopyTime;
1754 }
1755
1756 } else { // ---- (noFabHeader && useSynchronousReads) == false
1757
1758 int nReqs(0), ioProcNum(coordinatorProc);
1759 int nBoxes(hdr.m_ba.size());
1760 int totalIOReqs(nBoxes), nFiles(-1);
1761 std::vector<int> iDone(2);
1762 const int iDoneIndex(0), iDoneCount(1);
1763 std::set<int> busyProcs; // [whichProc]
1764 std::map<std::string, int> fileNames; // <filename, allreadsindex>
1765 std::multiset<int> availableFiles; // [whichFile] supports multiple reads/file
1766 int allReadsIndex(0);
1767 ParallelDescriptor::Message rmess;
1768 Vector<std::map<int,std::map<Long,int> > > allReads; // [file]<proc,<seek,index>>
1769
1770
1771 for(int i(0); i < nBoxes; ++i) { // count the files
1772 int whichProc(mf.DistributionMap()[i]);
1773 if(whichProc == myProc) {
1774 ++nReqs;
1775 }
1776 if(myProc == coordinatorProc) {
1777 std::string fname(hdr.m_fod[i].m_name);
1778 if(fileNames.insert(std::pair<std::string,int>(fname,allReadsIndex)).second)
1779 {
1780 ++allReadsIndex;
1781 }
1782 }
1783 }
1784
1785 if(myProc == coordinatorProc) { // fill availableFiles
1786 nFiles = fileNames.size();
1787 for(int i(0); i < nFiles; ++i) {
1788 for(int nOpens(0); nOpens < nOpensPerFile; ++nOpens) {
1789 availableFiles.insert(i);
1790 }
1791 }
1792 allReads.resize(nFiles);
1793 int whichProc;
1794 Long iSeekPos;
1795 std::map<std::string, int>::iterator fileNamesIter;
1796 for(int i(0); i < nBoxes; ++i) { // fill allReads maps
1797 whichProc = mf.DistributionMap()[i];
1798 iSeekPos = hdr.m_fod[i].m_head;
1799 std::string fname(hdr.m_fod[i].m_name);
1800 fileNamesIter = fileNames.find(fname);
1801 if(fileNamesIter != fileNames.end()) {
1802 int findex(fileNames.find(fname)->second);
1803 allReads[findex][whichProc].insert(std::pair<Long, int>(iSeekPos, i));
1804 } else {
1805 amrex::ErrorStream() << "**** Error: filename not found = " << fname << std::endl;
1806 amrex::Abort("**** Error in VisMF::Read");
1807 }
1808 }
1809 }
1810
1811 int readTag(ParallelDescriptor::SeqNum());
1812 int doneTag(ParallelDescriptor::SeqNum());
1813
1814 if(myProc == coordinatorProc) { // manage the file locks
1815 int reqsPending(0), iopFileIndex;
1816 std::deque<int> iopReads;
1817 MPI_Status status;
1818 int doneFlag;
1819 while(totalIOReqs > 0) {
1820 std::vector<int> vReads;
1821 std::multiset<int>::iterator aFilesIter;
1822 aFilesIter = availableFiles.begin();
1823 while(aFilesIter != availableFiles.end()) { // handle available files
1824 int arIndex(*aFilesIter);
1825 if(allReads[arIndex].empty()) {
1826 availableFiles.erase(arIndex);
1827 aFilesIter = availableFiles.begin();
1828 continue;
1829 }
1830 std::map<int,std::map<Long,int> >::iterator whichRead;
1831 for(whichRead = allReads[arIndex].begin();
1832 whichRead != allReads[arIndex].end(); ++whichRead)
1833 {
1834 int tryProc(whichRead->first);
1835 if(busyProcs.find(tryProc) == busyProcs.end()) { // tryProc not busy
1836 busyProcs.insert(tryProc);
1837 int nReads(whichRead->second.size());
1838 int ir(0);
1839 vReads.resize(nReads);
1840 std::map<Long,int>::iterator imiter;
1841 for(imiter = whichRead->second.begin();
1842 imiter != whichRead->second.end(); ++imiter)
1843 {
1844 vReads[ir] = imiter->second; // the mfindex
1845 ++ir;
1846 }
1847 if(tryProc == ioProcNum) {
1848 iopFileIndex = arIndex;
1849 for(int irr(0); irr < nReads; ++irr) {
1850 iopReads.push_back(vReads[irr]);
1851 }
1852 } else {
1853 ParallelDescriptor::Send(vReads, tryProc, readTag);
1854 ++messTotal;
1855 ++reqsPending;
1856 }
1857 availableFiles.erase(aFilesIter);
1858 aFilesIter = availableFiles.begin();
1859 break;
1860 }
1861 }
1862 if(whichRead == allReads[arIndex].end()) {
1863 ++aFilesIter;
1864 } else {
1865 allReads[arIndex].erase(whichRead);
1866 }
1867 } // end while(aFilesIter...)
1868
1869 while( ! iopReads.empty()) {
1870 int index(iopReads.front());
1871 VisMF::readFAB(mf,index, mf_name, hdr);
1872 --totalIOReqs;
1873 iopReads.pop_front();
1874 if(iopReads.empty()) {
1875 availableFiles.insert(iopFileIndex);
1876 busyProcs.erase(ioProcNum);
1877 }
1878 ParallelDescriptor::IProbe(MPI_ANY_SOURCE, doneTag, doneFlag, status);
1879 if(doneFlag) {
1880 break;
1881 }
1882 }
1883
1884 if(reqsPending > 0) {
1885 rmess = ParallelDescriptor::Recv(iDone, MPI_ANY_SOURCE, doneTag);
1886
1887 int index(iDone[iDoneIndex]);
1888 int procDone(rmess.pid());
1889 totalIOReqs -= iDone[iDoneCount];
1890 --reqsPending;
1891 busyProcs.erase(procDone);
1892 std::string fname(hdr.m_fod[index].m_name);
1893 int fileIndex(fileNames.find(fname)->second);
1894 availableFiles.insert(fileIndex);
1895 }
1896
1897 } // end while(totalIOReqs...)
1898
1899 } else {
1900 std::vector<int> recReads(nReqs, -1);
1901 while(nReqs > 0) {
1902 rmess = ParallelDescriptor::Recv(recReads, ioProcNum, readTag);
1903 for(int ir(0); ir < static_cast<int>(rmess.count()); ++ir) {
1904 int mfIndex(recReads[ir]);
1905 VisMF::readFAB(mf,mfIndex, mf_name, hdr);
1906 }
1907 nReqs -= rmess.count();
1908 iDone[iDoneIndex] = recReads[0];
1909 iDone[iDoneCount] = rmess.count();
1910 ParallelDescriptor::Send(iDone, ioProcNum, doneTag);
1911 }
1912 }
1913
1914 }
1915
1916#else
1917 for(MFIter mfi(mf); mfi.isValid(); ++mfi) {
1918 VisMF::readFAB(mf,mfi.index(), mf_name, hdr);
1919 }
1920#endif
1921
1922 if(VisMF::GetUsePersistentIFStreams()) {
1923 for(int idx(0); idx < hdr.m_fod.size(); ++idx) {
1924 std::string FullName(VisMF::DirName(mf_name));
1925 FullName += hdr.m_fod[idx].m_name;
1926 VisMF::DeleteStream(FullName);
1927 }
1928 }
1929
1930 if(myProc == coordinatorProc && verbose) {
1931 Real mfReadTime = amrex::second() - startTime;
1932 totalTime += mfReadTime;
1933 amrex::AllPrint() << "FARead :: nBoxes = " << hdr.m_ba.size()
1934 << " nMessages = " << messTotal << '\n'
1935 << "FARead :: hTime = " << (hEndTime - hStartTime) << '\n'
1936 << "FARead :: faCopyTime = " << faCopyTime << '\n'
1937 << "FARead :: mfReadTime = " << mfReadTime
1938 << " totalTime = " << totalTime << std::endl;
1939 }
1940
1941 BL_ASSERT(mf.ok());
1942
1943 if (Gpu::inLaunchRegion()) amrex::prefetchToDevice(mf);
1944}
