31const double PI = acos(-1.0);

33double perturbed_x(double q, double current_z, double final_z, double k)
34{
35 // Takes the unperturbed position of the particle on a regular grid, and
36 // returns the "perturbed" position, in comoving units.
37
38 return (q - (1.0 + final_z) / (1.0 + current_z) * sin(k * q) / k);
39}
40
41double perturbed_v_x(double q, double current_z, double final_z,
42 double k, double H_0)
43{
44 // Returns the "perturbed" velocity, in comoving units.
45 return (-H_0 * (1.0 + final_z) * sqrt(1.0 + current_z) * sin(k * q) / k);
46}
47
48double perturbed_rho(double q, double average_rho, double current_z,
49 double final_z, double k)
50{
51 // Returns the "perturbed" density, in proper units.
52 return (average_rho / (1.0 - (1.0 + final_z) / (1.0 + current_z)
53 * cos(k * q)));
54}

56double get_scale_factor(double current_z)
57{
58 // Take in the redshift, `z`, return the scale factor, `a`.
59 return 1.0 / (1.0 + current_z);
60}

62void write_particle(ofstream& p_file, double x, double y, double z,
63 double mass, double v_x, double v_y, double v_z)
64{
65 // Write the particle data, (x, y, z, mass, v_x, v_y, v_z), to the given
66 // file. This is just an abstraction for the format Nyx expects.
67 p_file << x << " " << y << " " << z << " " << mass << " " << v_x << " "
68 << v_y << " " << v_z << std::endl;
69}
