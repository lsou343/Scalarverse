121 double dominant_eigenvalue = -1e9;

123 PRAGMA_THREAD_ACROSS_BLOCKS_MAX(level,block,level->num_my_blocks,dominant_eigenvalue)
124 for(block=0;block<level->num_my_blocks;block++){
125 const int box = level->my_blocks[block].read.box;
126 const int ilo = level->my_blocks[block].read.i;
127 const int jlo = level->my_blocks[block].read.j;
128 const int klo = level->my_blocks[block].read.k;

132 int i,j,k;
133 const int jStride = level->my_boxes[box].jStride;
134 const int kStride = level->my_boxes[box].kStride;
135 const int ghosts = level->my_boxes[box].ghosts;
136 double h2inv = 1.0/(level->h*level->h);
137 double * __restrict__ alpha = level->my_boxes[box].vectors[VECTOR_ALPHA ] + ghosts*(1+jStride+kStride);
138 double * __restrict__ beta_i = level->my_boxes[box].vectors[VECTOR_BETA_I] + ghosts*(1+jStride+kStride);
139 double * __restrict__ beta_j = level->my_boxes[box].vectors[VECTOR_BETA_J] + ghosts*(1+jStride+kStride);
140 double * __restrict__ beta_k = level->my_boxes[box].vectors[VECTOR_BETA_K] + ghosts*(1+jStride+kStride);
141 double * __restrict__ Dinv = level->my_boxes[box].vectors[VECTOR_DINV ] + ghosts*(1+jStride+kStride);
142 double * __restrict__ L1inv = level->my_boxes[box].vectors[VECTOR_L1INV ] + ghosts*(1+jStride+kStride);
143 double block_eigenvalue = -1e9;
144
145 for(k=klo;k<khi;k++){
146 for(j=jlo;j<jhi;j++){
147 for(i=ilo;i<ihi;i++){
148 int ijk = i + j*jStride + k*kStride;
149
150 // used for quick linear approximation to zero dirichlet BC
151 double ilo_is_valid =1.0;
152 double ihi_is_valid =1.0;
153 double jlo_is_valid =1.0;
154 double jhi_is_valid =1.0;
155 double klo_is_valid =1.0;
156 double khi_is_valid =1.0;
157 if(level->boundary_condition.type != BC_PERIODIC){
158 if(level->my_boxes[box].low.i+i-1 < 0)ilo_is_valid = 0.0;
159 if(level->my_boxes[box].low.j+j-1 < 0)jlo_is_valid = 0.0;
160 if(level->my_boxes[box].low.k+k-1 < 0)klo_is_valid = 0.0;
161 if(level->my_boxes[box].low.i+i+1 >= level->dim.i)ihi_is_valid = 0.0;
162 if(level->my_boxes[box].low.j+j+1 >= level->dim.j)jhi_is_valid = 0.0;
163 if(level->my_boxes[box].low.k+k+1 >= level->dim.k)khi_is_valid = 0.0;
164 }
165
166 #ifdef STENCIL_VARIABLE_COEFFICIENT
167 // radius of Gershgorin disc is the sum of the absolute values of the off-diagonal elements...
168 double sumAbsAij = fabs(b*h2inv) * (
169 fabs( beta_i[ijk ]*ilo_is_valid )+
170 fabs( beta_j[ijk ]*jlo_is_valid )+
171 fabs( beta_k[ijk ]*klo_is_valid )+
172 fabs( beta_i[ijk+1 ]*ihi_is_valid )+
173 fabs( beta_j[ijk+jStride]*jhi_is_valid )+
174 fabs( beta_k[ijk+kStride]*khi_is_valid )
175 );
176

178 double Aii = a*alpha[ijk] - b*h2inv*(
179 beta_i[ijk ]*( ilo_is_valid-2.0 )+
180 beta_j[ijk ]*( jlo_is_valid-2.0 )+
181 beta_k[ijk ]*( klo_is_valid-2.0 )+
182 beta_i[ijk+1 ]*( ihi_is_valid-2.0 )+
183 beta_j[ijk+jStride]*( jhi_is_valid-2.0 )+
184 beta_k[ijk+kStride]*( khi_is_valid-2.0 )
185 );
186 #else // Constant coefficient versions with fused BC's...
187 // radius of Gershgorin disc is the sum of the absolute values of the off-diagonal elements...
188 double sumAbsAij = fabs(b*h2inv) * (
189 ilo_is_valid +
190 jlo_is_valid +
191 klo_is_valid +
192 ihi_is_valid +
193 jhi_is_valid +
194 khi_is_valid
195 );
196
197 // center of Gershgorin disc is the diagonal element...
198 double Aii = a - b*h2inv*(
199 ilo_is_valid +
200 jlo_is_valid +
201 klo_is_valid +
202 ihi_is_valid +
203 jhi_is_valid +
204 khi_is_valid - 12.0
205 );
206 #endif

208 // calculate Dinv = D^{-1}, L1inv = ( D+D^{L1} )^{-1}, and the dominant eigenvalue...
209 Dinv[ijk] = 1.0/Aii; // inverse of the diagonal Aii

214 }}}

216 }
