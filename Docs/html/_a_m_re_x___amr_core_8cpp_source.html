66void
67AmrCore::regrid (int lbase, Real time, bool)
68{
69 if (lbase >= max_level) return;
70
71 int new_finest;
72 Vector<BoxArray> new_grids(finest_level+2);
73 MakeNewGrids(lbase, time, new_finest, new_grids);
74
75 BL_ASSERT(new_finest <= finest_level+1);
76
77 bool coarse_ba_changed = false;
78 for (int lev = lbase+1; lev <= new_finest; ++lev)
79 {
80 if (lev <= finest_level) // an old level
81 {
82 bool ba_changed = (new_grids[lev] != grids[lev]);
83 if (ba_changed or coarse_ba_changed) {
84 BoxArray level_grids = grids[lev];
85 DistributionMapping level_dmap = dmap[lev];
86 if (ba_changed) {
87 level_grids = new_grids[lev];
88 level_dmap = DistributionMapping(level_grids);
89 }
90 const auto old_num_setdm = num_setdm;
91 RemakeLevel(lev, time, level_grids, level_dmap);
92 SetBoxArray(lev, level_grids);
93 if (old_num_setdm == num_setdm) {
94 SetDistributionMap(lev, level_dmap);
95 }
96 }
97 coarse_ba_changed = ba_changed;;
98 }
99 else // a new level
100 {
101 DistributionMapping new_dmap(new_grids[lev]);
102 const auto old_num_setdm = num_setdm;
103 MakeNewLevelFromCoarse(lev, time, new_grids[lev], new_dmap);
104 SetBoxArray(lev, new_grids[lev]);
105 if (old_num_setdm == num_setdm) {
106 SetDistributionMap(lev, new_dmap);
107 }
108 }
109 }
110
111 for (int lev = new_finest+1; lev <= finest_level; ++lev) {
112 ClearLevel(lev);
113 ClearBoxArray(lev);
114 ClearDistributionMap(lev);
115 }
116
117 finest_level = new_finest;
118}
