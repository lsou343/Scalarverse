143 attributes(device) function warpreducesum(x) result(y)
144 ! Reduce within a warp.
145 ! https://devblogs.nvidia.com/faster-parallel-reductions-kepler/
146
147 implicit none

171 attributes(device) function blockreducesum(x) result(y)
172 ! Reduce within a threadblock.
173 ! https://devblogs.nvidia.com/faster-parallel-reductions-kepler/
174
175 implicit none
176
177 real(amrex_real), intent(in) :: x
178
179 real(amrex_real) :: y
180
181 real(amrex_real), shared :: s(0:(AMREX_GPU_MAX_THREADS/warpsize) - 1)
182
183 integer :: lane, wid
184
185 lane = mod(threadidx%x-1, warpsize)
186 wid = (threadidx%x-1) / warpsize
187
188 y = x
189 y = warpreducesum(y)
190
191 ! syncthreads() prior to writing to shared memory is necessary
192 ! if this reduction call is occurring multiple times in a kernel,
193 ! and since we don't know how many times the user is calling it,
194 ! we do it always to be safe.
195
196 call syncthreads()
197
198 if (lane == 0) then
199 s(wid) = y
200 end if
201
202 call syncthreads()
203
204 if ((threadidx%x-1) < max(blockdim%x, warpsize) / warpsize) then
205 y = s(lane)
206 else
207 y = 0
208 end if
209
210 if (wid == 0) then
211 y = warpreducesum(y)
212 end if
213
214 end function blockreducesum
215#endif
