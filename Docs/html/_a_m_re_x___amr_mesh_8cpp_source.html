421void
422AmrMesh::ChopGrids (int lev, BoxArray& ba, int target_size) const
423{
424 for (int cnt = 1; cnt <= 4; cnt *= 2)
425 {
426 IntVect chunk = max_grid_size[lev] / cnt;
427
428 for (int j = AMREX_SPACEDIM-1; j >= 0 ; j--)
429 {
430 chunk[j] /= 2;
431
432 if ( (ba.size() < target_size) && (chunk[j]%blocking_factor[lev][j] == 0) )
433 {
434 ba.maxSize(chunk);
435 }
436 }
437 }

440BoxArray
441AmrMesh::MakeBaseGrids () const
442{
443 IntVect fac(2);
444 const Box& dom = geom[0].Domain();
445 const Box dom2 = amrex::refine(amrex::coarsen(dom,2),2);
446 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
447 if (dom.length(idim) != dom2.length(idim)) {
448 fac[idim] = 1;
449 }
450 }
451 BoxArray ba(amrex::coarsen(dom,fac));
452 ba.maxSize(max_grid_size[0]/fac);
453 ba.refine(fac);
454 // Boxes in ba have even number of cells in each direction
455 // unless the domain has odd number of cells in that direction.
456 if (refine_grid_layout) {
457 ChopGrids(0, ba, ParallelDescriptor::NProcs());
458 }
459 if (ba == grids[0]) {
460 ba = grids[0]; // to avoid duplicates
461 }
462 PostProcessBaseGrids(ba);
463 return ba;
464}

467void
468AmrMesh::MakeNewGrids (int lbase, Real time, int& new_finest, Vector<BoxArray>& new_grids)
469{
470 BL_PROFILE("AmrMesh::MakeNewGrids()");
471
472 BL_ASSERT(lbase < max_level);
473
474 // Add at most one new level
475 int max_crse = std::min(finest_level, max_level-1);
476
477 if (new_grids.size() < max_crse+2) new_grids.resize(max_crse+2);
478
479 //
480 // Construct problem domain at each level.
481 //
482 Vector<IntVect> bf_lev(max_level); // Blocking factor at each level.
483 Vector<IntVect> rr_lev(max_level);
484 Vector<Box> pc_domain(max_level); // Coarsened problem domain.
485
486 for (int i = 0; i <= max_crse; i++)
487 {
488 for (int n=0; n<AMREX_SPACEDIM; n++) {
489 bf_lev[i][n] = std::max(1,blocking_factor[i+1][n]/ref_ratio[i][n]);
490 }
491 }
492 for (int i = lbase; i < max_crse; i++)
493 {
494 for (int n=0; n<AMREX_SPACEDIM; n++)
495 rr_lev[i][n] = (ref_ratio[i][n]*bf_lev[i][n])/bf_lev[i+1][n];
496 }
497 for (int i = lbase; i <= max_crse; i++) {
498 pc_domain[i] = amrex::coarsen(Geom(i).Domain(),bf_lev[i]);
499 }
500 //
501 // Construct proper nesting domains.
502 //
503 Vector<BoxList> p_n(max_level); // Proper nesting domain.
504 Vector<BoxList> p_n_comp(max_level); // Complement proper nesting domain.
505
506 BoxList bl = grids[lbase].simplified_list();
507 bl.coarsen(bf_lev[lbase]);
508 p_n_comp[lbase].parallelComplementIn(pc_domain[lbase],bl);
509 p_n_comp[lbase].simplify();
510 p_n_comp[lbase].accrete(n_proper);
511 if (geom[lbase].isAnyPeriodic()) {
512 ProjPeriodic(p_n_comp[lbase], pc_domain[lbase],
513 geom[lbase].isPeriodic());
514 }
515 p_n[lbase].parallelComplementIn(pc_domain[lbase],p_n_comp[lbase]);
516 p_n[lbase].simplify();
517 bl.clear();
518
519 for (int i = lbase+1; i <= max_crse; i++)
520 {
521 p_n_comp[i] = p_n_comp[i-1];
522
523 // Need to simplify p_n_comp or the number of grids can too large for many levels.
524 p_n_comp[i].simplify();
525
526 p_n_comp[i].refine(rr_lev[i-1]);
527 p_n_comp[i].accrete(n_proper);
528
529 if (geom[i].isAnyPeriodic()) {
530 ProjPeriodic(p_n_comp[i], pc_domain[i], geom[i].isPeriodic());
531 }
532
533 p_n[i].parallelComplementIn(pc_domain[i],p_n_comp[i]);
534 p_n[i].simplify();
535 }
536
537 //
538 // Now generate grids from finest level down.
539 //
540 new_finest = lbase;
541
542 for (int levc = max_crse; levc >= lbase; levc--)
543 {
544 int levf = levc+1;
545 //
546 // Construct TagBoxArray with sufficient grow factor to contain
547 // new levels projected down to this level.
548 //
549 int ngrow = 0;
550
551 if (levf < new_finest)
552 {
553 BoxArray ba_proj = new_grids[levf+1].simplified();
554
555 ba_proj.coarsen(ref_ratio[levf]);
556 ba_proj.growcoarsen(n_proper, ref_ratio[levc]);
557
558 BoxArray levcBA = grids[levc].simplified();
559
560 while (!levcBA.contains(ba_proj))
561 {
562 levcBA.grow(1);
563 ++ngrow;
564 }
565 }
566 TagBoxArray tags(grids[levc],dmap[levc],n_error_buf[levc]+ngrow);
567
568 //
569 // Only use error estimation to tag cells for the creation of new grids
570 // if the grids at that level aren't already fixed.
571 //
572
573 if ( ! (useFixedCoarseGrids() && levc < useFixedUpToLevel()) ) {
574 ErrorEst(levc, tags, time, ngrow);
575 }
576
577 //
578 // If new grids have been constructed above this level, project
579 // those grids down and tag cells on intersections to ensure
580 // proper nesting.
581 //
582 // NOTE: this loop replaces the previous code:
583 // if (levf < new_finest)
584 // tags.setVal(ba_proj,TagBox::SET);
585 // The problem with this code is that it effectively
586 // "buffered the buffer cells", i.e., the grids at level
587 // levf+1 which were created by buffering with n_error_buf[levf][idim]
588 // are then coarsened down twice to define tagging at
589 // level levc, which will then also be buffered. This can
590 // create grids which are larger than necessary.
591 //
592 if (levf < new_finest)
593 {
594 // Replace this by n_error_buf that may be anisotropic
595 // int nerr = n_error_buf[levf];
596
597 BoxList bl_tagged = new_grids[levf+1].simplified_list();
598 bl_tagged.coarsen(ref_ratio[levf]);
599 //
600 // This grows the boxes by n_error_buf[levf][idir] if they touch the edge
601 // of the domain in preparation for them being shrunk by n_error_buf[levf][idir] later.
602 // We want the net effect to be that grids are NOT shrunk away
603 // from the edges of the domain.
604 //
605 for (BoxList::iterator blt = bl_tagged.begin(), End = bl_tagged.end();
606 blt != End;
607 ++blt)
608 {
609 for (int idir = 0; idir < AMREX_SPACEDIM; idir++)
610 {
611 if (blt->smallEnd(idir) == Geom(levf).Domain().smallEnd(idir))
612 blt->growLo(idir,n_error_buf[levf][idir]);
613 if (blt->bigEnd(idir) == Geom(levf).Domain().bigEnd(idir))
614 blt->growHi(idir,n_error_buf[levf][idir]);
615 }
616 }
617 Box mboxF = amrex::grow(bl_tagged.minimalBox(),1);
618 BoxList blFcomp;
619 blFcomp.parallelComplementIn(mboxF,bl_tagged);
620 blFcomp.simplify();
621 bl_tagged.clear();
622
623 const IntVect& iv = IntVect(AMREX_D_DECL(n_error_buf[levf][0]/ref_ratio[levf][0],
624 n_error_buf[levf][1]/ref_ratio[levf][1],
625 n_error_buf[levf][2]/ref_ratio[levf][2]));
626 blFcomp.accrete(iv);
627 BoxList blF;
628 blF.parallelComplementIn(mboxF,blFcomp);
629 BoxArray baF(std::move(blF));
630 baF.grow(n_proper);
631 //
632 // We need to do this in case the error buffering at
633 // levc will not be enough to cover the error buffering
634 // at levf which was just subtracted off.
635 //
636 for (int idir = 0; idir < AMREX_SPACEDIM; idir++)
637 {
638 if ( n_error_buf[levf][idir] > n_error_buf[levc][idir]*ref_ratio[levc][idir])
639 baF.grow(idir,n_error_buf[levf][idir] - n_error_buf[levc][idir]*ref_ratio[levc][idir]);
640 }
641
642 baF.coarsen(ref_ratio[levc]);
643
644 tags.setVal(baF,TagBox::SET);
645 }
646 //
647 // Buffer error cells.
648 //
649 tags.buffer(n_error_buf[levc]+ngrow);
650
651 if (useFixedCoarseGrids())
652 {
653 if (levc>=useFixedUpToLevel())
654 {
655 tags.setVal(GetAreaNotToTag(levc), TagBox::CLEAR);
656 }
657 else
658 {
659 new_finest = std::max(new_finest,levf);
660 }
661 }
662
663 //
664 // Coarsen the taglist by blocking_factor/ref_ratio.
665 //
666 int bl_max = 0;
667 for (int n=0; n<AMREX_SPACEDIM; n++) {
668 bl_max = std::max(bl_max,bf_lev[levc][n]);
669 }
670 if (bl_max >= 1) {
671 tags.coarsen(bf_lev[levc]);
672 } else {
673 amrex::Abort("blocking factor is too small relative to ref_ratio");
674 }
675 //
676 // Remove or add tagged points which violate/satisfy additional
677 // user-specified criteria.
678 //
679 ManualTagsPlacement(levc, tags, bf_lev);
680 //
681 // Map tagged points through periodic boundaries, if any.
682 //
683 tags.mapPeriodicRemoveDuplicates(Geometry(pc_domain[levc],
684 Geom(levc).ProbDomain(),
685 Geom(levc).CoordInt(),
686 Geom(levc).isPeriodic()));
687 //
688 // Remove cells outside proper nesting domain for this level.
689 //
690 tags.setVal(p_n_comp[levc],TagBox::CLEAR);
691 //
692 // Create initial cluster containing all tagged points.
693 //
694 Vector<IntVect> tagvec;
695 tags.collate(tagvec);
696 tags.clear();
697
698 if (tagvec.size() > 0)
699 {
700 //
701 // Created new level, now generate efficient grids.
702 //
703 if ( !(useFixedCoarseGrids() && levc<useFixedUpToLevel()) ) {
704 new_finest = std::max(new_finest,levf);
705 }
706
707 if (levf > useFixedUpToLevel()) {
708 BoxList new_bx;
709 if (ParallelDescriptor::IOProcessor()) {
710 BL_PROFILE("AmrMesh-cluster");
711 //
712 // Construct initial cluster.
713 //
714 ClusterList clist(&tagvec[0], tagvec.size());
715 if (use_new_chop) {
716 clist.new_chop(grid_eff);
717 } else {
718 clist.chop(grid_eff);
719 }
720 BoxDomain bd;
721 bd.add(p_n[levc]);
722 clist.intersect(bd);
723 bd.clear();
724 //
725 // Efficient properly nested Clusters have been constructed
726 // now generate list of grids at level levf.
727 //
728 clist.boxList(new_bx);
729 new_bx.refine(bf_lev[levc]);
730 new_bx.simplify();
731
732 if (new_bx.size()>0) {
733 // Chop new grids outside domain
734 new_bx.intersect(Geom(levc).Domain());
735 }
736 }
737 new_bx.Bcast(); // Broadcast the new BoxList to other processes
738
739 //
740 // Refine up to levf.
741 //
742 new_bx.refine(ref_ratio[levc]);
743 BL_ASSERT(new_bx.isDisjoint());
744
745 new_grids[levf] = BoxArray(std::move(new_bx), max_grid_size[levf]);
746 }
747 }
748 }
749
750#ifdef AMREX_DEBUG
751 if (!useFixedCoarseGrids()) {
752 // check proper nesting
753 for (int lev = lbase+1; lev <= new_finest; ++lev) {
754 BoxArray const& cba = (lev == lbase+1) ? grids[lev-1] : new_grids[lev-1];
755 BoxArray const& fba = amrex::coarsen(new_grids[lev],ref_ratio[lev-1]);
756 IntVect np = bf_lev[lev-1] * n_proper;
757 Box const& cdomain = Geom(lev-1).Domain();
758 for (int i = 0, N = fba.size(); i < N; ++i) {
759 Box const& fb = amrex::grow(fba[i],np) & cdomain;
760 if (!cba.contains(fb,true)) {
761 amrex::Abort("AmrMesh::MakeNewGrids: new grids not properly nested");
762 }
763 }
764 }
765 }
766#endif
767
768 for (int lev = lbase+1; lev <= new_finest; ++lev) {
769 if (new_grids[lev].empty())
770 {
771 if (!(useFixedCoarseGrids() && lev<useFixedUpToLevel()) ) {
772 amrex::Abort("AmrMesh::MakeNewGrids: how did this happen?");
773 }
774 }
775 else if (refine_grid_layout)
776 {
777 ChopGrids(lev,new_grids[lev],ParallelDescriptor::NProcs());
778 if (new_grids[lev] == grids[lev]) {
779 new_grids[lev] = grids[lev]; // to avoid dupliates
780 }
781 }
782 }

785void
786AmrMesh::MakeNewGrids (Real time)
787{
788 // define coarse level BoxArray and DistributionMap
789 {
790 finest_level = 0;
791
792 const BoxArray& ba = MakeBaseGrids();
793 DistributionMapping dm(ba);
794 const auto old_num_setdm = num_setdm;
795 const auto old_num_setba = num_setba;
796
797 MakeNewLevelFromScratch(0, time, ba, dm);
798
799 if (old_num_setba == num_setba) {
800 SetBoxArray(0, ba);
801 }
802 if (old_num_setdm == num_setdm) {
803 SetDistributionMap(0, dm);
804 }
805 }
806
807 if (max_level > 0) // build fine levels
808 {
809 Vector<BoxArray> new_grids(max_level+1);
810 new_grids[0] = grids[0];
811 do
812 {
813 int new_finest;
814
815 // Add (at most) one level at a time.
816 MakeNewGrids(finest_level,time,new_finest,new_grids);
817
818 if (new_finest <= finest_level) break;
819 finest_level = new_finest;
820
821 DistributionMapping dm(new_grids[new_finest]);
822 const auto old_num_setdm = num_setdm;
823
824 MakeNewLevelFromScratch(new_finest, time, new_grids[finest_level], dm);
825
826 SetBoxArray(new_finest, new_grids[new_finest]);
827 if (old_num_setdm == num_setdm) {
828 SetDistributionMap(new_finest, dm);
829 }
830 }
831 while (finest_level < max_level);
832
833 // Iterate grids to ensure fine grids encompass all interesting junk.
834 if (iterate_on_new_grids)
835 {
836 for (int it=0; it<4; ++it) // try at most 4 times
837 {
838 for (int i = 1; i <= finest_level; ++i) {
839 new_grids[i] = grids[i];
840 }
841
842 int new_finest;
843 MakeNewGrids(0, time, new_finest, new_grids);
844
845 if (new_finest < finest_level) break;
846 finest_level = new_finest;
847
848 bool grids_the_same = true;
849 for (int lev = 1; lev <= new_finest; ++lev) {
850 if (new_grids[lev] != grids[lev]) {
851 grids_the_same = false;
852 DistributionMapping dm(new_grids[lev]);
853 const auto old_num_setdm = num_setdm;
854
855 MakeNewLevelFromScratch(lev, time, new_grids[lev], dm);
856
857 SetBoxArray(lev, new_grids[lev]);
858 if (old_num_setdm == num_setdm) {
859 SetDistributionMap(lev, dm);
860 }
861 }
862 }
863 if (grids_the_same) break;
864 }
865 }
866 }

869void
870AmrMesh::ProjPeriodic (BoxList& blout, const Box& domain,
871 Array<int,AMREX_SPACEDIM> const& is_per)
872{
873 //
874 // Add periodic translates to blout.
875 //
876
877 BoxList blorig(blout);
878
879 int nist,njst,nkst;
880 int niend,njend,nkend;
881 nist = njst = nkst = 0;
882 niend = njend = nkend = 0;
883 AMREX_D_TERM( nist , =njst , =nkst ) = -1;
884 AMREX_D_TERM( niend , =njend , =nkend ) = +1;
885
886 int ri,rj,rk;
887 for (ri = nist; ri <= niend; ri++)
888 {
889 if (ri != 0 && !is_per[0])
890 continue;
891 if (ri != 0 && is_per[0])
892 blorig.shift(0,ri*domain.length(0));
893 for (rj = njst; rj <= njend; rj++)
894 {
895 if (rj != 0 && !is_per[1])
896 continue;
897 if (rj != 0 && is_per[1])
898 blorig.shift(1,rj*domain.length(1));
899 for (rk = nkst; rk <= nkend; rk++)
900 {
901 if (rk != 0 && !is_per[2])
902 continue;
903 if (rk != 0 && is_per[2])
904 blorig.shift(2,rk*domain.length(2));
905
906 BoxList tmp(blorig);
907 tmp.intersect(domain);
908 blout.catenate(tmp);
909
910 if (rk != 0 && is_per[2])
911 blorig.shift(2,-rk*domain.length(2));
912 }
913 if (rj != 0 && is_per[1])
914 blorig.shift(1,-rj*domain.length(1));
915 }
916 if (ri != 0 && is_per[0])
917 blorig.shift(0,-ri*domain.length(0));
918 }

921void
922AmrMesh::checkInput ()
923{
924 if (max_level < 0)
925 amrex::Error("checkInput: max_level not set");
926
927 //
928 // Check level dependent values.
929 //
930 for (int i = 0; i < max_level; i++)
931 {
932 if (MaxRefRatio(i) < 2 || MaxRefRatio(i) > 12)
933 amrex::Error("Amr::checkInput: bad ref_ratios");
934 }
935
936 const Box& domain = Geom(0).Domain();
937 if (!domain.ok())
938 amrex::Error("level 0 domain bad or not set");
939
940 //
941 // Check that domain size is a multiple of blocking_factor[0].
942 // (only check if blocking_factor <= max_grid_size)
943 //
944 for (int idim = 0; idim < AMREX_SPACEDIM; idim++)
945 {
946 int len = domain.length(idim);
947 if (blocking_factor[0][idim] <= max_grid_size[0][idim])
948 if (len%blocking_factor[0][idim] != 0)
949 {
950 amrex::Print() << "domain size in direction " << idim << " is " << len << std::endl;
951 amrex::Print() << "blocking_factor is " << blocking_factor[0][idim] << std::endl;
952 amrex::Error("domain size not divisible by blocking_factor");
953 }
954 }
955
956 //
957 // Check that blocking_factor is a power of 2.
958 //
959 for (int i = 0; i <= max_level; i++)
960 {
961 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
962 {
963 int k = blocking_factor[i][idim];
964 while ( k > 0 && (k%2 == 0) )
965 k /= 2;
966 if (k != 1)
967 amrex::Error("Amr::checkInput: blocking_factor not power of 2. You can bypass this by setting ParmParse runtime parameter amr.check_input=0, although we do not recommend it.");
968 }
969 }
970
971 //
972 // Check that max_grid_size is a multiple of blocking_factor at every level.
973 // (only check if blocking_factor <= max_grid_size)
974 //
975 for (int i = 0; i < max_level; i++)
976 {
977 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
978 if (blocking_factor[i][idim] <= max_grid_size[i][idim])
979 if (max_grid_size[i][idim]%blocking_factor[i][idim] != 0) {
980 {
981 amrex::Print() << "max_grid_size in direction " << idim
982 << " is " << max_grid_size[i][idim] << std::endl;
983 amrex::Print() << "blocking_factor is " << blocking_factor[i][idim] << std::endl;
984 amrex::Error("max_grid_size not divisible by blocking_factor");
985 }
986 }
987 }
988 }
989
990 // Make sure TagBoxArray has no overlapped valid cells after coarsening by block_factor/ref_ratio
991 for (int i = 0; i < max_level; ++i) {
992 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
993 int bf_lev = std::max(1,blocking_factor[i+1][idim]/ref_ratio[i][idim]);
994 int min_grid_size = std::min(blocking_factor[i][idim],max_grid_size[i][idim]);
995 if (min_grid_size % bf_lev != 0) {
996 amrex::Print() << "On level " << i << " in direction " << idim
997 << " max_grid_size is " << max_grid_size[i][idim]
998 << " blocking factor is " << blocking_factor[i][idim] << "\n"
999 << "On level " << i+1 << " in direction " << idim
1000 << " blocking_factor is " << blocking_factor[i+1][idim] << std::endl;
1001 amrex::Error("Coarse level blocking factor not a multiple of fine level blocking factor divided by ref ratio");
1002 }
1003 }
1004 }
1005
1006 if( ! (Geom(0).ProbDomain().volume() > 0.0) ) {
1007 amrex::Error("Amr::checkInput: bad physical problem size");
1008 }
1009
1010 if(verbose > 0) {
1011 amrex::Print() << "Successfully read inputs file ... " << '\n';
1012 }
1013}
