184 char* argv[])
185{
186 //
187 // Make sure to catch new failures.

191 if (argc < 6 || argc > 7) print_usage(argc,argv);

199 exit(-1);

202 ParallelDescriptor::StartParallel();

204 MultiFab mf0,mf1;
205 readMF(mf0,argv[1]);
206 readMF(mf1,argv[2]);
207
208 if (mf0.boxArray() != mf1.boxArray())
209 {
210 cerr << "BoxArray's incompatible" << endl;
211 return 0;
212 }
213
214 if (mf0.nGrow() != mf1.nGrow())
215 {
216 cerr << "nGrow's incompatible" << endl;
217 return 0;
218 }
219
220 int comp0 = atoi(argv[3]);
221 int comp1 = atoi(argv[4]);
222 int nComp = atoi(argv[5]);
223 if (mf0.nComp() < comp0 + nComp || mf1.nComp() < comp1 + nComp)
224 {
225 cerr << "nComp's incompatible" << endl;
226 cerr << "(need,have): (" << comp0 + nComp << "," << mf0.nComp()
227 << "), (" << comp1 + nComp << "," << mf1.nComp() << ")" << endl;
228 return 0;
229 }
230
231 MultiFab diff(mf0.boxArray(), nComp, mf0.nGrow(), Fab_allocate);
232 for (MultiFabIterator diffmfi(diff); diffmfi.isValid(); ++diffmfi)
233 {
234 DependentMultiFabIterator mf0mfi(diffmfi, mf0);
235 DependentMultiFabIterator mf1mfi(diffmfi, mf1);
236
237 const Box& validBox = mf0mfi.validbox();
238 diffmfi().setVal(0.0);
239 diffmfi().copy(mf0mfi(),validBox,comp0,validBox,0,nComp);
240 diffmfi().minus(mf1mfi(),validBox,validBox,comp1,0,nComp);
241 }
242
243 REAL norm0 = norm(diff,0);
244 REAL norm1 = norm(diff,1);
245 REAL norm2 = norm(diff,2);
246
247 if(ParallelDescriptor::IOProcessor())
248 {
249 cout << "Norms of diff (0,1,2): "
250 << norm0 << ", " << norm1 << ", " << norm2 << endl;
251 }
252
253 if (argc == 7)
254 {
255 writeMF(&diff,argv[6]);
256 return 1;
257
258 } else {
259
260 if (norm0 == 0 && norm1 == 0 && norm2 == 0)

262 cout << "MultiFabs equal!" << endl;
263 return 1;
264

266
267 return ArrayViewMultiFab(&diff);

276// What's the slowest way I can think of to compute all the norms??
277REAL
278norm ( const MultiFab& mfab,
279 int exponent )
280{
281 int srcComp = 0;
282 int numComp = mfab.nComp();
283 int numGrow = 0;
284 const BoxArray& grids = mfab.boxArray();
285
286 // Get a copy of the multifab, zero covered locations
287 MultiFab mftmp( grids, numComp, numGrow, Fab_allocate );
288 mftmp.copy( mfab );
289
290 Real myNorm = 0;
291 if ( exponent == 0 )
292 {
293 for ( MultiFabIterator mftmpmfi(mftmp); mftmpmfi.isValid(); ++mftmpmfi)
294 {
295 mftmpmfi().abs(mftmpmfi.validbox());
296 }
297 myNorm = mftmp.max( srcComp );
298 ParallelDescriptor::ReduceRealMax(myNorm);
299
300 } else if ( exponent == 1 )
301 {
302 for ( MultiFabIterator mftmpmfi(mftmp); mftmpmfi.isValid(); ++mftmpmfi)
303 {
304 mftmpmfi().abs();
305 myNorm += mftmpmfi().sum( srcComp, numComp );
306 }
307 ParallelDescriptor::ReduceRealSum(myNorm);
308
309 } else if ( exponent == 2 )
310 {
311 for ( MultiFabIterator mftmpmfi(mftmp); mftmpmfi.isValid(); ++mftmpmfi)
312 {
313 const Box& grid = mftmpmfi.validbox();
314 for ( IntVect idx = grid.smallEnd(); idx <= grid.bigEnd(); grid.next(idx) )
315 {
316 mftmpmfi()(idx) = mftmpmfi()(idx) * mftmpmfi()(idx);
317 }
318 }
319 int sum_exp = 1;
320 for ( MultiFabIterator mftmpmfi(mftmp); mftmpmfi.isValid(); ++mftmpmfi)
321 {
322 const Box& bx = mftmpmfi.validbox();
323 myNorm += mftmpmfi().norm( bx, sum_exp, srcComp, numComp );
324 }
325 ParallelDescriptor::ReduceRealSum(myNorm);
326 myNorm = sqrt( myNorm );
327
328 } else {
329
330 amrex::Error("Invalid exponent to norm function");
331 }
332
333 return myNorm;
334}
