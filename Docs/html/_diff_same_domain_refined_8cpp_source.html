29void
30PrintUsage (const char* progName)
31{
32 std::cout << "This utility performs a diff operation between two" << std::endl
33 << "plotfiles which have the same geometrical domain" << std::endl
34 << "but a factor of refinement between" << std::endl
35 << "the cells from each plotfile at the same level." << std::endl
36 << "For instance, it works to diff two plotfiles having" << std::endl
37 << " Plotfile 1: 25x25 base grid, Ref_Ratio = 2" << std::endl
38 << " Plotfile 2: 50x50 base grid, Ref_Ratio = 2" << std::endl
39 << "Should also work for," << std::endl
40 << " Plotfile 1: 25x25 base grid, Ref_Ratio = 2" << std::endl
41 << " Plotfile 2: 25x25 base grid, Ref_Ratio = 4" << std::endl
42 << "In both cases, the geometrical region which is refined" << std::endl
43 << "must be the same. So, this is generally good for" << std::endl
44 << "comparing cases ran using a fixed grid file." << std::endl;
45 std::cout << '\n';
46 std::cout << "Usage:" << '\n';
47 std::cout << progName << '\n';
48 std::cout << " infile1 = inputFileName1" << '\n';
49 std::cout << " reffile = refinedPlotFile" << '\n';
50 std::cout << " diffile = differenceFileName" << '\n';
51 std::cout << " (If not specified no file is written)" << '\n';
52 std::cout << " norm = integer norm (Ie. default is 2 for L2 norm)" << '\n';
53 std::cout << " [-help]" << '\n';
54 std::cout << " [-verbose]" << '\n';
55 std::cout << '\n';
56 exit(1);
57}

332amrDatasHaveSameDerives(const AmrData& amrd1,
333 const AmrData& amrd2)
334{
335 const Vector<std::string>& derives1 = amrd1.PlotVarNames();
336 const Vector<std::string>& derives2 = amrd2.PlotVarNames();

347IntVect
348getRefRatio(const Box& crse,
349 const Box& fine)
350{
351 // Compute refinement ratio between crse and fine boxes, return invalid
352 // IntVect if there is none suitable
353 ParmParse pp("");
354 Vector<int> rr_in(BL_SPACEDIM,-1);
355
356 int Nrr = pp.countval("ref_ratio");
357
358 BL_ASSERT(Nrr==0 || Nrr==BL_SPACEDIM || Nrr==1);
359 if (Nrr>0)
360 {
361 pp.queryarr("ref_ratio",rr_in,0,Nrr);
362 return IntVect(rr_in);
363 }
364
365 IntVect ref_ratio;
366 for (int i=0; i<BL_SPACEDIM; ++i)
367 ref_ratio[i] = fine.length(i) / crse.length(i);
368
369 // Check results
370 Box test1 = ::Box(fine).coarsen(ref_ratio);
371 Box test2 = ::Box(test1).refine(ref_ratio);
372 if (test1 != crse || test2 != fine)
373 ref_ratio = IntVect();
374 return ref_ratio;
375}
