70{

73 const auto& tile = pti.GetParticleTile();

75 const auto& aos = tile.GetArrayOfStructs();

77 const auto& geom = pti.Geom(pti.GetLevel());
78
79 const auto domain = geom.Domain();

81 const auto dxi = geom.InvCellSizeArray();
82
83 Box box = pti.tilebox();
84 box.grow(nGrow);
85
86 ReduceOps<ReduceOpSum> reduce_op;
87 ReduceData<int> reduce_data(reduce_op);
88 using ReduceTuple = typename decltype(reduce_data)::Type;
89
90 reduce_op.eval(np, reduce_data,
91 [=] AMREX_GPU_DEVICE (int i) -> ReduceTuple
92 {
93 const ParticleType& p = pstruct[i];
94 if ((p.id() < 0)) return false;
95 IntVect iv = IntVect(
96 AMREX_D_DECL(int(amrex::Math::floor((p.pos(0)-plo[0])*dxi[0])),
97 int(amrex::Math::floor((p.pos(1)-plo[1])*dxi[1])),
98 int(amrex::Math::floor((p.pos(2)-plo[2])*dxi[2]))));
99 iv += domain.smallEnd();
100 return !box.contains(iv);
101 });
102 int hv = amrex::get<0>(reduce_data.value());

162AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
163int getTileIndex (const IntVect& iv, const Box& box, const bool a_do_tiling,
164 const IntVect& a_tile_size, Box& tbx)
165{
166 if (a_do_tiling == false) {
167 tbx = box;
168 return 0;
169 } else {
170 //
171 // This function must be consistent with FabArrayBase::buildTileArray function!!!
172 //
173 auto tiling_1d = [](int i, int lo, int hi, int tilesize,
174 int& ntile, int& tileidx, int& tlo, int& thi) {
175 int ncells = hi-lo+1;
176 ntile = amrex::max(ncells/tilesize, 1);
177 int ts_right = ncells/ntile;
178 int ts_left = ts_right+1;
179 int nleft = ncells - ntile*ts_right;
180 int ii = i - lo;
181 int nbndry = nleft*ts_left;
182 if (ii < nbndry) {
183 tileidx = ii / ts_left; // tiles on the left of nbndry have size of ts_left
184 tlo = lo + tileidx * ts_left;
185 thi = tlo + ts_left - 1;
186 } else {
187 tileidx = nleft + (ii-nbndry) / ts_right; // tiles on the right: ts_right
188 tlo = lo + tileidx * ts_right + nleft;
189 thi = tlo + ts_right - 1;
190 }
191 };
192 const IntVect& small = box.smallEnd();
193 const IntVect& big = box.bigEnd();
194 IntVect ntiles, ivIndex, tilelo, tilehi;
195
196 AMREX_D_TERM(int iv0 = amrex::min(amrex::max(iv[0], small[0]), big[0]);,
197 int iv1 = amrex::min(amrex::max(iv[1], small[1]), big[1]);,
198 int iv2 = amrex::min(amrex::max(iv[2], small[2]), big[2]););
199
200 AMREX_D_TERM(tiling_1d(iv0, small[0], big[0], a_tile_size[0], ntiles[0], ivIndex[0], tilelo[0], tilehi[0]);,
201 tiling_1d(iv1, small[1], big[1], a_tile_size[1], ntiles[1], ivIndex[1], tilelo[1], tilehi[1]);,
202 tiling_1d(iv2, small[2], big[2], a_tile_size[2], ntiles[2], ivIndex[2], tilelo[2], tilehi[2]););
203
204 tbx = Box(tilelo, tilehi);
205
206 return AMREX_D_TERM(ivIndex[0], + ntiles[0]*ivIndex[1], + ntiles[0]*ntiles[1]*ivIndex[2]);
207 }

210AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
211int numTilesInBox (const Box& box, const bool a_do_tiling, const IntVect& a_tile_size)
212{
213 if (a_do_tiling == false) {
214 return 1;
215 } else {
216 //
217 // This function must be consistent with FabArrayBase::buildTileArray function!!!
218 //
219 auto tiling_1d = [](int lo, int hi, int tilesize, int& ntile) {
220 int ncells = hi-lo+1;
221 ntile = amrex::max(ncells/tilesize, 1);
222 };
223
224 const IntVect& small = box.smallEnd();
225 const IntVect& big = box.bigEnd();
226 IntVect ntiles;
227
228 AMREX_D_TERM(tiling_1d(small[0], big[0], a_tile_size[0], ntiles[0]);,
229 tiling_1d(small[1], big[1], a_tile_size[1], ntiles[1]);,
230 tiling_1d(small[2], big[2], a_tile_size[2], ntiles[2]););
231
232 return AMREX_D_TERM(ntiles[0], *=ntiles[1], *=ntiles[2]);

263template <typename P>
264AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
265bool enforcePeriodic (P& p,
266 amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& plo,
267 amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& phi,
268 amrex::GpuArray<int,AMREX_SPACEDIM> const& is_per) noexcept
269{
270 bool shifted = false;
271 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
272 {
273 if (not is_per[idim]) continue;
274 if (p.pos(idim) >= phi[idim]) {
275 while (p.pos(idim) >= phi[idim]) {
276 p.pos(idim) -= static_cast<ParticleReal>(phi[idim] - plo[idim]);
277 }
278 // clamp to avoid precision issues;
279 if (p.pos(idim) < plo[idim]) p.pos(idim) = static_cast<ParticleReal>(plo[idim]);
280 shifted = true;
281 }
282 else if (p.pos(idim) < plo[idim]) {
283 while (p.pos(idim) < plo[idim]) {
284 p.pos(idim) += static_cast<ParticleReal>(phi[idim] - plo[idim]);
285 }
286 // clamp to avoid precision issues;
287 if (p.pos(idim) == phi[idim]) p.pos(idim) = static_cast<ParticleReal>(plo[idim]);
288 if (p.pos(idim) > phi[idim]) p.pos(idim) = std::nextafter( (amrex::ParticleReal) phi[idim], (amrex::ParticleReal) plo[idim]);
289 shifted = true;
290 }
291 AMREX_ASSERT( (p.pos(idim) >= plo[idim] ) and ( p.pos(idim) < phi[idim] ));
292 }
293
294 return shifted;
295}

299template <typename PTile, typename PLocator>
300int
301partitionParticlesByDest (PTile& ptile, const PLocator& ploc, const ParticleBufferMap& pmap,
302 const Geometry& geom, int lev, int gid, int /*tid*/,
303 int lev_min, int lev_max, int nGrow)
304{
305 const auto plo = geom.ProbLoArray();
306 const auto phi = geom.ProbHiArray();
307 const auto is_per = geom.isPeriodicArray();
308
309 auto& aos = ptile.GetArrayOfStructs();
310 const int np = aos.numParticles();
311
312 if (np == 0) return 0;
313
314 auto getPID = pmap.getPIDFunctor();
315 auto p_ptr = &(aos[0]);
316
317 int pid = ParallelContext::MyProcSub();
318 constexpr int chunk_size = 256*256*256;
319 int num_chunks = std::max(1, (np + (chunk_size - 1)) / chunk_size);
320
321 PTile ptile_tmp;
322 ptile_tmp.define(ptile.NumRuntimeRealComps(), ptile.NumRuntimeIntComps());
323 ptile_tmp.resize(std::min(np, chunk_size));
324
325 auto src_data = ptile.getParticleTileData();
326 auto dst_data = ptile_tmp.getParticleTileData();
327
328 int last_offset = 0;
329 for (int ichunk = 0; ichunk < num_chunks; ++ichunk)
330 {
331 int this_offset = ichunk*chunk_size;
332 int this_chunk_size = std::min(chunk_size, np - this_offset);
333
334 int num_stay;
335 {
336 auto particle_stays = [=] AMREX_GPU_DEVICE (int i) -> int
337 {
338 int assigned_grid;
339 int assigned_lev;
340
341 auto& p = p_ptr[i+this_offset];
342
343 if (p.id() < 0 )
344 {
345 assigned_grid = -1;
346 assigned_lev = -1;
347 }
348 else
349 {
350 enforcePeriodic(p, plo, phi, is_per);
351 auto tup = ploc(p, lev_min, lev_max, nGrow);
352 assigned_grid = amrex::get<0>(tup);
353 assigned_lev = amrex::get<1>(tup);
354 }
355
356 return ((assigned_grid == gid) && (assigned_lev == lev) && (getPID(lev, gid) == pid));
357 };
358
359 num_stay = Scan::PrefixSum<int> (this_chunk_size,
360 [=] AMREX_GPU_DEVICE (int i) -> int
361 {
362 return particle_stays(i);
363 },
364 [=] AMREX_GPU_DEVICE (int i, int const& s)
365 {
366 if ( particle_stays(i) )
367 {
368 copyParticle(dst_data, src_data, i + this_offset, s);
369 }
370 else
371 {
372 copyParticle(dst_data, src_data, i + this_offset, this_chunk_size-1-(i-s));
373 }
374 },
375 Scan::Type::exclusive);
376 }
377
378 if (num_chunks == 1)
379 {
380 ptile.swap(ptile_tmp);
381 }
382 else
383 {
384 AMREX_FOR_1D(this_chunk_size, i,
385 {
386 copyParticle(src_data, dst_data, i, i + this_offset);
387 });
388 }
389
390 if ( ichunk > 0 )
391 {
392 int num_swap = std::min(this_offset - last_offset, num_stay);
393 AMREX_FOR_1D( num_swap, i,
394 {
395 swapParticle(src_data, src_data, last_offset + i,
396 this_offset + num_stay - 1 - i);
397 });
398 }
399
400 last_offset += num_stay;
401 }
402
403 return last_offset;
404}
