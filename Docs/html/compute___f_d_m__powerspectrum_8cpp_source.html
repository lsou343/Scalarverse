63const hacc::Distribution Nyx::initialize_fft_distribution(MultiFab& mf){
64
65 const BoxArray& ba = mf.boxArray();
66 const DistributionMapping& dm = mf.DistributionMap();
67
68 Vector<int> rank_mapping;
69 rank_mapping.resize(ba.size());
70
71 Vector<int> smallxEnds;
72 for (int ib = 0; ib < ba.size(); ++ib){
73 smallxEnds.push_back(ba[ib].smallEnd(0));
74 }
75 std::sort(smallxEnds.begin(), smallxEnds.end());
76 smallxEnds.erase( unique( smallxEnds.begin(), smallxEnds.end() ), smallxEnds.end() );
77
78 Vector<int> smallyEnds;
79 for (int ib = 0; ib < ba.size(); ++ib){
80 smallyEnds.push_back(ba[ib].smallEnd(1));
81 }
82 std::sort(smallyEnds.begin(), smallyEnds.end());
83 smallyEnds.erase( unique( smallyEnds.begin(), smallyEnds.end() ), smallyEnds.end() );
84
85 Vector<int> smallzEnds;
86 for (int ib = 0; ib < ba.size(); ++ib){
87 smallzEnds.push_back(ba[ib].smallEnd(2));
88 }
89 std::sort(smallzEnds.begin(), smallzEnds.end());
90 smallzEnds.erase( unique( smallzEnds.begin(), smallzEnds.end() ), smallzEnds.end() );
91
92 if ((smallxEnds.size()*smallyEnds.size()*smallzEnds.size()) != ba.size())
93 amrex::Error("GRIDS NEED TO BE DISTRIBUTED MORE REGULARLY");
94 if(ParallelDescriptor::NProcs() != ba.size())
95 amrex::Error("Number of MPI ranks has to be equal to the number of root level grids!");
96
97 for (int ib = 0; ib < ba.size(); ++ib)
98 {
99 int i = (std::find(smallxEnds.begin(), smallxEnds.end(), ba[ib].smallEnd(0))-smallxEnds.begin());
100 int j = (std::find(smallyEnds.begin(), smallyEnds.end(), ba[ib].smallEnd(1))-smallyEnds.begin());
101 int k = (std::find(smallzEnds.begin(), smallzEnds.end(), ba[ib].smallEnd(2))-smallzEnds.begin());
102 int local_index = i*smallzEnds.size()*smallyEnds.size() + j*smallzEnds.size() + k;
103 rank_mapping[local_index] = dm[ib];
104 }
105 int Ndims[3] = { smallzEnds.size(), smallyEnds.size(), smallxEnds.size() };
106 int n[3] = {Domain().length(2), Domain().length(1), Domain().length(0)};
107
108 return hacc::Distribution(MPI_COMM_WORLD,n,Ndims,&rank_mapping[0]);

112 std::vector<complex_t, hacc::AlignedAllocator<complex_t, ALIGN> > Nyx::initialize_fft_input_field(amrex::MultiFab &mf){
113
114 std::vector<complex_t, hacc::AlignedAllocator<complex_t, ALIGN> > a;
115
116 for (MFIter mfi(mf,false); mfi.isValid(); ++mfi){
117 Array4<Real> const& arr = mf.array(mfi);
118 const Box& bx = mfi.validbox();
119 const Dim3 lo = amrex::lbound(bx);
120 const Dim3 hi = amrex::ubound(bx);
121 const Dim3 w ={hi.x-lo.x+1,hi.y-lo.y+1,hi.z-lo.z+1};
122 const int gridsize = (hi.x-lo.x+1)*(hi.y-lo.y+1)*(hi.z-lo.z+1);
123 a.resize(gridsize);
124#ifdef _OPENMP
125#pragma omp parallel for
126#endif
127 for(size_t k=0; k<(size_t)w.z; k++) {
128 for(size_t j=0; j<(size_t)w.y; j++) {
129 AMREX_PRAGMA_SIMD
130 for(size_t i=0; i<(size_t)w.x; i++) {
131
132 size_t local_indx_threaded = (size_t)w.y*(size_t)w.z*i+(size_t)w.z*j+k;
133 complex_t temp(arr(i+lo.x,j+lo.y,k+lo.z,0),0);
134 a[local_indx_threaded] = temp;
135 }}}
136 }
137 return a;
138}

141void Nyx::actually_compute_powerspectrum(hacc::Dfft& dfft, std::vector<complex_t, hacc::AlignedAllocator<complex_t, ALIGN> >& a, amrex::Real time){
142 const Real pi = 4 * std::atan(1.0);
143 const Real tpi = 2 * pi;
144 const Real h = get_level(0).geom.CellSize(0);
145 const int *self = dfft.self_kspace();
146 const int *local_ng = dfft.local_ng_kspace();
147 const int *global_ng = dfft.global_ng();
148 const int nbins = std::ceil(std::sqrt(3)*std::floor(Domain().length(0)/2));
149 Vector<Real> spectrum(nbins,0.0);
150 Vector<Real> mode(nbins,0.0);
151 Vector<int> nmode(nbins,0);
152
153#ifdef _OPENMP
154#pragma omp parallel for
155#endif
156 for(size_t i=0; i<(size_t)local_ng[0]; i++) {
157 int global_i = local_ng[0]*self[0] + i;
158 if (global_i > global_ng[0]/2.){
159 global_i = global_i - global_ng[0];
160 }
161
162
163 for(size_t j=0; j<(size_t)local_ng[1]; j++) {
164 int global_j = local_ng[1]*self[1] + j;
165 if (global_j > global_ng[1]/2.){
166 global_j = global_j - global_ng[1];
167 }
168
169 AMREX_PRAGMA_SIMD
170 for(size_t k=0; k<(size_t)local_ng[2]; k++) {
171 int global_k = local_ng[2]*self[2] + k;
172 size_t local_indx_threaded = (size_t)local_ng[1]*(size_t)local_ng[2]*i+(size_t)local_ng[2]*j+k;
173 if (global_k > global_ng[2]/2.){
174 global_k = global_k - global_ng[2];
175 }
176
177 double kx = tpi * double(global_i)/double(global_ng[0]);
178 double ky = tpi * double(global_j)/double(global_ng[1]);
179 double kz = tpi * double(global_k)/double(global_ng[2]);
180 double k2 = (kx*kx + ky*ky + kz*kz)/h/h;
181
182 const int bin = std::floor(std::sqrt(global_i*global_i+global_j*global_j+global_k*global_k));
183
184 nmode[bin]++;
185 spectrum[bin] += std::pow(std::abs(a[local_indx_threaded]),2);
186 mode[bin] += std::sqrt(k2);
187 }
188 }
189 }
190
191 ParallelDescriptor::ReduceIntSum(nmode.dataPtr(), nmode.size(),
192 ParallelDescriptor::IOProcessorNumber());
193 ParallelDescriptor::ReduceRealSum(spectrum.dataPtr(), spectrum.size(),
194 ParallelDescriptor::IOProcessorNumber());
195 ParallelDescriptor::ReduceRealSum(mode.dataPtr(), mode.size(),
196 ParallelDescriptor::IOProcessorNumber());
197
198 if (ParallelDescriptor::IOProcessor()) {
199 if ( ! amrex::UtilCreateDirectory("./spectra", 0755)) {
200 amrex::CreateDirectoryFailed("./spectra");
201 }
202 std::string fileName("spectra/spectrum_");
203 std::stringstream stream;
204 stream << std::fixed << std::setprecision(5) << time;
205 fileName += stream.str();
206 std::ofstream spectrum_out(fileName.c_str());
207 for(int bin=0; bin < spectrum.size(); ++bin) {
208 spectrum[bin] *= time*time/(geom.ProbHi(0)-geom.ProbLo(0))/(geom.ProbHi(1)-geom.ProbLo(1))/(geom.ProbHi(2)-geom.ProbLo(2))*pow(h,6)/2.0;
209 spectrum_out << mode[bin] << " " << spectrum[bin] << " " << nmode[bin] << std::endl;
210 }
211 spectrum_out.close();
212 }
213}
