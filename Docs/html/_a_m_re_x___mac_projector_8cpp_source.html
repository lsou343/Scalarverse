48void MacProjector::initProjector(
49 const LPInfo& a_lpinfo,
50 const Vector<Array<MultiFab const*,AMREX_SPACEDIM> >& a_beta,
51 const Vector<iMultiFab const*>& a_overset_mask)
52{
53 const int nlevs = a_beta.size();
54 Vector<BoxArray> ba(nlevs);
55 Vector<DistributionMapping> dm(nlevs);
56 for (int ilev = 0; ilev < nlevs; ++ilev) {
57 ba[ilev] = amrex::convert(
58 a_beta[ilev][0]->boxArray(), IntVect::TheZeroVector());
59 dm[ilev] = a_beta[ilev][0]->DistributionMap();
60 }
61
62 m_rhs.resize(nlevs);
63 m_phi.resize(nlevs);
64 m_fluxes.resize(nlevs);
65 m_divu.resize(nlevs);
66
67#ifdef AMREX_USE_EB
68 bool has_eb = a_beta[0][0]->hasEBFabFactory();
69 if (has_eb) {
70 m_eb_factory.resize(nlevs, nullptr);
71 for (int ilev = 0; ilev < nlevs; ++ilev) {
72 m_eb_factory[ilev] = dynamic_cast<EBFArrayBoxFactory const*>(
73 &(a_beta[ilev][0]->Factory()));
74 m_rhs[ilev].define(
75 ba[ilev], dm[ilev], 1, 0, MFInfo(), a_beta[ilev][0]->Factory());
76 m_phi[ilev].define(
77 ba[ilev], dm[ilev], 1, 1, MFInfo(), a_beta[ilev][0]->Factory());
78 m_rhs[ilev].setVal(0.0);
79 m_phi[ilev].setVal(0.0);
80 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
81 m_fluxes[ilev][idim].define(
82 amrex::convert(ba[ilev], IntVect::TheDimensionVector(idim)),
83 dm[ilev], 1, 0, MFInfo(), a_beta[ilev][0]->Factory());
84 }
85 }
86
87 m_eb_abeclap.reset(
88 new MLEBABecLap(m_geom, ba, dm, a_lpinfo, m_eb_factory));
89 m_linop = m_eb_abeclap.get();
90
91 if (m_phi_loc == MLMG::Location::CellCentroid)
92 m_eb_abeclap->setPhiOnCentroid();
93
94 m_eb_abeclap->setScalars(0.0, 1.0);
95 for (int ilev = 0; ilev < nlevs; ++ilev) {
96 m_eb_abeclap->setBCoeffs(ilev, a_beta[ilev], m_beta_loc);
97 }
98 } else
99#endif
100 {
101 for (int ilev = 0; ilev < nlevs; ++ilev) {
102 m_rhs[ilev].define(ba[ilev], dm[ilev], 1, 0);
103 m_phi[ilev].define(ba[ilev], dm[ilev], 1, 1);
104 m_rhs[ilev].setVal(0.0);
105 m_phi[ilev].setVal(0.0);
106 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
107 m_fluxes[ilev][idim].define(
108 amrex::convert(ba[ilev], IntVect::TheDimensionVector(idim)),
109 dm[ilev], 1, 0);
110 }
111 }
112
113 if (a_overset_mask.empty())
114 m_abeclap.reset(new MLABecLaplacian(m_geom, ba, dm, a_lpinfo));
115 else
116 m_abeclap.reset(
117 new MLABecLaplacian(m_geom, ba, dm, a_overset_mask, a_lpinfo));
118
119 m_linop = m_abeclap.get();
120
121 m_abeclap->setScalars(0.0, 1.0);
122 for (int ilev = 0; ilev < nlevs; ++ilev) {
123 m_abeclap->setBCoeffs(ilev, a_beta[ilev]);
124 }
125 }
126
127 m_mlmg.reset(new MLMG(*m_linop));
128
129 setOptions();
130
131 m_needs_init = false;
132}

209void
210MacProjector::project (Real reltol, Real atol)
211{
212 const int nlevs = m_rhs.size();
213
214 averageDownVelocity();
215
216 for (int ilev = 0; ilev < nlevs; ++ilev)
217 {
218 Array<MultiFab const*, AMREX_SPACEDIM> u;
219 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
220 u[idim] = m_umac[ilev][idim];
221 }
222 MultiFab divu(m_rhs[ilev].boxArray(), m_rhs[ilev].DistributionMap(),
223 1, 0, MFInfo(), m_rhs[ilev].Factory());
224#ifdef AMREX_USE_EB
225 if (m_umac_loc != MLMG::Location::FaceCentroid)
226 {
227 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
228 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(m_umac[ilev][idim]->nGrow() > 0,
229 "MacProjector: with EB, umac must have at least one ghost cell if not already_on_centroid");
230 m_umac[ilev][idim]->FillBoundary(m_geom[ilev].periodicity());
231 }
232 }
233
234 EB_computeDivergence(divu, u, m_geom[ilev], (m_umac_loc == MLMG::Location::FaceCentroid));
235#else
236 computeDivergence(divu, u, m_geom[ilev]);
237#endif
238
239 // Setup RHS as (m_divu - divu) where m_divu is a user-provided source term
240 MultiFab::Copy(m_rhs[ilev], divu, 0, 0, 1, 0);
241 m_rhs[ilev].mult(-1.0);
242
243 if (m_divu[ilev].ok())
244 {
245 MultiFab::Add(m_rhs[ilev],m_divu[ilev],0,0,1,0);
246 }
247 }
248
249 m_mlmg->solve(amrex::GetVecOfPtrs(m_phi), amrex::GetVecOfConstPtrs(m_rhs), reltol, atol);
250
251 m_mlmg->getFluxes(amrex::GetVecOfArrOfPtrs(m_fluxes), m_umac_loc);
252
253 for (int ilev = 0; ilev < nlevs; ++ilev) {
254 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
255 MultiFab::Add(*m_umac[ilev][idim], m_fluxes[ilev][idim], 0, 0, 1, 0);
256#ifdef AMREX_USE_EB
257 EB_set_covered_faces(m_umac[ilev], 0.0);
258#endif
259 }
260 }
261
262 averageDownVelocity();
263}

332void
333MacProjector::setOptions ()
334{
335 // Default values
336 int maxorder(3);
337 int bottom_verbose(0);
338 int maxiter(200);
339 int bottom_maxiter(200);
340 Real bottom_rtol(1.0e-4);
341 Real bottom_atol(-1.0);
342 std::string bottom_solver("bicg");
343
344 int num_pre_smooth(2);
345 int num_post_smooth(2);
346
347 // Read from input file
348 ParmParse pp("mac_proj");
349 pp.query( "verbose" , m_verbose );
350 pp.query( "maxorder" , maxorder );
351 pp.query( "bottom_verbose", bottom_verbose );
352 pp.query( "maxiter" , maxiter );
353 pp.query( "bottom_maxiter", bottom_maxiter );
354 pp.query( "bottom_rtol" , bottom_rtol );
355 pp.query( "bottom_atol" , bottom_atol );
356 pp.query( "bottom_solver" , bottom_solver );
357
358 pp.query( "num_pre_smooth" , num_pre_smooth );
359 pp.query( "num_post_smooth" , num_post_smooth );
360
361 // Set default/input values
362 m_linop->setMaxOrder(maxorder);
363 m_mlmg->setVerbose(m_verbose);
364 m_mlmg->setBottomVerbose(bottom_verbose);
365 m_mlmg->setMaxIter(maxiter);
366 m_mlmg->setBottomMaxIter(bottom_maxiter);
367 m_mlmg->setBottomTolerance(bottom_rtol);
368 m_mlmg->setBottomToleranceAbs(bottom_atol);
369
370 m_mlmg->setPreSmooth(num_pre_smooth);
371 m_mlmg->setPostSmooth(num_post_smooth);
372
373 if (bottom_solver == "smoother")
374 {
375 m_mlmg->setBottomSolver(MLMG::BottomSolver::smoother);
376 }
377 else if (bottom_solver == "bicg")
378 {
379 m_mlmg->setBottomSolver(MLMG::BottomSolver::bicgstab);
380 }
381 else if (bottom_solver == "cg")
382 {
383 m_mlmg->setBottomSolver(MLMG::BottomSolver::cg);
384 }
385 else if (bottom_solver == "bicgcg")
386 {
387 m_mlmg->setBottomSolver(MLMG::BottomSolver::bicgcg);
388 }
389 else if (bottom_solver == "cgbicg")
390 {
391 m_mlmg->setBottomSolver(MLMG::BottomSolver::cgbicg);
392 }
393 else if (bottom_solver == "hypre")
394 {
395#ifdef AMREX_USE_HYPRE
396 m_mlmg->setBottomSolver(MLMG::BottomSolver::hypre);
397#else
398 amrex::Abort("AMReX was not built with HYPRE support");
399#endif
400 }

408

414#ifdef AMREX_USE_EB

418#else
419 average_down_faces(GetArrOfConstPtrs(m_umac[lev]),
420 m_umac[lev-1],
421 rr, m_geom[lev-1]);
422#endif
423 }
424}
