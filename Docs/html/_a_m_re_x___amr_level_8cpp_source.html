786
787 m_fbox.insert(m_fbox.end(),v2)->second.resize(m_amrlevel.level+1);
788 m_cbox.insert(m_cbox.end(),v2)->second.resize(m_amrlevel.level+1);
789
790 m_ba.insert(m_ba.end(),std::map<int,Box>::value_type(i,amrex::grow(m_leveldata.boxArray()[i],m_growsize)));
791 }
792
793 BoxList tempUnfillable(boxType);
794 BoxList unfillableThisLevel(boxType);
795 Vector<Box> unfilledThisLevel;
796 Vector<Box> crse_boxes;
797 Vector<IntVect> pshifts(27);
798
799 for (std::map<int,Box>::const_iterator it = m_ba.begin(), End = m_ba.end();
800 it != End;
801 ++it)
802 {
803 const int bxidx = it->first;
804 const Box& box = it->second;
805
806 unfilledThisLevel.clear();
807 unfilledThisLevel.push_back(box);
808
809 if (!topPDomain.contains(box))
810 {
811 unfilledThisLevel.back() &= topPDomain;
812
813 if (topLevel.geom.isAnyPeriodic())
814 {
815 //
816 // May need to add additional unique pieces of valid region
817 // in order to do periodic copies into ghost cells.
818 //
819 topLevel.geom.periodicShift(topPDomain,box,pshifts);
820
821 for (const auto& iv : pshifts)
822 {
823 Box shbox = box + iv;
824 shbox &= topPDomain;
825
826 if (boxType.nodeCentered())
827 {
828 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
829 {
830 if (iv[dir] > 0)
831 {
832 shbox.growHi(dir,-1);
833 }
834 else if (iv[dir] < 0)
835 {
836 shbox.growLo(dir,-1);
837 }
838 }
839 }
840
841 if (shbox.ok())
842 {
843 BoxList bl = amrex::boxDiff(shbox,box);
844
845 unfilledThisLevel.insert(unfilledThisLevel.end(), bl.begin(), bl.end());
846 }
847 }
848 }
849 }
850
851 // cells outside physical boundaries are not included in unfilledThisLevel
852
853 bool Done = false;
854
855 Vector< Vector<Box> >& TheCrseBoxes = m_cbox[bxidx];
856 Vector< Vector<Box> >& TheFineBoxes = m_fbox[bxidx];
857 Vector< Vector< Vector<FillBoxId> > >& TheFBIDs = m_fbid[bxidx];
858
859 for (int l = m_amrlevel.level; l >= 0 && !Done; --l)
860 {
861 unfillableThisLevel.clear();
862
863 AmrLevel& theAmrLevel = *amrLevels[l];
864 StateData& theState = theAmrLevel.state[m_index];
865 const Box& thePDomain = theState.getDomain();
866 const Geometry& theGeom = theAmrLevel.geom;
867 const bool is_periodic = theGeom.isAnyPeriodic();
868 const IntVect& fine_ratio = theAmrLevel.fine_ratio;
869 Vector<Box>& FineBoxes = TheFineBoxes[l];
870 //
871 // These are the boxes on this level contained in thePDomain
872 // that need to be filled in order to directly fill at the
873 // highest level or to interpolate up to the next higher level.
874 //
875 FineBoxes = unfilledThisLevel;
876 //
877 // Now build coarse boxes needed to interpolate to fine.
878 //
879 // If we're periodic and we're not at the finest level, we may
880 // need to get some additional data at this level in order to
881 // properly fill the CoarseBox()d versions of the fineboxes.
882 //
883 crse_boxes.clear();
884
885 for (const auto& fbx : FineBoxes)
886 {
887 crse_boxes.push_back(fbx);
888
889 if (l != m_amrlevel.level)
890 {
891 const Box& cbox = m_map->CoarseBox(fbx,fine_ratio);
892
893 crse_boxes.back() = cbox;
894
895 if (is_periodic && !thePDomain.contains(cbox))
896 {
897 theGeom.periodicShift(thePDomain,cbox,pshifts);
898
899 for (const auto& iv : pshifts)
900 {
901 Box shbox = cbox + iv;
902 shbox &= thePDomain;
903
904 if (boxType.nodeCentered())
905 {
906 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
907 {
908 if (iv[dir] > 0)
909 {
910 shbox.growHi(dir,-1);
911 }
912 else if (iv[dir] < 0)
913 {
914 shbox.growLo(dir,-1);
915 }
916 }
917 }
918
919 if (shbox.ok())
920 {
921 crse_boxes.push_back(shbox);
922 }
923 }
924 }
925 }
926 }
927
928 Vector< Vector<FillBoxId> >& FBIDs = TheFBIDs[l];
929 Vector<Box>& CrseBoxes = TheCrseBoxes[l];
930
931 FBIDs.resize(crse_boxes.size());
932 CrseBoxes.resize(crse_boxes.size());
933 //
934 // Now attempt to get as much coarse data as possible.
935 //
936 for (int i = 0, M = CrseBoxes.size(); i < M; i++)
937 {
938 BL_ASSERT(tempUnfillable.isEmpty());
939
940 CrseBoxes[i] = crse_boxes[i];
941
942 BL_ASSERT(CrseBoxes[i].intersects(thePDomain));
943
944 theState.InterpAddBox(m_mfcd,
945 m_mfid[l],
946 &tempUnfillable,
947 FBIDs[i],
948 CrseBoxes[i],
949 m_time,
950 m_scomp,
951 0,
952 m_ncomp,
953 extrap);
954
955 unfillableThisLevel.catenate(tempUnfillable);
956 }
957
958 unfillableThisLevel.intersect(thePDomain);
959
960 if (unfillableThisLevel.isEmpty())
961 {
962 Done = true;
963 }
964 else
965 {
966 unfilledThisLevel.clear();
967
968 unfilledThisLevel.insert(unfilledThisLevel.end(),
969 unfillableThisLevel.begin(),
970 unfillableThisLevel.end());
971 }
972 }
973 }
974
975 m_mfcd.CollectData();
976}

1171static
1172void
1173FixUpPhysCorners (FArrayBox& fab,
1174 AmrLevel& TheLevel,
1175 int state_indx,
1176 Real time,
1177 int scomp,
1178 int dcomp,
1179 int ncomp)
1180{
1181 StateData& TheState = TheLevel.get_state_data(state_indx);
1182 const Geometry& TheGeom = TheLevel.Geom();
1183 const Box& ProbDomain = TheState.getDomain();
1184
1185 if (!HasPhysBndry(fab.box(),ProbDomain,TheGeom)) return;
1186
1187 FArrayBox tmp;
1188
1189 Box GrownDomain = ProbDomain;
1190
1191 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
1192 {
1193 if (!TheGeom.isPeriodic(dir))
1194 {
1195 const int lo = ProbDomain.smallEnd(dir) - fab.box().smallEnd(dir);
1196 const int hi = fab.box().bigEnd(dir) - ProbDomain.bigEnd(dir);
1197 if (lo > 0) GrownDomain.growLo(dir,lo);
1198 if (hi > 0) GrownDomain.growHi(dir,hi);
1199 }
1200 }
1201
1202 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
1203 {
1204 if (!TheGeom.isPeriodic(dir)) continue;
1205
1206 Box lo_slab = fab.box();
1207 Box hi_slab = fab.box();
1208 lo_slab.shift(dir, ProbDomain.length(dir));
1209 hi_slab.shift(dir,-ProbDomain.length(dir));
1210 lo_slab &= GrownDomain;
1211 hi_slab &= GrownDomain;
1212
1213 if (lo_slab.ok())
1214 {
1215 lo_slab.shift(dir,-ProbDomain.length(dir));
1216
1217 BL_ASSERT(fab.box().contains(lo_slab));
1218 BL_ASSERT(HasPhysBndry(lo_slab,ProbDomain,TheGeom));
1219
1220 tmp.resize(lo_slab,ncomp);
1221 tmp.copy<RunOn::Host>(fab,dcomp,0,ncomp);
1222 tmp.shift(dir,ProbDomain.length(dir));
1223 TheLevel.setPhysBoundaryValues(tmp,
1224 state_indx,
1225 time,
1226 0,
1227 scomp,
1228 ncomp);
1229 tmp.shift(dir,-ProbDomain.length(dir));
1230 fab.copy<RunOn::Host>(tmp,0,dcomp,ncomp);
1231 }
1232
1233 if (hi_slab.ok())
1234 {
1235 hi_slab.shift(dir,ProbDomain.length(dir));
1236
1237 BL_ASSERT(fab.box().contains(hi_slab));
1238 BL_ASSERT(HasPhysBndry(hi_slab,ProbDomain,TheGeom));
1239
1240 tmp.resize(hi_slab,ncomp);
1241 tmp.copy<RunOn::Host>(fab,dcomp,0,ncomp);
1242 tmp.shift(dir,-ProbDomain.length(dir));
1243 TheLevel.setPhysBoundaryValues(tmp,
1244 state_indx,
1245 time,
1246 0,
1247 scomp,
1248 ncomp);
1249 tmp.shift(dir,ProbDomain.length(dir));
1250 fab.copy<RunOn::Host>(tmp,0,dcomp,ncomp);
1251 }
1252 }

1255void
1256FillPatchIteratorHelper::fill (FArrayBox& fab,
1257 int dcomp,
1258 int idx)
1259{
1260 BL_PROFILE("FillPatchIteratorHelper::fill()");
1261
1262 BL_ASSERT(fab.box() == m_ba[idx]);
1263 BL_ASSERT(fab.nComp() >= dcomp + m_ncomp);
1264
1265 Vector< Vector<std::unique_ptr<FArrayBox> > > cfab(m_amrlevel.level+1);
1266 Vector< Vector<Box> >& TheCrseBoxes = m_cbox[idx];
1267 Vector< Vector<Box> >& TheFineBoxes = m_fbox[idx];
1268 Vector< Vector< Vector<FillBoxId> > >& TheFBIDs = m_fbid[idx];
1269 const bool extrap = AmrLevel::desc_lst[m_index].extrap();
1270 auto& amrLevels = m_amrlevel.parent->getAmrLevels();
1271 //
1272 // Build all coarse fabs from which we'll interpolate and
1273 // fill them with coarse data as best we can.
1274 //
1275 for (int l = 0; l <= m_amrlevel.level; l++)
1276 {
1277 StateData& TheState = amrLevels[l]->state[m_index];
1278 const Vector<Box>& CrseBoxes = TheCrseBoxes[l];
1279 auto& CrseFabs = cfab[l];
1280 const Vector< Vector<FillBoxId> >& FBIDs = TheFBIDs[l];
1281 const int NC = CrseBoxes.size();
1282
1283 CrseFabs.resize(NC);
1284
1285 Box domain_box = amrex::convert(amrLevels[l]->Geom().Domain(), fab.box().ixType());
1286 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
1287 if (amrLevels[l]->Geom().isPeriodic(idim)) {
1288 int n = domain_box.length(idim);
1289 domain_box.grow(idim, n);
1290 }
1291 }
1292
1293 for (int i = 0; i < NC; i++)
1294 {
1295 BL_ASSERT(CrseBoxes[i].ok());
1296 CrseFabs[i].reset(new FArrayBox(CrseBoxes[i],m_ncomp));
1297 CrseFabs[i]->setComplement<RunOn::Host>(std::numeric_limits<Real>::quiet_NaN(), domain_box, 0, m_ncomp);
1298 }
1299
1300 for (int i = 0; i < NC; i++)
1301 {
1302 //
1303 // Set to special value we'll later check
1304 // to ensure we've filled the FABs at the coarse level.
1305 //
1306 TheState.InterpFillFab(m_mfcd,
1307 m_mfid[l],
1308 FBIDs[i],
1309 *CrseFabs[i],
1310 m_time,
1311 0,
1312 0,
1313 m_ncomp,
1314 extrap);
1315 }
1316 }
1317 //
1318 // Now work from the bottom up interpolating to next higher level.
1319 //
1320 for (int l = 0; l < m_amrlevel.level; l++)
1321 {
1322 auto& CrseFabs = cfab[l];
1323 AmrLevel& TheLevel = *amrLevels[l];
1324 StateData& TheState = TheLevel.state[m_index];
1325 const Box& ThePDomain = TheState.getDomain();
1326 const int NC = CrseFabs.size();
1327
1328 if (TheLevel.geom.isAnyPeriodic())
1329 {
1330 //
1331 // Fill CrseFabs with periodic data in preparation for interp().
1332 //
1333 for (int i = 0; i < NC; i++)
1334 {
1335 FArrayBox& dstfab = *CrseFabs[i];
1336
1337 if (ThePDomain.contains(dstfab.box())) continue;
1338
1339 Vector<IntVect> pshifts(27);
1340
1341 TheLevel.geom.periodicShift(ThePDomain,dstfab.box(),pshifts);
1342
1343 for (const auto& iv : pshifts)
1344 {
1345 Box fullsrcbox = dstfab.box() + iv;
1346 fullsrcbox &= ThePDomain;
1347
1348 for (int j = 0; j < NC; j++)
1349 {
1350 const FArrayBox& srcfab = *CrseFabs[j];
1351 const Box& srcbox = fullsrcbox & srcfab.box();
1352
1353 if (srcbox.ok())
1354 {
1355 const Box& dstbox = srcbox - iv;
1356
1357 dstfab.copy<RunOn::Host>(srcfab,srcbox,0,dstbox,0,m_ncomp);
1358 }
1359 }
1360 }
1361 }
1362 }
1363 //
1364 // Set non-periodic BCs in coarse data -- what we interpolate with.
1365 // This MUST come after the periodic fill mumbo-jumbo.
1366 for (int i = 0; i < NC; ++i)
1367 {
1368 if ( ! ThePDomain.contains(CrseFabs[i]->box()))
1369 {
1370 TheLevel.setPhysBoundaryValues(*CrseFabs[i],
1371 m_index,
1372 m_time,
1373 0,
1374 m_scomp,
1375 m_ncomp);
1376 }
1377 }
1378
1379 if (m_FixUpCorners)
1380 {
1381 for (int i = 0; i < NC; ++i)
1382 {
1383 FixUpPhysCorners(*CrseFabs[i],TheLevel,m_index,m_time,m_scomp,0,m_ncomp);
1384 }
1385 }
1386 //
1387 // Interpolate up to next level.
1388 //
1389 AmrLevel& crseAmrLevel = *amrLevels[l];
1390 AmrLevel& fineAmrLevel = *amrLevels[l+1];
1391 const IntVect& fine_ratio = crseAmrLevel.fine_ratio;
1392 const Vector<Box>& FineBoxes = TheFineBoxes[l];
1393 StateData& fState = fineAmrLevel.state[m_index];
1394 const Box& fDomain = fState.getDomain();
1395 auto& FinerCrseFabs = cfab[l+1];
1396 const Vector<BCRec>& theBCs = AmrLevel::desc_lst[m_index].getBCs();
1397 const int NF = FineBoxes.size();
1398
1399 for (int ifine = 0; ifine < NF; ++ifine)
1400 {
1401 Vector<BCRec> bcr(m_ncomp);
1402 FArrayBox finefab(FineBoxes[ifine],m_ncomp);
1403 FArrayBox crsefab(m_map->CoarseBox(finefab.box(),fine_ratio),m_ncomp);
1404 //
1405 // Fill crsefab from m_cbox via copy on intersect.
1406 //
1407 for (int j = 0; j < NC; j++) {
1408 crsefab.copy<RunOn::Host>(*CrseFabs[j]);
1409 }
1410 //
1411 // Get boundary conditions for the fine patch.
1412 //
1413 amrex::setBC(finefab.box(),
1414 fDomain,
1415 m_scomp,
1416 0,
1417 m_ncomp,
1418 theBCs,
1419 bcr);
1420 //
1421 // Interpolate up to fine patch.
1422 //
1423 m_map->interp(crsefab,
1424 0,
1425 finefab,
1426 0,
1427 m_ncomp,
1428 finefab.box(),
1429 fine_ratio,
1430 crseAmrLevel.geom,
1431 fineAmrLevel.geom,
1432 bcr,
1433 m_scomp,
1434 m_index,
1435 RunOn::Cpu);
1436 //
1437 // Copy intersect finefab into next level m_cboxes.
1438 //
1439 for (int j = 0, K = FinerCrseFabs.size(); j < K; ++j) {
1440 FinerCrseFabs[j]->copy<RunOn::Host>(finefab);
1441 }
1442 }
1443
1444 CrseFabs.clear();
1445 }
1446 //
1447 // Now for the finest level stuff.
1448 //
1449 StateData& FineState = m_amrlevel.state[m_index];
1450 const Box& FineDomain = FineState.getDomain();
1451 const Geometry& FineGeom = m_amrlevel.geom;
1452 auto& FinestCrseFabs = cfab[m_amrlevel.level];
1453 //
1454 // Copy intersect coarse into destination fab.
1455 //
1456 for (int i = 0, N = FinestCrseFabs.size(); i < N; ++i) {
1457 fab.copy<RunOn::Host>(*FinestCrseFabs[i],0,dcomp,m_ncomp);
1458 }
1459
1460 if (FineGeom.isAnyPeriodic() && !FineDomain.contains(fab.box()))
1461 {
1462 Vector<IntVect> pshifts(27);
1463
1464 FineGeom.periodicShift(FineDomain,fab.box(),pshifts);
1465
1466 for (int i = 0, N = FinestCrseFabs.size(); i < N; i++)
1467 {
1468 for (const auto& iv : pshifts)
1469 {
1470 fab.shift(iv);
1471
1472 Box src_dst = FinestCrseFabs[i]->box() & fab.box();
1473 src_dst &= FineDomain;
1474
1475 if (src_dst.ok())
1476 fab.copy<RunOn::Host>(*FinestCrseFabs[i],src_dst,0,src_dst,dcomp,m_ncomp);
1477
1478 fab.shift(-iv);
1479 }
1480 }

1536 Box domain_g = pdomain;
1537 for (int i = 0; i < AMREX_SPACEDIM; ++i) {
1538 if (geom.isPeriodic(i)) {
1539 domain_g.grow(i,nghost);
1540 }
1541 }
1542
1543 std::vector< std::pair<int,int> > ranges = desc.sameInterps(scomp,ncomp);
1544
1545 BL_ASSERT(desc.inRange(scomp, ncomp));
1546
1547 for (int i = 0; i < static_cast<int>(ranges.size()); i++)
1548 {
1549 const int SComp = ranges[i].first;
1550 const int NComp = ranges[i].second;
1551 Interpolater* mapper = desc.interp(SComp);
1552
1553 BoxArray crseBA(mf_BA.size());
1554
1555 for (int j = 0, N = crseBA.size(); j < N; ++j)
1556 {
1557 BL_ASSERT(mf_BA[j].ixType() == desc.getType());

1560 }
1561

1563#ifdef AMREX_USE_EB
1564 if (EB2::TopIndexSpaceIfPresent()) {
1565 auto cfactory = makeEBFabFactory(cgeom, crseBA, mf_DM, {0,0,0}, EBSupport::basic);
1566 crseMF.define(crseBA,mf_DM,NComp,0,MFInfo(),*cfactory);
1567 } else
1568#endif
1569 {
1570 crseMF.define(crseBA,mf_DM,NComp,0);
1571 }
1572
1573 if ( level == 1
1574 || amrex::ProperlyNested(crse_ratio, parent->blockingFactor(level),
1575 nghost, mf_BA.ixType(), mapper) )
1576 {
1577 StateData& statedata = clev.state[idx];
1578
1579 Vector<MultiFab*> smf;
1580 Vector<Real> stime;
1581 statedata.getData(smf,stime,time);
1582
1583 StateDataPhysBCFunct physbcf(statedata,SComp,cgeom);
1584
1585 crseMF.setDomainBndry(std::numeric_limits<Real>::quiet_NaN(), cgeom);
1586 amrex::FillPatchSingleLevel(crseMF,time,smf,stime,SComp,0,NComp,cgeom,physbcf,SComp);
1587 }
1588 else
1589 {
1590 FillPatch(clev,crseMF,0,time,idx,SComp,NComp,0);
1591 }
1592
1593#ifdef _OPENMP
1594#pragma omp parallel if (Gpu::notInLaunchRegion())
1595#endif
1596 for (MFIter mfi(mf); mfi.isValid(); ++mfi)
1597 {
1598 const Box& dbx = amrex::grow(mfi.validbox(),nghost) & domain_g;
1599
1600 Vector<BCRec> bcr(ncomp);
1601
1602 amrex::setBC(dbx,pdomain,SComp,0,NComp,desc.getBCs(),bcr);
1603
1604 mapper->interp(crseMF[mfi],
1605 0,
1606 mf[mfi],
1607 DComp,
1608 NComp,
1609 dbx,
1610 crse_ratio,
1611 cgeom,
1612 geom,
1613 bcr,
1614 SComp,
1615 idx, RunOn::Gpu);
1616 }
1617
1618 if (nghost > 0) {
1619 StateDataPhysBCFunct physbcf(state[idx],SComp,geom);
1620 physbcf.FillBoundary(mf, DComp, NComp, mf.nGrowVect(), time, SComp);
1621 }
1622
1623 DComp += NComp;
1624 }
