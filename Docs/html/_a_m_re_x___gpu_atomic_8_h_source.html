23 template <typename R, typename I, typename F>
24 AMREX_GPU_DEVICE AMREX_FORCE_INLINE
25 R atomic_op (R* const address, R const val, F const f) noexcept
26 {
27#if defined(__SYCL_DEVICE_ONLY__)
28 constexpr auto mo = sycl::memory_order::relaxed;
29 constexpr auto as = sycl::access::address_space::global_space;
30 static_assert(sizeof(R) == sizeof(I), "sizeof R != sizeof I");
31 I* const add_as_I = reinterpret_cast<I*>(address);
32 sycl::atomic<I,as> a{sycl::multi_ptr<I,as>(add_as_I)};
33 I old_I = a.load(mo), new_I;
34 do {
35 R const new_R = f(*(reinterpret_cast<R const*>(&old_I)), val);
36 new_I = *(reinterpret_cast<I const*>(&new_R));
37 } while (not a.compare_exchange_strong(old_I, new_I, mo));
38 return *(reinterpret_cast<R const*>(&old_I));
39#else
40 R old = *address;
41 *address = f(*address, val);
42 return old;
43#endif
44 }
45
46#elif defined(AMREX_USE_CUDA) || defined(AMREX_USE_HIP)

48 template <typename R, typename I, typename F>
49 AMREX_GPU_DEVICE AMREX_FORCE_INLINE
50 R atomic_op (R* const address, R const val, F const f) noexcept
51 {
52 static_assert(sizeof(R) == sizeof(I), "sizeof R != sizeof I");
53 I* const add_as_I = reinterpret_cast<I*>(address);
54 I old_I = *add_as_I, assumed_I;
55 do {
56 assumed_I = old_I;
57 R const new_R = f(*(reinterpret_cast<R const*>(&assumed_I)), val);
58 old_I = atomicCAS(add_as_I, assumed_I, *(reinterpret_cast<I const*>(&new_R)));
59 } while (assumed_I != old_I);
60 return *(reinterpret_cast<R const*>(&old_I));
61 }
