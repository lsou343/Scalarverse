115::Where (const P& p,
116 ParticleLocData& pld,
117 int lev_min,
118 int lev_max,
119 int nGrow,
120 int local_grid) const
121{
122
123 AMREX_ASSERT(m_gdb != 0);

125 if (lev_max == -1)
126 lev_max = finestLevel();

128 AMREX_ASSERT(lev_max <= finestLevel());
129
130 AMREX_ASSERT(nGrow == 0 || (nGrow >= 0 && lev_min == lev_max));

133
134 for (int lev = lev_max; lev >= lev_min; lev--) {
135 const IntVect& iv = Index(p, lev);

137 // The fact that we are here means this particle does not belong to any finer grids.
138 if (pld.m_grid >= 0) {
139 if (pld.m_grown_gridbox.contains(iv)) {
140 pld.m_cell = iv;
141 if (!pld.m_tilebox.contains(iv)) {
142 pld.m_tile = getTileIndex(iv, pld.m_gridbox, do_tiling, tile_size, pld.m_tilebox);
143 }
144 return true;
145 }
146 }
147 }
148
149 int grid;
150 const BoxArray& ba = ParticleBoxArray(lev);
151 AMREX_ASSERT(ba.ixType().cellCentered());
152
153 if (local_grid < 0) {
154 ba.intersections(Box(iv, iv), isects, true, nGrow);
155 grid = isects.empty() ? -1 : isects[0].first;
156 } else {
157 grid = (*redistribute_mask_ptr)[local_grid](iv, 0);
158 }
159
160 if (grid >= 0) {
161 const Box& bx = ba.getCellCenteredBox(grid);
162 pld.m_lev = lev;
163 pld.m_grid = grid;
164 pld.m_tile = getTileIndex(iv, bx, do_tiling, tile_size, pld.m_tilebox);
165 pld.m_cell = iv;
166 pld.m_gridbox = bx;
167 pld.m_grown_gridbox = amrex::grow(bx, nGrow);
168 return true;
169 }
170 }
171
172 return false;

177ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
178::EnforcePeriodicWhere (ParticleType& p,
179 ParticleLocData& pld,
180 int lev_min,
181 int lev_max,

184

188

194 // Create a copy "dummy" particle to check for periodic outs.
195 ParticleType p_prime = p;
196 if (PeriodicShift(p_prime)) {

334template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
335void
336ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>::locateParticle (ParticleType& p, ParticleLocData& pld,
337 int lev_min, int lev_max, int nGrow, int local_grid) const
338{
339 bool outside = AMREX_D_TERM(p.pos(0) < Geom(0).ProbLo(0)
340 || p.pos(0) >= Geom(0).ProbHi(0),
341 || p.pos(1) < Geom(0).ProbLo(1)
342 || p.pos(1) >= Geom(0).ProbHi(1),
343 || p.pos(2) < Geom(0).ProbLo(2)
344 || p.pos(2) >= Geom(0).ProbHi(2));
345
346 if (not outside)
347 {
348 if (Geom(0).outsideRoundoffDomain(AMREX_D_DECL(p.pos(0), p.pos(1), p.pos(2))))
349 {
350 RealBox roundoff_domain = Geom(0).RoundoffDomain();
351 for (int idim=0; idim < AMREX_SPACEDIM; ++idim)
352 {
353 if (p.pos(idim) <= roundoff_domain.lo(idim)) {
354 p.pos(idim) = std::nextafter((ParticleReal) roundoff_domain.lo(idim), (ParticleReal) roundoff_domain.hi(idim));
355 }
356 if (p.pos(idim) >= roundoff_domain.hi(idim)) {
357 p.pos(idim) = std::nextafter((ParticleReal) roundoff_domain.hi(idim), (ParticleReal) roundoff_domain.lo(idim));
358 }
359 }
360
361 AMREX_ASSERT(not Geom(0).outsideRoundoffDomain(AMREX_D_DECL(p.pos(0), p.pos(1), p.pos(2))));
362 }
363 }
364
365 bool success;
366 if (outside)
367 {
368 // Note that EnforcePeriodicWhere may shift the particle if it is successful.
369 success = EnforcePeriodicWhere(p, pld, lev_min, lev_max, local_grid);
370 if (!success && lev_min == 0)
371 {
372 // The particle has left the domain; invalidate it.
373 p.id() = -p.id();
374 success = true;
375 }
376 }
377 else
378 {
379 success = Where(p, pld, lev_min, lev_max, 0, local_grid);
380 }
381
382 if (!success)
383 {
384 success = (nGrow > 0) && Where(p, pld, lev_min, lev_min, nGrow);
385 pld.m_grown_gridbox = pld.m_gridbox; // reset grown box for subsequent calls.
386 }
387
388 if (!success)
389 {
390 amrex::Abort("ParticleContainer::locateParticle(): invalid particle.");
391 }
392}

679 else

681 // If mf is not defined on the particle_box_array, then we need
682 // to make a temporary mf_pointer here and copy it into mf at the end.
683 mf_pointer = new MultiFab(ParticleBoxArray(lev),
684 ParticleDistributionMap(lev),
685 mf.nComp(),mf.nGrow());
686 }
687
688 ParticleLocData pld;
689 for (auto& kv : pmap) {
690 int gid = kv.first.first;
691 const auto& pbox = kv.second.GetArrayOfStructs();
692 FArrayBox& fab = (*mf_pointer)[gid];
693 for (int k = 0; k < pbox.numParticles(); ++ k) {
694 const ParticleType& p = pbox[k];
695 if (p.id() > 0) {
696 Where(p, pld);
697 AMREX_ASSERT(pld.m_grid == gid);
698 fab(pld.m_cell) += 1;
699 num_particles_in_domain += 1;
700 }
701 }
702 }
703
704 // If mf is not defined on the particle_box_array, then we need
705 // to copy here from mf_pointer into mf. I believe that we don't
706 // need any information in ghost cells so we don't copy those.
707 if (mf_pointer != &mf)
708 {
709 mf.copy(*mf_pointer,0,0,mf.nComp());
710 delete mf_pointer;
711 }
712
713 if (!local) {
714 ParallelAllReduce::Sum(num_particles_in_domain, ParallelContext::CommunicatorSub());
715 }
716
717 return num_particles_in_domain;
718}

772template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
773void
774ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>
775::CreateVirtualParticles (int level, ParticleTileType& virts) const
776{
777 BL_PROFILE("ParticleContainer::CreateVirtualParticles()");
778 AMREX_ASSERT(level > 0);
779 AMREX_ASSERT(virts.empty());
780
781 if (level >= static_cast<int>(m_particles.size()))
782 return;
783
784 if (aggregation_type == "")
785 {
786 ParmParse pp("particles");
787 aggregation_type = "None";
788 pp.query("aggregation_type", aggregation_type);
789 aggregation_buffer = 2;
790 pp.query("aggregation_buffer", aggregation_buffer);
791 }
792
793 if (aggregation_type == "None");
794 else if (aggregation_type == "Cell");
795 else if (aggregation_type == "Flow") amrex::Abort("Flow aggregation not implemented");
796
797 if (aggregation_type == "None")
798 {
799 for(ParConstIterType pti(*this, level); pti.isValid(); ++pti)
800 {
801 const auto& ptile = ParticlesAt(level, pti);
802 const auto src = ptile.getConstParticleTileData();
803
804 auto& aos = ptile.GetArrayOfStructs();
805 for (int pindex = 0; pindex < ptile.numParticles(); ++pindex)
806 {
807 SuperParticleType p = src.getSuperParticle(pindex);
808 p.id() = VirtualParticleID;
809 virts.push_back(p);
810 }
811 }
812 }
813
814 if (aggregation_type == "Cell")
815 {
816 BoxList bl_buffer;
817 bl_buffer.complementIn(Geom(level).Domain(), ParticleBoxArray(level));
818 BoxArray buffer(std::move(bl_buffer));
819 buffer.grow(aggregation_buffer);
820
821 for(ParConstIterType pti(*this, level); pti.isValid(); ++pti)
822 {
823 const auto& ptile = ParticlesAt(level, pti);
824 const auto& aos = pti.GetArrayOfStructs();
825 const auto src = ptile.getConstParticleTileData();
826
827 std::map<IntVect,SuperParticleType> agg_map;
828
829 for (int pindex = 0; pindex < ptile.numParticles(); ++pindex)
830 {
831 IntVect cell = Index(aos[pindex], level);
832 if (buffer.contains(cell))
833 {
834 // It's in the no-aggregation buffer.
835 // Set its id to indicate that it's a virt.
836 SuperParticleType p = src.getSuperParticle(pindex);
837 p.id() = VirtualParticleID;
838 virts.push_back(p);
839 }
840 else
841 {
842 //
843 // Note that Cell aggregation assumes that p.rdata(0) is mass and
844 // that all other components should be combined in a mass-weighted
845 // average.
846 //
847 auto agg_map_it = agg_map.find(cell);
848
849 if (agg_map_it == agg_map.end())
850 {
851 //
852 // Add the particle.
853 //
854 SuperParticleType p = src.getSuperParticle(pindex);
855 p.id() = VirtualParticleID;
856 agg_map[cell] = p;
857 }
858 else
859 {
860 AMREX_ASSERT(agg_map_it != agg_map.end());
861 const SuperParticleType& pnew = src.getSuperParticle(pindex);
862 SuperParticleType& pold = agg_map_it->second;
863 const Real old_mass = pold.rdata(0);
864 const Real new_mass = pnew.rdata(0);
865 const Real total_mass = old_mass + new_mass;
866 //
867 // Set the position to the center of mass.
868 //
869 for (int i = 0; i < AMREX_SPACEDIM; i++)
870 {
871 pold.pos(i) = (old_mass*pold.pos(i) + new_mass*pnew.pos(i))/total_mass;
872 }
873 AMREX_ASSERT(this->Index(pold, level) == cell);
874 //
875 // Set the metadata (presumably velocity) to the mass-weighted average.
876 //
877 for (int i = 1; i < NStructReal; i++)
878 {
879 pold.rdata(i) = (old_mass*pold.rdata(i) + new_mass*pnew.rdata(i))/total_mass;
880 }
881 pold.rdata(0) = total_mass;
882 }
883 }
884 }
885
886 //
887 // Add the aggregated particles to the virtuals.
888 //
889 for (const auto& agg_particle : agg_map)
890 {
891 virts.push_back(agg_particle.second);
892 }
893 }
894 }
895}

1309 Vector<std::map<std::pair<int, int>, Vector<ParticleVector> > > tmp_local;
1310 Vector<std::map<std::pair<int, int>, Vector<StructOfArrays<NArrayReal, NArrayInt> > > > soa_local;
1311 tmp_local.resize(theEffectiveFinestLevel+1);
1312 soa_local.resize(theEffectiveFinestLevel+1);
1313
1314 // we resize these buffers outside the parallel region
1315 for (int lev = lev_min; lev <= lev_max; lev++) {
1316 for (MFIter mfi(*m_dummy_mf[lev], this->do_tiling ? this->tile_size : IntVect::TheZeroVector());
1317 mfi.isValid(); ++mfi) {
1318 auto index = std::make_pair(mfi.index(), mfi.LocalTileIndex());
1319 tmp_local[lev][index].resize(num_threads);
1320 soa_local[lev][index].resize(num_threads);
1321 for (int t = 0; t < num_threads; ++t) {
1322 soa_local[lev][index][t].define(m_num_runtime_real, m_num_runtime_int);
1323 }
1324 }
1325 }
1326 if (local) {
1327 for (int i = 0; i < neighbor_procs.size(); ++i)
1328 tmp_remote[neighbor_procs[i]].resize(num_threads);
1329 } else {
1330 for (int i = 0; i < ParallelContext::NProcsSub(); ++i)
1331 tmp_remote[i].resize(num_threads);
1332 }
1333
1334 // first pass: for each tile in parallel, in each thread copies the particles that
1335 // need to be moved into it's own, temporary buffer.
1336 for (int lev = lev_min; lev <= nlevs_particles; lev++) {
1337 auto& pmap = m_particles[lev];
1338
1339 Vector<std::pair<int, int> > grid_tile_ids;
1340 Vector<ParticleTileType*> ptile_ptrs;
1341 for (auto& kv : pmap)
1342 {
1343 grid_tile_ids.push_back(kv.first);
1344 ptile_ptrs.push_back(&(kv.second));
1345 }
1346
1347#ifdef _OPENMP
1348#pragma omp parallel for
1349#endif
1350 for (int pmap_it = 0; pmap_it < static_cast<int>(ptile_ptrs.size()); ++pmap_it)
1351 {
1352 int thread_num = OpenMP::get_thread_num();
1353 int grid = grid_tile_ids[pmap_it].first;
1354 int tile = grid_tile_ids[pmap_it].second;
1355 auto& aos = ptile_ptrs[pmap_it]->GetArrayOfStructs();
1356 auto& soa = ptile_ptrs[pmap_it]->GetStructOfArrays();
1357 AMREX_ASSERT_WITH_MESSAGE((NumRealComps() == 0 and NumIntComps() == 0)
1358 or aos.size() == soa.size(),
1359 "The AoS and SoA data on this tile are different sizes - "
1360 "perhaps particles have not been initialized correctly?");
1361 unsigned npart = aos.numParticles();
1362 ParticleLocData pld;
1363 if (npart != 0) {
1364 Long last = npart - 1;
1365 Long pindex = 0;
1366 while (pindex <= last) {
1367 ParticleType& p = aos[pindex];
1368
1369 if (p.id() < 0)
1370 {
1371 aos[pindex] = aos[last];
1372 for (int comp = 0; comp < NumRealComps(); comp++)
1373 soa.GetRealData(comp)[pindex] = soa.GetRealData(comp)[last];
1374 for (int comp = 0; comp < NumIntComps(); comp++)
1375 soa.GetIntData(comp)[pindex] = soa.GetIntData(comp)[last];
1376 correctCellVectors(last, pindex, grid, aos[pindex]);
1377 --last;
1378 continue;
1379 }
1380
1381 locateParticle(p, pld, lev_min, lev_max, nGrow, local ? grid : -1);
1382
1383 particlePostLocate(p, pld, lev);
1384
1385 if (p.id() < 0)
1386 {
1387 aos[pindex] = aos[last];
1388 for (int comp = 0; comp < NumRealComps(); comp++)
1389 soa.GetRealData(comp)[pindex] = soa.GetRealData(comp)[last];
1390 for (int comp = 0; comp < NumIntComps(); comp++)
1391 soa.GetIntData(comp)[pindex] = soa.GetIntData(comp)[last];
1392 correctCellVectors(last, pindex, grid, aos[pindex]);
1393 --last;
1394 continue;
1395 }
1396
1397 const int who = ParallelContext::global_to_local_rank(ParticleDistributionMap(pld.m_lev)[pld.m_grid]);
1398 if (who == MyProc) {
1399 if (pld.m_lev != lev || pld.m_grid != grid || pld.m_tile != tile) {
1400 // We own it but must shift it to another place.
1401 auto index = std::make_pair(pld.m_grid, pld.m_tile);
1402 AMREX_ASSERT(tmp_local[pld.m_lev][index].size() == num_threads);
1403 tmp_local[pld.m_lev][index][thread_num].push_back(p);
1404 for (int comp = 0; comp < NumRealComps(); ++comp) {
1405 RealVector& arr = soa_local[pld.m_lev][index][thread_num].GetRealData(comp);
1406 arr.push_back(soa.GetRealData(comp)[pindex]);
1407 }
1408 for (int comp = 0; comp < NumIntComps(); ++comp) {
1409 IntVector& arr = soa_local[pld.m_lev][index][thread_num].GetIntData(comp);
1410 arr.push_back(soa.GetIntData(comp)[pindex]);
1411 }
1412
1413 p.id() = -p.id(); // Invalidate the particle
1414 }
1415 }
1416 else {
1417 auto& particles_to_send = tmp_remote[who][thread_num];
1418 auto old_size = particles_to_send.size();
1419 auto new_size = old_size + superparticle_size;
1420 particles_to_send.resize(new_size);
1421 std::memcpy(&particles_to_send[old_size], &p, particle_size);
1422 char* dst = &particles_to_send[old_size] + particle_size;
1423 for (int comp = 0; comp < NumRealComps(); comp++) {
1424 if (h_communicate_real_comp[comp]) {
1425 std::memcpy(dst, &soa.GetRealData(comp)[pindex], sizeof(ParticleReal));
1426 dst += sizeof(ParticleReal);
1427 }
1428 }
1429 for (int comp = 0; comp < NumIntComps(); comp++) {
1430 if (h_communicate_int_comp[comp]) {
1431 std::memcpy(dst, &soa.GetIntData(comp)[pindex], sizeof(int));
1432 dst += sizeof(int);
1433 }
1434 }
1435
1436 p.id() = -p.id(); // Invalidate the particle
1437 }
1438
1439 if (p.id() < 0)
1440 {
1441 aos[pindex] = aos[last];
1442 for (int comp = 0; comp < NumRealComps(); comp++)
1443 soa.GetRealData(comp)[pindex] = soa.GetRealData(comp)[last];
1444 for (int comp = 0; comp < NumIntComps(); comp++)
1445 soa.GetIntData(comp)[pindex] = soa.GetIntData(comp)[last];
1446 correctCellVectors(last, pindex, grid, aos[pindex]);
1447 --last;
1448 continue;
1449 }
1450
1451 ++pindex;
1452 }
1453
1454 aos().erase(aos().begin() + last + 1, aos().begin() + npart);
1455 for (int comp = 0; comp < NumRealComps(); comp++) {
1456 RealVector& rdata = soa.GetRealData(comp);
1457 rdata.erase(rdata.begin() + last + 1, rdata.begin() + npart);
1458 }
1459 for (int comp = 0; comp < NumIntComps(); comp++) {
1460 IntVector& idata = soa.GetIntData(comp);
1461 idata.erase(idata.begin() + last + 1, idata.begin() + npart);
1462 }
1463 }
1464 }
1465 }
1466
1467 for (int lev = lev_min; lev <= lev_max; lev++) {
1468 auto& pmap = m_particles[lev];
1469 for (auto pmap_it = pmap.begin(); pmap_it != pmap.end(); /* no ++ */) {
1470
1471 // Remove any map entries for which the particle container is now empty.
1472 if (pmap_it->second.empty()) {
1473 pmap.erase(pmap_it++);
1474 }
1475 else {
1476 ++pmap_it;
1477 }
1478 }
1479 }
1480
1481 // Second pass - for each tile in parallel, collect the particles we are owed from all thread's buffers.
1482 for (int lev = lev_min; lev <= lev_max; lev++) {
1483 typename std::map<std::pair<int, int>, Vector<ParticleVector > >::iterator pmap_it;
1484
1485 Vector<std::pair<int, int> > grid_tile_ids;
1486 Vector<Vector<ParticleVector>* > pvec_ptrs;
1487
1488 // we need to create any missing map entries in serial here
1489 for (pmap_it=tmp_local[lev].begin(); pmap_it != tmp_local[lev].end(); pmap_it++)
1490 {
1491 m_particles[lev][pmap_it->first];
1492 grid_tile_ids.push_back(pmap_it->first);
1493 pvec_ptrs.push_back(&(pmap_it->second));
1494 }
1495
1496#ifdef _OPENMP
1497#pragma omp parallel for
1498#endif
1499 for (int pit = 0; pit < static_cast<int>(pvec_ptrs.size()); ++pit)
1500 {
1501 auto index = grid_tile_ids[pit];
1502 auto& ptile = DefineAndReturnParticleTile(lev, index.first, index.second);
1503 auto& aos = ptile.GetArrayOfStructs();
1504 auto& soa = ptile.GetStructOfArrays();
1505 auto& aos_tmp = *(pvec_ptrs[pit]);
1506 auto& soa_tmp = soa_local[lev][index];
1507 for (int i = 0; i < num_threads; ++i) {
1508 aos.insert(aos.end(), aos_tmp[i].begin(), aos_tmp[i].end());
1509 aos_tmp[i].erase(aos_tmp[i].begin(), aos_tmp[i].end());
1510 for (int comp = 0; comp < NumRealComps(); ++comp) {
1511 RealVector& arr = soa.GetRealData(comp);
1512 RealVector& tmp = soa_tmp[i].GetRealData(comp);
1513 arr.insert(arr.end(), tmp.begin(), tmp.end());
1514 tmp.erase(tmp.begin(), tmp.end());
1515 }
1516 for (int comp = 0; comp < NumIntComps(); ++comp) {
1517 IntVector& arr = soa.GetIntData(comp);
1518 IntVector& tmp = soa_tmp[i].GetIntData(comp);
1519 arr.insert(arr.end(), tmp.begin(), tmp.end());
1520 tmp.erase(tmp.begin(), tmp.end());
1521 }
1522 }
1523 }
1524 }
1525
1526 for (auto& map_it : tmp_remote) {
1527 int who = map_it.first;
1528 not_ours[who];
1529 }
1530
1531 Vector<int> dest_proc_ids;
1532 Vector<Vector<Vector<char> >* > pbuff_ptrs;
1533 for (auto& kv : tmp_remote)
1534 {
1535 dest_proc_ids.push_back(kv.first);
1536 pbuff_ptrs.push_back(&(kv.second));
1537 }
1538
1539#ifdef _OPENMP
1540#pragma omp parallel for
1541#endif
1542 for (int pmap_it = 0; pmap_it < static_cast<int>(pbuff_ptrs.size()); ++pmap_it)
1543 {
1544 int who = dest_proc_ids[pmap_it];
1545 Vector<Vector<char> >& tmp = *(pbuff_ptrs[pmap_it]);
1546 for (int i = 0; i < num_threads; ++i) {
1547 not_ours[who].insert(not_ours[who].end(), tmp[i].begin(), tmp[i].end());
1548 tmp[i].erase(tmp[i].begin(), tmp[i].end());
1549 }
1550 }
1551
1552 // remove any empty map entries from not_ours
1553 for (auto pmap_it = not_ours.begin(); pmap_it != not_ours.end(); /* no ++ */) {
1554 if (pmap_it->second.empty()) {
1555 not_ours.erase(pmap_it++);
1556 }
1557 else {
1558 ++pmap_it;
1559 }
1560 }
1561
1562 if (int(m_particles.size()) > theEffectiveFinestLevel+1) {
1563 // Looks like we lost an AmrLevel on a regrid.
1564 if (m_verbose > 0) {
1565 amrex::Print() << "ParticleContainer::Redistribute() resizing m_particles from "
1566 << m_particles.size() << " to " << theEffectiveFinestLevel+1 << '\n';
1567 }
1568 AMREX_ASSERT(int(m_particles.size()) >= 2);
1569
1570 m_particles.resize(theEffectiveFinestLevel + 1);
1571 m_dummy_mf.resize(theEffectiveFinestLevel + 1);
1572 }
1573
1574 if (ParallelContext::NProcsSub() == 1) {
1575 AMREX_ASSERT(not_ours.empty());
1576 }
1577 else {
1578 RedistributeMPI(not_ours, lev_min, lev_max, nGrow, local);
1579 }
1580
1581 AMREX_ASSERT(OK(lev_min, lev_max, nGrow));
1582
1583 if (m_verbose > 0) {
1584 Real stoptime = amrex::second() - strttime;
1585
1586 ByteSpread();
1587
1588#ifdef AMREX_LAZY
1589 Lazy::QueueReduction( [=] () mutable {
1590#endif
1591 ParallelReduce::Max(stoptime, ParallelContext::IOProcessorNumberSub(),
1592 ParallelContext::CommunicatorSub());
1593
1594 amrex::Print() << "ParticleContainer::Redistribute() time: " << stoptime << "\n\n";
1595#ifdef AMREX_LAZY
1596 });
1597#endif
1598 }
1599}

1614template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
1615void
1616ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>::
1617BuildRedistributeMask (int lev, int nghost) const
1618{
1619 BL_PROFILE("ParticleContainer::BuildRedistributeMask");
1620 AMREX_ASSERT(lev == 0);
1621
1622 if (redistribute_mask_ptr == nullptr ||
1623 redistribute_mask_nghost < nghost ||
1624 ! BoxArray::SameRefs(redistribute_mask_ptr->boxArray(), this->ParticleBoxArray(lev)) ||
1625 ! DistributionMapping::SameRefs(redistribute_mask_ptr->DistributionMap(), this->ParticleDistributionMap(lev)))
1626 {
1627 const Geometry& geom = this->Geom(lev);
1628 const BoxArray& ba = this->ParticleBoxArray(lev);
1629 const DistributionMapping& dmap = this->ParticleDistributionMap(lev);
1630
1631 redistribute_mask_nghost = nghost;
1632 redistribute_mask_ptr.reset(new iMultiFab(ba, dmap, 2, nghost));
1633 redistribute_mask_ptr->setVal(-1, nghost);
1634
1635 const auto tile_size_do = this->do_tiling ? this->tile_size : IntVect::TheZeroVector();
1636
1637#ifdef _OPENMP
1638#pragma omp parallel
1639#endif
1640 for (MFIter mfi(*redistribute_mask_ptr, tile_size_do); mfi.isValid(); ++mfi)
1641 {
1642 const Box& box = mfi.tilebox();
1643 const int grid_id = mfi.index();
1644 const int tile_id = mfi.LocalTileIndex();
1645 (*redistribute_mask_ptr)[mfi].template setVal<RunOn::Host>(grid_id, box, 0, 1);
1646 (*redistribute_mask_ptr)[mfi].template setVal<RunOn::Host>(tile_id, box, 1, 1);
1647 }
1648
1649 redistribute_mask_ptr->FillBoundary(geom.periodicity());
1650
1651 neighbor_procs.clear();
1652 for (MFIter mfi(*redistribute_mask_ptr, tile_size_do); mfi.isValid(); ++mfi)
1653 {
1654 const Box& box = mfi.growntilebox();
1655 for (IntVect iv = box.smallEnd(); iv <= box.bigEnd(); box.next(iv))
1656 {
1657 const int grid = (*redistribute_mask_ptr)[mfi](iv, 0);
1658 if (grid >= 0)
1659 {
1660 const int global_rank = this->ParticleDistributionMap(lev)[grid];
1661 const int rank = ParallelContext::global_to_local_rank(global_rank);
1662 if (rank != ParallelContext::MyProcSub())
1663 neighbor_procs.push_back(rank);
1664 }
1665 }
1666 }
1667 RemoveDuplicates(neighbor_procs);
1668 }
1669}

1671template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
1672void
1673ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>::
1674RedistributeMPI (std::map<int, Vector<char> >& not_ours,
1675 int lev_min, int lev_max, int nGrow, int local)
1676{
1677 BL_PROFILE("ParticleContainer::RedistributeMPI()");
1678 BL_PROFILE_VAR_NS("RedistributeMPI_locate", blp_locate);
1679 BL_PROFILE_VAR_NS("RedistributeMPI_copy", blp_copy);
1680
1681#ifdef AMREX_USE_MPI
1682
1683 using buffer_type = unsigned long long;
1684
1685 std::map<int, Vector<buffer_type> > mpi_snd_data;
1686 for (const auto& kv : not_ours)
1687 {
1688 int nbt = (kv.second.size() + sizeof(buffer_type)-1)/sizeof(buffer_type);
1689 mpi_snd_data[kv.first].resize(nbt);
1690 std::memcpy((char*) mpi_snd_data[kv.first].data(), kv.second.data(), kv.second.size());
1691 }
1692
1693 const int NProcs = ParallelContext::NProcsSub();
1694 const int NNeighborProcs = neighbor_procs.size();
1695
1696 // We may now have particles that are rightfully owned by another CPU.
1697 Vector<Long> Snds(NProcs, 0), Rcvs(NProcs, 0); // bytes!
1698
1699 Long NumSnds = 0;
1700 if (local > 0)
1701 {
1702 AMREX_ALWAYS_ASSERT(lev_min == 0);
1703 AMREX_ALWAYS_ASSERT(lev_max == 0);
1704 BuildRedistributeMask(0, local);
1705 NumSnds = doHandShakeLocal(not_ours, neighbor_procs, Snds, Rcvs);
1706 }
1707 else
1708 {
1709 NumSnds = doHandShake(not_ours, Snds, Rcvs);
1710 }
1711
1712 const int SeqNum = ParallelDescriptor::SeqNum();
1713
1714 if ((not local) and NumSnds == 0)
1715 return; // There's no parallel work to do.
1716
1717 if (local)
1718 {
1719 Long tot_snds_this_proc = 0;
1720 Long tot_rcvs_this_proc = 0;
1721 for (int i = 0; i < NNeighborProcs; ++i) {
1722 tot_snds_this_proc += Snds[neighbor_procs[i]];
1723 tot_rcvs_this_proc += Rcvs[neighbor_procs[i]];
1724 }
1725 if ( (tot_snds_this_proc == 0) and (tot_rcvs_this_proc == 0) ) {
1726 return; // There's no parallel work to do.
1727 }
1728 }
1729
1730 Vector<int> RcvProc;
1731 Vector<std::size_t> rOffset; // Offset (in bytes) in the receive buffer
1732
1733 std::size_t TotRcvInts = 0;
1734 std::size_t TotRcvBytes = 0;
1735 for (int i = 0; i < NProcs; ++i) {
1736 if (Rcvs[i] > 0) {
1737 RcvProc.push_back(i);
1738 rOffset.push_back(TotRcvInts);
1739 TotRcvBytes += Rcvs[i];
1740 int nbt = (Rcvs[i] + sizeof(buffer_type)-1)/sizeof(buffer_type);
1741 TotRcvInts += nbt;
1742 }
1743 }
1744
1745 const int nrcvs = RcvProc.size();
1746 Vector<MPI_Status> stats(nrcvs);
1747 Vector<MPI_Request> rreqs(nrcvs);
1748
1749 // Allocate data for rcvs as one big chunk.
1750 Vector<unsigned long long> recvdata(TotRcvInts);
1751
1752 // Post receives.
1753 for (int i = 0; i < nrcvs; ++i) {
1754 const auto Who = RcvProc[i];
1755 const auto offset = rOffset[i];
1756 const auto Cnt = (Rcvs[Who] + sizeof(buffer_type)-1)/sizeof(buffer_type);
1757 AMREX_ASSERT(Cnt > 0);
1758 AMREX_ASSERT(Cnt < size_t(std::numeric_limits<int>::max()));
1759 AMREX_ASSERT(Who >= 0 && Who < NProcs);
1760
1761 rreqs[i] = ParallelDescriptor::Arecv(&recvdata[offset], Cnt, Who, SeqNum,
1762 ParallelContext::CommunicatorSub()).req();
1763 }
1764
1765 // Send.
1766 for (const auto& kv : mpi_snd_data) {
1767 const auto Who = kv.first;
1768 const auto Cnt = kv.second.size();
1769
1770 AMREX_ASSERT(Cnt > 0);
1771 AMREX_ASSERT(Who >= 0 && Who < NProcs);
1772 AMREX_ASSERT(Cnt < std::numeric_limits<int>::max());
1773
1774 ParallelDescriptor::Send(kv.second.data(), Cnt, Who, SeqNum,
1775 ParallelContext::CommunicatorSub());
1776 }
1777
1778 if (nrcvs > 0) {
1779 ParallelDescriptor::Waitall(rreqs, stats);
1780
1781 BL_PROFILE_VAR_START(blp_locate);
1782
1783 int npart = TotRcvBytes / superparticle_size;
1784
1785 Vector<int> rcv_levs(npart);
1786 Vector<int> rcv_grid(npart);
1787 Vector<int> rcv_tile(npart);
1788
1789 int ipart = 0;
1790 ParticleLocData pld;
1791 for (int j = 0; j < nrcvs; ++j)
1792 {
1793 const auto offset = rOffset[j];
1794 const auto Who = RcvProc[j];
1795 const auto Cnt = Rcvs[Who] / superparticle_size;
1796 for (int i = 0; i < int(Cnt); ++i)
1797 {
1798 char* pbuf = ((char*) &recvdata[offset]) + i*superparticle_size;
1799 ParticleType p;
1800 std::memcpy(&p, pbuf, sizeof(ParticleType));
1801 locateParticle(p, pld, lev_min, lev_max, nGrow);
1802 rcv_levs[ipart] = pld.m_lev;
1803 rcv_grid[ipart] = pld.m_grid;
1804 rcv_tile[ipart] = pld.m_tile;
1805 ++ipart;
1806 }
1807 }
1808
1809 BL_PROFILE_VAR_STOP(blp_locate);
1810
1811 BL_PROFILE_VAR_START(blp_copy);
1812
1813#ifndef AMREX_USE_GPU
1814 ipart = 0;
1815 for (int i = 0; i < nrcvs; ++i)
1816 {
1817 const auto offset = rOffset[i];
1818 const auto Who = RcvProc[i];
1819 const auto Cnt = Rcvs[Who] / superparticle_size;
1820 for (int j = 0; j < int(Cnt); ++j)
1821 {
1822 auto& ptile = m_particles[rcv_levs[ipart]][std::make_pair(rcv_grid[ipart],
1823 rcv_tile[ipart])];
1824 char* pbuf = ((char*) &recvdata[offset]) + j*superparticle_size;
1825
1826 ParticleType p;
1827 std::memcpy(&p, pbuf, sizeof(ParticleType));
1828 pbuf += sizeof(ParticleType);
1829 ptile.push_back(p);
1830 for (int comp = 0; comp < NumRealComps(); ++comp) {
1831 if (h_communicate_real_comp[comp]) {
1832 ParticleReal rdata;
1833 std::memcpy(&rdata, pbuf, sizeof(ParticleReal));
1834 pbuf += sizeof(ParticleReal);
1835 ptile.push_back_real(comp, rdata);
1836 } else {
1837 ptile.push_back_real(comp, 0.0);
1838 }
1839 }
1840
1841 for (int comp = 0; comp < NumIntComps(); ++comp) {
1842 if (h_communicate_int_comp[comp]) {
1843 int idata;
1844 std::memcpy(&idata, pbuf, sizeof(int));
1845 pbuf += sizeof(int);
1846 ptile.push_back_int(comp, idata);
1847 } else {
1848 ptile.push_back_int(comp, 0);
1849 }
1850 }
1851 ++ipart;
1852 }
1853 }
1854#else
1855 Vector<std::map<std::pair<int, int>, Gpu::HostVector<ParticleType> > > host_particles;
1856 host_particles.reserve(15);
1857 host_particles.resize(finestLevel()+1);
1858
1859 Vector<std::map<std::pair<int, int>,
1860 std::vector<Gpu::HostVector<Real> > > > host_real_attribs;
1861 host_real_attribs.reserve(15);
1862 host_real_attribs.resize(finestLevel()+1);
1863
1864 Vector<std::map<std::pair<int, int>,
1865 std::vector<Gpu::HostVector<int> > > > host_int_attribs;
1866 host_int_attribs.reserve(15);
1867 host_int_attribs.resize(finestLevel()+1);
1868
1869 ipart = 0;
1870 for (int i = 0; i < nrcvs; ++i)
1871 {
