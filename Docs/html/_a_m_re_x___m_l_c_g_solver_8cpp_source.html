81int
82MLCGSolver::solve_bicgstab (MultiFab& sol,
83 const MultiFab& rhs,
84 Real eps_rel,
85 Real eps_abs)
86{
87 BL_PROFILE("MLCGSolver::bicgstab");
88
89 const int ncomp = sol.nComp();
90
91 const BoxArray& ba = sol.boxArray();
92 const DistributionMapping& dm = sol.DistributionMap();
93 const auto& factory = sol.Factory();
94
95 MultiFab ph(ba, dm, ncomp, sol.nGrow(), MFInfo(), factory);
96 MultiFab sh(ba, dm, ncomp, sol.nGrow(), MFInfo(), factory);
97 ph.setVal(0.0);
98 sh.setVal(0.0);
99
100 MultiFab sorig(ba, dm, ncomp, nghost, MFInfo(), factory);
101 MultiFab p (ba, dm, ncomp, nghost, MFInfo(), factory);
102 MultiFab r (ba, dm, ncomp, nghost, MFInfo(), factory);
103 MultiFab s (ba, dm, ncomp, nghost, MFInfo(), factory);
104 MultiFab rh (ba, dm, ncomp, nghost, MFInfo(), factory);
105 MultiFab v (ba, dm, ncomp, nghost, MFInfo(), factory);
106 MultiFab t (ba, dm, ncomp, nghost, MFInfo(), factory);
107
108 Lp.correctionResidual(amrlev, mglev, r, sol, rhs, MLLinOp::BCMode::Homogeneous);
109
110 // If singular remove mean from residual
111// if (Lp.isBottomSingular()) mlmg->makeSolvable(amrlev, mglev, r);
112
113 // Then normalize
114 Lp.normalize(amrlev, mglev, r);
115
116 MultiFab::Copy(sorig,sol,0,0,ncomp,nghost);
117 MultiFab::Copy(rh, r, 0,0,ncomp,nghost);
118
119 sol.setVal(0);
120
121 Real rnorm = norm_inf(r);
122 const Real rnorm0 = rnorm;
123
124 if ( verbose > 0 )
125 {
126 amrex::Print() << "MLCGSolver_BiCGStab: Initial error (error0) = " << rnorm0 << '\n';
127 }
128 int ret = 0;
129 iter = 1;
130 Real rho_1 = 0, alpha = 0, omega = 0;
131
132 if ( rnorm0 == 0 || rnorm0 < eps_abs )
133 {
134 if ( verbose > 0 )
135 {
136 amrex::Print() << "MLCGSolver_BiCGStab: niter = 0,"
137 << ", rnorm = " << rnorm
138 << ", eps_abs = " << eps_abs << std::endl;
139 }
140 return ret;
141 }
142
143 for (; iter <= maxiter; ++iter)
144 {
145 const Real rho = dotxy(rh,r);
146 if ( rho == 0 )
147 {
148 ret = 1; break;
149 }
150 if ( iter == 1 )
151 {
152 MultiFab::Copy(p,r,0,0,ncomp,nghost);
153 }
154 else
155 {
156 const Real beta = (rho/rho_1)*(alpha/omega);
157 sxay(p, p, -omega, v, nghost);
158 sxay(p, r, beta, p, nghost);
159 }
160 MultiFab::Copy(ph,p,0,0,ncomp,nghost);
161 Lp.apply(amrlev, mglev, v, ph, MLLinOp::BCMode::Homogeneous, MLLinOp::StateMode::Correction);
162 Lp.normalize(amrlev, mglev, v);
163
164 Real rhTv = dotxy(rh,v);
165 if ( rhTv != Real(0.0) )
166 {
167 alpha = rho/rhTv;
168 }
169 else
170 {
171 ret = 2; break;
172 }
173 sxay(sol, sol, alpha, ph, nghost);
174 sxay(s, r, -alpha, v, nghost);
175
176 //Subtract mean from s
177// if (Lp.isBottomSingular()) mlmg->makeSolvable(amrlev, mglev, s);
178
179 rnorm = norm_inf(s);
180
181 if ( verbose > 2 && ParallelDescriptor::IOProcessor() )
182 {
183 amrex::Print() << "MLCGSolver_BiCGStab: Half Iter "
184 << std::setw(11) << iter
185 << " rel. err. "
186 << rnorm/(rnorm0) << '\n';
187 }
188
189 if ( rnorm < eps_rel*rnorm0 || rnorm < eps_abs ) break;
190
191 MultiFab::Copy(sh,s,0,0,ncomp,nghost);
192 Lp.apply(amrlev, mglev, t, sh, MLLinOp::BCMode::Homogeneous, MLLinOp::StateMode::Correction);
193 Lp.normalize(amrlev, mglev, t);
194 //
195 // This is a little funky. I want to elide one of the reductions
196 // in the following two dotxy()s. We do that by calculating the "local"
197 // values and then reducing the two local values at the same time.
198 //
199 Real tvals[2] = { dotxy(t,t,true), dotxy(t,s,true) };
200
201 BL_PROFILE_VAR("MLCGSolver::ParallelAllReduce", blp_par);
202 ParallelAllReduce::Sum(tvals,2,Lp.BottomCommunicator());
203 BL_PROFILE_VAR_STOP(blp_par);
204
205 if ( tvals[0] != Real(0.0) )
206 {
207 omega = tvals[1]/tvals[0];
208 }
209 else
210 {
211 ret = 3; break;
212 }
213 sxay(sol, sol, omega, sh, nghost);
214 sxay(r, s, -omega, t, nghost);
215
216// if (Lp.isBottomSingular()) mlmg->makeSolvable(amrlev, mglev, r);
217
218 rnorm = norm_inf(r);
219
220 if ( verbose > 2 )
221 {
222 amrex::Print() << "MLCGSolver_BiCGStab: Iteration "
223 << std::setw(11) << iter
224 << " rel. err. "
225 << rnorm/(rnorm0) << '\n';
226 }
227
228 if ( rnorm < eps_rel*rnorm0 || rnorm < eps_abs ) break;
229
230 if ( omega == 0 )
231 {
232 ret = 4; break;
233 }
234 rho_1 = rho;
235 }
236
237 if ( verbose > 0 )
238 {
239 amrex::Print() << "MLCGSolver_BiCGStab: Final: Iteration "
240 << std::setw(4) << iter
241 << " rel. err. "
242 << rnorm/(rnorm0) << '\n';
243 }
244
245 if ( ret == 0 && rnorm > eps_rel*rnorm0 && rnorm > eps_abs)
246 {
247 if ( verbose > 0 && ParallelDescriptor::IOProcessor() )
248 amrex::Warning("MLCGSolver_BiCGStab:: failed to converge!");
249 ret = 8;
250 }
251
252 if ( ( ret == 0 || ret == 8 ) && (rnorm < rnorm0) )
253 {
254 sol.plus(sorig, 0, ncomp, nghost);
255 }
256 else
257 {
258 sol.setVal(0);
259 sol.plus(sorig, 0, ncomp, nghost);
260 }
261
262 return ret;
263}

266MLCGSolver::solve_cg (MultiFab& sol,
267 const MultiFab& rhs,
268 Real eps_rel,
269 Real eps_abs)
270{
271 BL_PROFILE("MLCGSolver::cg");
272
273 const int ncomp = sol.nComp();
274
275 const BoxArray& ba = sol.boxArray();
276 const DistributionMapping& dm = sol.DistributionMap();
277 const auto& factory = sol.Factory();
278
279 MultiFab p(ba, dm, ncomp, sol.nGrow(), MFInfo(), factory);
280 p.setVal(0.0);
281
282 MultiFab sorig(ba, dm, ncomp, nghost, MFInfo(), factory);
283 MultiFab r (ba, dm, ncomp, nghost, MFInfo(), factory);
284 MultiFab z (ba, dm, ncomp, nghost, MFInfo(), factory);
285 MultiFab q (ba, dm, ncomp, nghost, MFInfo(), factory);
286
287 MultiFab::Copy(sorig,sol,0,0,ncomp,nghost);
288
289 Lp.correctionResidual(amrlev, mglev, r, sol, rhs, MLLinOp::BCMode::Homogeneous);
290
291 sol.setVal(0);
292
293 Real rnorm = norm_inf(r);
294 const Real rnorm0 = rnorm;
295
296 if ( verbose > 0 )
297 {
298 amrex::Print() << "MLCGSolver_CG: Initial error (error0) : " << rnorm0 << '\n';
299 }
300
301 Real rho_1 = 0;
302 int ret = 0;
303 iter = 1;
304
305 if ( rnorm0 == 0 || rnorm0 < eps_abs )
306 {
307 if ( verbose > 0 ) {
308 amrex::Print() << "MLCGSolver_CG: niter = 0,"
309 << ", rnorm = " << rnorm
310 << ", eps_abs = " << eps_abs << std::endl;
311 }
312 return ret;
313 }
314
315 for (; iter <= maxiter; ++iter)
316 {
317 MultiFab::Copy(z,r,0,0,ncomp,nghost);
318
319 Real rho = dotxy(z,r);
320
321 if ( rho == 0 )
322 {
323 ret = 1; break;
324 }
325 if (iter == 1)
326 {
327 MultiFab::Copy(p,z,0,0,ncomp,nghost);
328 }
329 else
330 {
331 Real beta = rho/rho_1;
332 sxay(p, z, beta, p, nghost);
333 }
334 Lp.apply(amrlev, mglev, q, p, MLLinOp::BCMode::Homogeneous, MLLinOp::StateMode::Correction);
335
336 Real alpha;
337 Real pw = dotxy(p,q);
338 if ( pw != Real(0.0))
339 {
340 alpha = rho/pw;
341 }
342 else
343 {
344 ret = 1; break;
345 }
346
347 if ( verbose > 2 )
348 {
349 amrex::Print() << "MLCGSolver_cg:"
350 << " iter " << iter
351 << " rho " << rho
352 << " alpha " << alpha << '\n';
353 }
354 sxay(sol, sol, alpha, p, nghost);
355 sxay( r, r,-alpha, q, nghost);
356 rnorm = norm_inf(r);
357
358 if ( verbose > 2 )
359 {
360 amrex::Print() << "MLCGSolver_cg: Iteration"
361 << std::setw(4) << iter
362 << " rel. err. "
363 << rnorm/(rnorm0) << '\n';
364 }
365
366 if ( rnorm < eps_rel*rnorm0 || rnorm < eps_abs ) break;
367
368 rho_1 = rho;
369 }
370
371 if ( verbose > 0 )
372 {
373 amrex::Print() << "MLCGSolver_cg: Final Iteration"
374 << std::setw(4) << iter
375 << " rel. err. "
376 << rnorm/(rnorm0) << '\n';
377 }
378
379 if ( ret == 0 && rnorm > eps_rel*rnorm0 && rnorm > eps_abs )
380 {
381 if ( verbose > 0 && ParallelDescriptor::IOProcessor() )
382 amrex::Warning("MLCGSolver_cg: failed to converge!");
383 ret = 8;
384 }
385
386 if ( ( ret == 0 || ret == 8 ) && (rnorm < rnorm0) )
387 {
388 sol.plus(sorig, 0, ncomp, nghost);
389 }
