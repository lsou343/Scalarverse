1033void
1034Amr::checkInput ()
1035{
1036 if (max_level < 0)
1037 amrex::Error("checkInput: max_level not set");
1038 //
1039 // Check that blocking_factor is a power of 2.
1040 //
1041 for (int i = 0; i <= max_level; i++)
1042 {
1043 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim)
1044 {
1045 int k = blocking_factor[i][idim];
1046 while ( k > 0 && (k%2 == 0) )
1047 k /= 2;
1048 if (k != 1)
1049 amrex::Error("Amr::checkInput: blocking_factor not power of 2");
1050 }
1051 }
1052 //
1053 // Check level dependent values.
1054 //
1055 for (int i = 0; i < max_level; i++)
1056 {
1057 if (MaxRefRatio(i) < 2 || MaxRefRatio(i) > 12)
1058 amrex::Error("Amr::checkInput: bad ref_ratios");
1059 }
1060 const Box& domain = Geom(0).Domain();
1061 if (!domain.ok())
1062 amrex::Error("level 0 domain bad or not set");
1063 //
1064 // Check that domain size is a multiple of blocking_factor[0].
1065 //
1066 for (int i = 0; i < AMREX_SPACEDIM; i++)
1067 {
1068 int len = domain.length(i);
1069 if (len%blocking_factor[0][i] != 0)
1070 amrex::Error("domain size not divisible by blocking_factor");
1071 }
1072 //
1073 // Check that max_grid_size is even.
1074 //
1075 for (int i = 0; i <= max_level; i++)
1076 {
1077 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
1078 if (max_grid_size[i][idim]%2 != 0) {
1079 amrex::Error("max_grid_size is not even");
1080 }
1081 }
1082 }
1083
1084 //
1085 // Check that max_grid_size is a multiple of blocking_factor at every level.
1086 //
1087 for (int i = 0; i <= max_level; i++)
1088 {
1089 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
1090 if (max_grid_size[i][idim]%blocking_factor[i][idim] != 0) {
1091 amrex::Error("max_grid_size not divisible by blocking_factor");
1092 }
1093 }
1094 }
1095
1096 if( ! Geom(0).ProbDomain().ok()) {
1097 amrex::Error("Amr::checkInput: bad physical problem size");
1098 }
1099
1100 if(verbose > 0) {
1101 amrex::Print() << "Successfully read inputs file ... " << '\n';
1102 }

1143void
1144Amr::readProbinFile (int& a_init)
1145{
1146 BL_PROFILE("Amr::readProbinFile()");
1147 //
1148 // Populate integer array with name of probin file.
1149 //
1150 int probin_file_length = probin_file.length();
1151
1152 Vector<int> probin_file_name(probin_file_length);
1153
1154 for (int i = 0; i < probin_file_length; i++)
1155 probin_file_name[i] = probin_file[i];
1156
1157 if (verbose > 0)
1158 amrex::Print() << "Starting to call amrex_probinit ... \n";
1159
1160 const int nAtOnce = probinit_natonce;
1161 const int MyProc = ParallelDescriptor::MyProc();
1162 const int NProcs = ParallelDescriptor::NProcs();
1163 const int NSets = (NProcs + (nAtOnce - 1)) / nAtOnce;
1164 const int MySet = MyProc/nAtOnce;
1165
1166 Real piStart = 0, piEnd = 0, piStartAll = amrex::second();
1167
1168 for (int iSet = 0; iSet < NSets; ++iSet)
1169 {
1170 if (MySet == iSet)
1171 {
1172 //
1173 // Call the pesky probin reader.
1174 //
1175 piStart = amrex::second();
1176
1177#ifdef AMREX_DIMENSION_AGNOSTIC
1178
1179 amrex_probinit(&a_init,
1180 probin_file_name.dataPtr(),
1181 &probin_file_length,
1182 AMREX_ZFILL(Geom(0).ProbLo()),
1183 AMREX_ZFILL(Geom(0).ProbHi()));
1184
1185#else
1186
1187 amrex_probinit(&a_init,
1188 probin_file_name.dataPtr(),
1189 &probin_file_length,
1190 Geom(0).ProbLo(),
1191 Geom(0).ProbHi());
1192#endif
1193
1194 piEnd = amrex::second();
1195 const int iBuff = 0;
1196 const int wakeUpPID = (MyProc + nAtOnce);
1197 const int tag = (MyProc % nAtOnce);
1198 if (wakeUpPID < NProcs)
1199 ParallelDescriptor::Send(&iBuff, 1, wakeUpPID, tag);
1200 }
1201 if (MySet == (iSet + 1))
1202 {
1203 //
1204 // Next set waits.
1205 //
1206 int iBuff;
1207 int waitForPID = (MyProc - nAtOnce);
1208 int tag = (MyProc % nAtOnce);
1209 ParallelDescriptor::Recv(&iBuff, 1, waitForPID, tag);
1210 }
1211 }
1212
1213 if (verbose > 1)
1214 {
1215 const int IOProc = ParallelDescriptor::IOProcessorNumber();
1216 Real piTotal = piEnd - piStart;
1217 Real piTotalAll = amrex::second() - piStartAll;
1218
1219 ParallelDescriptor::ReduceRealMax(piTotal, IOProc);
1220 ParallelDescriptor::ReduceRealMax(piTotalAll, IOProc);
1221
1222 amrex::Print() << "amrex_probinit max time = " << piTotal << '\n'
1223 << "amrex_probinit total time = " << piTotalAll << '\n';
1224 }
1225
1226 if (verbose > 0)
1227 amrex::Print() << "Successfully run amrex_probinit\n";
1228}

1351void
1352Amr::restart (const std::string& filename)
1353{

1355 BL_PROFILE("Amr::restart()");
1356
1357 which_level_being_advanced = -1;
1358
1359 Real dRestartTime0 = amrex::second();
1360
1361 VisMF::SetMFFileInStreams(mffile_nstreams);
1362
1363 if (verbose > 0) {
1364 amrex::Print() << "restarting calculation from file: " << filename << "\n";
1365 }
1366
1367 if (record_run_info && ParallelDescriptor::IOProcessor()) {
1368 runlog << "RESTART from file = " << filename << '\n';
1369 }
1370 //
1371 // Init problem dependent data.
1372 //
1373 int linit = false;
1374
1375 if (!probin_file.empty()) {
1376 readProbinFile(linit);
1377 }
1378
1379 //
1380 // Start calculation from given restart file.
1381 //
1382 if (record_run_info && ParallelDescriptor::IOProcessor()) {
1383 runlog << "RESTART from file = " << filename << '\n';
1384 }
1385
1386 // ---- preread and broadcast all FabArray headers if this file exists
1387 std::map<std::string, Vector<char> > faHeaderMap;
1388 if(prereadFAHeaders) {
1389 // ---- broadcast the file with the names of the fabarray headers
1390 std::string faHeaderFilesName(filename + "/FabArrayHeaders.txt");
1391 Vector<char> faHeaderFileChars;
1392 bool bExitOnError(false); // ---- dont exit if this file does not exist
1393 ParallelDescriptor::ReadAndBcastFile(faHeaderFilesName, faHeaderFileChars,
1394 bExitOnError);
1395 if(faHeaderFileChars.size() > 0) { // ---- headers were read
1396 std::string faFileCharPtrString(faHeaderFileChars.dataPtr());
1397 std::istringstream fais(faFileCharPtrString, std::istringstream::in);
1398 while ( ! fais.eof()) { // ---- read and broadcast each header
1399 std::string faHeaderName;
1400 fais >> faHeaderName;
1401 if( ! fais.eof()) {
1402 std::string faHeaderFullName(filename + '/' + faHeaderName + "_H");
1403 Vector<char> &tempCharArray = faHeaderMap[faHeaderFullName];
1404 ParallelDescriptor::ReadAndBcastFile(faHeaderFullName, tempCharArray);
1405 if(verbose > 2) {
1406 amrex::Print()
1407 << ":::: faHeaderName faHeaderFullName tempCharArray.size() = " << faHeaderName
1408 << " " << faHeaderFullName << " " << tempCharArray.size() << "\n";
1409 }
1410 }
1411 }
1412 StateData::SetFAHeaderMapPtr(&faHeaderMap);
1413 }
1414 }
1415
1416 //
1417 // Open the checkpoint header file for reading.
1418 //
1419 std::string File(filename + "/Header");
1420
1421 VisMF::IO_Buffer io_buffer(VisMF::GetIOBufferSize());
1422
1423 Vector<char> fileCharPtr;
1424 ParallelDescriptor::ReadAndBcastFile(File, fileCharPtr);
1425 std::string fileCharPtrString(fileCharPtr.dataPtr());
1426 std::istringstream is(fileCharPtrString, std::istringstream::in);
1427 //
1428 // Read global data.
1429 //
1430 // Attempt to differentiate between old and new CheckPointFiles.
1431 //
1432 int spdim;
1433 bool new_checkpoint_format = false;
1434 std::string first_line;
1435
1436 std::getline(is,first_line);
1437
1438 if (first_line == CheckPointVersion)
1439 {
1440 new_checkpoint_format = true;
1441 is >> spdim;
1442 }
1443 else
1444 {
1445 spdim = atoi(first_line.c_str());
1446 }
1447
1448 if (spdim != AMREX_SPACEDIM)
1449 {
1450 amrex::ErrorStream() << "Amr::restart(): bad spacedim = " << spdim << '\n';
1451 amrex::Abort();
1452 }
1453
1454 is >> cumtime;
1455 int mx_lev;
1456 is >> mx_lev;
1457 is >> finest_level;
1458
1459 Vector<Box> inputs_domain(max_level+1);
1460 for (int lev = 0; lev <= max_level; ++lev)
1461 {
1462 Box bx(Geom(lev).Domain().smallEnd(),Geom(lev).Domain().bigEnd());
1463 inputs_domain[lev] = bx;
1464 }
1465
1466 if (max_level >= mx_lev) {
1467
1468 for (int i(0); i <= mx_lev; ++i) { is >> Geom(i); }
1469 for (int i(0); i < mx_lev; ++i) { is >> ref_ratio[i]; }
1470 for (int i(0); i <= mx_lev; ++i) { is >> dt_level[i]; }
1471
1472 if (new_checkpoint_format)
1473 {
1474 for (int i(0); i <= mx_lev; ++i) { is >> dt_min[i]; }
1475 }
1476 else
1477 {
1478 for (int i(0); i <= mx_lev; ++i) { dt_min[i] = dt_level[i]; }
1479 }
1480
1481 Vector<int> n_cycle_in;
1482 n_cycle_in.resize(mx_lev+1);
1483 for (int i(0); i <= mx_lev; ++i) { is >> n_cycle_in[i]; }
1484 bool any_changed = false;
1485
1486 for (int i(0); i <= mx_lev; ++i) {
1487 if (n_cycle[i] != n_cycle_in[i]) {
1488 any_changed = true;
1489 if (verbose > 0) {
1490 amrex::Print() << "Warning: n_cycle has changed at level " << i <<
1491 " from " << n_cycle_in[i] << " to " << n_cycle[i] << "\n";
1492 }
1493 }
1494 }
1495
1496 // If we change n_cycle then force a full regrid from level 0 up
1497 if (max_level > 0 && any_changed)
1498 {
1499 level_count[0] = regrid_int[0];
1500 if (verbose > 0) {
1501 amrex::Print() << "Warning: This forces a full regrid \n";
1502 }
1503 }
1504
1505
1506 for (int i(0); i <= mx_lev; ++i) { is >> level_steps[i]; }
1507 for (int i(0); i <= mx_lev; ++i) { is >> level_count[i]; }
1508
1509 //
1510 // Set bndry conditions.
1511 //
1512 if (max_level > mx_lev)
1513 {
1514 for (int i(mx_lev + 1); i <= max_level; ++i)
1515 {
1516 dt_level[i] = dt_level[i-1]/n_cycle[i];
1517 level_steps[i] = n_cycle[i]*level_steps[i-1];
1518 level_count[i] = 0;
1519 }
1520
1521 // This is just an error check
1522 if ( ! sub_cycle)
1523 {
1524 for (int i(1); i <= finest_level; ++i)
1525 {
1526 if (dt_level[i] != dt_level[i-1]) {
1527 amrex::Error("restart: must have same dt at all levels if not subcycling");
1528 }
1529 }
1530 }

1876Amr::timeStep (int level,
1877 Real time,
1878 int iteration,
1879 int niter,
1880 Real stop_time)
1881{
1882 BL_PROFILE("Amr::timeStep()");
1883 BL_COMM_PROFILE_NAMETAG("Amr::timeStep TOP");
1884
1885 // This is used so that the AmrLevel functions can know which level is being advanced
1886 // when regridding is called with possible lbase > level.
1887 which_level_being_advanced = level;
1888
1889
1890 // Update so that by default, we don't force a post-step regrid.
1891 amr_level[level]->setPostStepRegrid(0);
1892
1893 //
1894 // Allow regridding of level 0 calculation on restart.
1895 //
1896 if (max_level == 0 && regrid_on_restart)
1897 {
1898 regrid_level_0_on_restart();
1899 }
1900 else
1901 {
1902 int lev_top = std::min(finest_level, max_level-1);
1903 for (int i(level); i <= lev_top; ++i)
1904 {
1905 const int old_finest = finest_level;
1906
1907 if (okToRegrid(i))
1908 {
1909 regrid(i,time);
1910
1911 //
1912 // Compute new dt after regrid if at level 0 and compute_new_dt_on_regrid.
1913 //
1914 if ( compute_new_dt_on_regrid && (i == 0) )
1915 {
1916 int post_regrid_flag = 1;
1917 amr_level[0]->computeNewDt(finest_level,
1918 sub_cycle,
1919 n_cycle,
1920 ref_ratio,
1921 dt_min,
1922 dt_level,
1923 stop_time,
1924 post_regrid_flag);
1925 }
1926
1927 for (int k(i); k <= finest_level; ++k) {
1928 level_count[k] = 0;
1929 }
1930
1931 if (old_finest < finest_level)
1932 {
1933 //
1934 // The new levels will not have valid time steps
1935 // and iteration counts.
1936 //
1937 for (int k(old_finest + 1); k <= finest_level; ++k)
1938 {
1939 dt_level[k] = dt_level[k-1]/n_cycle[k];
1940 }
1941 }
1942 }
1943 if (old_finest > finest_level) {
1944 lev_top = std::min(finest_level, max_level - 1);
1945 }
1946 }
1947
1948 if (max_level == 0 && loadbalance_level0_int > 0 && loadbalance_with_workestimates)
1949 {
1950 if (level_steps[0] == 1 || level_count[0] >= loadbalance_level0_int) {
1951 LoadBalanceLevel0(time);
1952 level_count[0] = 0;
1953 }
1954 }
1955 }
1956 //
1957 // Check to see if should write plotfile.
1958 // This routine is here so it is done after the restart regrid.
1959 //
1960 if (plotfile_on_restart && ! (restart_chkfile.empty()) )
1961 {
1962 plotfile_on_restart = 0;
1963 writePlotFile();
1964 }
1965 //
1966 // Advance grids at this level.
1967 //
1968 if (verbose > 0)
1969 {
1970 amrex::Print() << "[Level " << level << " step " << level_steps[level]+1 << "] "
1971 << "ADVANCE with dt = " << dt_level[level] << "\n";
1972 }
1973
1974 Real dt_new = amr_level[level]->advance(time,dt_level[level],iteration,niter);
1975 BL_PROFILE_REGION_STOP("amr_level.advance");
1976
1977 dt_min[level] = iteration == 1 ? dt_new : std::min(dt_min[level],dt_new);
1978
1979 level_steps[level]++;
1980 level_count[level]++;
1981
1982 if (verbose > 0)
1983 {
1984 amrex::Print() << "[Level " << level << " step " << level_steps[level] << "] "
1985 << "Advanced " << amr_level[level]->countCells() << " cells\n";
1986 }
1987
1988 // If the level signified that it wants a regrid after the advance has
1989 // occurred, do that now.
1990 if (amr_level[level]->postStepRegrid()) {
1991
1992 int old_finest = finest_level;
1993
1994 regrid(level, time);
1995
1996 if (old_finest < finest_level)
1997 {
1998 //
1999 // The new levels will not have valid time steps.
2000 //
2001 for (int k = old_finest + 1; k <= finest_level; ++k)
2002 {
2003 dt_level[k] = dt_level[k-1] / n_cycle[k];
2004 }
2005 }
2006 }
2007
2008 //
2009 // Advance grids at higher level.
2010 //
2011 if (level < finest_level)
2012 {
2013 const int lev_fine = level+1;
2014
2015 if (sub_cycle)
2016 {
2017 const int ncycle = n_cycle[lev_fine];
2018
2019 BL_COMM_PROFILE_NAMETAG("Amr::timeStep timeStep subcycle");
2020 for (int i = 1; i <= ncycle; i++)
2021 timeStep(lev_fine,time+(i-1)*dt_level[lev_fine],i,ncycle,stop_time);
2022 }
2023 else
2024 {
2025 BL_COMM_PROFILE_NAMETAG("Amr::timeStep timeStep nosubcycle");
2026 timeStep(lev_fine,time,1,1,stop_time);
2027 }
2028 }
2029
2030 amr_level[level]->post_timestep(iteration);
2031
2032 // Set this back to negative so we know whether we are in fact in this routine
2033 which_level_being_advanced = -1;
2034}
