721Box::refine (const IntVect& ref_ratio) noexcept
722{
723 if (ref_ratio != 1) {
724 IntVect shft(1);
725 shft -= btype.ixType();
726 smallend *= ref_ratio;
727 bigend += shft;
728 bigend *= ref_ratio;
729 bigend -= shft;

736Box&
737Box::coarsen (const IntVect& ref_ratio) noexcept
738{
739 if (ref_ratio != 1)
740 {
741 smallend.coarsen(ref_ratio);
742
743 if (btype.any())
744 {
745 IntVect off(0);
746 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
747 {
748 if (btype[dir])
749 if (bigend[dir]%ref_ratio[dir])
750 off.setVal(dir,1);
751 }
752 bigend.coarsen(ref_ratio);
753 bigend += off;
754 }
755 else
756 {
757 bigend.coarsen(ref_ratio);
758 }
759 }
760
761 return *this;

884 auto lo = this->loVect3d();
885 Long k = offset / (len[0]*len[1]);
886 Long j = (offset - k*(len[0]*len[1])) / len[0];
887 Long i = (offset - k*(len[0]*len[1])) - j*len[0];
888 return {{static_cast<int>(i+lo[0]),
889 static_cast<int>(j+lo[1]),
890 static_cast<int>(k+lo[2])}};
891}

948// If CELL: chop_pnt included in high end.
949// If NODE: chop_pnt included in both Boxes.
950//
951
952AMREX_GPU_HOST_DEVICE
953inline
954Box
955Box::chop (int dir, int chop_pnt) noexcept
956{
957 //
958 // Define new high end Box including chop_pnt.
959 //
960 IntVect sm(smallend);
961 IntVect bg(bigend);
962 sm.setVal(dir,chop_pnt);
963 if (btype[dir])
964 {
965 //
966 // NODE centered Box.
967 //
968 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt < bigend[dir]);
969 //
970 // Shrink original Box to just contain chop_pnt.
971 //
972 bigend.setVal(dir,chop_pnt);
973 }
974 else
975 {
976 //
977 // CELL centered Box.
978 //
979 BL_ASSERT(chop_pnt > smallend[dir] && chop_pnt <= bigend[dir]);
980 //
981 // Shrink origional Box to one below chop_pnt.
982 //
983 bigend.setVal(dir,chop_pnt-1);
984 }
985 return Box(sm,bg,btype);
986}

989inline
990Box&
991Box::shiftHalf (int dir, int nzones) noexcept
992{
993 const int nbit = (nzones<0 ? -nzones : nzones)%2;
994 int nshift = nzones/2;
995 //
996 // Toggle btyp bit if nzones is odd.
997 //
998 const unsigned int bit_dir = btype[dir];
999 if (nbit)
1000 btype.flip(dir);
1001 if (nzones < 0)
1002 nshift -= (bit_dir ? nbit : 0);
1003 else
1004 nshift += (bit_dir ? 0 : nbit);
1005 smallend.shift(dir,nshift);
1006 bigend.shift(dir,nshift);
1007 return *this;
1008}
