60 apnorminv = 1.d0/apnorm
61 anrmx = -dapx * apnorminv ! unit vector pointing toward the wall
62 anrmy = -dapy * apnorminv
63 anrmz = -dapz * apnorminv

68 if (abs(anrmx).ge.abs(anrmy) .and. abs(anrmx).ge.abs(anrmz)) then
69 ! y-z plane: x = const
70 ! the equation for the line: x = bct(1) - d*anrmx
71 ! y = bct(2) - d*anrmy
72 ! z = bct(3) - d*anrmz

116 cz0 = 1.d0-zit*zit
117 czp = 0.5d0*zit*(zit+1.d0)

212 !
213 ! compute derivatives on the wall given that velocity is zero on the wall.
214 !
215 ddinv = 1.d0/(d1*d2*(d2-d1))
216 dudn = -ddinv*(d2*d2*u1-d1*d1*u2) ! note that the normal vector points toward the wall
217 dvdn = -ddinv*(d2*d2*v1-d1*d1*v2)
218 dwdn = -ddinv*(d2*d2*w1-d1*d1*w2)
219 !
220 ! transform them to d/dx, d/dy and d/dz given transverse derivatives are zero
221 dudx = dudn * anrmx
222 dudy = dudn * anrmy
223 dudz = dudn * anrmz
224 !
225 dvdx = dvdn * anrmx
226 dvdy = dvdn * anrmy
227 dvdz = dvdn * anrmz
228 !
229 dwdx = dwdn * anrmx
230 dwdy = dwdn * anrmy
231 dwdz = dwdn * anrmz
232
233 divu = dudx+dvdy+dwdz
234 tautmp = (xi(i,j,k)-twothirds*mu(i,j,k))*divu
235 tauxx = mu(i,j,k)*2.d0*dudx + tautmp
236 tauyy = mu(i,j,k)*2.d0*dvdy + tautmp
237 tauzz = mu(i,j,k)*2.d0*dwdz + tautmp
238 tauxy = mu(i,j,k)*(dudy+dvdx)
239 tauxz = mu(i,j,k)*(dudz+dwdx)
240 tauyz = mu(i,j,k)*(dwdy+dvdz)
241
242 ! dx == dy == dz
243 divw(2) = dxinv(1) * (dapx*tauxx + dapy*tauxy + dapz*tauxz)
244 divw(3) = dxinv(1) * (dapx*tauxy + dapy*tauyy + dapz*tauyz)
245 divw(4) = dxinv(1) * (dapx*tauxz + dapy*tauyz + dapz*tauzz)
246
247 end subroutine compute_diff_wallflux
