331Comm_nowait (FabArray<FAB>& mf, int scomp, int ncomp, FabArrayBase::CommMetaData const& cmd,
332 DTOS dtos)
333{
334#ifdef AMREX_USE_MPI
335 if (ParallelContext::NProcsSub() == 1)
336#endif
337 {
338 int N_locs = (*cmd.m_LocTags).size();
339 if (N_locs == 0) return CommHandler{};
340#ifdef AMREX_USE_GPU
341 if (Gpu::inLaunchRegion()) {
342 local_copy_gpu(mf, scomp, ncomp, cmd, dtos);
343 } else
344#endif
345 {
346 local_copy_cpu(mf, scomp, ncomp, cmd, dtos);
347 }
348 return CommHandler{};
349 }
350
351#ifdef AMREX_USE_MPI
352 //
353 // Do this before prematurely exiting if running in parallel.
354 // Otherwise sequence numbers will not match across MPI processes.
355 //
356 int SeqNum = ParallelDescriptor::SeqNum();
357
358 const int N_locs = cmd.m_LocTags->size();
359 const int N_rcvs = cmd.m_RcvTags->size();
360 const int N_snds = cmd.m_SndTags->size();
361
362 if (N_locs == 0 && N_rcvs == 0 && N_snds == 0) {
363 // No work to do.
364 return CommHandler{};
365 }
366
367 CommHandler handler{};
368 handler.mpi_tag = SeqNum;
369
370 if (N_rcvs > 0) {
371 mf.PostRcvs(*cmd.m_RcvTags, handler.the_recv_data, handler.recv_data, handler.recv_size,
372 handler.recv_from, handler.recv_reqs, ncomp, SeqNum);
373 }
374
375 if (N_snds > 0)
376 {
377 Vector<std::size_t> send_size;
378 Vector<int> send_rank;
379 Vector<const FabArrayBase::CopyComTagsContainer*> send_cctc;
380 mf.PrepareSendBuffers(*cmd.m_SndTags, handler.the_send_data, handler.send_data,
381 send_size, send_rank, handler.send_reqs, send_cctc, ncomp);
382
383#ifdef AMREX_USE_GPU
384 if (Gpu::inLaunchRegion()) {
385 FabArray<FAB>::pack_send_buffer_gpu(mf, scomp, ncomp, handler.send_data,
386 send_size, send_cctc);
387 } else
388#endif
389 {
390 FabArray<FAB>::pack_send_buffer_cpu(mf, scomp, ncomp, handler.send_data,
391 send_size, send_cctc);
392 }
393
394 FabArray<FAB>::PostSnds(handler.send_data, send_size, send_rank, handler.send_reqs, SeqNum);
395 }
396
397 if (N_locs > 0)
398 {
399#ifdef AMREX_USE_GPU
400 if (Gpu::inLaunchRegion()) {
401 local_copy_gpu(mf, scomp, ncomp, cmd, dtos);
402 } else
403#endif
404 {
405 local_copy_cpu(mf, scomp, ncomp, cmd, dtos);
406 }
407 }
408
409 return handler;
410#endif
