580TagBoxArray::collate (Vector<IntVect>& TheGlobalCollateSpace) const
581{
582 BL_PROFILE("TagBoxArray::collate()");
583
584 Vector<IntVect> TheLocalCollateSpace;
585#ifdef AMREX_USE_GPU
586 if (Gpu::inLaunchRegion()) {
587 local_collate_gpu(TheLocalCollateSpace);
588 } else
589#endif
590 {
591 local_collate_cpu(TheLocalCollateSpace);
592 }
593
594 Long count = TheLocalCollateSpace.size();
595
596 //
597 // The total number of tags system wide that must be collated.
598 //
599 Long numtags = count;
600 ParallelDescriptor::ReduceLongSum(numtags);
601
602 if (numtags == 0) {
603 TheGlobalCollateSpace.clear();
604 return;
605 } else if (numtags > static_cast<Long>(std::numeric_limits<int>::max())) {
606 // xxxxx todo
607 amrex::Abort("TagBoxArray::collate: Too many tags. Using a larger blocking factor might help. Please file an issue on github");
608 }
609
610#ifdef BL_USE_MPI
611 //
612 // On I/O proc. this holds all tags after they've been gather'd.
613 // On other procs. non-mempty signals size is not zero.
614 //
615 if (ParallelDescriptor::IOProcessor()) {
616 TheGlobalCollateSpace.resize(numtags);
617 } else {
618 TheGlobalCollateSpace.resize(1);
619 }
620
621 //
622 // Tell root CPU how many tags each CPU will be sending.
623 //
624 const int IOProcNumber = ParallelDescriptor::IOProcessorNumber();
625 const std::vector<int>& countvec = ParallelDescriptor::Gather(static_cast<int>(count),
626 IOProcNumber);
627 std::vector<int> offset(countvec.size(),0);
628 if (ParallelDescriptor::IOProcessor()) {
629 for (int i = 1, N = offset.size(); i < N; i++) {
630 offset[i] = offset[i-1] + countvec[i-1];
631 }
632 }
633 //
634 // Gather all the tags to IOProcNumber into TheGlobalCollateSpace.
635 //
636 const IntVect* psend = (count > 0) ? TheLocalCollateSpace.data() : nullptr;
637 IntVect* precv = TheGlobalCollateSpace.data();
638 ParallelDescriptor::Gatherv(psend, count, precv, countvec, offset, IOProcNumber);
639
640#else
641 TheGlobalCollateSpace = std::move(TheLocalCollateSpace);
642#endif

693void
694TagBoxArray::coarsen (const IntVect & ratio)
695{
696 // If team is used, all team workers need to go through all the fabs,
697 // including ones they don't own.
698 int teamsize = ParallelDescriptor::TeamSize();
699 unsigned char flags = (teamsize == 1) ? 0 : MFIter::AllBoxes;
700
701 IntVect new_n_grow;
702 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
703 new_n_grow[idim] = (n_grow[idim]+ratio[idim]-1)/ratio[idim];
704 }
705
706#if defined(_OPENMP)
707#pragma omp parallel if (teamsize == 1 && Gpu::notInLaunchRegion())
708#endif
709 for (MFIter mfi(*this,flags); mfi.isValid(); ++mfi)
710 {
711 Box const& cbox = amrex::grow(amrex::coarsen(mfi.validbox(),ratio),new_n_grow);
712 this->fabPtr(mfi)->coarsen(ratio,cbox);
713 }
714
715 boxarray.coarsen(ratio);
716 n_grow = new_n_grow;
717}

719bool
720TagBoxArray::hasTags (Box const& a_bx) const
721{
722 bool has_tags = false;
723#ifdef AMREX_USE_GPU
724 if (Gpu::inLaunchRegion()) {
725 ReduceOps<ReduceOpLogicalOr> reduce_op;
726 ReduceData<int> reduce_data(reduce_op);
727 using ReduceTuple = typename decltype(reduce_data)::Type;
728
729 for (MFIter mfi(*this); mfi.isValid(); ++mfi)
730 {
731 Box const& b = a_bx & mfi.fabbox();
732 if (b.ok()) {
733 const auto& arr = this->const_array(mfi);
734 reduce_op.eval(b, reduce_data,
735 [=] AMREX_GPU_DEVICE (int i, int j, int k) -> ReduceTuple
736 {
737 int tr = arr(i,j,k) != TagBox::CLEAR;
738 return {tr};
739 });
740 }
741 }
742
743 ReduceTuple hv = reduce_data.value();
744 has_tags = static_cast<bool>(amrex::get<0>(hv));
745 } else
746#endif
747 {
748#ifdef _OPENMP
749#pragma omp parallel reduction(||:has_tags)
750#endif
751 for (MFIter mfi(*this); mfi.isValid(); ++mfi)
752 {
753 Box const& b = a_bx & mfi.fabbox();
754 if (b.ok()) {
755 Array4<char const> const& arr = this->const_array(mfi);
756 AMREX_LOOP_3D(b, i, j, k,
757 {
758 has_tags = has_tags || (arr(i,j,k) != TagBox::CLEAR);
759 });
760 }
761 }
762 }
763
764 ParallelAllReduce::Or(has_tags, ParallelContext::CommunicatorSub());
765 return has_tags;
766}
