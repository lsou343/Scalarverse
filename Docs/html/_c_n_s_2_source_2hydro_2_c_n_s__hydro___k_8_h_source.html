12cns_ctoprim (int i, int j, int k,
13 amrex::Array4<amrex::Real const> const& u,
14 amrex::Array4<amrex::Real> const& q,
15 Parm const& parm) noexcept

17 using amrex::Real;
18
19 Real rho = amrex::max(u(i,j,k,URHO),parm.smallr);
20 Real rhoinv = Real(1.0)/rho;
21 Real ux = u(i,j,k,UMX)*rhoinv;
22 Real uy = u(i,j,k,UMY)*rhoinv;
23 Real uz = u(i,j,k,UMZ)*rhoinv;
24 Real kineng = Real(0.5)*rho*(ux*ux+uy*uy+uz*uz);
25 Real ei = u(i,j,k,UEDEN) - kineng;
26 if (ei <= Real(0.0)) ei = u(i,j,k,UEINT);
27 Real p = amrex::max((parm.eos_gamma-Real(1.0))*ei,parm.smallp);
28 ei *= rhoinv;
29
30 q(i,j,k,QRHO) = rho;
31 q(i,j,k,QU) = ux;
32 q(i,j,k,QV) = uy;
33 q(i,j,k,QW) = uz;
34 q(i,j,k,QEINT) = ei;
35 q(i,j,k,QPRES) = p;
36 q(i,j,k,QCS) = std::sqrt(parm.eos_gamma*p*rhoinv);
37 q(i,j,k,QTEMP) = Real(0.0);

58amrex::Real limiter (amrex::Real dlft, amrex::Real drgt) noexcept
59{
60 using amrex::Real;
61
62 Real dcen = Real(0.5)*(dlft+drgt);
63 Real dsgn = amrex::Math::copysign(Real(1.0), dcen);
64 Real slop = Real(2.0) * amrex::min(amrex::Math::abs(dlft),amrex::Math::abs(drgt));
65 Real dlim = (dlft*drgt >= Real(0.0)) ? slop : Real(0.0);
66 return dsgn * amrex::min(dlim,amrex::Math::abs(dcen));
67}

185inline
186void
187riemann (const amrex::Real gamma, const amrex::Real smallp, const amrex::Real smallr,
188 const amrex::Real rl, const amrex::Real ul, const amrex::Real pl,
189 const amrex::Real ut1l, const amrex::Real ut2l,
190 const amrex::Real rr, const amrex::Real ur, const amrex::Real pr,
191 const amrex::Real ut1r, const amrex::Real ut2r,
192 amrex::Real& flxrho, amrex::Real& flxu, amrex::Real& flxut,
193 amrex::Real& flxutt, amrex::Real& flxe) noexcept
194{
195 using amrex::Real;
196
197 constexpr Real weakwv = Real(1.e-3);
198 constexpr Real small = Real(1.e-6);
199
200 Real clsql = gamma*pl*rl;
201 Real clsqr = gamma*pr*rr;
202 Real wl = std::sqrt(clsql);
203 Real wr = std::sqrt(clsqr);
204 Real cleft = wl/rl;
205 Real cright = wr/rr;
206 Real ccsmall = small*(cleft+cright);
207
208 Real pstar = (wl*pr + wr*pl - wr*wl*(ur-ul))/(wl+wr);
209 pstar = amrex::max(pstar,smallp);
210 Real pstnm1 = pstar;
211
212 Real wlsq = (Real(0.5)*(gamma-Real(1.))*(pstar+pl)+pstar)*rl;
213 Real wrsq = (Real(0.5)*(gamma-Real(1.))*(pstar+pr)+pstar)*rr;
214
215 wl = std::sqrt(wlsq);
216 wr = std::sqrt(wrsq);
217 Real ustarp = ul - (pstar-pl)/wl;
218 Real ustarm = ur + (pstar-pr)/wr;
219
220 pstar = (wl*pr + wr*pl - wr*wl*(ur-ul))/(wl+wr);
221 pstar = amrex::max(pstar,smallp);
222
223 Real ustar;
224 for (int iter = 0; iter < 3; ++iter)
225 {
226 wlsq = (Real(0.5)*(gamma-Real(1.))*(pstar+pl)+pstar)*rl;
227 wrsq = (Real(0.5)*(gamma-Real(1.))*(pstar+pr)+pstar)*rr;
228
229 wl = Real(1.)/std::sqrt(wlsq);
230 wr = Real(1.)/std::sqrt(wrsq);
231
232 Real ustnm1 = ustarm;
233 Real ustnp1 = ustarp;
234
235 ustarm = ur - (pr - pstar)*wr;
236 ustarp = ul + (pl - pstar)*wl;
237
238 Real dpditer = amrex::Math::abs(pstnm1-pstar);
239 Real zp = amrex::Math::abs(ustarp-ustnp1);
240 if (zp-weakwv*cleft < Real(0.0) ) {
241 zp = dpditer*wl;
242 }
243 Real zm = amrex::Math::abs(ustarm-ustnm1);
244 if (zm-weakwv*cright < Real(0.0) ) {
245 zm = dpditer*wr;
246 }
247
248 Real zz = zp+zm;
249 Real denom = dpditer/ amrex::max(zz,ccsmall);
250 pstnm1 = pstar;
251 pstar = pstar - denom*(ustarm-ustarp);
252 pstar = amrex::max(pstar,smallp);
253 ustar = Real(0.5)*(ustarm+ustarp);
254 }
255
256 Real ro, uo, po, sgnm, utrans1, utrans2;
257 if (ustar > Real(0.)) {
258 ro = rl;
259 uo = ul;
260 po = pl;
261 sgnm = Real(1.);
262 utrans1 = ut1l;
263 utrans2 = ut2l;
264 } else if (ustar < Real(0.)) {
265 ro = rr;
266 uo = ur;
267 po = pr;
268 sgnm = Real(-1.);
269 utrans1 = ut1r;
270 utrans2 = ut2r;
271 } else {
272 uo = Real(0.5)*(ur+ul);
273 po = Real(0.5)*(pr+pl);
274 ro = Real(2.)*(rl*rr)/(rl+rr);
275 sgnm = Real(1.);
276 utrans1 = Real(0.5)*(ut1l+ut1r);
277 utrans2 = Real(0.5)*(ut2l+ut2r);
278 }
279 Real wosq = (Real(0.5)*(gamma-Real(1.))*(pstar+po)+pstar)*ro;
280 Real co = std::sqrt(gamma * po / ro);
281 Real wo = std::sqrt(wosq);
282 Real dpjmp = pstar-po;
283 Real rstar = ro/(Real(1.)-ro*dpjmp/wosq);
284 Real cstar = std::sqrt(gamma * pstar / rstar);
285 Real spout = co-sgnm*uo;
286 Real spin = cstar - sgnm*uo;
287 if(pstar >= po) {
288 spin = wo/ro-sgnm*uo;
289 spout = spin;
290 }
291 Real ss = amrex::max(spout-spin, spout+spin);
292 Real frac = Real(0.5)*(Real(1.)+(spin+spout)/amrex::max(ss,ccsmall));
293
294 Real rgdnv, ugdnv, pgdnv;
295 if (spout < Real(0.)) {
296 rgdnv = ro;
297 ugdnv = uo;
298 pgdnv = po;
299 } else if(spin >= Real(0.)) {
300 rgdnv = rstar;
301 ugdnv = ustar;
302 pgdnv = pstar;
303 } else {
304 rgdnv = frac*rstar + (Real(1.) - frac)* ro;
305 ugdnv = frac*ustar + (Real(1.) - frac)* uo;
306 pgdnv = frac*pstar + (Real(1.) - frac)* po;
307 }
308
309 flxrho = rgdnv*ugdnv;
310 flxu = rgdnv*ugdnv*ugdnv+pgdnv;
311 flxut = rgdnv*ugdnv*utrans1;
312 flxutt = rgdnv*ugdnv*utrans2;
313 flxe = ugdnv*(Real(0.5)*rgdnv*(ugdnv*ugdnv+utrans1*utrans1+utrans2*utrans2) + pgdnv/(gamma -Real(1.)) + pgdnv);
314}
