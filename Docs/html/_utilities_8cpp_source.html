118// -----------------------------------------------------------
119// Search for the closest index in an array to a given value
120// using the bisection technique.
121// INPUTS/OUTPUTS:
122// xtable(0:n-1) => array to search in (ascending order)
123// n => array size
124// x => x location
125// idxlo <=> output st. xtable(idxlo) <= x < xtable(idxlo+1)
126// -----------------------------------------------------------
127AMREX_GPU_HOST_DEVICE
128void
129locate(const amrex::Real* xtable, const int n, amrex::Real& x, int& idxlo)
130{
131 // If x is out of bounds, return boundary index
132 if (x >= xtable[n - 1]) {
133 idxlo = n - 1;
134 return;
135 } else if (x <= xtable[0]) {
136 idxlo = 0;
137 return;
138 }
139
140 // Do the bisection
141 idxlo = 0;
142 int idxhi = n - 1;
143 bool notdone = true;
144 while (notdone) {
145 if (idxhi - idxlo <= 1) {
146 notdone = false;
147 } else {
148 const int idxmid = (idxhi + idxlo) / 2;
149 if (x >= xtable[idxmid]) {
150 idxlo = idxmid;
151 } else {
152 idxhi = idxmid;
153 }
154 }
155 }
156}

161void limit_hydro_fluxes_on_small_dens(const int i,
162 const int j,
163 const int k,
164 int idir,
165 Array4<Real const> const& u,
166 Array4<Real const> const& q,
167 Array4<Real> const& flux,
168 Real small_dens, Real lcfl,
169 Real dx_dir, Real dt)
170{
171
172 // The following algorithm comes from Hu, Adams, and Shu (2013), JCP, 242, 169,
173 // "Positivity-preserving method for high-order conservative schemes solving
174 // compressible Euler equations." It has been modified to enforce not only positivity
175 // but also the stronger requirement that rho > small_dens. We do not limit on pressure
176 // (or, similarly, internal energy) because those cases are easily fixed by calls to
177 // reset_internal_energy that enforce a thermodynamic floor. The density limiter, by
178 // contrast, is very important because calls to enforce_minimum_density can yield
179 // hydrodynamic states that are inconsistent (there is no clear strategy for what to do
180 // when a density is negative).
181
182 const Real density_floor_tolerance = 1.1;
183
184 // The density floor is the small density, modified by a small factor.
185 // In practice numerical error can cause the density that is created
186 // by this flux limiter to be slightly lower than the target density,
187 // so we set the target to be slightly larger than the real density floor
188 // to avoid density resets.
189
190 Real density_floor = small_dens * density_floor_tolerance;
191
192 // We apply this flux limiter on a per-edge basis. So we can guarantee
193 // that any individual flux cannot cause a small density in one step,
194 // but with the above floor we cannot guarantee that the sum of the
195 // fluxes will enforce this constraint. The only way to guarantee that
196 // is if the density floor is increased by a factor of the number of
197 // edges, so that even if all edges are summed together, the density
198 // will still be at the floor. So we multiply the floor by a factor of
199 // 2 (two edges in each dimension) and a factor of AMREX_SPACEDIM.
200
201 density_floor *= AMREX_SPACEDIM * 2;
202
203 Real dtdx = dt / dx_dir;
204 Real alpha = 1.0 / AMREX_SPACEDIM;
205
206 // Grab the states on either side of the interface we are working with,
207 // depending on which dimension we're currently calling this with.
208
209 GpuArray<Real, QVAR> uR;
210 GpuArray<Real, QVAR> uL;
211 GpuArray<Real, QVAR> qR;
212 GpuArray<Real, QVAR> qL;
213 GpuArray<int, 3> idxL;
214
215 if (idir == 0) {
216 idxL = {i-1,j,k};
217 } else if (idir == 1) {
218 idxL = {i,j-1,k};
219 } else {
220 idxL = {i,j,k-1};
221 }
222
223 for (int n = 0; n < QVAR; ++n) {
224 uR[n] = u(i,j,k,n);
225 qR[n] = q(i,j,k,n);
226 uL[n] = u(idxL[0],idxL[1],idxL[2]);
227 qL[n] = q(idxL[0],idxL[1],idxL[2]);
228 }
229
230 // If an adjacent zone has a floor-violating density, set the flux to zero and move on.
231 // At that point, the only thing to do is wait for a reset at a later point.
232
233 if (uR[URHO] < density_floor || uL[URHO] < density_floor) {
234
235 for (int n = 0; n < QVAR; ++n) {
236 flux(i,j,k,n) = 0.0;
237 }
238
239 return;
240 }
241
242 // Construct cell-centered fluxes.
243 GpuArray<Real, QVAR> fluxL;
244 GpuArray<Real, QVAR> fluxR;
245
246 for (int n = 0; n < QVAR; ++n) {
247 fluxL[n] = 0.0;
248 fluxR[n] = 0.0;
249 }
250
251 fluxL[URHO] = uL[URHO] * qL[QU + idir];
252 fluxL[UMX] = uL[UMX] * qL[QU + idir];
253 fluxL[UMY] = uL[UMY] * qL[QU + idir];
254 fluxL[UMZ] = uL[UMZ] * qL[QU + idir];
255 fluxL[UEDEN] = (uL[UEDEN] + qL[QPRES]) * qL[QU + idir];
256 fluxL[UEINT] = uL[UEINT] * qL[QU + idir];
257
258 fluxR[URHO] = uR[URHO] * qR[QU + idir];
259 fluxR[UMX] = uR[UMX] * qR[QU + idir];
260 fluxR[UMY] = uR[UMY] * qR[QU + idir];
261 fluxR[UMZ] = uR[UMZ] * qR[QU + idir];
262 fluxR[UEDEN] = (uR[UEDEN] + qR[QPRES]) * qR[QU + idir];
263 fluxR[UEINT] = uR[UEINT] * qR[QU + idir];
264
265 fluxL[UMX + idir] = fluxL[UMX + idir] + qL[QPRES];
266 fluxR[UMX + idir] = fluxR[UMX + idir] + qR[QPRES];
267
268 // Construct the Lax-Friedrichs flux on the interface (Equation 12).
269 // Note that we are using the information from Equation 9 to obtain the
270 // effective maximum wave speed, (|u| + c)_max = CFL / lambda where
271 // lambda = dt/(dx * alpha); alpha = 1 in 1D and may be chosen somewhat
272 // freely in multi-D as long as alpha_x + alpha_y + alpha_z = 1.
273
274 GpuArray<Real, QVAR> fluxLF;
275 for (int n = 0; n < QVAR; ++n) {
276 fluxLF[n] = 0.5 * (fluxL[n] + fluxR[n] + (lcfl / dtdx / alpha) * (uL[n] - uR[n]));
277 }
278
279 // Coefficients of fluxes on either side of the interface.
280
281 Real flux_coefR = 2.0 * (dtdx / alpha);
282 Real flux_coefL = 2.0 * (dtdx / alpha);
283
284 // Obtain the one-sided update to the density, based on Hu et al., Eq. 11.
285 // If we would violate the floor, then we need to limit the flux. Since the
286 // flux adds to the density on one side and subtracts from the other, the floor
287 // can only be violated in at most one direction, so we'll do an if-else test
288 // below. This means that we can simplify the approach of Hu et al. -- whereas
289 // they constructed two thetas for each interface (corresponding to either side)
290 // we can complete the operation in one step with a single theta.
291
292 Real drhoL = flux_coefL * flux(i,j,k,URHO);
293 Real rhoL = uL[URHO] - drhoL;
294
295 Real drhoR = flux_coefR * flux(i,j,k,URHO);
296 Real rhoR = uR[URHO] + drhoR;
297
298 Real theta = 1.0;
299
300 if (rhoL < density_floor) {
301
302 // Obtain the final density corresponding to the LF flux.
303
304 Real drhoLF = flux_coefL * fluxLF[URHO];
305 Real rhoLF = uL[URHO] - drhoLF;
306
307 // Solve for theta from (1 - theta) * rhoLF + theta * rho = density_floor.
308
309 theta = amrex::min(theta, (density_floor - rhoLF) / (rhoL - rhoLF));
310
311 }
312 else if (rhoR < density_floor) {
313
314 Real drhoLF = flux_coefR * fluxLF[URHO];
315 Real rhoLF = uR[URHO] + drhoLF;
316
317 theta = amrex::min(theta, (density_floor - rhoLF) / (rhoR - rhoLF));
318
319 }
320
321 // Limit theta to the valid range (this will deal with roundoff issues).
322
323 theta = amrex::min(1.0, amrex::max(theta, 0.0));
324
325 // Assemble the limited flux (Equation 16).
326
327 for (int n = 0; n < QVAR; ++n) {
328 flux(i,j,k,n) = (1.0 - theta) * fluxLF[n] + theta * flux(i,j,k,n);
329 }
330
331 // Now, apply our requirement that the final flux cannot violate the density floor.
332
333 drhoR = flux_coefR * flux(i,j,k,URHO);
334 drhoL = flux_coefL * flux(i,j,k,URHO);
335
336 if (uR[URHO] + drhoR < density_floor) {
337 for (int n = 0; n < QVAR; ++n) {
338 flux(i,j,k,n) = flux(i,j,k,n) * std::abs((density_floor - uR[URHO]) / drhoR);
339 }
340 }
341 else if (uL[URHO] - drhoL < density_floor) {
342 for (int n = 0; n < QVAR; ++n) {
343 flux(i,j,k,n) = flux(i,j,k,n) * std::abs((density_floor - uL[URHO]) / drhoL);
344 }
345 }
346}
