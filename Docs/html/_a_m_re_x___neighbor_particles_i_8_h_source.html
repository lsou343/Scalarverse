219template <int NStructReal, int NStructInt>
220IntVect
221NeighborParticleContainer<NStructReal, NStructInt>
222::computeRefFac (const int src_lev, const int lev)
223{
224 IntVect ref_fac = IntVect(AMREX_D_DECL(1,1,1));
225 if (src_lev < lev) {
226 for (int l = src_lev; l < lev; ++l) {
227 ref_fac *= this->GetParGDB()->refRatio(l);
228 }
229 } else if (src_lev > lev) {
230 for (int l = src_lev; l > lev; --l) {
231 ref_fac *= this->GetParGDB()->refRatio(l-1);
232 }
233 ref_fac *= -1;
234 }
235 return ref_fac;

515template <int NStructReal, int NStructInt>
516void
517NeighborParticleContainer<NStructReal, NStructInt>::
518getNeighborTags (Vector<NeighborCopyTag>& tags, const ParticleType& p,
519 const IntVect& nGrow, const NeighborCopyTag& src_tag, const MyParIter& pti)
520{
521 Box shrink_box = pti.tilebox();
522 shrink_box.grow(-nGrow);
523
524 if (use_mask) {
525 const BaseFab<int>& mask = (*mask_ptr[src_tag.level])[src_tag.grid];
526 AMREX_ASSERT(this->finestLevel() == 0);
527 AMREX_ASSERT(src_tag.level == 0);
528
529 const int lev = 0;
530 const IntVect& iv = this->Index(p, lev);
531 if (shrink_box.contains(iv)) return;
532
533 const Periodicity& periodicity = this->Geom(lev).periodicity();
534 const Box& domain = this->Geom(lev).Domain();
535 const IntVect& lo = domain.smallEnd();
536 const IntVect& hi = domain.bigEnd();
537
538 // Figure out all our neighbors, removing duplicates
539 AMREX_D_TERM(
540 for (int ii = -nGrow[0]; ii < nGrow[0] + 1; ii += nGrow[0]) {,
541 for (int jj = -nGrow[1]; jj < nGrow[1] + 1; jj += nGrow[1]) {,
542 for (int kk = -nGrow[2]; kk < nGrow[2] + 1; kk += nGrow[2]) {)
543 if (AMREX_D_TERM((ii == 0), and (jj == 0), and (kk == 0))) continue;
544 IntVect shift(AMREX_D_DECL(ii, jj, kk));
545 IntVect neighbor_cell = iv + shift;
546
547 NeighborCopyTag tag;
548 tag.grid = mask(neighbor_cell, MaskComps::grid);
549 tag.tile = mask(neighbor_cell, MaskComps::tile);
550 tag.level = mask(neighbor_cell, MaskComps::level);
551 if (periodicity.isAnyPeriodic()) {
552 for (int dir = 0; dir < AMREX_SPACEDIM; ++dir) {
553 if (not periodicity.isPeriodic(dir)) continue;
554 if (neighbor_cell[dir] < lo[dir])
555 tag.periodic_shift[dir] = -1;
556 else if (neighbor_cell[dir] > hi[dir])
557 tag.periodic_shift[dir] = 1;
558 }
559 }
560
561 if (tag != src_tag) tags.push_back(tag);
562
563 AMREX_D_TERM(
564 },
565 },
566 })
567
568 RemoveDuplicates(tags);
569 return;
570 }
571 else
572 {
573 std::vector< std::pair<int, Box> > isects;
574 Box tbx;
575 for (int lev = 0; lev < this->numLevels(); ++lev)
576 {
577 IntVect ref_fac = computeRefFac(0, lev);
578 const Periodicity& periodicity = this->Geom(lev).periodicity();
579 const std::vector<IntVect>& pshifts = periodicity.shiftIntVect();
580 const BoxArray& ba = this->ParticleBoxArray(lev);
581 const IntVect& iv = this->Index(p, lev);
582 for (auto pit=pshifts.cbegin(); pit!=pshifts.cend(); ++pit)
583 {
584 Box pbox = amrex::grow(Box(iv, iv), ref_fac*nGrow) + (*pit);
585 bool first_only = false;
586 ba.intersections(pbox, isects, first_only, 0);
587 for (const auto& isec : isects)
588 {
589 const Box& grid_box = ba[isec.first];
590 for (IntVect cell = pbox.smallEnd(); cell <= pbox.bigEnd(); pbox.next(cell)) {
591 if ( !grid_box.contains(cell) ) continue;
592 int tile = getTileIndex(cell, grid_box,
593 this->do_tiling, this->tile_size, tbx);
594 auto nbor = NeighborCopyTag(lev, isec.first, tile);
595 nbor.periodic_shift = -1*(*pit);
596 if (src_tag != nbor) tags.push_back(nbor);
597 }
598 }
599 }
600 }
601
602 RemoveDuplicates(tags);
603 return;
604 }

660template <int NStructReal, int NStructInt>
661template <class CheckPair>
662void
663NeighborParticleContainer<NStructReal, NStructInt>::
664buildNeighborList (CheckPair&& check_pair, bool sort)
665{
666 AMREX_ASSERT(numParticlesOutOfRange(*this, m_num_neighbor_cells) == 0);
667
668 resizeContainers(this->numLevels());
669
670 for (int lev = 0; lev < this->numLevels(); ++lev)
671 {
672 m_neighbor_list[lev].clear();
673
674 for (MyParIter pti(*this, lev); pti.isValid(); ++pti) {
675 PairIndex index(pti.index(), pti.LocalTileIndex());
676 m_neighbor_list[lev][index];
677 }
678
679#ifndef AMREX_USE_GPU
680 neighbor_list[lev].clear();
681 for (MyParIter pti(*this, lev); pti.isValid(); ++pti) {
682 PairIndex index(pti.index(), pti.LocalTileIndex());
683 neighbor_list[lev][index];
684 }
685#endif
686
687 IntVect ref_fac = computeRefFac(0, lev);
688 auto& plev = this->GetParticles(lev);
689 const auto& geom = this->Geom(lev);
690
691#ifdef _OPENMP
692#pragma omp parallel if (Gpu::notInLaunchRegion())
693#endif
694 for (MyParIter pti(*this, lev); pti.isValid(); ++pti)
695 {
696 int gid = pti.index();
697 int tid = pti.LocalTileIndex();
698 auto index = std::make_pair(gid, tid);
699
700 auto& ptile = plev[index];
701
702 if (ptile.numParticles() == 0) continue;
703
704 Box bx = pti.tilebox();
705 bx.coarsen(ref_fac);
706 bx.grow(m_num_neighbor_cells);
707
708 m_neighbor_list[lev][index].build(ptile, bx, geom,
709 std::forward<CheckPair>(check_pair),
710 m_num_neighbor_cells);
711#ifndef AMREX_USE_GPU
712 const auto& counts = m_neighbor_list[lev][index].GetCounts();
713 const auto& list = m_neighbor_list[lev][index].GetList();
714
715 int li = 0;
716 for (int i = 0; i < ptile.numParticles(); ++i)
717 {
718 auto cnt = counts[i];
719 neighbor_list[lev][index].push_back(cnt);
720 for (size_t j = 0; j < cnt; ++j)
721 {
722 neighbor_list[lev][index].push_back(list[li++]+1);
723 }
724 }
725#endif
726 }
727 }
728}
