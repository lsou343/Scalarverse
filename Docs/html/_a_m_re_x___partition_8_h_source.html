17 int amrex_partition_helper (T const* AMREX_RESTRICT pv, T* AMREX_RESTRICT pv2, int n, F && f)
18 {
19 return Scan::PrefixSum<int> (n,
20 [=] AMREX_GPU_DEVICE (int i) -> int
21 {
22 return f(pv[i]);
23 },
24 [=] AMREX_GPU_DEVICE (int i, int const& s)
25 {
26 // We store true elements from the beginning and false
27 // elements reversely from the end. If all elements
28 // before pv[i] are true, the exclusive sum so far would
29 // be i. But the actual value is s.
30 if (f(pv[i])) {
31 // For true element, s spots from the beginning have
32 // been taken.
33 pv2[s] = pv[i];
34 } else {
35 // There are i-s elements before this element that
36 // are false. From the end, i-s spots have been
37 // taken.
38 pv2[n-1-(i-s)] = pv[i];
39 }
40 },
41 Scan::Type::exclusive);
42 }
