248 AMREX_ALWAYS_ASSERT(nlevs < 2); // relax this later
249
250 IntVect domain_lo(AMREX_D_DECL(0, 0, 0));
251 IntVect domain_hi(AMREX_D_DECL(ncells-1, ncells-1, ncells-1));
252
253 domains.resize(nlevs);
254 domains[0].setSmall(domain_lo);
255 domains[0].setBig(domain_hi);
256
257 ref_ratio.resize(nlevs-1);
258 for (int lev = 1; lev < nlevs; lev++)
259 ref_ratio[lev-1] = IntVect(AMREX_D_DECL(2, 2, 2));
260
261 grids.resize(nlevs);
262 grids[0].define(domains[0]);
263
264 // Now we make the refined level be the center eighth of the domain
265 if (nlevs > 1) {
266 int n_fine = ncells*ref_ratio[0][0];
267 IntVect refined_lo(D_DECL(n_fine/4,n_fine/4,n_fine/4));
268 IntVect refined_hi(D_DECL(3*n_fine/4-1,3*n_fine/4-1,3*n_fine/4-1));
269
270 // Build a box for the level 1 domain
271 Box refined_patch(refined_lo, refined_hi);
272 grids[1].define(refined_patch);
273 }
274
275 // break the BoxArrays at both levels into max_grid_size^3 boxes
276 for (int lev = 0; lev < nlevs; lev++) {
277 grids[lev].maxSize(max_grid_size);
278 }
279
280 for (int lev = 1; lev < nlevs; lev++) {
281 domains[lev] = amrex::refine(domains[lev-1], ref_ratio[lev-1]);
