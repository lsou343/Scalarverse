248AMREX_GPU_HOST_DEVICE AMREX_INLINE
249void update_dm_particle_single (amrex::ParticleContainer<4, 0>::SuperParticleType& p,
250 const int nc,
251 amrex::Array4<amrex::Real const> const& acc,
252 amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& plo,
253 amrex::GpuArray<amrex::Real,AMREX_SPACEDIM> const& dxi,
254 const amrex::Real& dt, const amrex::Real& a_prev,
255 const amrex::Real& a_cur, const int& do_move)
256{
257 amrex::Real half_dt = 0.5 * dt;
258 amrex::Real a_cur_inv = 1.0 / a_cur;
259 amrex::Real dt_a_cur_inv = dt * a_cur_inv;
260
261 amrex::Real lx = (p.pos(0) - plo[0]) * dxi[0] + 0.5;
262 amrex::Real ly = (p.pos(1) - plo[1]) * dxi[1] + 0.5;
263 amrex::Real lz = (p.pos(2) - plo[2]) * dxi[2] + 0.5;
264
265 int i = amrex::Math::floor(lx);
266 int j = amrex::Math::floor(ly);
267 int k = amrex::Math::floor(lz);
268
269 amrex::Real xint = lx - i;
270 amrex::Real yint = ly - j;
271 amrex::Real zint = lz - k;
272
273 amrex::Real sx[] = {1.-xint, xint};
274 amrex::Real sy[] = {1.-yint, yint};
275 amrex::Real sz[] = {1.-zint, zint};
276
277 for (int d=0; d < AMREX_SPACEDIM; ++d)
278 {
279 amrex::Real val = 0.0;
280 for (int kk = 0; kk<=1; ++kk)
281 {
282 for (int jj = 0; jj <= 1; ++jj)
283 {
284 for (int ii = 0; ii <= 1; ++ii)
285 {
286 val += sx[amrex::Math::abs(ii-1)]*
287 sy[amrex::Math::abs(jj-1)]*
288 sz[amrex::Math::abs(kk-1)]*acc(i-ii,j-jj,k-kk,d);
289 }
290 }
291 }
292
293
294 p.rdata(d+1)=a_prev*p.rdata(d+1)+half_dt * val;
295 p.rdata(d+1)*=a_cur_inv;
296 }
297
298 if (do_move == 1)
299 {
300 for (int comp=0; comp < nc; ++comp) {
301 p.pos(comp) = p.pos(comp) + dt_a_cur_inv * p.rdata(comp+1);
302 }
303 }

692 const int grid = kv.first.first;
693 AoS& pbox = kv.second.GetArrayOfStructs();
694 const int n = pbox.size();
695 const FArrayBox& dfab = mf[grid];
696
697#ifdef _OPENMP
698#pragma omp parallel for private(pld)
699#endif
700 for (int i = 0; i < n; i++)
701 {
702 ParticleType& p = pbox[i];
703
704 if (p.id() <= 0) continue;
705
706 Real disp[BL_SPACEDIM];
707 //
708 // Do CIC interpolation onto the particle positions.
709 // For CIC we need one ghost cell!
710 //
711 ParticleType::GetGravity(dfab, m_gdb->Geom(0), p, disp);
712
713 D_TERM(p.pos(0) += len[0]*disp[0];,
714 p.pos(1) += len[1]*disp[1];,
715 p.pos(2) += len[2]*disp[2];);
716 //
717 // Note: m_data[0] is mass, 1 is v_x, ...
718 //
719 D_TERM(p.rdata(1) = vel_fac[0]*disp[0];,
720 p.rdata(2) = vel_fac[1]*disp[1];,
721 p.rdata(3) = vel_fac[2]*disp[2];);

724 {
725 this->PeriodicShift(p);
726
727 if (!this->Where(p, pld))
728 amrex::Abort("DarkMatterParticleContainer::InitCosmo(): invalid particle");
729 }
730
731 this->Reset(p, true);
732 }
733 }

735 // Let Redistribute() sort out where the particles now belong.
736 //
737 Redistribute();
738

740 {
741 std::cout << "Done with particle displacement" << '\n';
742 }
743
744 if (m_verbose > 1)
745 {
746 Real runtime = ParallelDescriptor::second() - strttime;
747
748 ParallelDescriptor::ReduceRealMax(runtime, IOProc);
749
750 if (ParallelDescriptor::IOProcessor())
751 {

768void
769DarkMatterParticleContainer::InitFromBinaryMortonFile(const std::string& particle_directory,
770 int nextra, int skip_factor) {
771 BL_PROFILE("DarkMatterParticleContainer::InitFromBinaryMortonFile");
772
773 ParticleMortonFileHeader hdr;
774 ReadHeader(particle_directory, "Header", hdr);
775
776 uint64_t num_parts = hdr.NP;
777 int DM = hdr.DM;
778 int NX = hdr.NX;
779 int float_size = hdr.SZ;
780 int num_files = hdr.NF;
781 size_t psize = (DM + NX) * float_size;
782
783 std::string particle_file_base = particle_directory + "/particles.";
784 std::vector<std::string> file_names;
785 for (int i = 0; i < num_files; ++i)
786 file_names.push_back(get_file_name(particle_file_base, i));
787
788 const int lev = 0;
789 const BoxArray& ba = ParticleBoxArray(lev);
790 int num_boxes = ba.size();
791 uint64_t num_parts_per_box = num_parts / num_boxes;
792 uint64_t num_parts_per_file = num_parts / num_files;
793 uint64_t num_bytes_per_file = num_parts_per_file * psize;
794
795 std::vector<BoxMortonKey> box_morton_keys(num_boxes);
796 for (int i = 0; i < num_boxes; ++i) {
797 const Box& box = ba[i];
798 unsigned int x = box.smallEnd(0);
799 unsigned int y = box.smallEnd(1);
800 unsigned int z = box.smallEnd(2);
801 box_morton_keys[i].morton_id = get_morton_index(x, y, z);
802 box_morton_keys[i].box_id = i;
803 }
804
805 std::sort(box_morton_keys.begin(), box_morton_keys.end(), by_morton_id());
806
807 std::vector<int> file_indices(num_boxes);
808 for (int i = 0; i < num_boxes; ++i)
809 file_indices[box_morton_keys[i].box_id] = i;
810
811 ParticleType p;
812 for (MFIter mfi = MakeMFIter(lev, false); mfi.isValid(); ++mfi) { // no tiling
813 Box tile_box = mfi.tilebox();
814 const int grid = mfi.index();
815 const int tile = mfi.LocalTileIndex();
816 auto& particles = GetParticles(lev);
817
818 uint64_t start = file_indices[grid]*num_parts_per_box;
819 uint64_t stop = start + num_parts_per_box;
820
821 int file_num = start / num_parts_per_file;
822 uint64_t seek_pos = (start * psize ) % num_bytes_per_file;
823 std::string file_name = file_names[file_num];
824
825 std::ifstream ifs;
826 ifs.open(file_name.c_str(), std::ios::in|std::ios::binary);
827 if ( not ifs ) {
828 amrex::Print() << "Failed to open file " << file_name << " for reading. \n";
829 amrex::Abort();
830 }
831
832 ifs.seekg(seek_pos, std::ios::beg);
833
834 for (uint64_t i = start; i < stop; ++i) {
835 int next_file = i / num_parts_per_file;
836 if (next_file != file_num) {
837 file_num = next_file;
838 file_name = file_names[file_num];
839 ifs.close();
840 ifs.open(file_name.c_str(), std::ios::in|std::ios::binary);
841 if ( not ifs ) {
842 amrex::Print() << "Failed to open file " << file_name << " for reading. \n";
843 amrex::Abort();
844 }
845 }
846
847 float fpos[DM];
848 float fextra[NX];
849 ifs.read((char*)&fpos[0], DM*sizeof(float));
850 ifs.read((char*)&fextra[0], NX*sizeof(float));
851
852 if ( (i - start) % skip_factor == 0 ) {
853 AMREX_D_TERM(p.pos(0) = fpos[0];,
854 p.pos(1) = fpos[1];,
855 p.pos(2) = fpos[2];);
856
857 for (int comp = 0; comp < NX; comp++)
858 p.rdata(BL_SPACEDIM+comp) = fextra[comp];
859
860 p.rdata(BL_SPACEDIM) *= skip_factor;
861
862 p.id() = ParticleType::NextID();
863 p.cpu() = ParallelDescriptor::MyProc();
864 particles[std::make_pair(grid, tile)].push_back(p);
865 }
866 }
867 }
868
869 Redistribute();
870}

874DarkMatterParticleContainer::InitSphericalCollapse (amrex::MultiFab& mf, int lev, int nlevs, int comp, Real ratio, Real hubl, Real OmM)

943

946 // Add it to the appropriate PBox at the appropriate level.
947 //

951
952 if (ParallelDescriptor::IOProcessor() && m_verbose)
