16void get_position_unit_cell(Real* r, const IntVect& nppc, int i_part)
17{
18 int nx = nppc[0];
19#if AMREX_SPACEDIM > 1
20 int ny = nppc[1];
21#else
22 int ny = 1;
23#endif
24#if AMREX_SPACEDIM > 2
25 int nz = nppc[2];
26#else
27 int nz = 1;
28#endif
29
30 int ix_part = i_part/(ny * nz);
31 int iy_part = (i_part % (ny * nz)) % ny;
32 int iz_part = (i_part % (ny * nz)) / ny;
33
34 r[0] = (0.5+ix_part)/nx;
35 r[1] = (0.5+iy_part)/ny;
36 r[2] = (0.5+iz_part)/nz;
37}

324void testParallelContext ()
325{
326 BL_PROFILE("testParallelContext");
327
328 AMREX_ALWAYS_ASSERT(AMREX_SPACEDIM == 3);
329
330 // we always make two subcommunicators.
331 // one takes the left half of the domain in direction 0, the other the right.
332 int rank_n = ParallelContext::NProcsSub();
333 int myproc = ParallelContext::MyProcSub();
334 int task_me = myproc / (amrex::max(rank_n, 2) / 2);
335
336 if (task_me > 1) task_me = 1;
337
338 MPI_Comm new_comm;
339 MPI_Comm_split(ParallelContext::CommunicatorSub(), task_me, myproc, &new_comm);
340
341 const int io_rank = 0;
342 ParallelContext::push(new_comm, task_me, io_rank);
343
344 {
345 TestParams params;
346 get_test_params(params, "redistribute");
347
348 int is_per[BL_SPACEDIM];
349 for (int i = 0; i < BL_SPACEDIM; i++)
350 is_per[i] = params.is_periodic;
351
352 // Each comm gets a different domain
353 IntVect hs = params.size / 2;
354
355 RealBox real_box;
356 for (int n = 0; n < BL_SPACEDIM; n++)
357 {
358 Real physlo = (n == 0) ? task_me*hs[n] : 0.0;
359 Real physhi = (n == 0) ? task_me*hs[n] + hs[n] : params.size[n];
360 real_box.setLo(n, physlo);
361 real_box.setHi(n, physhi);
362 }
363
364 IntVect domain_lo(AMREX_D_DECL(task_me*hs[0], 0, 0));
365 IntVect domain_hi(AMREX_D_DECL(task_me*hs[0] + hs[0] - 1,
366 params.size[1] - 1,
367 params.size[2] - 1));
368 const Box base_domain(domain_lo, domain_hi);
369
370 Geometry geom;
371 geom.define(base_domain, &real_box, CoordSys::cartesian, is_per);
372
373 BoxArray ba;
374 ba.define(base_domain);
375 ba.maxSize(params.max_grid_size);
376
377 DistributionMapping dm;
378 dm.define(ba, ParallelContext::NProcsSub());
379
380 // by doing it this way, we have each subcommunicator making
381 // separate Redistribute calls at the same time.
382 if (task_me == 0)
383 {
384 TestParticleContainer pc(geom, dm, ba);
385
386 int npc = params.num_ppc;
387 IntVect nppc = IntVect(AMREX_D_DECL(npc, npc, npc));
388
389 pc.InitParticles(nppc);
390
391 pc.checkAnswer();
392
393 auto np_old = pc.TotalNumberOfParticles();
394
395 for (int i = 0; i < params.nsteps; ++i)
396 {
397 pc.moveParticles(params.move_dir, params.do_random);
398 pc.RedistributeLocal();
399 if (params.sort) pc.SortParticlesByCell();
400 pc.checkAnswer();
401 }
402
403 if (geom.isAllPeriodic()) AMREX_ALWAYS_ASSERT(np_old == pc.TotalNumberOfParticles());
404 }
405
406 if (task_me == 1)
407 {
408 TestParticleContainer pc(geom, dm, ba);
409
410 int npc = params.num_ppc;

412
413 pc.InitParticles(nppc);
414
415 pc.checkAnswer();

418
