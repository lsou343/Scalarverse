216NyxParticleContainer<NSR,NSI,NAR,NAI>::SetParticleVelocities (amrex::Vector<amrex::Real>& part_data)
217{
218 BL_PROFILE("NyxParticleContainer<NSR,NSI,NAR,NSI>::SetParticleVelocities()");
219 // This gives us the starting point into the part_data array
220 // If only one processor (or no MPI), then that's all we need
221 int cnt = 0;
222
223#if BL_USE_MPI
224 amrex::Vector<long> cnts(amrex::ParallelDescriptor::NProcs());
225
226 // This returns the number of particles on this processor
227 long lcnt = this->TotalNumberOfParticles(true,true);
228
229 // This accumulates the "lcnt" values into "cnts"
230 MPI_Gather(&lcnt,1,
231 amrex::ParallelDescriptor::Mpi_typemap<long>::type(),
232 cnts.dataPtr(),
233 1,
234 amrex::ParallelDescriptor::Mpi_typemap<long>::type(),
235 amrex::ParallelDescriptor::IOProcessorNumber(),
236 amrex::ParallelDescriptor::Communicator());
237
238 amrex::ParallelDescriptor::Bcast(cnts.dataPtr(), cnts.size(), amrex::ParallelDescriptor::IOProcessorNumber());
239
240 for (int iproc = 0; iproc < amrex::ParallelDescriptor::MyProc(); iproc++)
241 cnt += cnts[iproc];
242
243 // Each particle takes up (BL_SPACEDIM) Reals
244 cnt*= (BL_SPACEDIM);
245#endif
246
247 // This is the total number of particles on *all* processors
248 long npart = this->TotalNumberOfParticles(true,false);
249
250 // Velocities
251 if (part_data.size() != npart*(BL_SPACEDIM))
252 amrex::Abort("Sending in wrong size part_data to SetParticleVelocities");
253
254 for (int lev = 0; lev <= this->m_gdb->finestLevel(); lev++)
255 {
256 ParticleLevel& pmap = this->GetParticles(lev);
257
258 for (typename ParticleLevel::iterator pmap_it = pmap.begin(), pmapEnd = pmap.end(); pmap_it != pmapEnd; ++pmap_it)
259 {
260 AoS& pbx = pmap_it->second.GetArrayOfStructs();
261 const int n = pbx.size();
262
263 for (int i = 0; i < n; i++)
264 {
265 ParticleType& p = pbx[i];
266 if (p.id() > 0)
267 {
268 // Load velocities
269 for (int d=0; d < BL_SPACEDIM; d++)
270 p.rdata(d+1) = part_data[cnt+d];

272 // Update counter

288 amrex::Real* mom) const
289{
290 BL_PROFILE("NyxParticleContainer<NSR,NSI,NAR,NAI>::sumParticleMomentum()");
291 BL_ASSERT(NSR >= BL_SPACEDIM+1);
292 BL_ASSERT(lev >= 0 && lev < this->GetParticles().size());
293
294 const ParticleLevel& pmap = this->GetParticles(lev);
295
296 D_TERM(mom[0] = 0;, mom[1] = 0;, mom[2] = 0;);
297
298 for (typename ParticleLevel::const_iterator pmap_it = pmap.begin(), pmapEnd = pmap.end(); pmap_it != pmapEnd; ++pmap_it)
299 {
300 const AoS& pbox = pmap_it->second.GetArrayOfStructs();
301 const int n = pbox.size();
302
303 amrex::Real mom_0 = 0, mom_1 = 0, mom_2 = 0;
304
305#ifdef _OPENMP
306#pragma omp parallel for reduction(+:mom_0,mom_1,mom_2)
307#endif
308 for (int i = 0; i < n; i++)
309 {
310 const ParticleType& p = pbox[i];
311
312 if (p.id() > 0)
313 {
314 D_TERM(mom_0 += p.rdata(0) * p.rdata(1);,
315 mom_1 += p.rdata(0) * p.rdata(2);,
316 mom_2 += p.rdata(0) * p.rdata(3););
317 }
318 }
319
320 D_TERM(mom[0] += mom_0;, mom[1] += mom_1;, mom[2] += mom_2;);
321 }
322
323 amrex::ParallelDescriptor::ReduceRealSum(mom,BL_SPACEDIM);
324}

387 amrex::ReduceData<amrex::Real> reduce_data(reduce_op);

474 }
