24void MCNodalLinOp::Fapply (int amrlev, int mglev, MultiFab& a_out,const MultiFab& a_in) const
25{
26 BL_PROFILE("MCNodalLinOp::Fapply()");
27
28 a_out.setVal(0.0);
29 amrex::Box domain(m_geom[amrlev][mglev].Domain());
30 domain.convert(amrex::IntVect::TheNodeVector());
31 domain.grow(-1); // Shrink domain so we don't operate on any boundaries
32 const Real* DX = m_geom[amrlev][mglev].CellSize();
33
34 amrex::AsyncArray<Real> aa(coeff.data(), coeff.size());
35 amrex::Real const* dcoeff = aa.data();
36 const int N = getNComp();
37
38 for (MFIter mfi(a_out, amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi)
39 {
40 Box bx = mfi.tilebox();
41 bx.grow(1); // Expand to cover first layer of ghost nodes
42 bx = bx & domain; // Take intersection of box and the problem domain
43
44 amrex::Array4<const amrex::Real> const& in = a_in.array(mfi);
45 amrex::Array4<amrex::Real> const& out = a_out.array(mfi);
46
47 for (int n = 0; n < getNComp(); n++)
48 amrex::ParallelFor (bx,[=] AMREX_GPU_DEVICE(int i, int j, int k) {
49 out(i,j,k,n) = 0.0;
50 for (int m = 0; m < N; m++)
51 {
52 out(i,j,k,n) += dcoeff[ncomp*n + m] *
53 ( AMREX_D_TERM(- (in(i-1,j,k,m) - 2.0 * in(i,j,k,m) + in(i+1,j,k,m)) / DX[0] / DX[0],
54 - (in(i,j-1,k,m) - 2.0 * in(i,j,k,m) + in(i,j+1,k,m)) / DX[1] / DX[1],
55 - (in(i,j,k-1,m) - 2.0 * in(i,j,k,m) + in(i,j,k+1,m)) / DX[2] / DX[2])) ;
56 }
57
58 });
59 }

61void MCNodalLinOp::Diag (int amrlev, int mglev, MultiFab& a_diag)
62{
63 a_diag.setVal(1.0);
64 amrex::Box domain(m_geom[amrlev][mglev].Domain());
65 domain.convert(amrex::IntVect::TheNodeVector());
66 domain.grow(-1); // Shrink domain so we don't operate on any boundaries
67 const Real* DX = m_geom[amrlev][mglev].CellSize();
68
69 amrex::AsyncArray<Real> aa(coeff.data(), coeff.size());
70 amrex::Real const* dcoeff = aa.data();
71 const int N = getNComp();
72
73 for (MFIter mfi(a_diag, amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi)
74 {
75 Box bx = mfi.tilebox();
76 bx.grow(1); // Expand to cover first layer of ghost nodes
77 bx = bx & domain; // Take intersection of box and the problem domain
78
79 amrex::Array4<amrex::Real> const& diag = a_diag.array(mfi);
80
81 for (int n = 0; n < getNComp(); n++)
82 amrex::ParallelFor (bx,[=] AMREX_GPU_DEVICE(int i, int j, int k) {
83 diag(i,j,k,n) = dcoeff[N*n + n] *
84 ( AMREX_D_TERM(+ 2.0 / DX[0] / DX[0],
85 + 2.0 / DX[1] / DX[1],
86 + 2.0 / DX[2] / DX[2]) );
87 });
88 }

105void MCNodalLinOp::Fsmooth (int amrlev, int mglev, amrex::MultiFab& a_x, const amrex::MultiFab& a_b) const
106{
107 BL_PROFILE("MCNodalLinOp::Fsmooth()");
108
109 amrex::Box domain(m_geom[amrlev][mglev].Domain());
110 domain.convert(amrex::IntVect::TheNodeVector());
111 domain.grow(-1); // Shrink domain so we don't operate on any boundaries
112
113 //int ncomp = getNComp();
114 int nghost = getNGrow();
115
116 Real omega = 2./3.; // Damping factor (very important!)
117
118 MultiFab _Ax(a_x.boxArray(), a_x.DistributionMap(), ncomp, nghost);
119 MultiFab _Dx(a_x.boxArray(), a_x.DistributionMap(), ncomp, nghost);
120 MultiFab _Rx(a_x.boxArray(), a_x.DistributionMap(), ncomp, nghost);
121
122 if (!m_diagonal_computed) amrex::Abort("MCNodalLinOp::Diagonal() must be called before using Fsmooth");
123
124 // This is a JACOBI iteration, not Gauss-Seidel.
125 // So we need to do twice the number of iterations to get the same behavior as GS.
126 for (int ctr = 0; ctr < 2; ctr++)
127 {
128 Fapply(amrlev,mglev,_Ax,a_x); // find Ax
129
130 amrex::MultiFab::Copy(_Dx,a_x,0,0,ncomp,nghost); // Dx = x
131 amrex::MultiFab::Multiply(_Dx,*m_diag[amrlev][mglev],0,0,ncomp,nghost); // Dx *= diag (Dx = x*diag)
132
133 amrex::MultiFab::Copy(_Rx,_Ax,0,0,ncomp,nghost); // Rx = Ax
134 amrex::MultiFab::Subtract(_Rx,_Dx,0,0,ncomp,nghost); // Rx -= Dx (Rx = Ax - Dx)
135
136
137 //for (MFIter mfi(a_x, amrex::TilingIfNotGPU()); mfi.isValid(); ++mfi)
138 for (MFIter mfi(a_x, false); mfi.isValid(); ++mfi)
139 {
140 //Box bx = mfi.tilebox();
141 Box bx = mfi.validbox();
142 bx.grow(1); // Expand to cover first layer of ghost nodes
143 bx = bx & domain; // Take intersection of box and the problem domain
144
145 amrex::Array4<amrex::Real> const& x = a_x.array(mfi);
146
147 amrex::Array4<const amrex::Real> const& b = a_b.array(mfi);
148 amrex::Array4<const amrex::Real> const& Rx = _Rx.array(mfi);
149 amrex::Array4<const amrex::Real> const& diag = (*m_diag[amrlev][mglev]).array(mfi);
150
151 for (int n = 0; n < getNComp(); n++)
152 amrex::ParallelFor (bx,[=] AMREX_GPU_DEVICE(int i, int j, int k) {
153 x(i,j,k,n) = (1.-omega)*x(i,j,k,n) + omega*(b(i,j,k,n) - Rx(i,j,k,n))/diag(i,j,k,n);
154 });
155 }
156 }
157 amrex::Geometry geom = m_geom[amrlev][mglev];
158 a_x.setMultiGhost(true);
159 a_x.FillBoundary();
160 nodalSync(amrlev, mglev, a_x);
161}

172void MCNodalLinOp::define (const Vector<Geometry>& a_geom,
173 const Vector<BoxArray>& a_grids,
174 const Vector<DistributionMapping>& a_dmap,
175 const LPInfo& a_info,
176 const Vector<FabFactory<FArrayBox> const*>& a_factory)
177{
178 BL_PROFILE("MCNodalLinOp::~Operator()");
179
180 // This makes sure grids are node-centered;
181 Vector<BoxArray> cc_grids = a_grids;
182 for (auto& ba : cc_grids) {
183 ba.enclosedCells();
184 }
185
186 MLNodeLinOp::define(a_geom, a_grids, a_dmap, a_info, a_factory);
187
188 int nghost = 2;
189 // Resize the multifab containing the operator diagonal
190 m_diag.resize(m_num_amr_levels);
191 for (int amrlev = 0; amrlev < m_num_amr_levels; ++amrlev)

194
195 for (int mglev = 0; mglev < m_num_mg_levels[amrlev]; ++mglev)
196 {
197 m_diag[amrlev][mglev].reset(new MultiFab(amrex::convert(m_grids[amrlev][mglev], amrex::IntVect::TheNodeVector()),
198 m_dmap[amrlev][mglev], getNComp(), nghost));
199 }
200 }
201
202 m_lobc.resize(getNComp(),{{AMREX_D_DECL(BCType::Dirichlet,BCType::Dirichlet,BCType::Dirichlet)}});
203 m_hibc.resize(getNComp(),{{AMREX_D_DECL(BCType::Dirichlet,BCType::Dirichlet,BCType::Dirichlet)}});

207void MCNodalLinOp::buildMasks ()
208{
209 BL_PROFILE("MCNodalLinOp::buildMasks()");
210 if (m_masks_built) return;
211
212 m_masks_built = true;
213
214 m_is_bottom_singular = false;
215
216 {
217 std::vector< std::pair<int,Box> > isects;
218 IArrayBox ccfab;
219
220 for (int amrlev = 0; amrlev < m_num_amr_levels; ++amrlev)
221 {
222 for (int mglev = 0; mglev < m_num_mg_levels[amrlev]; ++mglev)
223 {
224 const Geometry& geom = m_geom[amrlev][mglev];
225 const auto& period = geom.periodicity();
226 const Box& ccdomain = geom.Domain();
227 const std::vector<IntVect>& pshifts = period.shiftIntVect();
228
229 Box ccdomain_p = ccdomain;
230 for (int idim = 0; idim < AMREX_SPACEDIM; ++idim) {
231 if (geom.isPeriodic(idim)) {
232 ccdomain_p.grow(idim, 1);
233 }
234 }
235
236 {
237 auto& dmask = *m_dirichlet_mask[amrlev][mglev];
238 const BoxArray& ccba = m_grids[amrlev][mglev];
239
240 for (MFIter mfi(dmask, MFItInfo().SetDynamic(true)); mfi.isValid(); ++mfi)
241 {
242 const Box& ndbx = mfi.validbox();
243 const Box& ccbx = amrex::enclosedCells(ndbx);
244 const Box& ccbxg1 = amrex::grow(ccbx,1);
245
246 ccfab.resize(ccbxg1);
247 ccfab.setVal<RunOn::Host>(1);
248 ccfab.setComplement<RunOn::Host>(2,ccdomain_p,0,1);
249
250 for (const auto& iv : pshifts)
251 {
252 ccba.intersections(ccbxg1+iv, isects);
253 for (const auto& is : isects)
254 {
255 ccfab.setVal<RunOn::Host>(0, is.second-iv, 0, 1);
256 }
257 }
258 }
259 }
260 }
261 }
262 }
263
264 for (int amrlev = 0; amrlev < m_num_amr_levels-1; ++amrlev)
265 {
266 iMultiFab& cc_mask = *m_cc_fine_mask[amrlev];
267 iMultiFab& nd_mask = *m_nd_fine_mask[amrlev];
268 LayoutData<int>& has_cf = *m_has_fine_bndry[amrlev];
269 const BoxArray& fba = m_grids[amrlev+1][0];
270 const BoxArray& cfba = amrex::coarsen(fba, AMRRefRatio(amrlev));
271
272 const Box& ccdom = m_geom[amrlev][0].Domain();
273
274 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(AMRRefRatio(amrlev) == 2, "ref_ratio != 0 not supported");
275
276 cc_mask.setVal(0); // coarse by default
277
278 const std::vector<IntVect>& pshifts = m_geom[amrlev][0].periodicity().shiftIntVect();
279
280 {
281 std::vector< std::pair<int,Box> > isects;
282
283 for (MFIter mfi(cc_mask); mfi.isValid(); ++mfi)
284 {
285 has_cf[mfi] = 0;
286 IArrayBox& fab = cc_mask[mfi];
287 const Box& bx = fab.box();
288 for (const auto& iv : pshifts)
289 {
290 cfba.intersections(bx+iv, isects);
291 for (const auto& is : isects)
292 {
293 fab.setVal<RunOn::Host>(1, is.second-iv, 0, 1);
294 }
295 if (!isects.empty()) has_cf[mfi] = 1;
296 }
297
298 mlndlap_fillbc_cc<int>(mfi.validbox(), fab.array(), ccdom,
299 LoBC(0), HiBC(0));
300 }
301 }
302
303 for (MFIter mfi(nd_mask,true); mfi.isValid(); ++mfi)
304 {
305 const Box& bx = mfi.tilebox();
306 Array4<int> const& nmsk = nd_mask.array(mfi);
307 Array4<int const> const& cmsk = cc_mask.const_array(mfi);
308 amrex::ParallelFor(bx, [=] (int i, int j, int k) noexcept
309 {
310 mlndlap_set_nodal_mask(i,j,k,nmsk,cmsk);
311 });
312 }
313 }
314
315 auto& has_cf = *m_has_fine_bndry[m_num_amr_levels-1];
316
317 for (MFIter mfi(has_cf); mfi.isValid(); ++mfi)
318 {
319 has_cf[mfi] = 0;
320 }
321
322 {
323 int amrlev = 0;
324 int mglev = m_num_mg_levels[amrlev]-1;
325 const iMultiFab& omask = *m_owner_mask[amrlev][mglev];
326 m_bottom_dot_mask.define(omask.boxArray(), omask.DistributionMap(), 1, 0);
327
328 const Geometry& geom = m_geom[amrlev][mglev];
329 Box nddomain = amrex::surroundingNodes(geom.Domain());
330
331 for (MFIter mfi(m_bottom_dot_mask,true); mfi.isValid(); ++mfi)
332 {
333 const Box& bx = mfi.tilebox();
334 Array4<Real> const& dfab = m_bottom_dot_mask.array(mfi);
335 Array4<int const> const& sfab = omask.const_array(mfi);
336 mlndlap_set_dot_mask(bx, dfab, sfab, nddomain, LoBC(0), HiBC(0));
337 }
338 }

442void MCNodalLinOp::interpolation (int amrlev, int fmglev, MultiFab& fine, const MultiFab& crse) const
443{
444 BL_PROFILE("MCNodalLinOp::interpolation()");
445 amrex::Box fdomain = m_geom[amrlev][fmglev].Domain(); fdomain.convert(amrex::IntVect::TheNodeVector());
446
447 bool need_parallel_copy = !amrex::isMFIterSafe(crse, fine);
448 MultiFab cfine;
449 const MultiFab* cmf = &crse;
450 if (need_parallel_copy) {
451 const BoxArray& ba = amrex::coarsen(fine.boxArray(), 2);
452 cfine.define(ba, fine.DistributionMap(), crse.nComp(), crse.nGrow());
453 cfine.ParallelCopy(crse);
454 cmf = &cfine;
455 }
456
457 for (MFIter mfi(fine, false); mfi.isValid(); ++mfi)
458 {
459 const Box& fine_bx = mfi.validbox() & fdomain;
460 const Box& course_bx = amrex::coarsen(fine_bx,2);
461 const Box& tmpbx = amrex::refine(course_bx,2);
462 FArrayBox tmpfab;
463 tmpfab.resize(tmpbx,fine.nComp());
464 tmpfab.setVal<RunOn::Host>(0.0);
465 const amrex::FArrayBox &crsefab = (*cmf)[mfi];
466
467 amrex::Array4<const amrex::Real> const& cdata = crsefab.const_array();
468 amrex::Array4<amrex::Real> const& fdata = tmpfab.array();
469
470 for (int n = 0; n < crse.nComp(); n++)
471 {
472 // I,J,K == coarse coordinates
473 // i,j,k == fine coordinates
474 amrex::ParallelFor (fine_bx,[=] AMREX_GPU_DEVICE(int i, int j, int k) {
475
476 int I=i/2, J=j/2, K=k/2;
477
478 if (i%2 == 0 && j%2 == 0 && k%2 ==0) // Coincident
479 fdata(i,j,k,n) = cdata(I,J,K,n);
480 else if (j%2 == 0 && k%2 == 0) // X Edge
481 fdata(i,j,k,n) = 0.5 * (cdata(I,J,K,n) + cdata(I+1,J,K,n));
482 else if (k%2 == 0 && i%2 == 0) // Y Edge
483 fdata(i,j,k,n) = 0.5 * (cdata(I,J,K,n) + cdata(I,J+1,K,n));
484 else if (i%2 == 0 && j%2 == 0) // Z Edge
485 fdata(i,j,k,n) = 0.5 * (cdata(I,J,K,n) + cdata(I,J,K+1,n));
486 else if (i%2 == 0) // X Face
487 fdata(i,j,k,n) = 0.25 * (cdata(I,J,K,n) + cdata(I,J+1,K,n) +
488 cdata(I,J,K+1,n) + cdata(I,J+1,K+1,n));
489 else if (j%2 == 0) // Y Face
490 fdata(i,j,k,n) = 0.25 * (cdata(I,J,K,n) + cdata(I,J,K+1,n) +
491 cdata(I+1,J,K,n) + cdata(I+1,J,K+1,n));
492 else if (k%2 == 0) // Z Face
493 fdata(i,j,k,n) = 0.25 * (cdata(I,J,K,n) + cdata(I+1,J,K,n) +
494 cdata(I,J+1,K,n) + cdata(I+1,J+1,K,n));
495 else // Center
496 fdata(i,j,k,n) = 0.125 * (cdata(I,J,K,n) +
497 cdata(I+1,J,K,n) + cdata(I,J+1,K,n) + cdata(I,J,K+1,n) +
498 cdata(I,J+1,K+1,n) + cdata(I+1,J,K+1,n) + cdata(I+1,J+1,K,n) +
499 cdata(I+1,J+1,K+1,n));
500 });
501 }
502 fine[mfi].plus<RunOn::Host>(tmpfab,fine_bx,fine_bx,0,0,fine.nComp());
503 }
504 amrex::Geometry geom = m_geom[amrlev][fmglev];
505 fine.setMultiGhost(true);
506 fine.FillBoundary();
507 nodalSync(amrlev, fmglev, fine);
508}

510void MCNodalLinOp::averageDownSolutionRHS (int camrlev, MultiFab& crse_sol, MultiFab&,
511 const MultiFab& fine_sol, const MultiFab&)
512{

527void MCNodalLinOp::reflux (int crse_amrlev,
528 MultiFab& res, const MultiFab&, const MultiFab&,
529 MultiFab& fine_res, MultiFab&, const MultiFab&) const
530{
531 BL_PROFILE("MCNodalLinOp::reflux()");
532
533 amrex::Box cdomain(m_geom[crse_amrlev][0].Domain());
534 cdomain.convert(amrex::IntVect::TheNodeVector());
535
536 const Geometry& cgeom = m_geom[crse_amrlev ][0];
537
538 const BoxArray& fba = fine_res.boxArray();
539 const DistributionMapping& fdm = fine_res.DistributionMap();
540
541 MultiFab fine_res_for_coarse(amrex::coarsen(fba, 2), fdm, ncomp, 2);
542 fine_res_for_coarse.ParallelCopy(res,0,0,ncomp,0,0,cgeom.periodicity());
543
544 applyBC(crse_amrlev+1, 0, fine_res, BCMode::Inhomogeneous, StateMode::Solution);
545
546 const int coarse_fine_node = 1;
547 const int fine_fine_node = 2;
548
549 amrex::iMultiFab nodemask(amrex::coarsen(fba,2), fdm, 1, 2);
550 nodemask.ParallelCopy(*m_nd_fine_mask[crse_amrlev],0,0,1,0,0,cgeom.periodicity());
551
552 amrex::iMultiFab cellmask(amrex::convert(amrex::coarsen(fba,2),amrex::IntVect::TheCellVector()), fdm, 1, 2);
553 cellmask.ParallelCopy(*m_cc_fine_mask[crse_amrlev],0,0,1,1,1,cgeom.periodicity());
554
555 for (MFIter mfi(fine_res_for_coarse, false); mfi.isValid(); ++mfi)
556 {
557 const Box& bx = mfi.validbox();
558
559 amrex::Array4<const int> const& nmask = nodemask.array(mfi);
560 amrex::Array4<amrex::Real> const& cdata = fine_res_for_coarse.array(mfi);
561 amrex::Array4<const amrex::Real> const& fdata = fine_res.array(mfi);
562
563 const Dim3 lo= amrex::lbound(cdomain), hi = amrex::ubound(cdomain);
564
565 for (int n = 0; n < fine_res.nComp(); n++)
566 {
567 // I,J,K == coarse coordinates
568 // i,j,k == fine coordinates
569 amrex::ParallelFor (bx,[=] AMREX_GPU_DEVICE(int I, int J, int K) {
570 int i=I*2, j=J*2, k=K*2;
571
572 if (nmask(I,J,K) == fine_fine_node || nmask(I,J,K) == coarse_fine_node)
573 {
574 if ((I == lo.x || I == hi.x) &&
575 (J == lo.y || J == hi.y) &&
576 (K == lo.z || K == hi.z)) // Corner
577 cdata(I,J,K,n) = fdata(i,j,k,n);
578 else if ((J == lo.y || J == hi.y) &&
579 (K == lo.z || K == hi.z)) // X edge
580 cdata(I,J,K,n) = 0.25*fdata(i-1,j,k,n) + 0.5*fdata(i,j,k,n) + 0.25*fdata(i+1,j,k,n);
581 else if ((K == lo.z || K == hi.z) &&
582 (I == lo.x || I == hi.x)) // Y edge
583 cdata(I,J,K,n) = 0.25*fdata(i,j-1,k,n) + 0.5*fdata(i,j,k,n) + 0.25*fdata(i,j+1,k,n);
584 else if ((I == lo.x || I == hi.x) &&
585 (J == lo.y || J == hi.y)) // Z edge
586 cdata(I,J,K,n) = 0.25*fdata(i,j,k-1,n) + 0.5*fdata(i,j,k,n) + 0.25*fdata(i,j,k+1,n);
587 else if (I == lo.x || I == hi.x) // X face
588 cdata(I,J,K,n) =
589 (+ fdata(i,j-1,k-1,n) + 2.0*fdata(i,j,k-1,n) + fdata(i,j+1,k-1,n)
590 + 2.0*fdata(i,j-1,k ,n) + 4.0*fdata(i,j,k ,n) + 2.0*fdata(i,j+1,k ,n)
591 + fdata(i,j-1,k+1,n) + 2.0*fdata(i,j,k+1,n) + fdata(i,j+1,k+1,n))/16.0;
592 else if (J == lo.y || J == hi.y) // Y face
593 cdata(I,J,K,n) =
594 (+ fdata(i-1,j,k-1,n) + 2.0*fdata(i-1,j,k,n) + fdata(i-1,j,k+1,n)
595 + 2.0*fdata(i ,j,k-1,n) + 4.0*fdata(i ,j,k,n) + 2.0*fdata(i ,j,k+1,n)
596 + fdata(i+1,j,k-1,n) + 2.0*fdata(i+1,j,k,n) + fdata(i+1,j,k+1,n))/16.0;
597 else if (K == lo.z || K == hi.z) // Z face
598 cdata(I,J,K,n) =
599 (+ fdata(i-1,j-1,k,n) + 2.0*fdata(i,j-1,k,n) + fdata(i+1,j-1,k,n)
600 + 2.0*fdata(i-1,j ,k,n) + 4.0*fdata(i,j ,k,n) + 2.0*fdata(i+1,j ,k,n)
601 + fdata(i-1,j+1,k,n) + 2.0*fdata(i,j+1,k,n) + fdata(i+1,j+1,k,n))/16.0;
602 else // Interior
603 cdata(I,J,K,n) =
604 (fdata(i-1,j-1,k-1,n) + fdata(i-1,j-1,k+1,n) + fdata(i-1,j+1,k-1,n) + fdata(i-1,j+1,k+1,n) +
605 fdata(i+1,j-1,k-1,n) + fdata(i+1,j-1,k+1,n) + fdata(i+1,j+1,k-1,n) + fdata(i+1,j+1,k+1,n)) / 64.0
606 +
607 (fdata(i,j-1,k-1,n) + fdata(i,j-1,k+1,n) + fdata(i,j+1,k-1,n) + fdata(i,j+1,k+1,n) +
608 fdata(i-1,j,k-1,n) + fdata(i+1,j,k-1,n) + fdata(i-1,j,k+1,n) + fdata(i+1,j,k+1,n) +
609 fdata(i-1,j-1,k,n) + fdata(i-1,j+1,k,n) + fdata(i+1,j-1,k,n) + fdata(i+1,j+1,k,n)) / 32.0
610 +
611 (fdata(i-1,j,k,n) + fdata(i,j-1,k,n) + fdata(i,j,k-1,n) +
612 fdata(i+1,j,k,n) + fdata(i,j+1,k,n) + fdata(i,j,k+1,n)) / 16.0
613 +
614 fdata(i,j,k,n) / 8.0;
615 }
616
617 });
618 }
619 }
620
621 // Copy the fine residual restricted onto the coarse grid
622 // into the final residual.
623 res.ParallelCopy(fine_res_for_coarse,0,0,ncomp,0,0,cgeom.periodicity());
624
625 const int mglev = 0;
626
627 // Sync up ghost nodes
628 amrex::Geometry geom = m_geom[crse_amrlev][mglev];
629 res.setMultiGhost(true);
630 res.FillBoundary();
631 nodalSync(crse_amrlev,mglev, res);
632}
