39int qsortRP(const void *a, const void*b){
40 RP_type *rpa = (RP_type*)a;
41 RP_type *rpb = (RP_type*)b;
42 // sort first by sendRank
43 if(rpa->sendRank < rpb->sendRank)return(-1);
44 if(rpa->sendRank > rpb->sendRank)return( 1);
45 // then by sendBoxID
46 if(rpa->sendBoxID < rpb->sendBoxID)return(-1);
47 if(rpa->sendBoxID > rpb->sendBoxID)return( 1);
48 return(0);
49}

654 for(bj=0;bj<all_grids->levels[level-1]->boxes_in.j/all_grids->levels[level]->boxes_in.j;bj++){

657 int fineBox_j = (all_grids->levels[level-1]->boxes_in.j/all_grids->levels[level]->boxes_in.j)*coarseBox_j + bj;
658 int fineBox_k = (all_grids->levels[level-1]->boxes_in.k/all_grids->levels[level]->boxes_in.k)*coarseBox_k + bk;
659 int fineBoxID = fineBox_i + fineBox_j*all_grids->levels[level-1]->boxes_in.i + fineBox_k*all_grids->levels[level-1]->boxes_in.i*all_grids->levels[level-1]->boxes_in.j;
660 if(all_grids->levels[level-1]->rank_of_box[fineBoxID] != all_grids->levels[level]->my_rank){

663 fineBoxes[numFineBoxesRemote].sendBox = -1; // I don't know the off-node box index

669 fineBoxes[numFineBoxesRemote].k = bk*all_grids->levels[level-1]->box_dim/2;

671 numFineBoxesRemote++;
672 }
673 }}}
674 } // my (coarse) boxes
675 // sort boxes by sendRank(==my rank) then by sendBoxID... ensures the sends and receive buffers are always sorted by sendBoxID...
676 qsort(fineBoxes,numFineBoxesRemote,sizeof(RP_type),qsortRP );
677 // sort the lists of neighboring ranks and remove duplicates...
678 qsort(fineRanks,numFineBoxesRemote,sizeof( int),qsortInt);
679 int numFineRanks=0;
680 int _rank=-1;int neighbor=0;
681 for(neighbor=0;neighbor<numFineBoxesRemote;neighbor++)if(fineRanks[neighbor] != _rank){_rank=fineRanks[neighbor];fineRanks[numFineRanks++]=fineRanks[neighbor];}
682
683 // allocate structures...
684 all_grids->levels[level]->restriction[restrictionType].num_recvs = numFineRanks;
685 all_grids->levels[level]->restriction[restrictionType].recv_ranks = (int*)malloc(numFineRanks*sizeof(int));
686 all_grids->levels[level]->restriction[restrictionType].recv_sizes = (int*)malloc(numFineRanks*sizeof(int));
687 all_grids->levels[level]->restriction[restrictionType].recv_buffers = (double**)malloc(numFineRanks*sizeof(double*));
688 if(numFineRanks>0){
689 if(all_grids->levels[level]->restriction[restrictionType].recv_ranks ==NULL){fprintf(stderr,"malloc failed - all_grids->levels[%d]->restriction[restrictionType].recv_ranks \n",level);exit(0);}
690 if(all_grids->levels[level]->restriction[restrictionType].recv_sizes ==NULL){fprintf(stderr,"malloc failed - all_grids->levels[%d]->restriction[restrictionType].recv_sizes \n",level);exit(0);}
691 if(all_grids->levels[level]->restriction[restrictionType].recv_buffers==NULL){fprintf(stderr,"malloc failed - all_grids->levels[%d]->restriction[restrictionType].recv_buffers\n",level);exit(0);}
692 }
693
694 int elementSize;
695 int restrict_dim_i=-1;
696 int restrict_dim_j=-1;
697 int restrict_dim_k=-1;
698 switch(restrictionType){
699 case RESTRICT_CELL : restrict_dim_i = ( all_grids->levels[level-1]->box_dim/2);
700 restrict_dim_j = ( all_grids->levels[level-1]->box_dim/2);
701 restrict_dim_k = ( all_grids->levels[level-1]->box_dim/2);break;
702 case RESTRICT_FACE_I : restrict_dim_i = (1+all_grids->levels[level-1]->box_dim/2);
703 restrict_dim_j = ( all_grids->levels[level-1]->box_dim/2);
704 restrict_dim_k = ( all_grids->levels[level-1]->box_dim/2);break;
705 case RESTRICT_FACE_J : restrict_dim_i = ( all_grids->levels[level-1]->box_dim/2);
706 restrict_dim_j = (1+all_grids->levels[level-1]->box_dim/2);
707 restrict_dim_k = ( all_grids->levels[level-1]->box_dim/2);break;
708 case RESTRICT_FACE_K : restrict_dim_i = ( all_grids->levels[level-1]->box_dim/2);
709 restrict_dim_j = ( all_grids->levels[level-1]->box_dim/2);
710 restrict_dim_k = (1+all_grids->levels[level-1]->box_dim/2);break;
711 }
712 elementSize = restrict_dim_i*restrict_dim_j*restrict_dim_k;
713
714 double * all_recv_buffers = (double*)malloc(numFineBoxesRemote*elementSize*sizeof(double));
715 if(numFineBoxesRemote*elementSize>0)
716 if(all_recv_buffers==NULL){fprintf(stderr,"malloc failed - restriction/all_recv_buffers\n");exit(0);}
717 memset(all_recv_buffers,0,numFineBoxesRemote*elementSize*sizeof(double)); // DO NOT DELETE... you must initialize to 0 to avoid getting something like 0.0*NaN and corrupting the solve
718 //printf("level=%d, rank=%2d, recv_buffers=%6d\n",level,all_grids->my_rank,numFineBoxesRemote*elementSize*sizeof(double));
719
720 // for each neighbor, construct the unpack list and allocate the MPI recv buffer...
721 for(neighbor=0;neighbor<numFineRanks;neighbor++){
722 int fineBox;
723 int offset = 0;
724 all_grids->levels[level]->restriction[restrictionType].recv_buffers[neighbor] = all_recv_buffers;
725 for(fineBox=0;fineBox<numFineBoxesRemote;fineBox++)if(fineBoxes[fineBox].sendRank==fineRanks[neighbor]){
726 // unpack MPI recv buffer...
727 append_block_to_list( &(all_grids->levels[level]->restriction[restrictionType].blocks[2]),
728 &(all_grids->levels[level]->restriction[restrictionType].allocated_blocks[2]),
729 &(all_grids->levels[level]->restriction[restrictionType].num_blocks[2]),
730 /* dim.i = */ restrict_dim_i,
731 /* dim.j = */ restrict_dim_j,
732 /* dim.k = */ restrict_dim_k,
733 /* read.box = */ -1,
734 /* read.ptr = */ all_grids->levels[level]->restriction[restrictionType].recv_buffers[neighbor],
735 /* read.i = */ offset,
736 /* read.j = */ 0,
737 /* read.k = */ 0,
738 /* read.jStride = */ restrict_dim_i,
739 /* read.kStride = */ restrict_dim_i*restrict_dim_j,
740 /* read.scale = */ 1,
741 /* write.box = */ fineBoxes[fineBox].recvBox,
742 /* write.ptr = */ NULL,
743 /* write.i = */ fineBoxes[fineBox].i,
744 /* write.j = */ fineBoxes[fineBox].j,
745 /* write.k = */ fineBoxes[fineBox].k,
746 /* write.jStride = */ all_grids->levels[level]->my_boxes[fineBoxes[fineBox].recvBox].jStride,
747 /* write.kStride = */ all_grids->levels[level]->my_boxes[fineBoxes[fineBox].recvBox].kStride,
748 /* write.scale = */ 1,
749 /* blockcopy_i = */ BLOCKCOPY_TILE_I, // default
750 /* blockcopy_j = */ BLOCKCOPY_TILE_J, // default
751 /* blockcopy_k = */ BLOCKCOPY_TILE_K, // default
752 /* subtype = */ 0
753 );
754 offset+=elementSize;
755 }
756 all_grids->levels[level]->restriction[restrictionType].recv_ranks[neighbor] = fineRanks[neighbor];
757 all_grids->levels[level]->restriction[restrictionType].recv_sizes[neighbor] = offset;
758 all_recv_buffers+=offset;
759 } // neighbor
760
761 // free temporary storage...
762 free(fineBoxes);
763 free(fineRanks);
764 } // recv/unpack

769 } // level loop
770
771
772 #ifdef USE_MPI
773 for(level=0;level<all_grids->num_levels;level++){
774 all_grids->levels[level]->restriction[restrictionType].requests = NULL;
775 all_grids->levels[level]->restriction[restrictionType].status = NULL;
776 if(level<all_grids->num_levels-1){ // bottom never calls restriction()
777 // by convention, level_f allocates a combined array of requests for both level_f sends and level_c recvs...
778 int nMessages = all_grids->levels[level+1]->restriction[restrictionType].num_recvs + all_grids->levels[level]->restriction[restrictionType].num_sends;
779 all_grids->levels[level]->restriction[restrictionType].requests = (MPI_Request*)malloc(nMessages*sizeof(MPI_Request));
780 all_grids->levels[level]->restriction[restrictionType].status = (MPI_Status *)malloc(nMessages*sizeof(MPI_Status ));
781 }
782 }
783 #endif
784}

788// given a fine grid input, build a hiearchy of MG levels
789// level 0 simply points to fine_grid. All other levels are created
790// rebuild the restriction/interpolation lists for each coarse grid level
791// rebuild the operator on each coarse grid level
792// add extra vectors to the coarse grid once here instead of on every call to the coarse grid solve
793// NOTE, this routine presumes the fine_grid domain is cubical... fine_grid->dim.i==fine_grid->dim.j==fine_grid->dim.k
794// NOTE, as this function is not timed, it has not been optimzied for performance
795void MGBuild(mg_type *all_grids, level_type *fine_grid, double a, double b, int minCoarseGridDim, const MPI_Comm comm){
796 int maxLevels=100; // i.e. maximum problem size is (2^100)^3
797 int nProcs[100];
798 int dim_i[100];
799 int boxes_in_i[100];
800 int box_dim[100];
801 int box_ghosts[100];
802 all_grids->my_rank = fine_grid->my_rank;
803 all_grids->timers.MGBuild = 0;
804 double _timeStartMGBuild = getTime();
805
806 // calculate how deep we can make the v-cycle...
807 int level=1;
808 int coarse_dim = fine_grid->dim.i;
809//if(fine_grid->dim.j<coarse_dim)coarse_dim = fine_grid->dim.j;
810//if(fine_grid->dim.k<coarse_dim)coarse_dim = fine_grid->dim.k;
811 while( (coarse_dim>=2*minCoarseGridDim) && ((coarse_dim&0x1)==0) ){ // grid dimension is even and big enough...
812 level++;
813 coarse_dim = coarse_dim / 2;
814 }if(level<maxLevels)maxLevels=level;
815
816 nProcs[0] = fine_grid->num_ranks;
817 dim_i[0] = fine_grid->dim.i;
818 boxes_in_i[0] = fine_grid->boxes_in.i;
819 box_dim[0] = fine_grid->box_dim;
820 box_ghosts[0] = fine_grid->box_ghosts;
821
822 // build the list of levels...
823 all_grids->levels = (level_type**)malloc(maxLevels*sizeof(level_type*));
824 if(all_grids->levels == NULL){fprintf(stderr,"malloc failed - MGBuild/all_grids->levels\n");exit(0);}
825 all_grids->num_levels=1;
826 all_grids->levels[0] = fine_grid;
827
828
829 // build a table to guide the construction of the v-cycle...
830 int doRestrict=1;if(maxLevels<2)doRestrict=0; // i.e. can't restrict if there is only one level !!!
831 #ifdef USE_UCYCLES
832 while(doRestrict){
833 level = all_grids->num_levels;
834 doRestrict=0;
835 if( (box_dim[level-1] % 2 == 0) ){
836 nProcs[level] = nProcs[level-1];
837 dim_i[level] = dim_i[level-1]/2;
838 box_dim[level] = box_dim[level-1]/2;
839 boxes_in_i[level] = boxes_in_i[level-1];
840 box_ghosts[level] = box_ghosts[level-1];
841 doRestrict = 1;
842 }
843 if(box_dim[level] < box_ghosts[level])doRestrict=0;
844 if(dim_i[level]<minCoarseGridDim)doRestrict=0;
845 if(doRestrict)all_grids->num_levels++;
846 }
847 #else // TRUE V-Cycle...
848 while(doRestrict){
849 level = all_grids->num_levels;
850 doRestrict=0;
851 int fine_box_dim = box_dim[level-1];
852 int fine_nProcs = nProcs[level-1];
853 int fine_dim_i = dim_i[level-1];
854 int fine_boxes_in_i = boxes_in_i[level-1];
855 if( (fine_box_dim % 2 == 0) && (fine_box_dim > MG_AGGLOMERATION_START) && ((fine_box_dim/2)>=stencil_get_radius()) ){ // Boxes are too big to agglomerate
856 nProcs[level] = fine_nProcs;
857 dim_i[level] = fine_dim_i/2;
858 box_dim[level] = fine_box_dim/2; // FIX, verify its not less than the stencil radius
859 boxes_in_i[level] = fine_boxes_in_i;
860 box_ghosts[level] = box_ghosts[level-1];
861 doRestrict = 1;
862 }else
863 if( (fine_boxes_in_i % 2 == 0) && ((fine_box_dim)>=stencil_get_radius()) ){ // 8:1 box agglomeration
864 nProcs[level] = fine_nProcs;
865 dim_i[level] = fine_dim_i/2;
866 box_dim[level] = fine_box_dim;
867 boxes_in_i[level] = fine_boxes_in_i/2;
868 box_ghosts[level] = box_ghosts[level-1];
869 doRestrict = 1;
870 }else
871 if( (coarse_dim != 1) && (fine_dim_i == 2*coarse_dim) && ((fine_dim_i/2)>=stencil_get_radius()) ){ // agglomerate everything
872 nProcs[level] = 1;
873 dim_i[level] = fine_dim_i/2;
874 box_dim[level] = fine_dim_i/2; // FIX, verify its not less than the stencil radius
875 boxes_in_i[level] = 1;
876 box_ghosts[level] = box_ghosts[level-1];
877 doRestrict = 1;
878 }else
879 if( (coarse_dim != 1) && (fine_dim_i == 4*coarse_dim) && ((fine_box_dim/2)>=stencil_get_radius()) ){ // restrict box dimension, and run on fewer ranks
880 nProcs[level] = coarse_dim<fine_nProcs ? coarse_dim : fine_nProcs;
881 dim_i[level] = fine_dim_i/2;
882 box_dim[level] = fine_box_dim/2; // FIX, verify its not less than the stencil radius
883 boxes_in_i[level] = fine_boxes_in_i;
884 box_ghosts[level] = box_ghosts[level-1];
885 doRestrict = 1;
886 }else
887 if( (coarse_dim != 1) && (fine_dim_i == 8*coarse_dim) && ((fine_box_dim/2)>=stencil_get_radius()) ){ // restrict box dimension, and run on fewer ranks
888 nProcs[level] = coarse_dim*coarse_dim<fine_nProcs ? coarse_dim*coarse_dim : fine_nProcs;
889 dim_i[level] = fine_dim_i/2;
890 box_dim[level] = fine_box_dim/2; // FIX, verify its not less than the stencil radius
891 boxes_in_i[level] = fine_boxes_in_i;
892 box_ghosts[level] = box_ghosts[level-1];
893 doRestrict = 1;
894 }else
895 if( (fine_box_dim % 2 == 0) && ((fine_box_dim/2)>=stencil_get_radius()) ){ // restrict box dimension, and run on the same number of ranks
896 nProcs[level] = fine_nProcs;
897 dim_i[level] = fine_dim_i/2;
898 box_dim[level] = fine_box_dim/2; // FIX, verify its not less than the stencil radius
899 boxes_in_i[level] = fine_boxes_in_i;
900 box_ghosts[level] = box_ghosts[level-1];
901 doRestrict = 1;
902 }
903 if(dim_i[level]<minCoarseGridDim)doRestrict=0;
904 if(doRestrict)all_grids->num_levels++;
905 }
906 #endif
907
908
909 // now build all the coarsened levels...
910 for(level=1;level<all_grids->num_levels;level++){
911 all_grids->levels[level] = (level_type*)malloc(sizeof(level_type));
912 if(all_grids->levels[level] == NULL){fprintf(stderr,"malloc failed - MGBuild/doRestrict\n");exit(0);}
913 create_level(all_grids->levels[level],boxes_in_i[level],box_dim[level],box_ghosts[level],all_grids->levels[level-1]->numVectors,all_grids->levels[level-1]->boundary_condition.type,all_grids->levels[level-1]->my_rank,nProcs[level], comm);
914 all_grids->levels[level]->h = 2.0*all_grids->levels[level-1]->h;
915 }
916
917
918 // bottom solver (level = all_grids->num_levels-1) gets extra vectors...
919 create_vectors(all_grids->levels[all_grids->num_levels-1],all_grids->levels[all_grids->num_levels-1]->numVectors + IterativeSolver_NumVectors() );
920
921
922 // build the restriction and interpolation communicators...
923 if(all_grids->my_rank==0){fprintf(stdout,"\n Building restriction and interpolation lists... ");fflush(stdout);}
924 build_restriction(all_grids,RESTRICT_CELL ); // cell-centered
925 build_restriction(all_grids,RESTRICT_FACE_I); // face-centered, normal to i
926 build_restriction(all_grids,RESTRICT_FACE_J); // face-centered, normal to j
927 build_restriction(all_grids,RESTRICT_FACE_K); // face-centered, normal to k
928 build_interpolation(all_grids);
929 if(all_grids->my_rank==0){fprintf(stdout,"done\n");fflush(stdout);}
930
931
932 // build subcommunicators...
933 #ifdef USE_MPI
934 #ifdef USE_SUBCOMM
935 if(all_grids->my_rank==0){fprintf(stdout,"\n");}
936 for(level=1;level<all_grids->num_levels;level++){
937 double comm_split_start = MPI_Wtime();
938 if(all_grids->my_rank==0){fprintf(stdout," Building MPI subcommunicator for level %d... ",level);fflush(stdout);}
939 all_grids->levels[level]->active=0;
940 int ll;for(ll=level;ll<all_grids->num_levels;ll++)if(all_grids->levels[ll]->num_my_boxes>0)all_grids->levels[level]->active=1;
941 MPI_Comm_split(comm, all_grids->levels[level]->active, all_grids->levels[level]->my_rank, &all_grids->levels[level]->MPI_COMM_ALLREDUCE);
942 double comm_split_end = MPI_Wtime();
943 double comm_split_time_send = comm_split_end-comm_split_start;
944 double comm_split_time = 0;
945 MPI_Allreduce(&comm_split_time_send,&comm_split_time,1,MPI_DOUBLE,MPI_MAX,all_grids->levels[level]->MPI_COMM_ALLREDUCE);
946 if(all_grids->my_rank==0){fprintf(stdout,"done (%0.6f seconds)\n",comm_split_time);fflush(stdout);}
947 }
948 #endif
949 #endif
950
951
952 // rebuild various coefficients for the operator... must occur after build_restriction !!!
953 if(all_grids->my_rank==0){fprintf(stdout,"\n");}
954 for(level=1;level<all_grids->num_levels;level++){
955 rebuild_operator(all_grids->levels[level],(level>0)?all_grids->levels[level-1]:NULL,a,b);
956 }
957 if(all_grids->my_rank==0){fprintf(stdout,"\n");}
958
959
960 // quick tests for Poisson, Neumann, etc...
961 for(level=0;level<all_grids->num_levels;level++){
962 all_grids->levels[level]->must_subtract_mean = 0;
963 int alpha_is_zero = (dot(all_grids->levels[level],VECTOR_ALPHA,VECTOR_ALPHA) == 0.0);
964 // For Poisson with Periodic Boundary Conditions, by convention we assume the solution sums to zero. Eliminate any constants from the solution by subtracting the mean.
965 if( (all_grids->levels[level]->boundary_condition.type==BC_PERIODIC) && ((a==0) || (alpha_is_zero==1)) )all_grids->levels[level]->must_subtract_mean = 1;
966 }
967
968
969 all_grids->timers.MGBuild += (double)(getTime()-_timeStartMGBuild);
970}

1052// perform a richardson error analysis to infer the order of the operator/solver
1053void richardson_error(mg_type *all_grids, int levelh, int u_id){
1054 // in FV...
1055 // +-------+ +---+---+ +-------+ +-------+
1056 // | | | a | b | | | |a+b+c+d|
1057 // | u^2h | - +---+---+ = | u^2h | - | --- |
1058 // | | | c | d | | | | 4 |
1059 // +-------+ +---+---+ +-------+ +-------+
1060 //
1061 restriction(all_grids->levels[levelh+1],VECTOR_TEMP,all_grids->levels[levelh ],u_id,RESTRICT_CELL); // temp^2h = R u^h
1062 restriction(all_grids->levels[levelh+2],VECTOR_TEMP,all_grids->levels[levelh+1],u_id,RESTRICT_CELL); // temp^4h = R u^2h
1063 add_vectors(all_grids->levels[levelh+1],VECTOR_TEMP,1.0,u_id,-1.0,VECTOR_TEMP); // temp^2h = u^2h - temp^2h = u^2h - R u^h
1064 add_vectors(all_grids->levels[levelh+2],VECTOR_TEMP,1.0,u_id,-1.0,VECTOR_TEMP); // temp^2h = u^4h - temp^4h = u^4h - R u^2h
1065 double norm_of_u2h_minus_uh = norm(all_grids->levels[levelh+1],VECTOR_TEMP); // || u^2h - R u^h ||max
1066 double norm_of_u4h_minus_u2h = norm(all_grids->levels[levelh+2],VECTOR_TEMP); // || u^4h - R u^2h ||max
1067 // estimate the error^h using ||u^2h - R u^h||
1068 if(all_grids->my_rank==0){fprintf(stdout," h=%0.15e ||error||=%0.15e\n",all_grids->levels[levelh]->h,norm_of_u2h_minus_uh);fflush(stdout);}
1069 // log( ||u^4h - R u^2h|| / ||u^2h - R u^h|| ) / log(2) is an estimate of the order of the method (e.g. 4th order)
1070 if(all_grids->my_rank==0){fprintf(stdout," order=%0.3f\n",log(norm_of_u4h_minus_u2h / norm_of_u2h_minus_uh) / log(2) );fflush(stdout);}
1071}

1338void MGPCG(mg_type *all_grids, int onLevel, int x_id, int F_id, double a, double b, double dtol, double rtol){
1339 // Algorithm 9.1 in Iterative Methods for Sparse Linear Systems(Yousef Saad) using a MG V-Cycle as M^{-1}
1340 level_type * level = all_grids->levels[onLevel];
1341 if(!level->active)return;
1342
1343 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1344 // CG with a MG preconditioner, every level needs 3 extra vectors (p, Ap, z)
1345 int l;
1346 for(l=0;l<all_grids->num_levels;l++){
1347 create_vectors(all_grids->levels[l],VECTORS_RESERVED+3);
1348 }
1349
1350 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1351 // Test for Poisson with Periodic BCs
1352 for(l=0;l<all_grids->num_levels;l++){
1353 if(all_grids->levels[l]->must_subtract_mean==-1){
1354 all_grids->levels[l]->must_subtract_mean=0;
1355 int alpha_is_zero = (dot(all_grids->levels[l],VECTOR_ALPHA,VECTOR_ALPHA) == 0.0);
1356 if( (all_grids->levels[l]->boundary_condition.type==BC_PERIODIC) && ((a==0) || (alpha_is_zero)) )all_grids->levels[l]->must_subtract_mean = 1;
1357 }
1358 }
1359
1360 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1361 int r_id = VECTOR_R;
1362 int p_id = VECTORS_RESERVED+0;
1363 int Ap_id = VECTORS_RESERVED+1;
1364 int z_id = VECTORS_RESERVED+2;
1365
1366 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1367 #ifdef _OPENMP
1368 double MGPCG_Start_Time = omp_get_wtime();
1369 #elif USE_MPI
1370 double MGPCG_Start_Time = MPI_Wtime();
1371 #endif
1372 if(all_grids->levels[onLevel]->my_rank==0){fprintf(stdout,"MGPCG... ");}
1373 double _timeStartMGSolve = getTime();
1374 all_grids->MGSolves_performed++;
1375 int jMax=20;
1376 int j=0;
1377 int CGFailed = 0;
1378 int CGConverged = 0;
1379
1380 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1381 zero_vector(level,x_id); // x[] = 0
1382 residual(level,r_id,x_id,F_id,a,b); // r[] = F_id[] - A(x_id)
1383 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1384 if(level->must_subtract_mean == 1){
1385 double mean_of_r = mean(level,r_id);
1386 shift_vector(level,r_id,r_id,-mean_of_r);
1387 }
1388 //- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
1389 double norm_of_r0 = norm(level,r_id); // the norm of the initial residual...
1390 if(norm_of_r0 == 0.0){CGConverged=1;} // entered CG with exact solution
1391 level->vcycles_from_this_level++; //
