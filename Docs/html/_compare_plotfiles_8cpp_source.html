11void
12PrintUsage (const char* progName)
13{
14 Print() << std::endl
15 << "This utility performs a diff operation between two" << std::endl
16 << "plotfiles that have the same geometrical domain and nodality" << std::endl
17 << "(supports all nodality types; cell, face, edge, node)" << std::endl
18 << "but possibly a factor of refinement between the cells," << std::endl
19 << "and outputs the L0, L1, and L2 norms" << std::endl
20 << "L1 = sum(|diff_ijk|)/npts_coarsedomain" << std::endl
21 << "L2 = sqrt[sum(diff_ijk^2)]/sqrt(npts_coarsedomain)" << std::endl
22 << "(only single-level supported)" << std::endl << std::endl;
23
24 Print() << "Usage:" << '\n';
25 Print() << progName << '\n';
26 Print() << " infile1 = inputFileName1" << '\n';
27 Print() << " reffile = refinedPlotFile" << '\n';
28 Print() << " diffile = differenceFileName" << '\n';
29 Print() << " (If not specified no file is written)" << '\n' << '\n';
30
31 Print() << "You can either point to the plotfile base directory itself, e.g." << std::endl
32 << " infile=plt00000" << std::endl
33 << "Or the raw data itself, e.g." << std::endl
34 << " infile=plt00000/Level_0/Cell" << std::endl
35 << "the latter is useful for some applications that dump out raw" << std::endl
36 << "nodal data within a plotfile directory." << std::endl
37 << "The program will first try appending 'Level_0/Cell'" << std::endl
38 << "onto the specified filenames." << std::endl
39 << "If that _H file doesn't exist, it tries using the full specified name" << std::endl << std::endl;
40
41 exit(1);
42}

182 const Array4<Real >& coarse = mf_c2.array(mfi);

192#endif
193 for (int jj=0; jj<rr_j; ++jj) {
194 for (int ii=0; ii<rr_i; ++ii) {

217 Real norm0 = mf_c2.norm0(i);
218 Real norm1 = mf_c2.norm1(i,period);
219 Real norm2 = mf_c2.norm2(i,period);
220 Print() << "(comp,L0,L1,L2) " << i << " "
