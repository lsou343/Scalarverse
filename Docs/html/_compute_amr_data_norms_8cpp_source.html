18void
19ComputeAmrDataNorms (AmrData& amrData,
20 Vector<Real>& norm0,
21 Vector<Real>& norm1,
22 Vector<Real>& norm2,
23 bool verbose)
24{
25 std::string oFile, iFileDir, oFileDir;
26
27 if (verbose)
28 {
29 ParmParse pp;
30 pp.query("outfile", oFile);
31 if (oFile.empty())
32 amrex::Abort("You must specify `outfile' if run in verbose mode");
33 }
34
35 int finestLevel = amrData.FinestLevel();
36 int nComp = amrData.NComp();
37
38 norm0.clear(); norm0.resize(nComp,0.0);
39 norm1.clear(); norm1.resize(nComp,0.0);
40 norm2.clear(); norm2.resize(nComp,0.0);
41
42 Vector<int> refMult(finestLevel + 1, 1);
43 for (int iLevel=finestLevel-1; iLevel>=0; --iLevel)
44 {
45 int ref_ratio = amrData.RefRatio()[iLevel];
46 int vol = 1;
47 for (int i=0; i<BL_SPACEDIM; ++i)
48 vol *= ref_ratio;
49 for (int jLevel=0; jLevel<=iLevel; ++jLevel)
50 refMult[jLevel] *= vol;
51 }
52
53 //
54 // Compute the norms
55 //
56 long total_volume = 0;
57 Vector<MultiFab*> error(finestLevel+1);
58
59 for (int iLevel = finestLevel; iLevel>=0; --iLevel)
60 {
61 const BoxArray& ba = amrData.boxArray(iLevel);
62 DistributionMapping dm {ba};
63
64 error[iLevel] = new MultiFab(ba, dm, nComp, 0);
65 for (int iComp=0; iComp<nComp; ++iComp)
66 {
67 MultiFab& data = amrData.GetGrids(iLevel,iComp);
68 error[iLevel]->copy(data,0,iComp,1);
69 }
70
71 // Zero out the error covered by fine grid
72 long covered_volume = 0;
73 if (iLevel != finestLevel)
74 {
75 int ref_ratio = amrData.RefRatio()[iLevel];
76 BoxArray baF = ::BoxArray(amrData.boxArray(iLevel+1)).coarsen(ref_ratio);
77 for (MFIter mfi(*error[iLevel]); mfi.isValid(); ++mfi)
78 {
79 for (int iGrid=0; iGrid<baF.size(); ++iGrid)
80 {
81 Box ovlp = baF[iGrid] & mfi.validbox();
82 if (ovlp.ok())
83 {
84 (*error[iLevel])[mfi].setVal(0.0, ovlp, 0, nComp);
85 covered_volume += ovlp.numPts()*refMult[iLevel];
86 }
87 }
88 }
89 ParallelDescriptor::ReduceLongSum(covered_volume);
90 }
91
92 // Compute volume at this level
93 Real level_volume = 0.0;
94 for (int iGrid=0; iGrid<ba.size(); ++iGrid)
95 level_volume += ba[iGrid].numPts()*refMult[iLevel];
96 level_volume -= covered_volume;
97
98 if (level_volume > 0.0)
99 {
100 // Convert volume in numPts to volume in number of fine cells
101 total_volume += long(level_volume);
102
103 // Get norms at this level
104 Vector<Real> n0(nComp,0.0), n1(nComp,0.0), n2(nComp,0.0);
105 for (MFIter mfi(*error[iLevel]); mfi.isValid(); ++mfi)
106 {
107 FArrayBox& fab = (*error[iLevel])[mfi];
108 const Box& fabbox = mfi.validbox();
109 FArrayBox vwFab(fabbox,nComp);
110 FArrayBox vwFabSqrd(fabbox,nComp);
111
112 // compute volume-weighted error
113 vwFab.copy(fab,0,0,nComp);
114 vwFab.mult(refMult[iLevel]);
115
116 vwFabSqrd.copy(fab,0,0,nComp);
117 vwFabSqrd.mult(fab,0,0,nComp);
118 vwFabSqrd.mult(refMult[iLevel]);
119
120 for (int iComp=0; iComp<nComp; ++iComp)
121 {
122 n0[iComp] = std::max(n0[iComp], fab.norm(fabbox, 0, iComp, 1));
123 n1[iComp] += vwFab.norm(fabbox, 1, iComp, 1);
124 n2[iComp] += vwFabSqrd.norm(fabbox, 1, iComp, 1);
125 }
126 }
127
128 // Do necessary communication, then blend this level's norms
129 // in with the running global values
130 for (int iComp=0; iComp<nComp; ++iComp)
131 {
132 ParallelDescriptor::ReduceRealMax(n0[iComp]);
133 ParallelDescriptor::ReduceRealSum(n1[iComp]);
134 ParallelDescriptor::ReduceRealSum(n2[iComp]);
135
136 norm0[iComp] = std::max(norm0[iComp],n0[iComp]);
137 norm1[iComp] += n1[iComp];
138 norm2[iComp] += n2[iComp];
139 }
140 }
141 }
142 for (int iComp=0; iComp<nComp; ++iComp)
143 {
144 norm1[iComp] /= total_volume;
145 norm2[iComp] = sqrt(norm2[iComp] / total_volume);
146 }
147
148 if (ParallelDescriptor::IOProcessor() && verbose)
149 {
150 std::cout << "Writing zeroed state to " << oFile << '\n';
151 WritePlotFile(error, amrData, oFile, verbose);
152 }
153
154 // Clean up memory
155 for (int iLevel = 0; iLevel <= finestLevel; ++iLevel)
156 delete error[iLevel];
157}
