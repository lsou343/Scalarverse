856
857#ifdef CONST_SPECIES

863 int iufs = FirstSpec;
864 int nspec = NumSpec;

866#pragma omp parallel

869 {
870 const auto state = S_new.array(mfi);
871
872 const Box& tbx = mfi.tilebox();

874 reduce_op.eval(tbx, reduce_data,
875 [state,nspec,iden,iufs]

878 Real sum = state(i,j,k,iufs);
879 for (int n = 1; n < nspec; n++)
880 sum += state(i,j,k,iufs+n);
881
882 sum /= state(i,j,k,iden);
883
884 Real x = amrex::Math::abs(amrex::Math::abs(sum) - 1.);

886 });
