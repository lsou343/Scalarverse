201void
202NodalProjector::project ( Real a_rtol, Real a_atol )
203{
204 BL_PROFILE("NodalProjector::project");
205 AMREX_ALWAYS_ASSERT(!m_need_bcs);
206
207 if (m_verbose > 0)
208 amrex::Print() << "Nodal Projection:" << std::endl;
209
210 //
211 // Average fine grid velocity values down to the coarse grid
212 // By doing this operation at this time we:
213 //
214 // 1) fill regions covered by a finer grid with some "valid" data, i.e. not NaNs
215 // for example. This is required internally by MLMG.
216 // 2) make the velocity field ready for projection on the entirety of each level.
217 //
218 averageDown(m_vel);
219
220 // Set matrix coefficients
221 for (int lev = 0; lev < m_sigma.size(); ++lev)
222 {
223 m_linop -> setSigma(lev, *m_sigma[lev]);
224 }
225
226 // Compute RHS if necessary
227 if (!m_has_rhs)
228 {
229 computeRHS( GetVecOfPtrs(m_rhs), m_vel, m_S_cc, m_S_nd );
230 }
231
232 // Print diagnostics
233 if (m_verbose > 0)
234 {
235 amrex::Print() << " >> Before projection:" << std::endl;
236 printInfo();
237 amrex::Print() << std::endl;
238 }
239
240 // Solve
241 // phi comes out already averaged-down and ready to be used by caller if needed
242 m_mlmg -> solve( GetVecOfPtrs(m_phi), GetVecOfConstPtrs(m_rhs), a_rtol, a_atol );
243
244 // Get fluxes -- fluxes = - sigma * grad(phi)
245 m_mlmg -> getFluxes( GetVecOfPtrs(m_fluxes) );
246
247 // At this time, the fluxes are "correct" only on regions not covered by finer grids.
248 // We average the fluxes down so that they are "correct" everywhere in each level.
249 // This is necessary because the caller can access grad(phi) and may use it for
250 // computations involving the whole level.
251 averageDown(GetVecOfPtrs(m_fluxes));
252
253 // Compute sync residual BEFORE performing projection
254 computeSyncResidual();
255
256 // Perform projection
257 for (int lev(0); lev < m_phi.size(); ++lev)
258 {
259 if (m_has_alpha)
260 {
261 // fluxes -> fluxes/alpha = - ( sigma / alpha ) * grad(phi)
262 for (int n = 0; n < AMREX_SPACEDIM; ++n)
263 {
264 MultiFab::Divide( m_fluxes[lev], *m_alpha[lev], 0, n, 1, 0 );
265 }
266 }
267
268 //
269 // vel = vel + fluxes = vel - ( sigma / alpha ) * grad(phi)
270 //
271 // Since we already averaged-down the velocity field and -grad(phi),
272 // we perform the projection by simply adding the two of them.
273 // In virtue of the linearity of the operations involved, this is equivalent
274 // to averaging down the velocity only once AFTER summing -grad(phi)
275 //
276 MultiFab::Add( *m_vel[lev], m_fluxes[lev], 0, 0, AMREX_SPACEDIM, 0);
277
278 // set m_fluxes = grad(phi)
279 m_fluxes[lev].mult(-1.0);
280 for (int n = 0; n < AMREX_SPACEDIM; ++n)
281 {
282 if (m_has_alpha)
283 {
284 MultiFab::Multiply(m_fluxes[lev], *m_alpha[lev], 0, n, 1, 0);
285 }
286 MultiFab::Divide(m_fluxes[lev], *m_sigma[lev], 0, n, 1, 0);
287 }
288
289 }
290
291 // Print diagnostics
292 if ( (m_verbose > 0) && (!m_has_rhs))
293 {
294 computeRHS( GetVecOfPtrs(m_rhs), m_vel, m_S_cc, m_S_nd );
295 amrex::Print() << " >> After projection:" << std::endl;
296 printInfo();
297 amrex::Print() << std::endl;
298 }

407// Set velocity in ghost cells to zero except for inflow
408// 1) At non-inflow faces, the normal component of velocity will be completely zero'd
409// 2) The normal velocity at corners -- even periodic corners -- just outside inflow faces will be zero'd
410void
411NodalProjector::setCoarseBoundaryVelocityForSync ()
412{
413
414 const BoxArray& grids = m_vel[0]->boxArray();
415 const Box& domainBox = m_geom[0].Domain();
416
417 for (int idir=0; idir < AMREX_SPACEDIM; idir++)
418 {
419 if (m_bc_lo[idir] != LinOpBCType::inflow && m_bc_hi[idir] != LinOpBCType::inflow)
420 {
421 m_vel[0]->setBndry(0.0, idir, 1);
422 }
423 else
424 {
425#ifdef _OPENMP
426#pragma omp parallel if (Gpu::notInLaunchRegion())
427#endif
428 for (MFIter mfi(*m_vel[0]); mfi.isValid(); ++mfi)
429 {
430 int i = mfi.index();
431 FArrayBox& v_fab = (*m_vel[0])[mfi];
432
433 const Box& reg = grids[i];
434 const Box& bxg1 = amrex::grow(reg,1);
435 BoxList bxlist(reg);
436
437 if (m_bc_lo[idir] == LinOpBCType::inflow && reg.smallEnd(idir) == domainBox.smallEnd(idir))
438 {
439 Box bx; // bx is the region we *protect* from zero'ing
440 bx = amrex::adjCellLo(reg, idir);
441 bxlist.push_back(bx);
442 }
443
444 if (m_bc_hi[idir] == LinOpBCType::inflow && reg.bigEnd(idir) == domainBox.bigEnd(idir)) {
445 Box bx; // bx is the region we *protect* from zero'ing
446 bx = amrex::adjCellHi(reg, idir);
447 bxlist.push_back(bx);
448 }
449
450 BoxList bxlist2 = amrex::complementIn(bxg1, bxlist);
451
452 for (BoxList::iterator it=bxlist2.begin(); it != bxlist2.end(); ++it)
453 {
454 Box ovlp = *it & v_fab.box();
455 if (ovlp.ok())
