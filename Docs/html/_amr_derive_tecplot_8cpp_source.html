28struct Node
29{
30 enum typeEnum{INIT=0, COVERED=1, VALID=2};
31 Node()
32 : level(-1), iv(IntVect(AMREX_D_DECL(-1,-1,-1))), grid(-1), type(Node::INIT) {}
33 Node(const IntVect& idx, int lev, int grd, typeEnum typ = INIT)
34 : level(lev), iv(idx), grid(grd), type(typ) {}
35 inline bool operator< (const Node& rhs) const
36 {
37 if (level < rhs.level) return true;
38 if ((level == rhs.level) && iv < rhs.iv) return true;
39 return false;
40 }
41 inline bool operator!= (const Node& rhs) const
42 {
43 return ((*this) < rhs || rhs < (*this));
44 }
45 int level;
46 IntVect iv;
47 int grid;
48 typeEnum type;
49};

127static
128BoxArray
129GetBndryCells (const BoxArray& ba,
130 int ngrow,
131 const Geometry& geom)
132{
133 //
134 // First get list of all ghost cells.
135 //
136 BoxList gcells, bcells;
137
138 for (int i = 0; i < ba.size(); ++i)
139 gcells.join(amrex::boxDiff(amrex::grow(ba[i],ngrow),ba[i]));
140 //
141 // Now strip out intersections with original BoxArray.
142 //
143 for (BoxList::const_iterator it = gcells.begin(); it != gcells.end(); ++it)
144 {
145 std::vector< std::pair<int,Box> > isects = ba.intersections(*it);
146
147 if (isects.empty())
148 bcells.push_back(*it);
149 else
150 {
151 //
152 // Collect all the intersection pieces.
153 //
154 BoxList pieces;
155 for (int i = 0; i < isects.size(); i++)
156 pieces.push_back(isects[i].second);
157 BoxList leftover = amrex::complementIn(*it,pieces);
158 bcells.catenate(leftover);
159 }
160 }
161 //
162 // Now strip out overlaps.
163 //
164 gcells.clear();
165 gcells = amrex::removeOverlap(bcells);
166 bcells.clear();
167
168 if (geom.isAnyPeriodic())
169 {
170 Vector<IntVect> pshifts(27);
171
172 const Box& domain = geom.Domain();
173
174 for (BoxList::const_iterator it = gcells.begin(); it != gcells.end(); ++it)
175 {
176 if (!domain.contains(*it))
177 {
178 //
179 // Add in periodic ghost cells shifted to valid region.
180 //
181 geom.periodicShift(domain, *it, pshifts);
182
183 for (int i = 0; i < pshifts.size(); i++)
184 {
185 const Box& shftbox = *it + pshifts[i];
186
187 const Box& ovlp = domain & shftbox;
188 BoxList bl = amrex::complementIn(ovlp,BoxList(ba));
189 bcells.catenate(bl);
190 }
191 }
192 }
193
194 gcells.catenate(bcells);
195 }
196
197 return BoxArray(gcells);
198}
