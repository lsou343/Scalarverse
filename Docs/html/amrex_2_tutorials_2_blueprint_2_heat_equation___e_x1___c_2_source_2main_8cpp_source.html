34void main_main ()
35{
36 // What time is it now? We'll use this to compute total run time.
37 Real strt_time = ParallelDescriptor::second();
38
39 // AMREX_SPACEDIM: number of dimensions
40 int n_cell, max_grid_size, nsteps, plot_int;
41 Vector<int> is_periodic(AMREX_SPACEDIM,1); // periodic in all direction by default
42
43 // inputs parameters
44 {
45 // ParmParse is way of reading inputs from the inputs file
46 ParmParse pp;
47
48 // We need to get n_cell from the inputs file - this is the number of cells on each side of
49 // a square (or cubic) domain.
50 pp.get("n_cell",n_cell);
51
52 // The domain is broken into boxes of size max_grid_size
53 pp.get("max_grid_size",max_grid_size);
54
55 // Default plot_int to -1, allow us to set it to something else in the inputs file
56 // If plot_int < 0 then no plot files will be writtenq
57 plot_int = -1;
58 pp.query("plot_int",plot_int);
59
60 // Default nsteps to 0, allow us to set it to something else in the inputs file
61 nsteps = 10;
62 pp.query("nsteps",nsteps);
63
64 pp.queryarr("is_periodic", is_periodic);
65 }
66
67 // make BoxArray and Geometry
68 BoxArray ba;
69 Geometry geom;
70 {
71 IntVect dom_lo(AMREX_D_DECL( 0, 0, 0));
72 IntVect dom_hi(AMREX_D_DECL(n_cell-1, n_cell-1, n_cell-1));
73 Box domain(dom_lo, dom_hi);
74
75 // Initialize the boxarray "ba" from the single box "bx"
76 ba.define(domain);
77 // Break up boxarray "ba" into chunks no larger than "max_grid_size" along a direction
78 ba.maxSize(max_grid_size);
79
80 // This defines the physical box, [-1,1] in each direction.
81 RealBox real_box({AMREX_D_DECL(-1.0,-1.0,-1.0)},
82 {AMREX_D_DECL( 1.0, 1.0, 1.0)});
83
84 // This defines a Geometry object
85 geom.define(domain,&real_box,CoordSys::cartesian,is_periodic.data());
86 }
87
88 // Nghost = number of ghost cells for each array
89 int Nghost = 1;
90
91 // Ncomp = number of components for each array
92 int Ncomp = 1;
93
94 // How Boxes are distrubuted among MPI processes
95 DistributionMapping dm(ba);
96
97 // we allocate two phi multifabs; one will store the old state, the other the new.
98 MultiFab phi_old(ba, dm, Ncomp, Nghost);
99 MultiFab phi_new(ba, dm, Ncomp, Nghost);
100
101 // Initialize phi_new by calling a Fortran routine.
102 // MFIter = MultiFab Iterator
103 for ( MFIter mfi(phi_new); mfi.isValid(); ++mfi )
104 {
105 const Box& bx = mfi.validbox();
106
107 init_phi(BL_TO_FORTRAN_BOX(bx),
108 BL_TO_FORTRAN_ANYD(phi_new[mfi]),
109 geom.CellSize(), geom.ProbLo(), geom.ProbHi());
110 }
111
112 // compute the time step
113 const Real* dx = geom.CellSize();
114 Real dt = 0.9*dx[0]*dx[0] / (2.0*AMREX_SPACEDIM);
115
116 // time = starting time in the simulation
117 Real time = 0.0;
118
119 int rank = ParallelDescriptor::MyProc(); // Return the rank
120 int ntasks = ParallelDescriptor::NProcs(); // Return the number of processes
121
123 // Setup Ascent
125 // Create an instance of Ascent
126 Ascent ascent;
127 Node open_opts;
128
129 // for the MPI case, provide the mpi comm
130#ifdef BL_USE_MPI
131 open_opts["mpi_comm"] = MPI_Comm_c2f(ParallelDescriptor::Communicator());
132#endif
133
134 ascent.open(open_opts);
135
136 // Write a plotfile of the initial data if plot_int > 0 (plot_int was defined in the inputs file)
137
138 if (plot_int > 0)
139 {
140 int n = 0;
141 const std::string& pltfile = amrex::Concatenate("plt",n,5);
142 WriteSingleLevelPlotfile(pltfile, phi_new, {"phi"}, geom, time, 0);
143
145 // Wrap our AMReX Mesh into a Conduit Mesh Blueprint Tree
147 conduit::Node bp_mesh;
148 SingleLevelToBlueprint( phi_new, {"phi"}, geom, time, n, bp_mesh);
149
150
152 // Save the Blueprint Mesh to a set of files that we can
153 // view in VisIt.
154 // (For debugging and to demonstrate how to do this w/o Ascent)
156 WriteBlueprintFiles( bp_mesh, "bp_heateq_ex_", n);
157
159 // Render with Ascent
161
162 Node scenes;
163 scenes["s1/plots/p1/type"] = "pseudocolor";
164 scenes["s1/plots/p1/field"] = "phi";
165 // Set the output file name (ascent will add ".png")
166 const std::string& png_out = amrex::Concatenate("ascent_render_",0,5);
167 scenes["s1/image_prefix"] = png_out;
168
169 // setup actions
170 Node actions;
171 Node &add_act = actions.append();
172 add_act["action"] = "add_scenes";
173 add_act["scenes"] = scenes;
174 actions.append()["action"] = "execute";
175 actions.append()["action"] = "reset";
176
177
178 ascent.publish(bp_mesh);
179 ascent.execute(actions);
180 }
181
182 // build the flux multifabs
183 std::array<MultiFab, AMREX_SPACEDIM> flux;
184 for (int dir = 0; dir < AMREX_SPACEDIM; dir++)
185 {
186 // flux(dir) has one component, zero ghost cells, and is nodal in direction dir
187 BoxArray edge_ba = ba;
188 edge_ba.surroundingNodes(dir);
189 flux[dir].define(edge_ba, dm, 1, 0);
190 }
191
192
193
194
195 for (int n = 1; n <= nsteps; ++n)
196 {
197 MultiFab::Copy(phi_old, phi_new, 0, 0, 1, 0);
198
199 // new_phi = old_phi + dt * (something)
200 advance(phi_old, phi_new, flux, dt, geom);
201 time = time + dt;
202
203
204 // Tell the I/O Processor to write out which step we're doing
205 amrex::Print() << "Advanced step " << n << "\n";
206
207 // Write a plotfile of the current data and write a conduit blueprint
208 // file as well
209 if ( plot_int >= 0 && n%plot_int == 0)
210 {
211 const std::string& pltfile = amrex::Concatenate("plt",n,5);
212 WriteSingleLevelPlotfile(pltfile, phi_new, {"phi"}, geom, time, n);
213
215 // Wrap our AMReX Mesh into a Conduit Mesh Blueprint Tree
217 conduit::Node bp_mesh;
218 SingleLevelToBlueprint( phi_new, {"phi"}, geom, time, n, bp_mesh);
219
220
222 // Save the Blueprint Mesh to a set of files that we can
223 // view in VisIt.
224 // (For debugging and to demonstrate how to do this w/o Ascent)
226 WriteBlueprintFiles( bp_mesh, "bp_heateq_ex_", n);
227
229 // Render with Ascent
231
232 // add a scene with a pseudocolor plot
233 Node scenes;
234 scenes["s1/plots/p1/type"] = "pseudocolor";
235 scenes["s1/plots/p1/field"] = "phi";
236 // Set the output file name (ascent will add ".png")
237 const std::string& png_out = amrex::Concatenate("ascent_render_",n,5);
238 scenes["s1/image_prefix"] = png_out;
239
240 // setup actions
241 Node actions;
242 Node &add_act = actions.append();
243 add_act["action"] = "add_scenes";
244 add_act["scenes"] = scenes;
245 actions.append()["action"] = "execute";
246 actions.append()["action"] = "reset";
247
248
249 ascent.publish(bp_mesh);
250 ascent.execute(actions);
251
252 }
253 }
254
255 ascent.close();
256
257 // Call the timer again and compute the maximum difference between the start time and stop time
258 // over all processors
259 Real stop_time = ParallelDescriptor::second() - strt_time;
260 const int IOProc = ParallelDescriptor::IOProcessorNumber();
261 ParallelDescriptor::ReduceRealMax(stop_time,IOProc);
262
263 // Tell the I/O Processor to write out the "run time"
264 amrex::Print() << "Run time = " << stop_time << std::endl;
265}
