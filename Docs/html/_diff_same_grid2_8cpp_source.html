50static
51bool
52amrDatasHaveSameDerives(const AmrData& amrd1,
53 const AmrData& amrd2)
54{
55 const Vector<std::string>& derives1 = amrd1.PlotVarNames();
56 const Vector<std::string>& derives2 = amrd2.PlotVarNames();
57 int length = derives1.size();
58 if (length != derives2.size())
59 return false;
60 for (int i=0; i<length; ++i)
61 if (derives1[i] != derives2[i])
62 return false;
63 return true;
64}
