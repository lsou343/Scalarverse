441void
442DistributionMapping::RoundRobinProcessorMap (const BoxArray& boxes, int nprocs)
443{
444 BL_ASSERT(boxes.size() > 0);
445 BL_ASSERT(m_ref->m_pmap.size() == boxes.size());
446 //
447 // Create ordering of boxes from largest to smallest.
448 // When we round-robin the boxes we want to go from largest
449 // to smallest box, starting from the CPU having the least
450 // amount of FAB data to the one having the most. This "should"
451 // help even out the FAB data distribution when running on large
452 // numbers of CPUs, where the lower levels of the calculation are
453 // using RoundRobin to lay out fewer than NProc boxes across
454 // the CPUs.
455 //
456 std::vector<LIpair> LIpairV;
457
458 const int N = boxes.size();
459
460 LIpairV.reserve(N);
461
462 for (int i = 0; i < N; ++i)
463 {
464 LIpairV.push_back(LIpair(boxes[i].numPts(),i));
465 }
466
467 Sort(LIpairV, true);
468
469 RoundRobinDoIt(boxes.size(), nprocs, &LIpairV);
470}

558void
559knapsack (const std::vector<Long>& wgts,
560 int nprocs,
561 std::vector< std::vector<int> >& result,
562 Real& efficiency,
563 bool do_full_knapsack,
564 int nmax)
565{
566 BL_PROFILE("knapsack()");
567
568 //
569 // Sort balls by size largest first.
570 //
571 result.resize(nprocs);
572
573 Vector<WeightedBox> lb;
574 lb.reserve(wgts.size());
575 for (unsigned int i = 0, N = wgts.size(); i < N; ++i)
576 {
577 lb.push_back(WeightedBox(i, wgts[i]));
578 }
579 std::sort(lb.begin(), lb.end());
580 //
581 // For each ball, starting with heaviest, assign ball to the lightest bin.
582 //
583 std::priority_queue<WeightedBoxList> wblq;
584 Vector<std::unique_ptr<Vector<WeightedBox> > > raii_vwb(nprocs);
585 for (int i = 0; i < nprocs; ++i)
586 {
587 raii_vwb[i].reset(new Vector<WeightedBox>);
588 wblq.push(WeightedBoxList(raii_vwb[i].get()));
589 }
590 Vector<WeightedBoxList> wblv;
591 wblv.reserve(nprocs);
592 for (unsigned int i = 0, N = wgts.size(); i < N; ++i)
593 {
594 if (!wblq.empty()) {
595 WeightedBoxList wbl = wblq.top();
596 wblq.pop();
597 wbl.push_back(lb[i]);
598 if (wbl.size() < nmax) {
599 wblq.push(wbl);
600 } else {
601 wblv.push_back(wbl);
602 }
603 } else {
604 int ip = static_cast<int>(i) % nprocs;
605 wblv[ip].push_back(lb[i]);
606 }
607 }
608
609 Real max_weight = 0;
610 Real sum_weight = 0;
611 for (auto const& wbl : wblv)
612 {
613 Real wgt = wbl.weight();
614 sum_weight += wgt;
615 max_weight = std::max(wgt, max_weight);
616 }
617
618 while (!wblq.empty())
619 {
620 WeightedBoxList wbl = wblq.top();
621 wblq.pop();
622 if (wbl.size() > 0) {
623 Real wgt = wbl.weight();
624 sum_weight += wgt;
625 max_weight = std::max(wgt, max_weight);
626 wblv.push_back(wbl);
627 }
628 }
629
630 efficiency = sum_weight/(nprocs*max_weight);
631
632 std::sort(wblv.begin(), wblv.end());
633
634 if (efficiency < max_efficiency && do_full_knapsack
635 && wblv.size() > 1 && wblv.begin()->size() > 1)
636 {
637 BL_PROFILE_VAR("knapsack()swap", swap);
638top: ;
639
640 if (efficiency < max_efficiency && wblv.begin()->size() > 1)
641 {
642 auto bl_top = wblv.begin();
643 auto bl_bottom = wblv.end()-1;
644 Long w_top = bl_top->weight();
645 Long w_bottom = bl_bottom->weight();
646 for (auto ball_1 = bl_top->begin(); ball_1 != bl_top->end(); ++ball_1)
647 {
648 for (auto ball_2 = bl_bottom->begin(); ball_2 != bl_bottom->end(); ++ball_2)
649 {
650 // should we swap ball 1 and ball 2?
651 Long dw = ball_1->weight() - ball_2->weight();
652 Long w_top_new = w_top - dw;
653 Long w_bottom_new = w_bottom + dw;
654 if (w_top_new < w_top && w_bottom_new < w_top)
655 {
656 std::swap(*ball_1, *ball_2);
657 bl_top->addWeight(-dw);
658 bl_bottom->addWeight(dw);
659
660 if (bl_top+1 == bl_bottom) // they are next to each other
661 {
662 if (*bl_bottom < *bl_top) {
663 std::swap(*bl_top, *bl_bottom);
664 }
665 }
666 else
667 {
668 // bubble up
669 auto it = std::lower_bound(bl_top+1, bl_bottom, *bl_bottom);
670 std::rotate(it, bl_bottom, bl_bottom+1);
671
672 // sink down
673 it = std::lower_bound(bl_top+1, bl_bottom+1, *bl_top);
674 std::rotate(bl_top, bl_top+1, it);
675 }
676
677 max_weight = bl_top->weight();
678 efficiency = sum_weight / (nprocs*max_weight);
679 goto top;
680 }
681 }
682 }
683 }
684
685 BL_ASSERT(std::is_sorted(wblv.begin(), wblv.end()));
686 }
687
688 for (int i = 0, N = wblv.size(); i < N; ++i)
689 {
690 const WeightedBoxList& wbl = wblv[i];
691
692 result[i].reserve(wbl.size());
693 for (auto const& wb : wbl)
694 {
695 result[i].push_back(wb.boxid());
696 }
697 }

973 AMREX_FORCE_INLINE
974 SFCToken makeSFCToken (int box_index, IntVect const& iv)
975 {
976 SFCToken token;
977 token.m_box = box_index;
978
979#if (AMREX_SPACEDIM == 3)
980
981 constexpr int imin = -(1 << 29);
982 AMREX_ASSERT_WITH_MESSAGE(AMREX_D_TERM(iv[0] >= imin && iv[0] < -imin,
983 && iv[1] >= imin && iv[1] < -imin,
984 && iv[2] >= imin && iv[2] < -imin),
985 "SFCToken: index out of range");
986 uint32_t x = iv[0] - imin;
987 uint32_t y = iv[1] - imin;
988 uint32_t z = iv[2] - imin;
989 // extract lowest 10 bits and make space for interleaving
990 token.m_morton[0] = make_space(x & 0x3FF)
991 | (make_space(y & 0x3FF) << 1)
992 | (make_space(z & 0x3FF) << 2);
993 x = x >> 10;
994 y = y >> 10;
995 z = z >> 10;
996 token.m_morton[1] = make_space(x & 0x3FF)
997 | (make_space(y & 0x3FF) << 1)
998 | (make_space(z & 0x3FF) << 2);
999 x = x >> 10;
1000 y = y >> 10;
1001 z = z >> 10;
1002 token.m_morton[2] = make_space(x & 0x3FF)
1003 | (make_space(y & 0x3FF) << 1)
1004 | (make_space(z & 0x3FF) << 2);
1005
1006#elif (AMREX_SPACEDIM == 2)
1007
1008 constexpr uint32_t offset = 1u << 31;
1009 static_assert(static_cast<uint32_t>(std::numeric_limits<int>::max())+1 == offset,
1010 "INT_MAX != (1<<31)-1");
1011 uint32_t x = (iv[0] >= 0) ? static_cast<uint32_t>(iv[0]) + offset
1012 : static_cast<uint32_t>(iv[0]-std::numeric_limits<int>::lowest());
1013 uint32_t y = (iv[1] >= 0) ? static_cast<uint32_t>(iv[1]) + offset
1014 : static_cast<uint32_t>(iv[1]-std::numeric_limits<int>::lowest());
1015 // extract lowest 16 bits and make sapce for interleaving
1016 token.m_morton[0] = make_space(x & 0xFFFF)
1017 | (make_space(y & 0xFFFF) << 1);
1018 x = x >> 16;
1019 y = y >> 16;
1020 token.m_morton[1] = make_space(x) | (make_space(y) << 1);
1021
1022#elif (AMREX_SPACEDIM == 1)
1023
1024 constexpr uint32_t offset = 1u << 31;
1025 static_assert(static_cast<uint32_t>(std::numeric_limits<int>::max())+1 == offset,
1026 "INT_MAX != (1<<31)-1");
1027 token.m_morton[0] = (iv[0] >= 0) ? static_cast<uint32_t>(iv[0]) + offset
1028 : static_cast<uint32_t>(iv[0]-std::numeric_limits<int>::lowest());
1029
1030#else
1031 static_assert(false,"AMREX_SPACEDIM != 1, 2 or 3");
1032#endif
1033
1034 return token;
1035 }

1122void
1123DistributionMapping::SFCProcessorMapDoIt (const BoxArray& boxes,
1124 const std::vector<Long>& wgts,
1125 int /* nprocs */,
1126 bool sort,
1127 Real* eff)
1128{
1129 if (flag_verbose_mapper) {
1130 Print() << "DM: SFCProcessorMapDoIt called..." << std::endl;
1131 }
1132
1133 BL_PROFILE("DistributionMapping::SFCProcessorMapDoIt()");
1134
1135 int nprocs = ParallelContext::NProcsSub();
1136
1137 int nteams = nprocs;
1138 int nworkers = 1;
1139#if defined(BL_USE_TEAM)
1140 nteams = ParallelDescriptor::NTeams();
1141 nworkers = ParallelDescriptor::TeamSize();
1142#else
1143 if (node_size > 0) {
1144 nteams = nprocs/node_size;
1145 nworkers = node_size;
1146 if (nworkers*nteams != nprocs) {
1147 nteams = nprocs;
1148 nworkers = 1;
1149 }
1150 }
1151#endif
1152
1153 if (flag_verbose_mapper) {
1154 Print() << " (nprocs, nteams, nworkers) = ("
1155 << nprocs << ", " << nteams << ", " << nworkers << ")\n";
1156 }
1157
1158 const int N = boxes.size();
1159 std::vector<SFCToken> tokens;
1160 tokens.reserve(N);
1161 for (int i = 0; i < N; ++i)
1162 {
1163 const Box& bx = boxes[i];
1164 tokens.push_back(makeSFCToken(i, bx.smallEnd()));
1165 }
1166 //
1167 // Put'm in Morton space filling curve order.
1168 //
1169 std::sort(tokens.begin(), tokens.end(), SFCToken::Compare());
1170 //
1171 // Split'm up as equitably as possible per team.
1172 //
1173 Real volperteam = 0;
1174 for (Long wt : wgts) {
1175 volperteam += wt;
1176 }
1177 volperteam /= nteams;
1178
1179 std::vector< std::vector<int> > vec(nteams);
1180
1181 Distribute(tokens,wgts,nteams,volperteam,vec);
1182
1183 // vec has a size of nteams and vec[] holds a vector of box ids.
1184
1185 tokens.clear();
1186
1187 std::vector<LIpair> LIpairV;
1188
1189 LIpairV.reserve(nteams);
1190
1191 for (int i = 0; i < nteams; ++i)
1192 {
1193 Long wgt = 0;
1194 const std::vector<int>& vi = vec[i];
1195 for (int j = 0, M = vi.size(); j < M; ++j)
1196 wgt += wgts[vi[j]];
1197
1198 LIpairV.push_back(LIpair(wgt,i));
1199 }
1200
1201 if (sort) Sort(LIpairV, true);
1202
1203 if (flag_verbose_mapper) {
1204 for (const auto &p : LIpairV) {
1205 Print() << " Bucket " << p.second << " contains " << p.first << std::endl;
1206 }
1207 }
1208
1209 // LIpairV has a size of nteams and LIpairV[] is pair whose first is weight
1210 // and second is an index into vec. LIpairV is sorted by weight such that
1211 // LIpairV is the heaviest.
1212
1213 Vector<int> ord;
1214 Vector<Vector<int> > wrkerord;
1215
1216 if (nteams == nprocs) {
1217 if (sort) {
1218 LeastUsedCPUs(nprocs,ord);
1219 } else {
1220 ord.resize(nprocs);
1221 std::iota(ord.begin(), ord.end(), 0);
1222 }
1223 } else {
1224 if (sort) {
1225 LeastUsedTeams(ord,wrkerord,nteams,nworkers);
1226 } else {
1227 ord.resize(nteams);
1228 std::iota(ord.begin(), ord.end(), 0);
1229 wrkerord.resize(nteams);
1230 for (auto& v : wrkerord) {
1231 v.resize(nworkers);
1232 std::iota(v.begin(), v.end(), 0);
1233 }
1234 }
1235 }
1236
1237 // ord is a vector of process (or team) ids, sorted from least used to more heavily used.
1238 // wrkerord is a vector of sorted worker ids.
1239
1240 for (int i = 0; i < nteams; ++i)
1241 {
1242 const int tid = ord[i]; // tid is team id
1243 const int ivec = LIpairV[i].second; // index into vec
1244 const std::vector<int>& vi = vec[ivec]; // this vector contains boxes assigned to this team
1245 const int Nbx = vi.size(); // # of boxes assigned to this team
1246
1247 if (flag_verbose_mapper) {
1248 Print() << "Mapping bucket " << LIpairV[i].second << " to rank " << ord[i] << std::endl;
1249 }
1250
1251 if (nteams == nprocs) { // In this case, team id is process id.
1252 for (int j = 0; j < Nbx; ++j)
1253 {
1254 m_ref->m_pmap[vi[j]] = ParallelContext::local_to_global_rank(tid);
1255 }
1256 }
1257 else // We would like to do knapsack within the team workers
1258 {
1259 std::vector<Long> local_wgts;
1260 for (int j = 0; j < Nbx; ++j) {
1261 local_wgts.push_back(wgts[vi[j]]);
1262 }
1263
1264 std::vector<std::vector<int> > kpres;
1265 Real kpeff;
1266 knapsack(local_wgts, nworkers, kpres, kpeff, true, N);
1267
1268 // kpres has a size of nworkers. kpres[] contains a vector of indices into vi.
1269
1270 // sort the knapsacked chunks
1271 std::vector<LIpair> ww;
1272 for (int w = 0; w < nworkers; ++w) {
1273 Long wgt = 0;
1274 for (std::vector<int>::const_iterator it = kpres[w].begin();
1275 it != kpres[w].end(); ++it)
1276 {
1277 wgt += local_wgts[*it];
1278 }
1279 ww.push_back(LIpair(wgt,w));
1280 }
1281 Sort(ww,true);
1282
1283 // ww is a sorted vector of pair whose first is the weight and second is a index
1284 // into kpres.
1285
1286 const Vector<int>& sorted_workers = wrkerord[i];
1287
1288 const int leadrank = tid * nworkers;
1289
1290 for (int w = 0; w < nworkers; ++w)
1291 {
1292 const int cpu = leadrank + sorted_workers[w];
1293 int ikp = ww[w].second;
1294 const std::vector<int>& js = kpres[ikp];
1295 for (std::vector<int>::const_iterator it = js.begin(); it!=js.end(); ++it)
1296 m_ref->m_pmap[vi[*it]] = cpu;
1297 }
1298 }
1299 }
1300
1301 if (eff || verbose)
1302 {
1303 Real sum_wgt = 0, max_wgt = 0;
1304 for (int i = 0; i < nteams; ++i)
1305 {
1306 const Long W = LIpairV[i].first;
1307 if (W > max_wgt) max_wgt = W;
1308 sum_wgt += W;
1309 }
1310 Real efficiency = (sum_wgt/(nteams*max_wgt));
1311 if (eff) *eff = efficiency;
1312
1313 if (verbose)
1314 {
1315 amrex::Print() << "SFC efficiency: " << efficiency << '\n';
1316 }
1317 }

1551void
1552DistributionMapping::ComputeDistributionMappingEfficiency (const DistributionMapping& dm,
1553 const Vector<Real>& cost,
1554 Real* efficiency)
1555{
1556 const int nprocs = ParallelDescriptor::NProcs();
1557
1558 // This will store mapping from processor to the costs of FABs it controls,
1559 // (proc) --> ([cost_FAB_1, cost_FAB_2, ... ]),
1560 // for each proc
1561 Vector<Vector<Real>> rankToCosts(nprocs);
1562
1563 // Count the number of costs belonging to each rank
1564 Vector<int> cnt(nprocs);
1565 for (int i=0; i<dm.size(); ++i)
1566 {
1567 ++cnt[dm[i]];
1568 }
1569
1570 for (int i=0; i<rankToCosts.size(); ++i)
1571 {
1572 rankToCosts[i].reserve(cnt[i]);
1573 }
1574
1575 for (int i=0; i<cost.size(); ++i)
1576 {
1577 rankToCosts[dm[i]].push_back(cost[i]);
1578 }
1579
1580 Real maxCost = -1.0;
1581
1582 // This will store mapping from (proc) --> (sum of cost) for each proc
1583 Vector<Real> rankToCost(nprocs);
1584 for (int i=0; i<nprocs; ++i)
1585 {
1586 const Real rwSum = std::accumulate(rankToCosts[i].begin(),
1587 rankToCosts[i].end(), 0.0);
1588 rankToCost[i] = rwSum;
1589 maxCost = std::max(maxCost, rwSum);
1590 }
1591
1592 // Write `efficiency` (number between 0 and 1), the mean cost per processor
1593 // (normalized to the max cost)
1594 *efficiency = (std::accumulate(rankToCost.begin(),
1595 rankToCost.end(), 0.0) / (nprocs*maxCost));
1596}
