293 // it should be the direction in which the mesh spacing is much larger
294 // than in the other directions
295 // int idir = 1;
296
297 // This should be moved outside the kernel!
298 if (dhy <= dhx) amrex::Abort("dhy is supposed to be much larger than dhx");
299
300 int ilen = hi.y - lo.y + 1;

302 // This should be moved outside the kernel!
303 if (ilen > 32) amrex::Abort("abec_gsrb_with_line_solve is hard-wired to be no longer than 32");

307 Array1D<Real,0,31> c_ls;
308 Array1D<Real,0,31> r_ls;
309 Array1D<Real,0,31> u_ls;
310 Array1D<Real,0,31> gam;
311
312 for (int n = 0; n < nc; ++n) {
313 AMREX_PRAGMA_SIMD
314 for (int i = lo.x; i <= hi.x; ++i) {
315 if ((i+redblack)%2 == 0) {
316 for (int j = lo.y; j <= hi.y; ++j) {
317 Real gamma = alpha*a(i,j,0)
318 + dhx*(bX(i,j,0,n)+bX(i+1,j,0,n))
319 + dhy*(bY(i,j,0,n)+bY(i,j+1,0,n));
320
321 Real cf0 = (i == vlo.x and m0(vlo.x-1,j,0) > 0)
322 ? f0(vlo.x,j,0,n) : 0.0;
323 Real cf1 = (j == vlo.y and m1(i,vlo.y-1,0) > 0)
324 ? f1(i,vlo.y,0,n) : 0.0;
325 Real cf2 = (i == vhi.x and m2(vhi.x+1,j,0) > 0)
326 ? f2(vhi.x,j,0,n) : 0.0;
327 Real cf3 = (j == vhi.y and m3(i,vhi.y+1,0) > 0)
328 ? f3(i,vhi.y,0,n) : 0.0;
329
330 Real g_m_d = gamma
331 - (dhx*(bX(i,j,0,n)*cf0 + bX(i+1,j,0,n)*cf2)
332 + dhy*(bY(i,j,0,n)*cf1 + bY(i,j+1,0,n)*cf3));
333
334 Real rho = dhx*( bX(i ,j,0,n)*phi(i-1,j,0,n)
335 + bX(i+1,j,0,n)*phi(i+1,j,0,n) );
336
337 // We have already accounted for this external boundary in the coefficient of phi(i,j,k,n)
338 if (i == vlo.x and m0(vlo.x-1,j,0) > 0)
339 rho -= dhx*bX(i ,j,0,n)*phi(i-1,j,0,n);
340 if (i == vhi.x and m3(vhi.x+1,j,0) > 0)
341 rho -= dhx*bX(i+1,j,0,n)*phi(i+1,j,0,n);
342

344 b_ls(j-lo.y) = g_m_d;
345 c_ls(j-lo.y) = -dhy*bY(i,j+1,0,n);
346 u_ls(j-lo.y) = 0.;
347 r_ls(j-lo.y) = rhs(i,j,0,n) + rho;
348
349 if (j == lo.y) {
350 a_ls(j-lo.y) = 0.;
351 if (!(m1(i,vlo.y-1,0) > 0)) r_ls(j-lo.y) += dhy*bY(i,j,0,n)*phi(i,j-1,0,n);
352 }
353 if (j == hi.y) {
354 c_ls(j-lo.y) = 0.;

356 }
357 }
358// This is the tridiagonal solve
359 {
360 Real bet = b_ls(0);
361 u_ls(0) = r_ls(0) / bet;
362
363 for (int jj = 1; jj <= ilen-1; jj++) {
364 gam(jj) = c_ls(jj-1) / bet;
365 bet = b_ls(jj) - a_ls(jj)*gam(jj);
366 if (bet == 0) amrex::Abort(">>>TRIDIAG FAILED");
367 u_ls(jj) = (r_ls(jj)-a_ls(jj)*u_ls(jj-1)) / bet;
368 }
369
370 for (int jj = ilen-2; jj >= 0; jj--) {
371 u_ls(jj) = u_ls(jj) - gam(jj+1)*u_ls(jj+1);
372 }
373 }
374
375 for (int j = lo.y; j <= hi.y; ++j) {
376 phi(i,j,0,n) = u_ls(j-lo.y);
377 }
378 }
379 }
