161// CastroRadiation is the only code that uses this function.
162FabSet&
163FabSet::linComb (Real a, const MultiFab& mfa, int a_comp,
164 Real b, const MultiFab& mfb, int b_comp,
165 int dcomp, int ncomp, int ngrow)
166{
167 BL_PROFILE("FabSet::linComb()");
168 BL_ASSERT(ngrow <= mfa.nGrow());
169 BL_ASSERT(ngrow <= mfb.nGrow());
170 BL_ASSERT(mfa.boxArray() == mfb.boxArray());
171 BL_ASSERT(boxArray() != mfa.boxArray());
172
173 MultiFab bdrya(boxArray(),DistributionMap(),ncomp,0,MFInfo(),FArrayBoxFactory());
174 MultiFab bdryb(boxArray(),DistributionMap(),ncomp,0,MFInfo(),FArrayBoxFactory());
175
176#ifdef _OPENMP
177#pragma omp parallel if (Gpu::notInLaunchRegion())
178#endif
179 for (MFIter mfi(bdrya); mfi.isValid(); ++mfi) // tiling is not safe for this BoxArray
180 {
181 const Box& bx = mfi.validbox();
182 auto afab = bdrya.array(mfi);
183 auto bfab = bdryb.array(mfi);
184#ifdef AMREX_USE_FLOAT
185 const Real huge = 1.e30f;
186#else
187 const Real huge = 1.e200;
188#endif
189 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
190 {
191 afab(i,j,k,n) = huge;
192 bfab(i,j,k,n) = huge;
193 });
194 }
195
196 bdrya.copy(mfa,a_comp,0,ncomp,ngrow,0);
197 bdryb.copy(mfb,b_comp,0,ncomp,ngrow,0);
198
199#ifdef _OPENMP
200#pragma omp parallel if (Gpu::notInLaunchRegion())
201#endif
202 for (FabSetIter fsi(*this); fsi.isValid(); ++fsi)
203 {
204 const Box& bx = fsi.validbox();
205 auto const afab = bdrya.array(fsi);
206 auto const bfab = bdryb.array(fsi);
207 auto dfab = this->array(fsi);
208 AMREX_HOST_DEVICE_PARALLEL_FOR_4D ( bx, ncomp, i, j, k, n,
209 {
210 dfab(i,j,k,n+dcomp) = a*afab(i,j,k,n) + b*bfab(i,j,k,n);
211 });
212 }
