236#endif

238 derive_names.push_back(it->name());

261 //
262 for (i = 0; i < plot_var_map.size(); i++)
263 {
264 int typ = plot_var_map[i].first;
265 int comp = plot_var_map[i].second;
266 varnames.push_back(desc_lst[typ].name(comp));
267 this_dat = &state[typ].newData();
268 MultiFab::Copy(plotMF, *this_dat, comp, cnt, 1, nGrow);
269 cnt++;
270 }
271 //
272 // Cull data from derived variables.
273 //
274 if (derive_names.size() > 0)
275 {
276 for (std::list<DeriveRec>::const_iterator it = derive_lst.dlist().begin();
277 it != derive_lst.dlist().end(); ++it)
278 {
279 varnames.push_back(it->name());
280 const auto& derive_dat = derive(it->name(), cur_time, nGrow);
281 MultiFab::Copy(plotMF, *derive_dat, 0, cnt, 1, nGrow);
282 cnt++;
283 }
284 }
285
286 WriteSingleLevelPlotfileHDF5(dir_final,
287 plotMF, varnames,
288 Geom(), cur_time, nStep());
289// const std::string &versionName,
290// const std::string &levelPrefix,
291// const std::string &mfPrefix,
292// const Vector<std::string>& extra_dirs)
293 }

295#endif
