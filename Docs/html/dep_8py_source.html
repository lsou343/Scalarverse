3# automatically generate Makefile dependencies for Fortran 90 source.
4#
5# this will output all the dependency pairs amongst the source files.
6#
7# tips for debugging:
8#
9# -- do `make > out` when building your code. Any error messages
10# about modules not found will be sent to stderr, so you will
11# easily see them. Usually this is a case of the regex not
12# capturing how the module is defined
13#
14# -- add `--debug` to the `dep.py` commandline (e.g., in
15# `C_mk/Make.rules`). This will output a file called
16# `dependencies.out` that shows details about what files are
17# parsed, what modules they define, and what modules they require.
18#
19# -- a system-provided module won't be found in your file.
20# (e.g. iso_c_binding). Add any system-provided modules to the
21# `IGNORES` list below

44# regular expression for "{}module{}name", where {} can be any number
45# of spaces. We use 4 groups here, denoted by (), so the name of the
46# module is the 4th group
47module_re = re.compile(r"^(\s*)([Mm][Oo][Dd][Uu][Ll][Ee])(\s+)((?:[a-z][a-z_0-9]+))",
48 re.IGNORECASE|re.DOTALL)
49
50# regular expression for "{}module{}procedure{}name"
51module_proc_re = re.compile(r"(\s*)(module)(\s+)(procedure)(\s+)((?:[a-z][a-z_0-9]+))",
52 re.IGNORECASE|re.DOTALL)
53
54# regular expression for "{}use{}modulename...". Note this will work for
55# use modulename, only: stuff, other stuff'
56# see (txt2re.com)
57use_re = re.compile(r"^(\s*)([Uu][Ss][Ee])(\s+)((?:[a-z_][a-z_0-9]+))",
58 re.IGNORECASE|re.DOTALL)

121 def needed_modules(self):
122 """determine what modules this file needs. Assume only one use
123 statement per line. Ignore any only clauses. We work off
124 of the preprocessed file if it exists."""
125
126 depends = []

128 with io.open(self.search_namesearch_name(), "r", encoding="latin-1") as f:

136 rebreak = use_re.search(line)

222 # on its own; otherwise output the dependency line
