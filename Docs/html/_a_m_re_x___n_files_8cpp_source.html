362 } else { // ---- writing

396

399 fileStream.flush();

401 finishedWriting = true;
402
403 // ---- tell the decider we are done
404 ParallelDescriptor::Send(&myProc, 1, deciderProc, deciderTag);
405
406 // ---- wait to find out who will coordinate
407 ParallelDescriptor::Recv(&coordinatorProc, 1, deciderProc, coordinatorTag);
408
409 if(myProc == coordinatorProc) {
410 Vector<std::deque<int> > procsToWrite(nOutFiles); // ---- [fileNumber](procsToWriteToFileNumber)
411 // ---- populate with the static nfiles sets
412 for(int i(0); i < nProcs; ++i) {
413 int procSet(WhichSetPosition(i, nProcs, nOutFiles, groupSets));
414 int whichFileNumber(NFilesIter::FileNumber(nOutFiles, i, groupSets));
415 // ---- procSet == 0 have already written their data
416 if(procSet == 0) {
417 fileNumbersWriteOrder[whichFileNumber].push_back(i);
418 --remainingWriters;
419 }
420 if(procSet != 0) {
421 procsToWrite[whichFileNumber].push_back(i);
422 }
423 }
424
425 // ---- signal each remaining processor when to write and to which file
426 std::set<int> availableFileNumbers;
427 availableFileNumbers.insert(fileNumber); // ---- the coordinators file number
428
429 // ---- recv incoming available files
430 while(remainingWriters > 0) {
431
432 int nextProcToWrite(-1), nextFileNumberToWrite, nextFileNumberAvailable;
433 std::set<int>::iterator ait = availableFileNumbers.begin();
434 nextFileNumberToWrite = *ait;
435 availableFileNumbers.erase(nextFileNumberToWrite);
436
437 for(int nfn(0); nfn < procsToWrite.size(); ++nfn) {
438 // ---- start with the current next file number
439 // ---- get a proc from another file number if the queue is empty
440 int tempNFN((nextFileNumberToWrite + nfn) % procsToWrite.size());
441 if(procsToWrite[tempNFN].size() > 0) {
442 nextProcToWrite = procsToWrite[tempNFN].front();
443 procsToWrite[tempNFN].pop_front();
444 break; // ---- found one
445 }
446 }
447 if(nextProcToWrite == -1) {
448 --remainingWriters;
449// amrex::Print() << myProc << "::IOIOIOIO: nptw == -1 rW = " << remainingWriters << std::endl;
450 } else {
451
452 fileNumbersWriteOrder[nextFileNumberToWrite].push_back(nextProcToWrite);
453
454 ParallelDescriptor::Send(&nextFileNumberToWrite, 1, nextProcToWrite, writeTag);
455
456 rmess = ParallelDescriptor::Recv(&nextFileNumberAvailable, 1, MPI_ANY_SOURCE, doneTag);
457 availableFileNumbers.insert(nextFileNumberAvailable);
458 --remainingWriters;
459 }
460 }
461 unreadMessages.push_back(std::make_pair(doneTag, setZeroProcs.size() - 1));
462
463 } else { // ---- tell the coordinatorProc we are done writing
464 ParallelDescriptor::Send(&fileNumber, 1, coordinatorProc, doneTag);
465 }
466
467 }
468
469 if( ! finishedWriting) { // ---- the deciderProc drops through to here
470 fileStream.flush();
471 fileStream.close();
472 finishedWriting = true;
473
474 // ---- signal we are finished
475 ParallelDescriptor::Send(&fileNumber, 1, coordinatorProc, doneTag);
476 }
477
478 }
479
480 }
481
482#else
483 if(isReading) {
484 fileStream.close();
485 finishedReading = true;
486 } else { // ---- writing
487 fileStream.flush();
488 fileStream.close();
489 finishedWriting = true;
490 }
491#endif
492
493 return *this;
