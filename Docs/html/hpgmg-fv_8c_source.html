50void bench_hpgmg(mg_type *all_grids, int onLevel, double a, double b, double dtol, double rtol){
51 int doTiming;
52 int minSolves = 10; // do at least minSolves MGSolves
53 double timePerSolve = 0;
54
55 for(doTiming=0;doTiming<=1;doTiming++){ // first pass warms up, second pass times
56
57 #ifdef USE_HPM // IBM performance counters for BGQ...
58 if( (doTiming==1) && (onLevel==0) )HPM_Start("FMGSolve()");
59 #endif
60
61 #ifdef USE_MPI
62 double minTime = 60.0; // minimum time in seconds that the benchmark should run
63 double startTime = MPI_Wtime();
64 if(doTiming==1){
65 if((minTime/timePerSolve)>minSolves)minSolves=(minTime/timePerSolve); // if one needs to do more than minSolves to run for minTime, change minSolves
66 }
67 #endif
68
69 if(all_grids->levels[onLevel]->my_rank==0){
70 if(doTiming==0){fprintf(stdout,"\n\n===== Warming up by running %d solves ==========================================\n",minSolves);}
71 else{fprintf(stdout,"\n\n===== Running %d solves ========================================================\n",minSolves);}
72 fflush(stdout);
73 }
74
75 int numSolves = 0; // solves completed
76 MGResetTimers(all_grids);
77 while( (numSolves<minSolves) ){
78 zero_vector(all_grids->levels[onLevel],VECTOR_U);
79 #ifdef USE_FCYCLES
80 FMGSolve(all_grids,onLevel,VECTOR_U,VECTOR_F,a,b,dtol,rtol);
81 #else
82 MGSolve(all_grids,onLevel,VECTOR_U,VECTOR_F,a,b,dtol,rtol);
83 #endif
84 numSolves++;
85 }
86
87 #ifdef USE_MPI
88 if(doTiming==0){
89 double endTime = MPI_Wtime();
90 timePerSolve = (endTime-startTime)/numSolves;
91 MPI_Bcast(&timePerSolve,1,MPI_DOUBLE,0,MPI_COMM_WORLD); // after warmup, process 0 broadcasts the average time per solve (consensus)
92 }
93 #endif
94
95 #ifdef USE_HPM // IBM performance counters for BGQ...
96 if( (doTiming==1) && (onLevel==0) )HPM_Stop("FMGSolve()");
97 #endif
98 }
99}
