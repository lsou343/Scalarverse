141ParIterBase<is_const, NStructReal, NStructInt, NArrayReal, NArrayInt>::ParIterBase
142 (ContainerRef pc, int level, MFItInfo& info)
143 :
144 MFIter(*pc.m_dummy_mf[level], pc.do_tiling ? info.EnableTiling(pc.tile_size) : info),
145 m_level(level),
146 m_pariter_index(0),
147 m_pc(pc)
148{
149 auto& particles = pc.GetParticles(level);
150
151 int start = dynamic ? 0 : beginIndex;
152 for (int i = start; i < endIndex; ++i)
153 {
154 int grid = (*index_map)[i];
155 int tile = local_tile_index_map ? (*local_tile_index_map)[i] : 0;
156 auto key = std::make_pair(grid,tile);
157 auto f = particles.find(key);
158 if (f != particles.end() && f->second.numParticles() > 0)
159 {
160 m_valid_index.push_back(i);
161 m_particle_tiles.push_back(&(f->second));
162 }
163 }
164
165 if (m_valid_index.empty())
166 {
167 endIndex = beginIndex;
168 }
169 else
170 {
171 currentIndex = beginIndex = m_valid_index.front();
172 if (dynamic) {
173#ifdef _OPENMP
174 int ind = omp_get_thread_num();
175 m_pariter_index += ind;
176 if (ind < m_valid_index.size()) {
177 currentIndex = beginIndex = m_valid_index[ind];
178 } else {
179 currentIndex = endIndex;
180 }
181 for (int i = 0; i < omp_get_num_threads(); ++i) {
182 m_valid_index.push_back(endIndex);
183 }
184#endif
185 }
186 m_valid_index.push_back(endIndex);
187 }
