42IntVect
43getRefRatio(const Box& crse,
44 const Box& fine)
45{
46 // Compute refinement ratio between crse and fine boxes, return invalid
47 // IntVect if there is none suitable
48 IntVect ref_ratio;
49 for (int i=0; i<BL_SPACEDIM; ++i)
50 ref_ratio[i] = fine.length(i) / crse.length(i);
51
52 // Check results
53 Box test1 = ::Box(fine).coarsen(ref_ratio);
54 Box test2 = ::Box(test1).refine(ref_ratio);
55 if (test1 != crse || test2 != fine)
56 ref_ratio = IntVect();
57 return ref_ratio;
58}
