142 void forward(complex_t const *in) {
143
144 if (PlansMade != true) Error() << "Dfft buffers not set";
145 int nprocs;
146 MPI_Comm_size(MPI_COMM_WORLD, &nprocs);
147 if(ALLOW_MONOLITHIC && nprocs==1)
148 {
149 int64_t n = local_ng_rspace(0)*local_ng_rspace(1)*local_ng_rspace(2);
150 for(int64_t i=0;i<n;i++)
151 m_fs[i] = in[i];
152 fftw_execute(m_plan_f_monolithic);
153 return;
154 }
155
156#if DFFT_TIMING
157 double start, stop;
158 double tdist=0.0, tdfft=0.0;
159 start = MPI_Wtime();
160#endif
161
162 distribution_3_to_2(&in[0], &m_fs[0], &d.m_d, 0); // in --> fs
163
164#if DFFT_TIMING
165 stop = MPI_Wtime();
166 tdist += stop-start;
167#if DFFT_TIMING > 1
168 printTimingStats(d.parent_comm(), "DFFT fd32", stop-start);
169#endif
170 MPI_Barrier(d.parent_comm());
171 start = MPI_Wtime();
172#endif
173
174 fftw_execute(m_plan_f_x); // fs --> fo
175
176#if DFFT_TIMING
177 stop = MPI_Wtime();
178 tdfft += stop-start;
179#if DFFT_TIMING > 1
180 printTimingStats(d.parent_comm(), "DFFT f1dx", stop-start);
181#endif
182 MPI_Barrier(d.parent_comm());
183 start = MPI_Wtime();
184#endif
185
186 distribution_2_to_3(&m_fo[0], &m_fs[0], &d.m_d, 0); // fo --> fs
187
188#if DFFT_TIMING
189 stop = MPI_Wtime();
190 tdist += stop-start;
191#if DFFT_TIMING > 1
192 printTimingStats(d.parent_comm(), "DFFT fd23", stop-start);
193#endif
194 MPI_Barrier(d.parent_comm());
195 start = MPI_Wtime();
196#endif
197
198 distribution_3_to_2(&m_fs[0], &m_fo[0], &d.m_d, 1); // fs --> fo
199
200#if DFFT_TIMING
201 stop = MPI_Wtime();
202 tdist += stop-start;
203#if DFFT_TIMING > 1
204 printTimingStats(d.parent_comm(), "DFFT fd32", stop-start);
205#endif
206 MPI_Barrier(d.parent_comm());
207 start = MPI_Wtime();
208#endif
209
210 fftw_execute(m_plan_f_y); // fo --> fs
211
212#if DFFT_TIMING
213 stop = MPI_Wtime();
214 tdfft += stop-start;
215#if DFFT_TIMING > 1
216 printTimingStats(d.parent_comm(), "DFFT f1dy", stop-start);
217#endif
218 MPI_Barrier(d.parent_comm());
219 start = MPI_Wtime();
220#endif
221
222 distribution_2_to_3(&m_fs[0], &m_fo[0], &d.m_d, 1); // fs --> fo
223
224#if DFFT_TIMING
225 stop = MPI_Wtime();
226 tdist += stop-start;
227#if DFFT_TIMING > 1
228 printTimingStats(d.parent_comm(), "DFFT fd23", stop-start);
229#endif
230 MPI_Barrier(d.parent_comm());
231 start = MPI_Wtime();
232#endif
233
234 distribution_3_to_2(&m_fo[0], &m_fs[0], &d.m_d, 2); // fo --> fs
235
236#if DFFT_TIMING
237 stop = MPI_Wtime();
238 tdist += stop-start;
239#if DFFT_TIMING > 1
240 printTimingStats(d.parent_comm(), "DFFT fd32", stop-start);
241#endif
242 MPI_Barrier(d.parent_comm());
243 start = MPI_Wtime();
244#endif
245
246 fftw_execute(m_plan_f_z); // fs --> fo
247
248#if DFFT_TIMING
249 stop = MPI_Wtime();
250 tdfft += stop-start;
251#if DFFT_TIMING > 1
252 printTimingStats(d.parent_comm(), "DFFT f1dz", stop-start);
253#endif
254 printTimingStats(d.parent_comm(), "DFFT fdist", tdist);
255 printTimingStats(d.parent_comm(), "DFFT fdfft", tdfft);
256#endif

258 }

262 void forward(float const *in, size_t ghost0, size_t ghost1) {
263
264 if (PlansMade != true) Error() << "Dfft buffers not set";
265
266#if DFFT_TIMING
267 double start = MPI_Wtime();
268#endif
269
270 // copy from overloaded in to m_fo
271 const int *local_dim_r = local_ng_rspace();
272 int indexc = 0;
273 int indexf = ghost0*(ghost0 + local_dim_r[2] + ghost1)*(ghost0 + local_dim_r[1] + ghost1);
274 for(int local_r0=0; local_r0 < local_dim_r[0]; ++local_r0) {
275 indexf += ghost0*(ghost0 + local_dim_r[2] + ghost1);
276 for(int local_r1=0; local_r1 < local_dim_r[1]; ++local_r1) {
277 indexf += ghost0;
278 for(int local_r2=0; local_r2 < local_dim_r[2]; ++local_r2) {
279 m_fo[indexc] = in[indexf];
280 indexf++;
281 indexc++;
282 } // 2
283 indexf += ghost1;
284 } // 1
285 indexf += ghost1*(ghost0 + local_dim_r[2] + ghost1);
286 } // 0
287
288#if DFFT_TIMING
289 double stop = MPI_Wtime();
290 printTimingStats(d.parent_comm(), "DFFT fcopy", stop-start);
291#endif
292
293 // foward FFT from m_fo into m_fo
294 forward(&m_fo[0]);
295 }

425 void backward(float *out, size_t ghost0, size_t ghost1) {
426
427 // backward FFT from m_bi into m_bi
428 backward(&m_bi[0]);
429
430#if DFFT_TIMING
431 double start = MPI_Wtime();
432#endif
433
434 // copy from m_bi to overloaded out
435 const int *local_dim_r = local_ng_rspace();
436 int indexc = 0;
437 int indexf = ghost0*(ghost0 + local_dim_r[2] + ghost1)*(ghost0 + local_dim_r[1] + ghost1);
438 for(int local_r0=0; local_r0 < local_dim_r[0]; ++local_r0) {
439 indexf += ghost0*(ghost0 + local_dim_r[2] + ghost1);
440 for(int local_r1=0; local_r1 < local_dim_r[1]; ++local_r1) {
441 indexf += ghost0;
442 for(int local_r2=0; local_r2 < local_dim_r[2]; ++local_r2) {
443 out[indexf] = std::real(m_bi[indexc]);
444 indexf++;
445 indexc++;
446 } // 2
447 indexf += ghost1;
448 } // 1
449 indexf += ghost1*(ghost0 + local_dim_r[2] + ghost1);
450 } // 0
451
452#if DFFT_TIMING
453 double stop = MPI_Wtime();
454 printTimingStats(d.parent_comm(), "DFFT bcopy", stop-start);
455#endif
456
457 }

473 // arrays given at class initialization are hard-wired to the fft plans
474 Dfft(Distribution &dist,
475 complex_t *forward_output,
476 complex_t *forward_scratch,
477 complex_t *backward_input,
478 complex_t *backward_scratch,
479 unsigned int flags = FFTW_MEASURE)
480 : d(dist), PlansMade(false)
481 {
482 makePlans(forward_output,
483 forward_scratch,
484 backward_input,
485 backward_scratch,
486 flags);
487 }

489
490
491 // array rules:
492 // forward/backward method calls over-write these arrays
493 // forward method has separate input array argument
494 // backward method has separate output array argument
495 // forward_scratch != forward_output && backward_scratch != backward_input
496 // no required relationship between forward and backward arrays
497
498 // actually, can these transforms be done in-place?
499 // for now leave as-is, but test later
500
501 // FFTW_MEASURE = 0
502 void makePlans(complex_t *forward_output,
503 complex_t *forward_scratch,
504 complex_t *backward_input,
505 complex_t *backward_scratch,
506 unsigned int flags = FFTW_MEASURE)
507 {
508 if(forward_output == forward_scratch)
509 Error() << "Dfft::setBuffers() forward_output == forward_scratch";
510 if(backward_input == backward_scratch)
511 Error() << "Dfft::setBuffers() backward_input == backward_scratch";
512
513 m_fo = forward_output;
514 m_fs = forward_scratch;
515 m_bi = backward_input;
516 m_bs = backward_scratch;
517
518#if DFFT_TIMING
519 double start = MPI_Wtime();
520#endif
521
522#ifdef _OPENMP
523 fftw_init_threads();
524 fftw_plan_with_nthreads(omp_get_max_threads());
525#endif
526 int nprocs;
527 MPI_Comm_size(MPI_COMM_WORLD, &nprocs);
528 if(ALLOW_MONOLITHIC && nprocs==1)
529 {
530 m_plan_f_monolithic = fftw_plan_dft_3d(local_ng_rspace(0),local_ng_rspace(1),local_ng_rspace(2),
531 FFTW_ADDR(m_fs),FFTW_ADDR(m_fo),
532 FFTW_FORWARD, FFTW_MEASURE);
533 m_plan_b_monolithic = fftw_plan_dft_3d(local_ng_rspace(0),local_ng_rspace(1),local_ng_rspace(2),
534 FFTW_ADDR(m_bi),FFTW_ADDR(m_bs),
535 FFTW_BACKWARD, FFTW_MEASURE);
536 }
537 // fs --> fo
538 m_plan_f_x = fftw_plan_many_dft(1, // rank
539 &(d.m_d.process_topology_2_x.n[0]), // const int *n,
540 d.m_d.process_topology_2_x.n[1] * d.m_d.process_topology_2_x.n[2], // howmany
541 FFTW_ADDR(m_fs),
542 NULL, // const int *inembed,
543 1, // int istride,
544 d.m_d.process_topology_2_x.n[0], // int idist,
545 FFTW_ADDR(m_fo),
546 NULL, // const int *onembed,
547 1, // int ostride,
548 d.m_d.process_topology_2_x.n[0], // int odist,
549 FFTW_FORWARD, // int sign,
550 flags); // unsigned flags
551
552 // fo --> fs
553 m_plan_f_y = fftw_plan_many_dft(1, // rank
554 &(d.m_d.process_topology_2_y.n[1]), // const int *n,
555 d.m_d.process_topology_2_y.n[0] * d.m_d.process_topology_2_y.n[2], // howmany
556 FFTW_ADDR(m_fo),
557 NULL, // const int *inembed,
558 1, // int istride,
559 d.m_d.process_topology_2_y.n[1], // int idist,
560 FFTW_ADDR(m_fs),
561 NULL, // const int *onembed,
562 1, // int ostride,
563 d.m_d.process_topology_2_y.n[1], // int odist,
564 FFTW_FORWARD, // int sign,
565 flags); // unsigned flags
566
567 // fs --> fo
568 m_plan_f_z = fftw_plan_many_dft(1, // rank
569 &(d.m_d.process_topology_2_z.n[2]), // const int *n,
570 d.m_d.process_topology_2_z.n[1] * d.m_d.process_topology_2_z.n[0], // howmany
571 FFTW_ADDR(m_fs),
572 NULL, // const int *inembed,
573 1, // int istride,
574 d.m_d.process_topology_2_z.n[2], // int idist,
575 FFTW_ADDR(m_fo),
576 NULL, // const int *onembed,
577 1, // int ostride,
578 d.m_d.process_topology_2_z.n[2], // int odist,
579 FFTW_FORWARD, // int sign,
580 flags); // unsigned flags
581
582 // bi --> bs
583 m_plan_b_z = fftw_plan_many_dft(1, // rank
584 &(d.m_d.process_topology_2_z.n[2]), // const int *n,
585 d.m_d.process_topology_2_z.n[1] * d.m_d.process_topology_2_z.n[0], // howmany
586 FFTW_ADDR(m_bi),
587 NULL, // const int *inembed,
588 1, // int istride,
589 d.m_d.process_topology_2_z.n[2], // int idist,
590 FFTW_ADDR(m_bs),
591 NULL, // const int *onembed,
592 1, // int ostride,
593 d.m_d.process_topology_2_z.n[2], // int odist,
594 FFTW_BACKWARD, // int sign,
595 flags); // unsigned flags
596
597 // bs --> bi
598 m_plan_b_y = fftw_plan_many_dft(1, // rank
599 &(d.m_d.process_topology_2_y.n[1]), // const int *n,
600 d.m_d.process_topology_2_y.n[0] * d.m_d.process_topology_2_y.n[2], // howmany
601 FFTW_ADDR(m_bs),
602 NULL, // const int *inembed,
603 1, // int istride,
604 d.m_d.process_topology_2_y.n[1], // int idist,
605 FFTW_ADDR(m_bi),
606 NULL, // const int *onembed,
607 1, // int ostride,
608 d.m_d.process_topology_2_y.n[1], // int odist,
609 FFTW_BACKWARD, // int sign,
610 flags); // unsigned flags
611
612 // bi --> bs
613 m_plan_b_x = fftw_plan_many_dft(1, // rank
614 &(d.m_d.process_topology_2_x.n[0]), // const int *n,
615 d.m_d.process_topology_2_x.n[1] * d.m_d.process_topology_2_x.n[2], // howmany
616 FFTW_ADDR(m_bi),
617 NULL, // const int *inembed,
618 1, // int istride,
619 d.m_d.process_topology_2_x.n[0], // int idist,
620 FFTW_ADDR(m_bs),
621 NULL, // const int *onembed,
622 1, // int ostride,
623 d.m_d.process_topology_2_x.n[0], // int odist,
624 FFTW_BACKWARD, // int sign,
625 flags); // unsigned flags
626
627#if DFFT_TIMING
628 double stop = MPI_Wtime();
629 printTimingStats(d.parent_comm(), "DFFT init", stop-start);
630#endif
631
632 PlansMade = true;
633 }
