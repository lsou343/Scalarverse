91void build_geometry_and_grids (Vector<Geometry>& geom, Vector<BoxArray>& grids)
92{
93 ParmParse pp;
94 pp.query("n_cell", n_cell);
95 pp.query("max_level", max_level);
96 pp.query("max_grid_size", max_grid_size);
97 pp.query("ref_ratio", ref_ratio);
98 pp.query("boxes", boxes_file);
99
100 if (!boxes_file.empty())
101 {
102 Box dmn;
103 const BoxArray& ba = readBoxList(boxes_file, dmn);
104
105 const BoxArray& uncovered = amrex::complementIn(dmn, ba);
106
107 if (uncovered.empty() && dmn.isSquare() && dmn.smallEnd() == IntVect::TheZeroVector())
108 {
109 max_level = 0;
110 nlevels = max_level + 1;
111 n_cell = dmn.longside();
112
113 geom.resize(nlevels);
114 grids.resize(nlevels);
115
116 grids[0] = ba;
117 }
118 else
119 {
120 amrex::Print() << "Add a coarse level\n";
121 max_level = 1;
122 nlevels = max_level + 1;
123
124 AMREX_ALWAYS_ASSERT_WITH_MESSAGE(dmn.coarsenable(ref_ratio), "Domain must be coarsenable");

126 dmn.setSmall(IntVect::TheZeroVector());
127 n_cell = dmn.longside();
128
129 geom.resize(nlevels);
130 grids.resize(nlevels);
131
132 IntVect dom0_lo {IntVect::TheZeroVector()};
133 IntVect dom0_hi {AMREX_D_DECL(n_cell-1, n_cell-1, n_cell-1)};
134 Box dom0 {dom0_lo, dom0_hi};
135 grids[0] = BoxArray{dom0};
136 grids[0].maxSize(max_grid_size);
137
138 grids[1] = ba;
139 }
140 }
141 else
142 {
143 nlevels = max_level + 1;
144 geom.resize(nlevels);
145 grids.resize(nlevels);
146
147 IntVect dom0_lo {IntVect::TheZeroVector()};
148 IntVect dom0_hi {AMREX_D_DECL(n_cell-1, n_cell-1, n_cell-1)};

150 Box dom0 {dom0_lo, dom0_hi};
151 BoxArray ba0{dom0};
152
153 grids[0] = ba0;
154 grids[0].maxSize(max_grid_size);
155

168
169 const int coord = 0; // Cartesian coordinates

176 IntVect dom0_lo {IntVect::TheZeroVector()};
177 IntVect dom0_hi {AMREX_D_DECL(n_cell-1, n_cell-1, n_cell-1)};
178 Box dom0 {dom0_lo, dom0_hi};
179
180 geom[0].define(dom0, &real_box, coord, is_periodic.data());
181 for (int ilev=1, n=grids.size(); ilev < n; ++ilev)
182 {
183 dom0.refine(ref_ratio);
184 geom[ilev].define(dom0, &real_box, coord, is_periodic.data());

189readBoxList (const std::string& file, Box& domain)
190{
191 BoxList retval;
192
193 std::ifstream boxspec;
194
195 boxspec.open(file.c_str(), std::ios::in);
196
197 if( !boxspec )
198 {
199 std::string msg = "readBoxList: unable to open ";
200 msg += file;
201 amrex::Error(msg.c_str());
202 }
203 boxspec >> domain;
204
205 int numbox = 0;
206 boxspec >> numbox;
207
208 for ( int i=0; i<numbox; i++ )
209 {
210 Box tmpbox;
211 boxspec >> tmpbox;
212 if( !domain.contains(tmpbox) )
213 {
214 std::cerr << "readBoxList: bogus box " << tmpbox << '\n';
215 exit(1);
216 }
217 retval.push_back(tmpbox);
218 }
219
220 return BoxArray(retval);
221}
