321iMultiFab::sum (int comp, int nghost, bool local) const
322{
323 AMREX_ASSERT(nghost >= 0 && nghost <= n_grow.min());
324
325 Long sm = 0;
326
327#ifdef AMREX_USE_GPU
328 if (Gpu::inLaunchRegion())
329 {
330 ReduceOps<ReduceOpSum> reduce_op;
331 ReduceData<Long> reduce_data(reduce_op);
332 using ReduceTuple = typename decltype(reduce_data)::Type;
333
334 for (MFIter mfi(*this); mfi.isValid(); ++mfi)
335 {
336 const Box& bx = amrex::grow(mfi.validbox(),nghost);
337 const auto& arr = this->const_array(mfi);
338 reduce_op.eval(bx, reduce_data,
339 [=] AMREX_GPU_DEVICE (int i, int j, int k) -> ReduceTuple
340 {
341 return { static_cast<Long>(arr(i,j,k,comp)) };
342 });
343 }
344
345 ReduceTuple hv = reduce_data.value();
346 sm = amrex::get<0>(hv);
347 }
348 else
349#endif
350 {
351#ifdef _OPENMP
352#pragma omp parallel if (!system::regtest_reduction) reduction(+:sm)
353#endif
354 for (MFIter mfi(*this,true); mfi.isValid(); ++mfi)
355 {
356 const Box& bx = mfi.growntilebox(nghost);
357 Array4<int const> const& fab = this->const_array(mfi);
358 AMREX_LOOP_3D(bx, i, j, k,
359 {
360 sm += fab(i,j,k,comp);
361 });
362 }
363 }
364
365 if (!local) ParallelAllReduce::Sum(sm, ParallelContext::CommunicatorSub());
366
367 return sm;

527std::unique_ptr<iMultiFab>
528OwnerMask (FabArrayBase const& mf, const Periodicity& period, const IntVect& ngrow)
529{
530 BL_PROFILE("OwnerMask()");
531
532 const BoxArray& ba = mf.boxArray();
533 const DistributionMapping& dm = mf.DistributionMap();
534
535 const int owner = 1;
536 const int nonowner = 0;
537
538 std::unique_ptr<iMultiFab> p{new iMultiFab(ba,dm,1,ngrow, MFInfo(),
539 DefaultFabFactory<IArrayBox>())};
540 const std::vector<IntVect>& pshifts = period.shiftIntVect();
541
542 Vector<Array4BoxTag<int> > tags;
543
544 bool run_on_gpu = Gpu::inLaunchRegion();
545#ifdef _OPENMP
546#pragma omp parallel if (!run_on_gpu)
547#endif
548 {
549 std::vector< std::pair<int,Box> > isects;
550
551 for (MFIter mfi(*p); mfi.isValid(); ++mfi)
552 {
553 const Box& bx = (*p)[mfi].box();
554 auto arr = p->array(mfi);
555 const int idx = mfi.index();
556
557 AMREX_HOST_DEVICE_PARALLEL_FOR_3D(bx, i, j, k,
558 {
559 arr(i,j,k) = owner;
560 });
561
562 for (const auto& iv : pshifts)
563 {
564 ba.intersections(bx+iv, isects, false, ngrow);
565 for (const auto& is : isects)
566 {
567 const int oi = is.first;
568 const Box& obx = is.second-iv;
569 if ((oi < idx) || (oi == idx && iv < IntVect::TheZeroVector()))
570 {
571 if (run_on_gpu) {
572 tags.push_back({arr,obx});
573 } else {
574 // cannot use amrex::Loop because of a gcc bug.
575 const auto lo = amrex::lbound(obx);
576 const auto hi = amrex::ubound(obx);
577 for (int k = lo.z; k <= hi.z; ++k) {
578 for (int j = lo.y; j <= hi.y; ++j) {
579 AMREX_PRAGMA_SIMD
580 for (int i = lo.x; i <= hi.x; ++i) {
581 arr(i,j,k) = nonowner;
582 }}}
583 }
584 }
585 }
586 }
587 }
588 }
589
590#ifdef AMREX_USE_GPU
591 amrex::ParallelFor(tags, 1,
592 [=] AMREX_GPU_DEVICE (int i, int j, int k, int n, Array4BoxTag<int> const& tag) noexcept
593 {
594 tag.dfab(i,j,k,n) = nonowner;
595 });
596#endif
597
598 return p;
599}
