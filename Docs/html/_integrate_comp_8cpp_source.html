1/*
2 A very simple example of reading a plotfile and doing a simple analysis. Here, we want
3 to do a volume integral of a component specified by name.
4
5 The twist here is to demonstrate what this might look like if the amr data were coming down a
6 pipe (such as SENSEI, e.g.). So, we read the plotfile as usual, but mine it for the required
7 data, then put that data into a couple of structs that then are queried from that point forward.
8
9 The output is an array of numbers. */

29struct AMReXMeshHierarchy
30{
31/*
32 A minimal class to describe the AMR hierarchy for analysis routines
33 */
34public:
35 AMReXMeshHierarchy() {}
36 void define(const AmrData& ad) {
37 finestLevel = ad.FinestLevel();
38 int nlevs = finestLevel + 1;
39 ba.resize(nlevs);
40 probSize = ad.ProbSize();
41 probDomain = ad.ProbDomain();
42 refRatio = ad.RefRatio();
43 for (int lev=0; lev<nlevs; ++lev) {
44 ba[lev] = &ad.boxArray(lev);
45 }
46 }
47 int FinestLevel() const {return finestLevel;}
48 const BoxArray &boxArray(int level) const {return *ba[level];}
49 const Vector<int> &RefRatio() const {return refRatio;}
50 const Vector<Real> &ProbSize() const {return probSize;}
51 const Vector<Box> &ProbDomain() const {return probDomain;}
52
53protected:
54 int finestLevel;
55 std::vector<const BoxArray*> ba;
56 Vector<int> refRatio;
57 Vector<Real> probSize;
58 Vector<Box> probDomain;
59};

61struct AMReXDataHierarchy
62{
63/*
64 Data on a AMReXMeshHierarchy, currently pointing to MultiFabs of
65 named variables managed by an AmrData object.
66*/
67public:
68 AMReXDataHierarchy(AmrData& ad, const Vector<std::string>& varNames) {
69 mesh.define(ad);
70 const Vector<std::string>& plotVarNames = ad.PlotVarNames();
71 int nComp = varNames.size();
72 int nlevs = mesh.FinestLevel() + 1;
73 for (int i=0; i<nComp; ++i) {
74 int idx = -1;
75 for (int j=0; j<plotVarNames.size() && idx<0; ++j) {
76 if (plotVarNames[j] == varNames[i]) {idx = j;}
77 }
78 if (ParallelDescriptor::IOProcessor() && idx<0) {
79 Abort("Cannot find variable="+varNames[i]+" in pltfile");
80 }
81 std::vector<MultiFab*> mfs(nlevs);
82 for (int lev=0; lev<nlevs; ++lev) {
83 mfs[lev] = &ad.GetGrids(lev,idx); // Note: This lazily triggers a MultiFab read in the AmrData
84 }
85 varMap[varNames[i]] = std::make_pair(0,mfs);
86 }
87 }
88
89 MultiFab &GetGrids(int level, const std::string& name) {
90 if (varMap.find(name) == varMap.end()) {
91 Abort("Unknown component requested");
92 }
93 return *(varMap[name].second[level]);
94 }
95
96 const AMReXMeshHierarchy& Mesh() const {return mesh;}
97
98protected:
99 AMReXMeshHierarchy mesh;
100 std::map<std::string,std::pair<int,std::vector<MultiFab*>>> varMap;
101};

103int
104main (int argc,
105 char* argv[])
106{
107 amrex::Initialize(argc,argv);
108 {
109 if (argc < 2)
110 print_usage(argc,argv);
111
112 ParmParse pp;
113
114 if (pp.contains("help"))
115 print_usage(argc,argv);
116
117 // Create the AmrData object from a pltfile on disk
118 std::string infile; pp.get("infile",infile);
119 DataServices::SetBatchMode();
120 Amrvis::FileType fileType(Amrvis::NEWPLT);
121 DataServices dataServices(infile, fileType);
122 if( ! dataServices.AmrDataOk()) {
123 DataServices::Dispatch(DataServices::ExitRequest, NULL);
124 }
125 AmrData& amrData = dataServices.AmrDataRef();
126
127 int nv = pp.countval("varNames");
128 Vector<std::string> varNames(nv); pp.getarr("varNames",varNames,0,nv);
129
130 // Make a data struct for just the variables needed
131 AMReXDataHierarchy data(amrData,varNames);
132 const AMReXMeshHierarchy& mesh = data.Mesh();
133
134
135
136 // Compute the volume integrals
137 const int nGrow = 0;
138 const int finestLevel = mesh.FinestLevel();
139 const int nLev = finestLevel + 1;
140 const int nComp = varNames.size();
141
142 Vector<Real> integrals(nComp,0); // Results, initialized to zero
143 for (int lev=0; lev<nLev; ++lev) {
144 const BoxArray ba = mesh.boxArray(lev);
145
146 // Make boxes that are projection of finer ones (if exist)
147 const BoxArray baf = lev < finestLevel
148 ? BoxArray(mesh.boxArray(lev+1)).coarsen(mesh.RefRatio()[lev])
149 : BoxArray();
150
151 // Compute volume of a cell at this level
152 Real vol=1.0;
153 for (int d=0; d<AMREX_SPACEDIM; ++d) {
154 vol *= mesh.ProbSize()[d] / mesh.ProbDomain()[lev].length(d);
155 }
156
157 // For each component listed...
158 for (int n=0; n<nComp; ++n) {
159
160 // Make copy of original data because we will modify here
161 const MultiFab &pfData = data.GetGrids(lev,varNames[n]);
162 const DistributionMapping dmap(pfData.DistributionMap());
163 MultiFab mf(ba,dmap,1,nGrow);
164 MultiFab::Copy(mf,pfData,0,0,1,nGrow);
165
166 Real sm = 0;
167#ifdef _OPENMP
168#pragma omp parallel reduction(+:sm)
169#endif
170 for (MFIter mfi(mf,true); mfi.isValid(); ++mfi) {
171 FArrayBox& myFab = mf[mfi];
172 const Box& box = mfi.tilebox();
173
174 // Zero out covered cells
175 if (lev < finestLevel) {
176 std::vector< std::pair<int,Box> > isects = baf.intersections(box);
177 for (int ii = 0; ii < isects.size(); ii++) {
178 myFab.setVal(0,isects[ii].second,0,1);
179 }
180 }
181
182 // Get sum over tile, including zeroed covered cells
183 sm += myFab.sum(box,0,1);
184 }
185
186 // Accumulate to this ranks total
187 integrals[n] += sm * vol;
188
189 }
190 }
191
192 // Accumulate over ranks
193 ParallelDescriptor::ReduceRealSum(&(integrals[0]),nComp);
194
195 // Integrals to stdout
196 for (int n=0; n<nComp; ++n) {
197 Print() << integrals[n] << " ";
198 }
199 Print() << std::endl;
200 }
201 amrex::Finalize();
202 return 0;
203}
