75ByteOrder ()
76{
77 const int BigEndian = 1000;
78 const int SmallEndian = 0;
79
80 union
81 {
82 long m_Long;
83 char m_Char[sizeof(long)];
84 }
85 SwapTest;

88
89 return SwapTest.m_Char[0] == 1 ? SmallEndian : BigEndian;

92static
93void
94WriteFab (std::ostream& os,
95 const FArrayBox& fab,
96 const char* name)
97
98{
99 int nx = fab.box().length(0);
100 int ny = fab.box().length(1);
101 int dim = BL_SPACEDIM;
102
103
104#if (BL_SPACEDIM == 2)
105 for (int i = 0; i < nx; i++)
106 for (int j = 0; j < ny; j++)
107 {
108 int index = j*nx + i;
109 const Real * ptr = fab.dataPtr();
110 os.write((char*)(ptr+index),sizeof(Real));
111 }
112
113#elif (BL_SPACEDIM == 3)
114 int nz = fab.box().length(2);
115 os.write((char*)&nz,sizeof(int));
116
117 for (int k = 0; k < nz; k++)
118 for (int i = 0; i < nx; i++)
119 for (int j = 0; j < ny; j++)
120 {
121 int index = k*(nx*ny) + j*nx + i;
122 const Real * ptr = fab.dataPtr();
123 os.write((char*)(ptr+index),sizeof(Real));
124 }
125#endif
126}

145static
146void
147Write (AmrData& amrData,
148 const std::string& iFile_name,
149 const std::list<std::string> plot_vars)
150{
151 int finest_level = amrData.FinestLevel();
152
153 VisMF::IO_Buffer io_buffer(VisMF::IO_Buffer_Size);
154
155 char buf[128];
156
157 for (int icomp = 0; icomp < amrData.NComp(); icomp++)
158 {
159 //
160 // Write one component per file.
161 //
162 const std::string& CompName = amrData.PlotVarNames()[icomp];
163 if (isPlotVar(CompName))
164 {
165
166 std::string file = iFile_name;
167 file += '_';
168 file += CompName;
169 file += ".mat";
170
171 std::ofstream os;
172
173 os.rdbuf()->pubsetbuf(io_buffer.dataPtr(), io_buffer.size());
174
175 os.open(file.c_str(), std::ios::out|std::ios::binary);
176
177 if (os.fail()) {
178 amrex::FileOpenFailed(file);
179 }
180
181 int dim = BL_SPACEDIM;
182 os.write((char*)&dim,sizeof(int));
183
184 int num_levels = finest_level+1;
185 os.write((char*)&num_levels,sizeof(int));
186
187// Write the number of grids at each level
188 for (int iLevel = 0; iLevel <= finest_level; ++iLevel)
189 {
190 MultiFab& mf = amrData.GetGrids(iLevel,icomp);
191 const BoxArray& ba = mf.boxArray();
192 int num_grids = ba.size();
193 os.write((char*)&num_grids,sizeof(int));
194 }
195
196// Write the (Real) physical locations of each grid at each level
197 for (int iLevel = 0; iLevel <= finest_level; ++iLevel)
198 {
199 MultiFab& mf = amrData.GetGrids(iLevel,icomp);
200 const BoxArray& ba = mf.boxArray();
201 int num_grids = ba.size();
202 for (int i = 0; i < num_grids; ++i)
203 {
204 for (int idim = 0; idim < BL_SPACEDIM; ++idim)
205 {
206 Real xlo = amrData.GridLocLo()[iLevel][i][idim];
207 Real xhi = amrData.GridLocHi()[iLevel][i][idim];
208 os.write((char*)&xlo,sizeof(Real));
209 os.write((char*)&xhi,sizeof(Real));
210 }
211 }
212 }
213
214// Write the (integer) dimensions of each grid at each level
215 for (int iLevel = 0; iLevel <= finest_level; ++iLevel)
216 {
217 MultiFab& mf = amrData.GetGrids(iLevel,icomp);
218 const BoxArray& ba = mf.boxArray();
219 int num_grids = ba.size();
220 for (int i = 0; i < num_grids; ++i)
221 {
222 for (int idim = 0; idim < BL_SPACEDIM; ++idim)
223 {
224 int n = ba[i].length(idim);
225 os.write((char*)&n,sizeof(int));
226 }
227 }
228 }
229
230// Write the (Real) actual data of each grid at each level
231 for (int iLevel = 0; iLevel <= finest_level; ++iLevel)
232 {
233 MultiFab& mf = amrData.GetGrids(iLevel,icomp);
234 const BoxArray& ba = mf.boxArray();
235 int num_grids = ba.size();
236 for (int ig = 0; ig < num_grids; ++ig)
237 {
238 sprintf(buf,
239 "%s_%d_%d",
240 CompName.c_str(),
241 iLevel,
242 ig);
243 WriteFab(os, mf[ig], buf);
244 }
245 }
246
247 os.close();
248
249 if (os.fail()) {
250 amrex::FileOpenFailed(file);
251 }
252 }
253 }
254}

256int
257main (int argc,
258 char* argv[])
259{
260 amrex::Initialize(argc,argv);
261 {
262 if (argc == 1)
263 PrintUsage(argv[0]);
264
265 if (ParallelDescriptor::NProcs() > 1) {
266 amrex::Error("This is an inherently serial program");
267 }
268
269 ParmParse pp;
270
271 if (pp.contains("help"))
272 PrintUsage(argv[0]);
273 //
274 // MatLab expects native floating-point format.
275 //
276 FArrayBox::setFormat(FABio::FAB_NATIVE);
277 //
278 // Scan the arguments.
279 //
280 std::string iFile;
281 pp.query("infile", iFile);
282 if (iFile.empty()) {
283 amrex::Abort("You must specify `infile'");
284 }
285
286 DataServices::SetBatchMode();
287 Amrvis::FileType fileType(Amrvis::NEWPLT);
288
289 DataServices dataServices(iFile, fileType);
290
291 if (!dataServices.AmrDataOk())
292 {
293 //
294 // This calls ParallelDescriptor::EndParallel() and exit()
295 //
296 DataServices::Dispatch(DataServices::ExitRequest, NULL);
297 }
298 AmrData& amrData = dataServices.AmrDataRef();
299
300 std::string plot_var;
301 pp.query("plot_vars", plot_var);
302
303 int npv = pp.countval("plot_vars");
304
305 if (npv == 0) {
306 for (int i = 0; i < amrData.PlotVarNames().size(); i++)
307 {
308 plot_vars.push_back(amrData.PlotVarNames()[i]);
309 }
310
311 } else {
312
313 for (int i = 0; i < npv; i++)
314 {
315 pp.get("plot_vars", plot_var, i);
316 plot_vars.push_back(plot_var);
317 }
318
319 }
320
321 Write(amrData, iFile, plot_vars);
322 }
323 Finalize();
324}
